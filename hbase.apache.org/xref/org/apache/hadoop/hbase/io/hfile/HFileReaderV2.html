84    /**
85     * Opens a HFile. You must load the index before you can use it by calling
86     * {@link #loadFileInfo()}.
87     *
88     * @param path Path to HFile.
89     * @param trailer File trailer.
90     * @param fsdis input stream. Caller is responsible for closing the passed
91     *          stream.
92     * @param size Length of the stream.
93     * @param closeIStream Whether to close the stream.
94     * @param cacheConf Cache configuration.
95     * @param preferredEncodingInCache the encoding to use in cache in case we
96     *          have a choice. If the file is already encoded on disk, we will
97     *          still use its on-disk encoding in cache.
98     */
99    public HFileReaderV2(Path path, FixedFileTrailer trailer,
100       final FSDataInputStream fsdis, final FSDataInputStream fsdisNoFsChecksum,
101       final long size,
102       final boolean closeIStream, final CacheConfig cacheConf,
103       DataBlockEncoding preferredEncodingInCache, final HFileSystem hfs)
104       throws IOException {
105     super(path, trailer, fsdis, fsdisNoFsChecksum, size, 
106           closeIStream, cacheConf, hfs);
107     trailer.expectMajorVersion(2);
108     validateMinorVersion(path, trailer.getMinorVersion());
109     HFileBlock.FSReaderV2 fsBlockReaderV2 = new HFileBlock.FSReaderV2(fsdis,
110         fsdisNoFsChecksum,
111         compressAlgo, fileSize, trailer.getMinorVersion(), hfs, path);
112     this.fsBlockReader = fsBlockReaderV2; // upcast
113 
114     // Comparator class name is stored in the trailer in version 2.
115     comparator = trailer.createComparator();
116     dataBlockIndexReader = new HFileBlockIndex.BlockIndexReader(comparator,
117         trailer.getNumDataIndexLevels(), this);
118     metaBlockIndexReader = new HFileBlockIndex.BlockIndexReader(
119         Bytes.BYTES_RAWCOMPARATOR, 1);
120 
121     // Parse load-on-open data.
122 
123     HFileBlock.BlockIterator blockIter = fsBlockReaderV2.blockRange(
124         trailer.getLoadOnOpenDataOffset(),
125         fileSize - trailer.getTrailerSize());
126 
127     // Data index. We also read statistics about the block index written after
128     // the root level.
129     dataBlockIndexReader.readMultiLevelIndexRoot(
130         blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
131         trailer.getDataIndexCount());
132 
133     // Meta index.
134     metaBlockIndexReader.readRootIndex(
135         blockIter.nextBlockWithBlockType(BlockType.ROOT_INDEX),
136         trailer.getMetaIndexCount());
137 
138     // File info
139     fileInfo = new FileInfo();
140     fileInfo.read(blockIter.nextBlockWithBlockType(BlockType.FILE_INFO).getByteStream());
141     lastKey = fileInfo.get(FileInfo.LASTKEY);
142     avgKeyLen = Bytes.toInt(fileInfo.get(FileInfo.AVG_KEY_LEN));
143     avgValueLen = Bytes.toInt(fileInfo.get(FileInfo.AVG_VALUE_LEN));
144     byte [] keyValueFormatVersion =
145         fileInfo.get(HFileWriterV2.KEY_VALUE_VERSION);
146     includesMemstoreTS = keyValueFormatVersion != null &&
147         Bytes.toInt(keyValueFormatVersion) ==
148             HFileWriterV2.KEY_VALUE_VER_WITH_MEMSTORE;
149     fsBlockReaderV2.setIncludesMemstoreTS(includesMemstoreTS);
150 
151     // Read data block encoding algorithm name from file info.
152     dataBlockEncoder = HFileDataBlockEncoderImpl.createFromFileInfo(fileInfo,
153         preferredEncodingInCache);
154     fsBlockReaderV2.setDataBlockEncoder(dataBlockEncoder);
155 
156     // Store all other load-on-open blocks for further consumption.
157     HFileBlock b;
158     while ((b = blockIter.nextBlock()) != null) {
159       loadOnOpenBlocks.add(b);
160     }
161   }

163   /**
164    * Create a Scanner on this file. No seeks or reads are done on creation. Call
165    * {@link HFileScanner#seekTo(byte[])} to position an start the read. There is
166    * nothing to clean up in a Scanner. Letting go of your references to the
167    * scanner is sufficient.
168    *
169    * @param cacheBlocks True if we should cache blocks read in by this scanner.
170    * @param pread Use positional read rather than seek+read if true (pread is
171    *          better for random reads, seek+read is better scanning).
172    * @param isCompaction is scanner being used for a compaction?
173    * @return Scanner on this file.
174    */
175    @Override
176    public HFileScanner getScanner(boolean cacheBlocks, final boolean pread,
177       final boolean isCompaction) {
178     // check if we want to use data block encoding in memory
179     if (dataBlockEncoder.useEncodedScanner(isCompaction)) {
180       return new EncodedScannerV2(this, cacheBlocks, pread, isCompaction,
181           includesMemstoreTS);
182     }
183 
184     return new ScannerV2(this, cacheBlocks, pread, isCompaction);
185   }

187   /**
188    * @param metaBlockName
189    * @param cacheBlock Add block to cache, if found
190    * @return block wrapped in a ByteBuffer, with header skipped
191    * @throws IOException
192    */
193   @Override
194   public ByteBuffer getMetaBlock(String metaBlockName, boolean cacheBlock)
195       throws IOException {
196     if (trailer.getMetaIndexCount() == 0) {
197       return null; // there are no meta blocks
198     }
199     if (metaBlockIndexReader == null) {
200       throw new IOException("Meta index not loaded");
201     }
202 
203     byte[] mbname = Bytes.toBytes(metaBlockName);
204     int block = metaBlockIndexReader.rootBlockContainingKey(mbname, 0,
205         mbname.length);
206     if (block == -1)
207       return null;
208     long blockSize = metaBlockIndexReader.getRootBlockDataSize(block);
209     long startTimeNs = System.nanoTime();
210 
211     // Per meta key from any given file, synchronize reads for said block. This
212     // is OK to do for meta blocks because the meta block index is always
213     // single-level.
214     synchronized (metaBlockIndexReader.getRootBlockKey(block)) {
215       // Check cache for block. If found return.
216       long metaBlockOffset = metaBlockIndexReader.getRootBlockOffset(block);
217       BlockCacheKey cacheKey = new BlockCacheKey(name, metaBlockOffset,
218           DataBlockEncoding.NONE, BlockType.META);
219 
220       cacheBlock &= cacheConf.shouldCacheDataOnRead();
221       if (cacheConf.isBlockCacheEnabled()) {
222         HFileBlock cachedBlock =
223           (HFileBlock) cacheConf.getBlockCache().getBlock(cacheKey, cacheBlock, false);
224         if (cachedBlock != null) {
225           // Return a distinct 'shallow copy' of the block,
226           // so pos does not get messed by the scanner
227           return cachedBlock.getBufferWithoutHeader();
228         }
229         // Cache Miss, please load.
230       }
231 
232       HFileBlock metaBlock = fsBlockReader.readBlockData(metaBlockOffset,
233           blockSize, -1, true);
234 
235       final long delta = System.nanoTime() - startTimeNs;
236       HFile.offerReadLatency(delta, true);
237 
238       // Cache the block
239       if (cacheBlock) {
240         cacheConf.getBlockCache().cacheBlock(cacheKey, metaBlock,
241             cacheConf.isInMemory());
242       }
243 
244       return metaBlock.getBufferWithoutHeader();
245     }
246   }

248   /**
249    * Read in a file block.
250    * @param dataBlockOffset offset to read.
251    * @param onDiskBlockSize size of the block
252    * @param cacheBlock
253    * @param pread Use positional read instead of seek+read (positional is
254    *          better doing random reads whereas seek+read is better scanning).
255    * @param isCompaction is this block being read as part of a compaction
256    * @param expectedBlockType the block type we are expecting to read with this
257    *          read operation, or null to read whatever block type is available
258    *          and avoid checking (that might reduce caching efficiency of
259    *          encoded data blocks)
260    * @return Block wrapped in a ByteBuffer.
261    * @throws IOException
262    */
263   @Override
264   public HFileBlock readBlock(long dataBlockOffset, long onDiskBlockSize,
265       final boolean cacheBlock, boolean pread, final boolean isCompaction,
266       BlockType expectedBlockType)
267       throws IOException {
268     if (dataBlockIndexReader == null) {
269       throw new IOException("Block index not loaded");
270     }
271     if (dataBlockOffset < 0
272         || dataBlockOffset >= trailer.getLoadOnOpenDataOffset()) {
273       throw new IOException("Requested block is out of range: "
274           + dataBlockOffset + ", lastDataBlockOffset: "
275           + trailer.getLastDataBlockOffset());
276     }
277     // For any given block from any given file, synchronize reads for said
278     // block.
279     // Without a cache, this synchronizing is needless overhead, but really
280     // the other choice is to duplicate work (which the cache would prevent you
281     // from doing).
282 
283     BlockCacheKey cacheKey =
284         new BlockCacheKey(name, dataBlockOffset,
285             dataBlockEncoder.getEffectiveEncodingInCache(isCompaction),
286             expectedBlockType);
287 
288     boolean useLock = false;
289     IdLock.Entry lockEntry = null;
290     try {
291       while (true) {
292 
293         if (useLock) {
294           lockEntry = offsetLock.getLockEntry(dataBlockOffset);
295         }
296 
297         // Check cache for block. If found return.
298         if (cacheConf.isBlockCacheEnabled()) {
299           // Try and get the block from the block cache. If the useLock variable is true then this
300           // is the second time through the loop and it should not be counted as a block cache miss.
301           HFileBlock cachedBlock = (HFileBlock) cacheConf.getBlockCache().getBlock(cacheKey,
302               cacheBlock, useLock);
303           if (cachedBlock != null) {
304             if (cachedBlock.getBlockType() == BlockType.DATA) {
305               HFile.dataBlockReadCnt.incrementAndGet();
306             }
307 
308             validateBlockType(cachedBlock, expectedBlockType);
309 
310             // Validate encoding type for encoded blocks. We include encoding
311             // type in the cache key, and we expect it to match on a cache hit.
312             if (cachedBlock.getBlockType() == BlockType.ENCODED_DATA
313                 && cachedBlock.getDataBlockEncoding() != dataBlockEncoder.getEncodingInCache()) {
314               throw new IOException("Cached block under key " + cacheKey + " "
315                   + "has wrong encoding: " + cachedBlock.getDataBlockEncoding() + " (expected: "
316                   + dataBlockEncoder.getEncodingInCache() + ")");
317             }
318             return cachedBlock;
319           }
320           // Carry on, please load.
321         }
322         if (!useLock) {
323           // check cache again with lock
324           useLock = true;
325           continue;
326         }
327 
328         // Load block from filesystem.
329         long startTimeNs = System.nanoTime();
330         HFileBlock hfileBlock = fsBlockReader.readBlockData(dataBlockOffset, onDiskBlockSize, -1,
331             pread);
332         hfileBlock = dataBlockEncoder.diskToCacheFormat(hfileBlock, isCompaction);
333         validateBlockType(hfileBlock, expectedBlockType);
334 
335         final long delta = System.nanoTime() - startTimeNs;
336         HFile.offerReadLatency(delta, pread);
337 
338         // Cache the block if necessary
339         if (cacheBlock && cacheConf.shouldCacheBlockOnRead(hfileBlock.getBlockType().getCategory())) {
340           cacheConf.getBlockCache().cacheBlock(cacheKey, hfileBlock, cacheConf.isInMemory());
341         }
342 
343         if (hfileBlock.getBlockType() == BlockType.DATA) {
344           HFile.dataBlockReadCnt.incrementAndGet();
345         }
346 
347         return hfileBlock;
348       }
349     } finally {
350       if (lockEntry != null) {
351         offsetLock.releaseLockEntry(lockEntry);
352       }
353     }
354   }

356   /**
357    * Compares the actual type of a block retrieved from cache or disk with its
358    * expected type and throws an exception in case of a mismatch. Expected
359    * block type of {@link BlockType#DATA} is considered to match the actual
360    * block type [@link {@link BlockType#ENCODED_DATA} as well.
361    * @param block a block retrieved from cache or disk
362    * @param expectedBlockType the expected block type, or null to skip the
363    *          check
364    */
365   private void validateBlockType(HFileBlock block,
366       BlockType expectedBlockType) throws IOException {
367     if (expectedBlockType == null) {
368       return;
369     }
370     BlockType actualBlockType = block.getBlockType();
371     if (actualBlockType == BlockType.ENCODED_DATA &&
372         expectedBlockType == BlockType.DATA) {
373       // We consider DATA to match ENCODED_DATA for the purpose of this
374       // verification.
375       return;
376     }
377     if (actualBlockType != expectedBlockType) {
378       throw new IOException("Expected block type " + expectedBlockType + ", " +
379           "but got " + actualBlockType + ": " + block);
380     }
381   }

408   public void close(boolean evictOnClose) throws IOException {
409     if (evictOnClose && cacheConf.isBlockCacheEnabled()) {
410       int numEvicted = cacheConf.getBlockCache().evictBlocksByHfileName(name);
411       if (LOG.isTraceEnabled()) {
412         LOG.trace("On close, file=" + name + " evicted=" + numEvicted
413           + " block(s)");
414       }
415     }
416     if (closeIStream) {
417       if (istream != istreamNoFsChecksum && istreamNoFsChecksum != null) {
418         istreamNoFsChecksum.close();
419         istreamNoFsChecksum = null;
420       }
421       if (istream != null) {
422         istream.close();
423         istream = null;
424       }
425     }

428   protected abstract static class AbstractScannerV2
429       extends AbstractHFileReader.Scanner {
430     protected HFileBlock block;
431 
432     /**
433      * The next indexed key is to keep track of the indexed key of the next data block.
434      * If the nextIndexedKey is HConstants.NO_NEXT_INDEXED_KEY, it means that the
435      * current data block is the last data block.
436      *
437      * If the nextIndexedKey is null, it means the nextIndexedKey has not been loaded yet.
438      */
439     protected byte[] nextIndexedKey;
440 
441     public AbstractScannerV2(HFileReaderV2 r, boolean cacheBlocks,
442         final boolean pread, final boolean isCompaction) {
443       super(r, cacheBlocks, pread, isCompaction);
444     }
445 
446     /**
447      * An internal API function. Seek to the given key, optionally rewinding to
448      * the first key of the block before doing the seek.
449      *
450      * @param key key byte array
451      * @param offset key offset in the key byte array
452      * @param length key length
453      * @param rewind whether to rewind to the first key of the block before
454      *        doing the seek. If this is false, we are assuming we never go
455      *        back, otherwise the result is undefined.
456      * @return -1 if the key is earlier than the first key of the file,
457      *         0 if we are at the given key, and 1 if we are past the given key
458      * @throws IOException
459      */
460     protected int seekTo(byte[] key, int offset, int length, boolean rewind)
461         throws IOException {
462       HFileBlockIndex.BlockIndexReader indexReader =
463           reader.getDataBlockIndexReader();
464       BlockWithScanInfo blockWithScanInfo =
465         indexReader.loadDataBlockWithScanInfo(key, offset, length, block,
466             cacheBlocks, pread, isCompaction);
467       if (blockWithScanInfo == null || blockWithScanInfo.getHFileBlock() == null) {
468         // This happens if the key e.g. falls before the beginning of the file.
469         return -1;
470       }
471       return loadBlockAndSeekToKey(blockWithScanInfo.getHFileBlock(),
472           blockWithScanInfo.getNextIndexedKey(), rewind, key, offset, length, false);
473     }
474 
475     protected abstract ByteBuffer getFirstKeyInBlock(HFileBlock curBlock);
476 
477     protected abstract int loadBlockAndSeekToKey(HFileBlock seekToBlock, byte[] nextIndexedKey,
478         boolean rewind, byte[] key, int offset, int length, boolean seekBefore)
479         throws IOException;
480 
481     @Override
482     public int seekTo(byte[] key, int offset, int length) throws IOException {
483       // Always rewind to the first key of the block, because the given key
484       // might be before or after the current key.
485       return seekTo(key, offset, length, true);
486     }
487 
488     @Override
489     public int reseekTo(byte[] key, int offset, int length) throws IOException {
490       int compared;
491       if (isSeeked()) {
492         ByteBuffer bb = getKey();
493         compared = reader.getComparator().compare(key, offset,
494             length, bb.array(), bb.arrayOffset(), bb.limit());
495         if (compared < 1) {
496           // If the required key is less than or equal to current key, then
497           // don't do anything.
498           return compared;
499         } else {
500           if (this.nextIndexedKey != null &&
501               (this.nextIndexedKey == HConstants.NO_NEXT_INDEXED_KEY ||
502                reader.getComparator().compare(key, offset, length,
503                    nextIndexedKey, 0, nextIndexedKey.length) < 0)) {
504             // The reader shall continue to scan the current data block instead of querying the
505             // block index as long as it knows the target key is strictly smaller than
506             // the next indexed key or the current data block is the last data block.
507             return loadBlockAndSeekToKey(this.block, this.nextIndexedKey,
508                 false, key, offset, length, false);
509           }
510         }
511       }
512       // Don't rewind on a reseek operation, because reseek implies that we are
513       // always going forward in the file.
514       return seekTo(key, offset, length, false);
515     }
516 
517     @Override
518     public boolean seekBefore(byte[] key, int offset, int length)
519         throws IOException {
520       HFileBlock seekToBlock =
521           reader.getDataBlockIndexReader().seekToDataBlock(key, offset, length,
522               block, cacheBlocks, pread, isCompaction);
523       if (seekToBlock == null) {
524         return false;
525       }
526       ByteBuffer firstKey = getFirstKeyInBlock(seekToBlock);
527 
528       if (reader.getComparator().compare(firstKey.array(),
529           firstKey.arrayOffset(), firstKey.limit(), key, offset, length) == 0)
530       {
531         long previousBlockOffset = seekToBlock.getPrevBlockOffset();
532         // The key we are interested in
533         if (previousBlockOffset == -1) {
534           // we have a 'problem', the key we want is the first of the file.
535           return false;
536         }
537 
538         // It is important that we compute and pass onDiskSize to the block
539         // reader so that it does not have to read the header separately to
540         // figure out the size.
541         seekToBlock = reader.readBlock(previousBlockOffset,
542             seekToBlock.getOffset() - previousBlockOffset, cacheBlocks,
543             pread, isCompaction, BlockType.DATA);
544         // TODO shortcut: seek forward in this block to the last key of the
545         // block.
546       }
547       byte[] firstKeyInCurrentBlock = Bytes.getBytes(firstKey);
548       loadBlockAndSeekToKey(seekToBlock, firstKeyInCurrentBlock, true, key, offset, length, true);
549       return true;
550     }

553     /**
554      * Scans blocks in the "scanned" section of the {@link HFile} until the next
555      * data block is found.
556      *
557      * @return the next block, or null if there are no more data blocks
558      * @throws IOException
559      */
560     protected HFileBlock readNextDataBlock() throws IOException {
561       long lastDataBlockOffset = reader.getTrailer().getLastDataBlockOffset();
562       if (block == null)
563         return null;
564 
565       HFileBlock curBlock = block;
566 
567       do {
568         if (curBlock.getOffset() >= lastDataBlockOffset)
569           return null;
570 
571         if (curBlock.getOffset() < 0) {
572           throw new IOException("Invalid block file offset: " + block);
573         }
574 
575         // We are reading the next block without block type validation, because
576         // it might turn out to be a non-data block.
577         curBlock = reader.readBlock(curBlock.getOffset()
578             + curBlock.getOnDiskSizeWithHeader(),
579             curBlock.getNextBlockOnDiskSizeWithHeader(), cacheBlocks, pread,
580             isCompaction, null);
581       } while (!(curBlock.getBlockType().equals(BlockType.DATA) ||
582           curBlock.getBlockType().equals(BlockType.ENCODED_DATA)));
583 
584       return curBlock;
585     }
586   }

588   /**
589    * Implementation of {@link HFileScanner} interface.
590    */
591   protected static class ScannerV2 extends AbstractScannerV2 {
592     private HFileReaderV2 reader;
593 
594     public ScannerV2(HFileReaderV2 r, boolean cacheBlocks,
595         final boolean pread, final boolean isCompaction) {
596       super(r, cacheBlocks, pread, isCompaction);
597       this.reader = r;
598     }
599 
600     @Override
601     public KeyValue getKeyValue() {
602       if (!isSeeked())
603         return null;
604 
605       KeyValue ret = new KeyValue(blockBuffer.array(),
606           blockBuffer.arrayOffset() + blockBuffer.position(),
607           KEY_VALUE_LEN_SIZE + currKeyLen + currValueLen,
608           currKeyLen);
609       if (this.reader.shouldIncludeMemstoreTS()) {
610         ret.setMemstoreTS(currMemstoreTS);
611       }
612       return ret;
613     }
614 
615     @Override
616     public ByteBuffer getKey() {
617       assertSeeked();
618       return ByteBuffer.wrap(
619           blockBuffer.array(),
620           blockBuffer.arrayOffset() + blockBuffer.position()
621               + KEY_VALUE_LEN_SIZE, currKeyLen).slice();
622     }
623 
624     @Override
625     public ByteBuffer getValue() {
626       assertSeeked();
627       return ByteBuffer.wrap(
628           blockBuffer.array(),
629           blockBuffer.arrayOffset() + blockBuffer.position()
630               + KEY_VALUE_LEN_SIZE + currKeyLen, currValueLen).slice();
631     }
632 
633     private void setNonSeekedState() {
634       block = null;
635       blockBuffer = null;
636       currKeyLen = 0;
637       currValueLen = 0;
638       currMemstoreTS = 0;
639       currMemstoreTSLen = 0;
640     }

642     /**
643      * Go to the next key/value in the block section. Loads the next block if
644      * necessary. If successful, {@link #getKey()} and {@link #getValue()} can
645      * be called.
646      *
647      * @return true if successfully navigated to the next key/value
648      */
649     @Override
650     public boolean next() throws IOException {
651       assertSeeked();
652 
653       try {
654         blockBuffer.position(blockBuffer.position() + KEY_VALUE_LEN_SIZE
655             + currKeyLen + currValueLen + currMemstoreTSLen);
656       } catch (IllegalArgumentException e) {
657         LOG.error("Current pos = " + blockBuffer.position()
658             + "; currKeyLen = " + currKeyLen + "; currValLen = "
659             + currValueLen + "; block limit = " + blockBuffer.limit()
660             + "; HFile name = " + reader.getName()
661             + "; currBlock currBlockOffset = " + block.getOffset());
662         throw e;
663       }
664 
665       if (blockBuffer.remaining() <= 0) {
666         long lastDataBlockOffset =
667             reader.getTrailer().getLastDataBlockOffset();
668 
669         if (block.getOffset() >= lastDataBlockOffset) {
670           setNonSeekedState();
671           return false;
672         }
673 
674         // read the next block
675         HFileBlock nextBlock = readNextDataBlock();
676         if (nextBlock == null) {
677           setNonSeekedState();
678           return false;
679         }
680 
681         updateCurrBlock(nextBlock);
682         return true;
683       }
684 
685       // We are still in the same block.
686       readKeyValueLen();
687       return true;
688     }

690     /**
691      * Positions this scanner at the start of the file.
692      *
693      * @return false if empty file; i.e. a call to next would return false and
694      *         the current key and value are undefined.
695      * @throws IOException
696      */
697     @Override
698     public boolean seekTo() throws IOException {
699       if (reader == null) {
700         return false;
701       }
702 
703       if (reader.getTrailer().getEntryCount() == 0) {
704         // No data blocks.
705         return false;
706       }
707 
708       long firstDataBlockOffset =
709           reader.getTrailer().getFirstDataBlockOffset();
710       if (block != null && block.getOffset() == firstDataBlockOffset) {
711         blockBuffer.rewind();
712         readKeyValueLen();
713         return true;
714       }
715 
716       block = reader.readBlock(firstDataBlockOffset, -1, cacheBlocks, pread,
717           isCompaction, BlockType.DATA);
718       if (block.getOffset() < 0) {
719         throw new IOException("Invalid block offset: " + block.getOffset());
720       }
721       updateCurrBlock(block);
722       return true;
723     }

741      * Updates the current block to be the given {@link HFileBlock}. Seeks to
742      * the the first key/value pair.
743      *
744      * @param newBlock the block to make current
745      */

747       block = newBlock;
748 
749       // sanity check
750       if (block.getBlockType() != BlockType.DATA) {
751         throw new IllegalStateException("ScannerV2 works only on data " +
752             "blocks, got " + block.getBlockType() + "; " +
753             "fileName=" + reader.name + ", " +

755             "isCompaction=" + isCompaction);
756       }

763       this.nextIndexedKey = null;
764     }

766     private final void readKeyValueLen() {
767       blockBuffer.mark();
768       currKeyLen = blockBuffer.getInt();
769       currValueLen = blockBuffer.getInt();
770       blockBuffer.reset();
771       if (this.reader.shouldIncludeMemstoreTS()) {
772         try {
773           int memstoreTSOffset = blockBuffer.arrayOffset()
774               + blockBuffer.position() + KEY_VALUE_LEN_SIZE + currKeyLen
775               + currValueLen;
776           currMemstoreTS = Bytes.readVLong(blockBuffer.array(),
777               memstoreTSOffset);
778           currMemstoreTSLen = WritableUtils.getVIntSize(currMemstoreTS);
779         } catch (Exception e) {
780           throw new RuntimeException("Error reading memstore timestamp", e);
781         }
782       }
783 
784       if (currKeyLen < 0 || currValueLen < 0
785           || currKeyLen > blockBuffer.limit()
786           || currValueLen > blockBuffer.limit()) {
787         throw new IllegalStateException("Invalid currKeyLen " + currKeyLen
788             + " or currValueLen " + currValueLen + ". Block offset: "
789             + block.getOffset() + ", block length: " + blockBuffer.limit()
790             + ", position: " + blockBuffer.position() + " (without header).");
791       }
792     }

794     /**
795      * Within a loaded block, seek looking for the last key that is smaller
796      * than (or equal to?) the key we are interested in.
797      *
798      * A note on the seekBefore: if you have seekBefore = true, AND the first
799      * key in the block = key, then you'll get thrown exceptions. The caller has
800      * to check for that case and load the previous block as appropriate.
801      *
802      * @param key the key to find
803      * @param seekBefore find the key before the given key in case of exact
804      *          match.
805      * @return 0 in case of an exact key match, 1 in case of an inexact match
806      */
807     private int blockSeek(byte[] key, int offset, int length,
808         boolean seekBefore) {
809       int klen, vlen;
810       long memstoreTS = 0;
811       int memstoreTSLen = 0;
812       int lastKeyValueSize = -1;
813       do {
814         blockBuffer.mark();
815         klen = blockBuffer.getInt();
816         vlen = blockBuffer.getInt();
817         blockBuffer.reset();
818         if (this.reader.shouldIncludeMemstoreTS()) {
819           try {
820             int memstoreTSOffset = blockBuffer.arrayOffset()
821                 + blockBuffer.position() + KEY_VALUE_LEN_SIZE + klen + vlen;
822             memstoreTS = Bytes.readVLong(blockBuffer.array(),
823                 memstoreTSOffset);
824             memstoreTSLen = WritableUtils.getVIntSize(memstoreTS);
825           } catch (Exception e) {
826             throw new RuntimeException("Error reading memstore timestamp", e);
827           }
828         }
829 
830         int keyOffset = blockBuffer.arrayOffset() + blockBuffer.position()
831             + KEY_VALUE_LEN_SIZE;
832         int comp = reader.getComparator().compare(key, offset, length,
833             blockBuffer.array(), keyOffset, klen);
834 
835         if (comp == 0) {
836           if (seekBefore) {
837             if (lastKeyValueSize < 0) {
838               throw new IllegalStateException("blockSeek with seekBefore "
839                   + "at the first key of the block: key="
840                   + Bytes.toStringBinary(key) + ", blockOffset="
841                   + block.getOffset() + ", onDiskSize="
842                   + block.getOnDiskSizeWithHeader());
843             }
844             blockBuffer.position(blockBuffer.position() - lastKeyValueSize);
845             readKeyValueLen();
846             return 1; // non exact match.
847           }
848           currKeyLen = klen;
849           currValueLen = vlen;
850           if (this.reader.shouldIncludeMemstoreTS()) {
851             currMemstoreTS = memstoreTS;
852             currMemstoreTSLen = memstoreTSLen;
853           }
854           return 0; // indicate exact match
855         }
856 
857         if (comp < 0) {
858           if (lastKeyValueSize > 0)
859             blockBuffer.position(blockBuffer.position() - lastKeyValueSize);
860           readKeyValueLen();
861           return 1;
862         }
863 
864         // The size of this key/value tuple, including key/value length fields.
865         lastKeyValueSize = klen + vlen + memstoreTSLen + KEY_VALUE_LEN_SIZE;
866         blockBuffer.position(blockBuffer.position() + lastKeyValueSize);
867       } while (blockBuffer.remaining() > 0);
868 
869       // Seek to the last key we successfully read. This will happen if this is
870       // the last key/value pair in the file, in which case the following call
871       // to next() has to return false.
872       blockBuffer.position(blockBuffer.position() - lastKeyValueSize);
873       readKeyValueLen();
874       return 1; // didn't exactly find it.
875     }
