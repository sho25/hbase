60  /**
61   * Executes region split as a "transaction".  Call {@link #prepare()} to setup
62   * the transaction, {@link #execute(Server, RegionServerServices)} to run the
63   * transaction and {@link #rollback(Server, RegionServerServices)} to cleanup if execute fails.
64   *
65   * <p>Here is an example of how you would use this class:
66   * <pre>
67   *  SplitTransaction st = new SplitTransaction(this.conf, parent, midKey)
68   *  if (!st.prepare()) return;
69   *  try {
70   *    st.execute(server, services);
71   *  } catch (IOException ioe) {
72   *    try {
73   *      st.rollback(server, services);
74   *      return;
75   *    } catch (RuntimeException e) {
76   *      myAbortable.abort("Failed split, abort");
77   *    }
78   *  }
79   * </Pre>
80   * <p>This class is not thread safe.  Caller needs ensure split is run by
81   * one thread only.
82   */
83  @InterfaceAudience.Private
84  public class SplitTransaction {
85    private static final Log LOG = LogFactory.getLog(SplitTransaction.class);
86    private static final String SPLITDIR = ".splits";

103   /**
104    * Types to add to the transaction journal.
105    * Each enum is a step in the split transaction. Used to figure how much
106    * we need to rollback.
107    */
108   enum JournalEntry {
109     /**
110      * Set region as in transition, set it into SPLITTING state.
111      */
112     SET_SPLITTING_IN_ZK,
113     /**
114      * We created the temporary split data directory.
115      */
116     CREATE_SPLIT_DIR,
117     /**
118      * Closed the parent region.
119      */
120     CLOSED_PARENT_REGION,
121     /**
122      * The parent has been taken out of the server's online regions list.
123      */
124     OFFLINED_PARENT,
125     /**
126      * Started in on creation of the first daughter region.
127      */
128     STARTED_REGION_A_CREATION,
129     /**
130      * Started in on the creation of the second daughter region.
131      */
132     STARTED_REGION_B_CREATION,
133     /**
134      * Point of no return.
135      * If we got here, then transaction is not recoverable other than by
136      * crashing out the regionserver.
137      */
138     PONR
139   }

157   /**
158    * Does checks on split inputs.
159    * @return <code>true</code> if the region is splittable else
160    * <code>false</code> if it is not (e.g. its already closed, etc.).
161    */
162   public boolean prepare() {
163     if (!this.parent.isSplittable()) return false;
164     // Split key can be null if this region is unsplittable; i.e. has refs.
165     if (this.splitrow == null) return false;
166     HRegionInfo hri = this.parent.getRegionInfo();
167     parent.prepareToSplit();
168     // Check splitrow.
169     byte [] startKey = hri.getStartKey();
170     byte [] endKey = hri.getEndKey();
171     if (Bytes.equals(startKey, splitrow) ||
172         !this.parent.getRegionInfo().containsRow(splitrow)) {
173       LOG.info("Split row is not inside region key range or is equal to " +
174           "startkey: " + Bytes.toStringBinary(this.splitrow));
175       return false;
176     }
177     long rid = getDaughterRegionIdTimestamp(hri);
178     this.hri_a = new HRegionInfo(hri.getTableName(), startKey, this.splitrow,
179       false, rid);
180     this.hri_b = new HRegionInfo(hri.getTableName(), this.splitrow, endKey,
181       false, rid);
182     return true;
183   }
184 
185   /**
186    * Calculate daughter regionid to use.
187    * @param hri Parent {@link HRegionInfo}
188    * @return Daughter region id (timestamp) to use.
189    */
190   private static long getDaughterRegionIdTimestamp(final HRegionInfo hri) {
191     long rid = EnvironmentEdgeManager.currentTimeMillis();
192     // Regionid is timestamp.  Can't be less than that of parent else will insert
193     // at wrong location in .META. (See HBASE-710).
194     if (rid < hri.getRegionId()) {
195       LOG.warn("Clock skew; parent regions id is " + hri.getRegionId() +
196         " but current time here is " + rid);
197       rid = hri.getRegionId() + 1;
198     }
199     return rid;
200   }

203       "Failed to close region: already closed by another thread");

205   /**
206    * Prepare the regions and region files.
207    * @param server Hosting server instance.  Can be null when testing (won't try
208    * and update in zk if a null server)
209    * @param services Used to online/offline regions.
210    * @throws IOException If thrown, transaction failed. Call {@link #rollback(Server, RegionServerServices)}
211    * @return Regions created
212    */
213   /* package */PairOfSameType<HRegion> createDaughters(final Server server,
214       final RegionServerServices services) throws IOException {
215     LOG.info("Starting split of region " + this.parent);
216     if ((server != null && server.isStopped()) ||
217         (services != null && services.isStopping())) {
218       throw new IOException("Server is stopped or stopping");
219     }
220     assert !this.parent.lock.writeLock().isHeldByCurrentThread(): "Unsafe to hold write lock while performing RPCs";
221 
222     // Coprocessor callback
223     if (this.parent.getCoprocessorHost() != null) {
224       this.parent.getCoprocessorHost().preSplit();
225     }
226     
227     // Coprocessor callback
228     if (this.parent.getCoprocessorHost() != null) {
229       this.parent.getCoprocessorHost().preSplit(this.splitrow);
230     }
231 
232     // If true, no cluster to write meta edits to or to update znodes in.
233     boolean testing = server == null? true:
234         server.getConfiguration().getBoolean("hbase.testing.nocluster", false);
235     this.fileSplitTimeout = testing ? this.fileSplitTimeout :
236         server.getConfiguration().getLong("hbase.regionserver.fileSplitTimeout",
237           this.fileSplitTimeout);
238 
239     // Set ephemeral SPLITTING znode up in zk.  Mocked servers sometimes don't
240     // have zookeeper so don't do zk stuff if server or zookeeper is null
241     if (server != null && server.getZooKeeper() != null) {
242       try {
243         createNodeSplitting(server.getZooKeeper(),
244           this.parent.getRegionInfo(), server.getServerName());
245       } catch (KeeperException e) {
246         throw new IOException("Failed creating SPLITTING znode on " +
247           this.parent.getRegionNameAsString(), e);
248       }
249     }
250     this.journal.add(JournalEntry.SET_SPLITTING_IN_ZK);
251     if (server != null && server.getZooKeeper() != null) {
252       try {
253         // Transition node from SPLITTING to SPLITTING after creating the split node.
254         // Master will get the callback for node change only if the transition is successful.
255         // Note that if the transition fails then the rollback will delete the created znode
256         // as the journal entry SET_SPLITTING_IN_ZK is added.
257         // TODO : May be we can add some new state to znode and handle the new state incase of success/failure
258         this.znodeVersion = transitionNodeSplitting(server.getZooKeeper(),
259             this.parent.getRegionInfo(), server.getServerName(), -1);
260       } catch (KeeperException e) {
261         throw new IOException("Failed setting SPLITTING znode on "
262             + this.parent.getRegionNameAsString(), e);
263       }
264     }
265 
266     createSplitDir(this.parent.getFilesystem(), this.splitdir);
267     this.journal.add(JournalEntry.CREATE_SPLIT_DIR);
268 
269     List<StoreFile> hstoreFilesToSplit = null;
270     Exception exceptionToThrow = null;
271     try{
272       hstoreFilesToSplit = this.parent.close(false);
273     } catch (Exception e) {
274       exceptionToThrow = e;
275     }
276     if (exceptionToThrow == null && hstoreFilesToSplit == null) {
277       // The region was closed by a concurrent thread.  We can't continue
278       // with the split, instead we must just abandon the split.  If we
279       // reopen or split this could cause problems because the region has
280       // probably already been moved to a different server, or is in the
281       // process of moving to a different server.
282       exceptionToThrow = closedByOtherException;
283     }
284     if (exceptionToThrow != closedByOtherException) {
285       this.journal.add(JournalEntry.CLOSED_PARENT_REGION);
286     }
287     if (exceptionToThrow != null) {
288       if (exceptionToThrow instanceof IOException) throw (IOException)exceptionToThrow;
289       throw new IOException(exceptionToThrow);
290     }
291 
292         
293     if (hstoreFilesToSplit.size() == 0) {
294       String errorMsg = "No store files to split for the region "+this.parent.getRegionInfo();
295       LOG.error(errorMsg);
296       throw new IOException(errorMsg);
297     }
298     if (!testing) {
299       services.removeFromOnlineRegions(this.parent.getRegionInfo().getEncodedName(), null);
300     }
301     this.journal.add(JournalEntry.OFFLINED_PARENT);
302 
303     // TODO: If splitStoreFiles were multithreaded would we complete steps in
304     // less elapsed time?  St.Ack 20100920
305     //
306     // splitStoreFiles creates daughter region dirs under the parent splits dir
307     // Nothing to unroll here if failure -- clean up of CREATE_SPLIT_DIR will
308     // clean this up.
309     splitStoreFiles(this.splitdir, hstoreFilesToSplit);
310 
311     // Log to the journal that we are creating region A, the first daughter
312     // region.  We could fail halfway through.  If we do, we could have left
313     // stuff in fs that needs cleanup -- a storefile or two.  Thats why we
314     // add entry to journal BEFORE rather than AFTER the change.
315     this.journal.add(JournalEntry.STARTED_REGION_A_CREATION);
316     HRegion a = createDaughterRegion(this.hri_a, this.parent.rsServices);
317 
318     // Ditto
319     this.journal.add(JournalEntry.STARTED_REGION_B_CREATION);
320     HRegion b = createDaughterRegion(this.hri_b, this.parent.rsServices);
321 
322     // This is the point of no return.  Adding subsequent edits to .META. as we
323     // do below when we do the daughter opens adding each to .META. can fail in
324     // various interesting ways the most interesting of which is a timeout
325     // BUT the edits all go through (See HBASE-3872).  IF we reach the PONR
326     // then subsequent failures need to crash out this regionserver; the
327     // server shutdown processing should be able to fix-up the incomplete split.
328     // The offlined parent will have the daughters as extra columns.  If
329     // we leave the daughter regions in place and do not remove them when we
330     // crash out, then they will have their references to the parent in place
331     // still and the server shutdown fixup of .META. will point to these
332     // regions.
333     // We should add PONR JournalEntry before offlineParentInMeta,so even if
334     // OfflineParentInMeta timeout,this will cause regionserver exit,and then
335     // master ServerShutdownHandler will fix daughter & avoid data loss. (See
336     // HBase-4562).
337     this.journal.add(JournalEntry.PONR);
338 
339     // Edit parent in meta.  Offlines parent region and adds splita and splitb.
340     if (!testing) {
341       MetaEditor.offlineParentInMeta(server.getCatalogTracker(),
342         this.parent.getRegionInfo(), a.getRegionInfo(), b.getRegionInfo());
343     }
344     return new PairOfSameType<HRegion>(a, b);
345   }

347   /**
348    * Perform time consuming opening of the daughter regions.
349    * @param server Hosting server instance.  Can be null when testing (won't try
350    * and update in zk if a null server)
351    * @param services Used to online/offline regions.
352    * @param a first daughter region
353    * @param a second daughter region
354    * @throws IOException If thrown, transaction failed. Call {@link #rollback(Server, RegionServerServices)}
355    */
356   /* package */void openDaughters(final Server server,
357       final RegionServerServices services, HRegion a, HRegion b)
358       throws IOException {
359     boolean stopped = server != null && server.isStopped();
360     boolean stopping = services != null && services.isStopping();
361     // TODO: Is this check needed here?
362     if (stopped || stopping) {
363       LOG.info("Not opening daughters " +
364           b.getRegionInfo().getRegionNameAsString() +
365           " and " +
366           a.getRegionInfo().getRegionNameAsString() +
367           " because stopping=" + stopping + ", stopped=" + stopped);
368     } else {
369       // Open daughters in parallel.
370       DaughterOpener aOpener = new DaughterOpener(server, a);
371       DaughterOpener bOpener = new DaughterOpener(server, b);
372       aOpener.start();
373       bOpener.start();
374       try {
375         aOpener.join();
376         bOpener.join();
377       } catch (InterruptedException e) {
378         Thread.currentThread().interrupt();
379         throw new IOException("Interrupted " + e.getMessage());
380       }
381       if (aOpener.getException() != null) {
382         throw new IOException("Failed " +
383           aOpener.getName(), aOpener.getException());
384       }
385       if (bOpener.getException() != null) {
386         throw new IOException("Failed " +
387           bOpener.getName(), bOpener.getException());
388       }
389       if (services != null) {
390         try {
391           // add 2nd daughter first (see HBASE-4335)
392           services.postOpenDeployTasks(b, server.getCatalogTracker(), true);
393           // Should add it to OnlineRegions
394           services.addToOnlineRegions(b);
395           services.postOpenDeployTasks(a, server.getCatalogTracker(), true);
396           services.addToOnlineRegions(a);
397         } catch (KeeperException ke) {
398           throw new IOException(ke);
399         }
400       }
401     }
402   }

404   /**
405    * Finish off split transaction, transition the zknode
406    * @param server Hosting server instance.  Can be null when testing (won't try
407    * and update in zk if a null server)
408    * @param services Used to online/offline regions.
409    * @param a first daughter region
410    * @param a second daughter region
411    * @throws IOException If thrown, transaction failed. Call {@link #rollback(Server, RegionServerServices)}
412    */
413   /* package */void transitionZKNode(final Server server,
414       final RegionServerServices services, HRegion a, HRegion b)
415       throws IOException {
416     // Tell master about split by updating zk.  If we fail, abort.
417     if (server != null && server.getZooKeeper() != null) {
418       try {
419         this.znodeVersion = transitionNodeSplit(server.getZooKeeper(),
420           parent.getRegionInfo(), a.getRegionInfo(), b.getRegionInfo(),
421           server.getServerName(), this.znodeVersion);
422 
423         int spins = 0;
424         // Now wait for the master to process the split. We know it's done
425         // when the znode is deleted. The reason we keep tickling the znode is
426         // that it's possible for the master to miss an event.
427         do {
428           if (spins % 10 == 0) {
429             LOG.debug("Still waiting on the master to process the split for " +
430                 this.parent.getRegionInfo().getEncodedName());
431           }
432           Thread.sleep(100);
433           // When this returns -1 it means the znode doesn't exist
434           this.znodeVersion = tickleNodeSplit(server.getZooKeeper(),
435             parent.getRegionInfo(), a.getRegionInfo(), b.getRegionInfo(),
436             server.getServerName(), this.znodeVersion);
437           spins++;
438         } while (this.znodeVersion != -1 && !server.isStopped()
439             && !services.isStopping());
440       } catch (Exception e) {
441         if (e instanceof InterruptedException) {
442           Thread.currentThread().interrupt();
443         }
444         throw new IOException("Failed telling master about split", e);
445       }
446     }
447 
448     // Coprocessor callback
449     if (this.parent.getCoprocessorHost() != null) {
450       this.parent.getCoprocessorHost().postSplit(a,b);
451     }
452 
453     // Leaving here, the splitdir with its dross will be in place but since the
454     // split was successful, just leave it; it'll be cleaned when parent is
455     // deleted and cleaned up.
456   }

458   /**
459    * Run the transaction.
460    * @param server Hosting server instance.  Can be null when testing (won't try
461    * and update in zk if a null server)
462    * @param services Used to online/offline regions.
463    * @throws IOException If thrown, transaction failed. Call {@link #rollback(Server, RegionServerServices)}
464    * @return Regions created
465    * @throws IOException
466    * @see #rollback(Server, RegionServerServices)
467    */
468   public PairOfSameType<HRegion> execute(final Server server,
469       final RegionServerServices services)
470   throws IOException {
471     PairOfSameType<HRegion> regions = createDaughters(server, services);
472     openDaughters(server, services, regions.getFirst(), regions.getSecond());
473     transitionZKNode(server, services, regions.getFirst(), regions.getSecond());
474     return regions;
475   }

477   /*
478    * Open daughter region in its own thread.
479    * If we fail, abort this hosting server.
480    */
481   class DaughterOpener extends HasThread {
482     private final Server server;
483     private final HRegion r;
484     private Throwable t = null;
485 
486     DaughterOpener(final Server s, final HRegion r) {
487       super((s == null? "null-services": s.getServerName()) +
488         "-daughterOpener=" + r.getRegionInfo().getEncodedName());
489       setDaemon(true);
490       this.server = s;
491       this.r = r;
492     }
493 
494     /**
495      * @return Null if open succeeded else exception that causes us fail open.
496      * Call it after this thread exits else you may get wrong view on result.
497      */
498     Throwable getException() {
499       return this.t;
500     }
501 
502     @Override
503     public void run() {
504       try {
505         openDaughterRegion(this.server, r);
506       } catch (Throwable t) {
507         this.t = t;
508       }
509     }
510   }

513    * Open daughter regions, add them to online list and update meta.

519   void openDaughterRegion(final Server server, final HRegion daughter)

524     daughter.openHRegion(reporter);

527   static class LoggingProgressable implements CancelableProgressable {
528     private final HRegionInfo hri;
529     private long lastLog = -1;
530     private final long interval;
531 
532     LoggingProgressable(final HRegionInfo hri, final Configuration c) {
533       this.hri = hri;
534       this.interval = c.getLong("hbase.regionserver.split.daughter.open.log.interval",
535         10000);
536     }
537 
538     @Override
539     public boolean progress() {
540       long now = System.currentTimeMillis();
541       if (now - lastLog > this.interval) {
542         LOG.info("Opening " + this.hri.getRegionNameAsString());
543         this.lastLog = now;
544       }
545       return true;
546     }
547   }

553   /**
554    * @param fs Filesystem to use
555    * @param splitdir Directory to store temporary split data in
556    * @throws IOException If <code>splitdir</code> already exists or we fail
557    * to create it.
558    * @see #cleanupSplitDir(FileSystem, Path)
559    */
560   private static void createSplitDir(final FileSystem fs, final Path splitdir)
561   throws IOException {
562     if (fs.exists(splitdir)) {
563       LOG.info("The " + splitdir
564           + " directory exists.  Hence deleting it to recreate it");
565       if (!fs.delete(splitdir, true)) {
566         throw new IOException("Failed deletion of " + splitdir
567             + " before creating them again.");
568       }
569     }
570     if (!fs.mkdirs(splitdir)) throw new IOException("Failed create of " + splitdir);
571   }
572 
573   private static void cleanupSplitDir(final FileSystem fs, final Path splitdir)
574   throws IOException {
575     // Splitdir may have been cleaned up by reopen of the parent dir.
576     deleteDir(fs, splitdir, false);
577   }

579   /**
580    * @param fs Filesystem to use
581    * @param dir Directory to delete
582    * @param mustPreExist If true, we'll throw exception if <code>dir</code>
583    * does not preexist, else we'll just pass.
584    * @throws IOException Thrown if we fail to delete passed <code>dir</code>
585    */
586   private static void deleteDir(final FileSystem fs, final Path dir,
587       final boolean mustPreExist)
588   throws IOException {
589     if (!fs.exists(dir)) {
590       if (mustPreExist) throw new IOException(dir.toString() + " does not exist!");
591     } else if (!fs.delete(dir, true)) {
592       throw new IOException("Failed delete of " + dir);
593     }
594   }

596   private void splitStoreFiles(final Path splitdir,
597     final List<StoreFile> hstoreFilesToSplit)
598   throws IOException {
599     if (hstoreFilesToSplit == null) {
600       // Could be null because close didn't succeed -- for now consider it fatal
601       throw new IOException("Close returned empty list of StoreFiles");
602     }
603     // The following code sets up a thread pool executor with as many slots as
604     // there's files to split. It then fires up everything, waits for
605     // completion and finally checks for any exception
606     int nbFiles = hstoreFilesToSplit.size();
607     ThreadFactoryBuilder builder = new ThreadFactoryBuilder();
608     builder.setNameFormat("StoreFileSplitter-%1$d");
609     ThreadFactory factory = builder.build();
610     ThreadPoolExecutor threadPool =
611       (ThreadPoolExecutor) Executors.newFixedThreadPool(nbFiles, factory);
612     List<Future<Void>> futures = new ArrayList<Future<Void>>(nbFiles);
613 
614      // Split each store file.
615     for (StoreFile sf: hstoreFilesToSplit) {
616       //splitStoreFile(sf, splitdir);
617       StoreFileSplitter sfs = new StoreFileSplitter(sf, splitdir);
618       futures.add(threadPool.submit(sfs));
619     }
620     // Shutdown the pool
621     threadPool.shutdown();
622 
623     // Wait for all the tasks to finish
624     try {
625       boolean stillRunning = !threadPool.awaitTermination(
626           this.fileSplitTimeout, TimeUnit.MILLISECONDS);
627       if (stillRunning) {
628         threadPool.shutdownNow();
629         // wait for the thread to shutdown completely.
630         while (!threadPool.isTerminated()) {
631           Thread.sleep(50);
632         }
633         throw new IOException("Took too long to split the" +
634             " files and create the references, aborting split");
635       }
636     } catch (InterruptedException e) {
637       Thread.currentThread().interrupt();
638       throw new IOException("Interrupted while waiting for file splitters", e);
639     }
640 
641     // Look for any exception
642     for (Future<Void> future: futures) {
643       try {
644         future.get();
645       } catch (InterruptedException e) {
646         Thread.currentThread().interrupt();
647         throw new IOException(
648             "Interrupted while trying to get the results of file splitters", e);
649       } catch (ExecutionException e) {
650         throw new IOException(e);
651       }
652     }
653   }

738   /**
739    * @param server Hosting server instance (May be null when testing).
740    * @param services
741    * @throws IOException If thrown, rollback failed.  Take drastic action.
742    * @return True if we successfully rolled back, false if we got to the point
743    * of no return and so now need to abort the server to minimize damage.
744    */
745   public boolean rollback(final Server server, final RegionServerServices services)
746   throws IOException {
747     // Coprocessor callback
748     if (this.parent.getCoprocessorHost() != null) {
749       this.parent.getCoprocessorHost().preRollBackSplit();
750     }
751     
752     boolean result = true;
753     FileSystem fs = this.parent.getFilesystem();
754     ListIterator<JournalEntry> iterator =
755       this.journal.listIterator(this.journal.size());
756     // Iterate in reverse.
757     while (iterator.hasPrevious()) {
758       JournalEntry je = iterator.previous();
759       switch(je) {
760 
761       case SET_SPLITTING_IN_ZK:
762         if (server != null && server.getZooKeeper() != null) {
763           cleanZK(server, this.parent.getRegionInfo());
764         }
765         break;
766 
767       case CREATE_SPLIT_DIR:
768     	this.parent.writestate.writesEnabled = true;
769         cleanupSplitDir(fs, this.splitdir);
770         break;
771 
772       case CLOSED_PARENT_REGION:
773         try {
774           // So, this returns a seqid but if we just closed and then reopened, we
775           // should be ok. On close, we flushed using sequenceid obtained from
776           // hosting regionserver so no need to propagate the sequenceid returned
777           // out of initialize below up into regionserver as we normally do.
778           // TODO: Verify.
779           this.parent.initialize();
780         } catch (IOException e) {
781           LOG.error("Failed rollbacking CLOSED_PARENT_REGION of region " +
782             this.parent.getRegionNameAsString(), e);
783           throw new RuntimeException(e);
784         }
785         break;
786 
787       case STARTED_REGION_A_CREATION:
788         cleanupDaughterRegion(fs, this.parent.getTableDir(),
789           this.hri_a.getEncodedName());
790         break;
791 
792       case STARTED_REGION_B_CREATION:
793         cleanupDaughterRegion(fs, this.parent.getTableDir(),
794           this.hri_b.getEncodedName());
795         break;
796 
797       case OFFLINED_PARENT:
798         if (services != null) services.addToOnlineRegions(this.parent);
799         break;
800 
801       case PONR:
802         // We got to the point-of-no-return so we need to just abort. Return
803         // immediately.  Do not clean up created daughter regions.  They need
804         // to be in place so we don't delete the parent region mistakenly.
805         // See HBASE-3872.
806         return false;
807 
808       default:
809         throw new RuntimeException("Unhandled journal entry: " + je);
810       }
811     }
812     // Coprocessor callback
813     if (this.parent.getCoprocessorHost() != null) {
814       this.parent.getCoprocessorHost().postRollBackSplit();
815     }
816     return result;
817   }

832   /**
833    * Clean up any split detritus that may have been left around from previous
834    * split attempts.
835    * Call this method on initial region deploy.  Cleans up any mess
836    * left by previous deploys of passed <code>r</code> region.
837    * @param r
838    * @throws IOException
839    */
840   static void cleanupAnySplitDetritus(final HRegion r) throws IOException {
841     Path splitdir = getSplitDir(r);
842     FileSystem fs = r.getFilesystem();
843     if (!fs.exists(splitdir)) return;
844     // Look at the splitdir.  It could have the encoded names of the daughter
845     // regions we tried to make.  See if the daughter regions actually got made
846     // out under the tabledir.  If here under splitdir still, then the split did
847     // not complete.  Try and do cleanup.  This code WILL NOT catch the case
848     // where we successfully created daughter a but regionserver crashed during
849     // the creation of region b.  In this case, there'll be an orphan daughter
850     // dir in the filesystem.  TOOD: Fix.
851     FileStatus [] daughters = fs.listStatus(splitdir, new FSUtils.DirFilter(fs));
852     for (int i = 0; i < daughters.length; i++) {
853       cleanupDaughterRegion(fs, r.getTableDir(),
854         daughters[i].getPath().getName());
855     }
856     cleanupSplitDir(r.getFilesystem(), splitdir);
857     LOG.info("Cleaned up old failed split transaction detritus: " + splitdir);
858   }
