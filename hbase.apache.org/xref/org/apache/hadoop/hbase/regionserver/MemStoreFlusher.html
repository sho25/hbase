52  /**
53   * Thread that flushes cache on request
54   *
55   * NOTE: This class extends Thread rather than Chore because the sleep time
56   * can be interrupted when there is something to do, rather than the Chore
57   * sleep time which is invariant.
58   *
59   * @see FlushRequester
60   */
61  @InterfaceAudience.Private
62  class MemStoreFlusher extends HasThread implements FlushRequester {
63    static final Log LOG = LogFactory.getLog(MemStoreFlusher.class);
64    // These two data members go together.  Any entry in the one must have
65    // a corresponding entry in the other.

67      new DelayQueue<FlushQueueEntry>();

94    public MemStoreFlusher(final Configuration conf,
95        final HRegionServer server) {
96      super();
97      this.server = server;
98      this.threadWakeFrequency =
99        conf.getLong(HConstants.THREAD_WAKE_FREQUENCY, 10 * 1000);
100     long max = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage().getMax();
101     this.globalMemStoreLimit = globalMemStoreLimit(max, DEFAULT_UPPER,
102       UPPER_KEY, conf);
103     long lower = globalMemStoreLimit(max, DEFAULT_LOWER, LOWER_KEY, conf);
104     if (lower > this.globalMemStoreLimit) {
105       lower = this.globalMemStoreLimit;
106       LOG.info("Setting globalMemStoreLimitLowMark == globalMemStoreLimit " +
107         "because supplied " + LOWER_KEY + " was > " + UPPER_KEY);
108     }
109     this.globalMemStoreLimitLowMark = lower;
110     this.blockingStoreFilesNumber =
111       conf.getInt("hbase.hstore.blockingStoreFiles", 7);
112     this.blockingWaitTime = conf.getInt("hbase.hstore.blockingWaitTime",
113       90000);
114     LOG.info("globalMemStoreLimit=" +
115       StringUtils.humanReadableInt(this.globalMemStoreLimit) +
116       ", globalMemStoreLimitLowMark=" +
117       StringUtils.humanReadableInt(this.globalMemStoreLimitLowMark) +
118       ", maxHeap=" + StringUtils.humanReadableInt(max));
119   }

130   static long globalMemStoreLimit(final long max,
131      final float defaultLimit, final String key, final Configuration c) {
132     float limit = c.getFloat(key, defaultLimit);
133     return getMemStoreLimit(max, limit, defaultLimit);

136   static long getMemStoreLimit(final long max, final float limit,
137       final float defaultLimit) {
138     float effectiveLimit = limit;
139     if (limit >= 0.9f || limit < 0.1f) {
140       LOG.warn("Setting global memstore limit to default of " + defaultLimit +
141         " because supplied value outside allowed range of 0.1 -> 0.9");
142       effectiveLimit = defaultLimit;
143     }
144     return (long)(max * effectiveLimit);
145   }

151   /**
152    * The memstore across all regions has exceeded the low water mark. Pick
153    * one region to flush and flush it synchronously (this is called from the
154    * flush thread)
155    * @return true if successful
156    */
157   private boolean flushOneForGlobalPressure() {
158     SortedMap<Long, HRegion> regionsBySize =
159         server.getCopyOfOnlineRegionsSortedBySize();
160 
161     Set<HRegion> excludedRegions = new HashSet<HRegion>();
162 
163     boolean flushedOne = false;
164     while (!flushedOne) {
165       // Find the biggest region that doesn't have too many storefiles
166       // (might be null!)
167       HRegion bestFlushableRegion = getBiggestMemstoreRegion(
168           regionsBySize, excludedRegions, true);
169       // Find the biggest region, total, even if it might have too many flushes.
170       HRegion bestAnyRegion = getBiggestMemstoreRegion(
171           regionsBySize, excludedRegions, false);
172 
173       if (bestAnyRegion == null) {
174         LOG.error("Above memory mark but there are no flushable regions!");
175         return false;
176       }
177 
178       HRegion regionToFlush;
179       if (bestFlushableRegion != null &&
180           bestAnyRegion.memstoreSize.get() > 2 * bestFlushableRegion.memstoreSize.get()) {
181         // Even if it's not supposed to be flushed, pick a region if it's more than twice
182         // as big as the best flushable one - otherwise when we're under pressure we make
183         // lots of little flushes and cause lots of compactions, etc, which just makes
184         // life worse!
185         if (LOG.isDebugEnabled()) {
186           LOG.debug("Under global heap pressure: " +
187             "Region " + bestAnyRegion.getRegionNameAsString() + " has too many " +
188             "store files, but is " +
189             StringUtils.humanReadableInt(bestAnyRegion.memstoreSize.get()) +
190             " vs best flushable region's " +
191             StringUtils.humanReadableInt(bestFlushableRegion.memstoreSize.get()) +
192             ". Choosing the bigger.");
193         }
194         regionToFlush = bestAnyRegion;
195       } else {
196         if (bestFlushableRegion == null) {
197           regionToFlush = bestAnyRegion;
198         } else {
199           regionToFlush = bestFlushableRegion;
200         }
201       }
202 
203       Preconditions.checkState(regionToFlush.memstoreSize.get() > 0);
204 
205       LOG.info("Flush of region " + regionToFlush + " due to global heap pressure");
206       flushedOne = flushRegion(regionToFlush, true);
207       if (!flushedOne) {
208         LOG.info("Excluding unflushable region " + regionToFlush +
209           " - trying to find a different region to flush.");
210         excludedRegions.add(regionToFlush);
211       }
212     }
213     return true;
214   }

216   @Override
217   public void run() {
218     while (!this.server.isStopped()) {
219       FlushQueueEntry fqe = null;
220       try {
221         wakeupPending.set(false); // allow someone to wake us up again
222         fqe = flushQueue.poll(threadWakeFrequency, TimeUnit.MILLISECONDS);
223         if (fqe == null || fqe instanceof WakeupFlushThread) {
224           if (isAboveLowWaterMark()) {
225             LOG.debug("Flush thread woke up because memory above low water=" +
226               StringUtils.humanReadableInt(this.globalMemStoreLimitLowMark));
227             if (!flushOneForGlobalPressure()) {
228               // Wasn't able to flush any region, but we're above low water mark
229               // This is unlikely to happen, but might happen when closing the
230               // entire server - another thread is flushing regions. We'll just
231               // sleep a little bit to avoid spinning, and then pretend that
232               // we flushed one, so anyone blocked will check again
233               lock.lock();
234               try {
235                 Thread.sleep(1000);
236                 flushOccurred.signalAll();
237               } finally {
238                 lock.unlock();
239               }
240             }
241             // Enqueue another one of these tokens so we'll wake up again
242             wakeupFlushThread();
243           }
244           continue;
245         }
246         FlushRegionEntry fre = (FlushRegionEntry)fqe;
247         if (!flushRegion(fre)) {
248           break;
249         }
250       } catch (InterruptedException ex) {
251         continue;
252       } catch (ConcurrentModificationException ex) {
253         continue;
254       } catch (Exception ex) {
255         LOG.error("Cache flusher failed for entry " + fqe, ex);
256         if (!server.checkFileSystem()) {
257           break;
258         }
259       }
260     }
261     this.regionsInQueue.clear();
262     this.flushQueue.clear();
263 
264     // Signal anyone waiting, so they see the close flag
265     lock.lock();
266     try {
267       flushOccurred.signalAll();
268     } finally {
269       lock.unlock();
270     }
271     LOG.info(getName() + " exiting");
272   }
273 
274   private void wakeupFlushThread() {
275     if (wakeupPending.compareAndSet(false, true)) {
276       flushQueue.add(new WakeupFlushThread());
277     }
278   }

280   private HRegion getBiggestMemstoreRegion(
281       SortedMap<Long, HRegion> regionsBySize,
282       Set<HRegion> excludedRegions,
283       boolean checkStoreFileCount) {
284     synchronized (regionsInQueue) {
285       for (HRegion region : regionsBySize.values()) {
286         if (excludedRegions.contains(region)) {
287           continue;
288         }
289 
290         if (checkStoreFileCount && isTooManyStoreFiles(region)) {
291           continue;
292         }
293         return region;
294       }
295     }
296     return null;
297   }

308    * Return true if we're above the high watermark

315   public void requestFlush(HRegion r) {
316     synchronized (regionsInQueue) {
317       if (!regionsInQueue.containsKey(r)) {
318         // This entry has no delay so it will be added at the top of the flush
319         // queue.  It'll come out near immediately.
320         FlushRegionEntry fqe = new FlushRegionEntry(r);
321         this.regionsInQueue.put(r, fqe);
322         this.flushQueue.add(fqe);
323       }
324     }
325   }

343   /*
344    * A flushRegion that checks store file count.  If too many, puts the flush
345    * on delay queue to retry later.
346    * @param fqe
347    * @return true if the region was successfully flushed, false otherwise. If
348    * false, there will be accompanying log messages explaining why the log was
349    * not flushed.
350    */
351   private boolean flushRegion(final FlushRegionEntry fqe) {
352     HRegion region = fqe.region;
353     if (!region.getRegionInfo().isMetaRegion() &&
354         isTooManyStoreFiles(region)) {
355       if (fqe.isMaximumWait(this.blockingWaitTime)) {
356         LOG.info("Waited " + (System.currentTimeMillis() - fqe.createTime) +
357           "ms on a compaction to clean up 'too many store files'; waited " +
358           "long enough... proceeding with flush of " +
359           region.getRegionNameAsString());
360       } else {
361         // If this is first time we've been put off, then emit a log message.
362         if (fqe.getRequeueCount() <= 0) {
363           // Note: We don't impose blockingStoreFiles constraint on meta regions
364           LOG.warn("Region " + region.getRegionNameAsString() + " has too many " +
365             "store files; delaying flush up to " + this.blockingWaitTime + "ms");
366           if (!this.server.compactSplitThread.requestSplit(region)) {
367             try {
368               this.server.compactSplitThread.requestCompaction(region, getName());
369             } catch (IOException e) {
370               LOG.error(
371                 "Cache flush failed for region " + Bytes.toStringBinary(region.getRegionName()),
372                 RemoteExceptionHandler.checkIOException(e));
373             }
374           }
375         }
376 
377         // Put back on the queue.  Have it come back out of the queue
378         // after a delay of this.blockingWaitTime / 100 ms.
379         this.flushQueue.add(fqe.requeue(this.blockingWaitTime / 100));
380         // Tell a lie, it's not flushed but it's ok
381         return true;
382       }
383     }
384     return flushRegion(region, false);
385   }

387   /*
388    * Flush a region.
389    * @param region Region to flush.
390    * @param emergencyFlush Set if we are being force flushed. If true the region
391    * needs to be removed from the flush queue. If false, when we were called
392    * from the main flusher run loop and we got the entry to flush by calling
393    * poll on the flush queue (which removed it).
394    *
395    * @return true if the region was successfully flushed, false otherwise. If
396    * false, there will be accompanying log messages explaining why the log was
397    * not flushed.
398    */
399   private boolean flushRegion(final HRegion region, final boolean emergencyFlush) {
400     synchronized (this.regionsInQueue) {
401       FlushRegionEntry fqe = this.regionsInQueue.remove(region);
402       if (fqe != null && emergencyFlush) {
403         // Need to remove from region from delay queue.  When NOT an
404         // emergencyFlush, then item was removed via a flushQueue.poll.
405         flushQueue.remove(fqe);
406      }
407      lock.lock();
408     }
409     try {
410       boolean shouldCompact = region.flushcache();
411       // We just want to check the size
412       boolean shouldSplit = region.checkSplit() != null;
413       if (shouldSplit) {
414         this.server.compactSplitThread.requestSplit(region);
415       } else if (shouldCompact) {
416         server.compactSplitThread.requestCompaction(region, getName());
417       }
418 
419     } catch (DroppedSnapshotException ex) {
420       // Cache flush can fail in a few places. If it fails in a critical
421       // section, we get a DroppedSnapshotException and a replay of hlog
422       // is required. Currently the only way to do this is a restart of
423       // the server. Abort because hdfs is probably bad (HBASE-644 is a case
424       // where hdfs was bad but passed the hdfs check).
425       server.abort("Replay of HLog required. Forcing server shutdown", ex);
426       return false;
427     } catch (IOException ex) {
428       LOG.error("Cache flush failed" +
429         (region != null ? (" for region " + Bytes.toStringBinary(region.getRegionName())) : ""),
430         RemoteExceptionHandler.checkIOException(ex));
431       if (!server.checkFileSystem()) {
432         return false;
433       }
434     } finally {
435       try {
436         flushOccurred.signalAll();
437       } finally {
438         lock.unlock();
439       }
440     }
441     return true;
442   }

444   private boolean isTooManyStoreFiles(HRegion region) {
445     for (Store hstore : region.stores.values()) {
446       if (hstore.getStorefilesCount() > this.blockingStoreFilesNumber) {
447         return true;
448       }
449     }
450     return false;
451   }

453   /**
454    * Check if the regionserver's memstore memory usage is greater than the
455    * limit. If so, flush regions with the biggest memstores until we're down
456    * to the lower limit. This method blocks callers until we're down to a safe
457    * amount of memstore consumption.
458    */
459   public void reclaimMemStoreMemory() {
460     if (isAboveHighWaterMark()) {
461       lock.lock();
462       try {
463         boolean blocked = false;
464         long startTime = 0;
465         while (isAboveHighWaterMark() && !server.isStopped()) {
466           if(!blocked){
467             startTime = EnvironmentEdgeManager.currentTimeMillis();
468             LOG.info("Blocking updates on " + server.toString() +
469             ": the global memstore size " +
470             StringUtils.humanReadableInt(server.getRegionServerAccounting().getGlobalMemstoreSize()) +
471             " is >= than blocking " +
472             StringUtils.humanReadableInt(globalMemStoreLimit) + " size");
473           }
474           blocked = true;
475           wakeupFlushThread();
476           try {
477             // we should be able to wait forever, but we've seen a bug where
478             // we miss a notify, so put a 5 second bound on it at least.
479             flushOccurred.await(5, TimeUnit.SECONDS);
480           } catch (InterruptedException ie) {
481             Thread.currentThread().interrupt();
482           }
483         }
484         if(blocked){
485           final long totalTime = EnvironmentEdgeManager.currentTimeMillis() - startTime;
486           if(totalTime > 0){
487             this.updatesBlockedMsHighWater.add(totalTime);
488           }
489           LOG.info("Unblocking updates for server " + server.toString());
490         }
491       } finally {
492         lock.unlock();
493       }
494     } else if (isAboveLowWaterMark()) {
495       wakeupFlushThread();
496     }
497   }

535   /**
536    * Datastructure used in the flush queue.  Holds region and retry count.
537    * Keeps tabs on how old this object is.  Implements {@link Delayed}.  On
538    * construction, the delay is zero. When added to a delay queue, we'll come
539    * out near immediately.  Call {@link #requeue(long)} passing delay in
540    * milliseconds before readding to delay queue if you want it to stay there
541    * a while.
542    */
543   static class FlushRegionEntry implements FlushQueueEntry {
544     private final HRegion region;
545 
546     private final long createTime;
547     private long whenToExpire;
548     private int requeueCount = 0;
549 
550     FlushRegionEntry(final HRegion r) {
551       this.region = r;
552       this.createTime = System.currentTimeMillis();
553       this.whenToExpire = this.createTime;
554     }
555 
556     /**
557      * @param maximumWait
558      * @return True if we have been delayed > <code>maximumWait</code> milliseconds.
559      */
560     public boolean isMaximumWait(final long maximumWait) {
561       return (System.currentTimeMillis() - this.createTime) > maximumWait;
562     }
563 
564     /**
565      * @return Count of times {@link #resetDelay()} was called; i.e this is
566      * number of times we've been requeued.
567      */
568     public int getRequeueCount() {
569       return this.requeueCount;
570     }
571 
572     /**
573      * @param when When to expire, when to come up out of the queue.
574      * Specify in milliseconds.  This method adds System.currentTimeMillis()
575      * to whatever you pass.
576      * @return This.
577      */
578     public FlushRegionEntry requeue(final long when) {
579       this.whenToExpire = System.currentTimeMillis() + when;
580       this.requeueCount++;
581       return this;
582     }
583 
584     @Override
585     public long getDelay(TimeUnit unit) {
586       return unit.convert(this.whenToExpire - System.currentTimeMillis(),
587           TimeUnit.MILLISECONDS);
588     }
589 
590     @Override
591     public int compareTo(Delayed other) {
592       return Long.valueOf(getDelay(TimeUnit.MILLISECONDS) -
593         other.getDelay(TimeUnit.MILLISECONDS)).intValue();
594     }
595 
596     @Override
597     public String toString() {
598       return "[flush region " + Bytes.toStringBinary(region.getRegionName()) + "]";
599     }
600   }
