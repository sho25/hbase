61    /**
62     * @param candidateFiles candidate files, ordered from oldest to newest
63     * @return subset copy of candidate list that meets compaction criteria
64     * @throws java.io.IOException
65     */
66    public CompactSelection selectCompaction(List<StoreFile> candidateFiles,
67        boolean isUserCompaction, boolean forceMajor)
68      throws IOException {
69      // Prelimanry compaction subject to filters
70      CompactSelection candidateSelection = new CompactSelection(candidateFiles);
71      long cfTtl = this.storeConfig.getStoreFileTtl();
72      if (!forceMajor) {
73        // If there are expired files, only select them so that compaction deletes them
74        if (comConf.shouldDeleteExpired() && (cfTtl != Long.MAX_VALUE)) {
75          CompactSelection expiredSelection = selectExpiredStoreFiles(
76            candidateSelection, EnvironmentEdgeManager.currentTimeMillis() - cfTtl);
77          if (expiredSelection != null) {
78            return expiredSelection;
79          }
80        }
81        candidateSelection = skipLargeFiles(candidateSelection);
82      }
83  
84      // Force a major compaction if this is a user-requested major compaction,
85      // or if we do not have too many files to compact and this was requested
86      // as a major compaction.
87      // Or, if there are any references among the candidates.
88      boolean majorCompaction = (
89        (forceMajor && isUserCompaction)
90        || ((forceMajor || isMajorCompaction(candidateSelection.getFilesToCompact()))
91            && (candidateSelection.getFilesToCompact().size() < comConf.getMaxFilesToCompact()))
92        || StoreUtils.hasReferences(candidateSelection.getFilesToCompact())
93        );
94  
95      if (!majorCompaction) {
96        // we're doing a minor compaction, let's see what files are applicable
97        candidateSelection = filterBulk(candidateSelection);
98        candidateSelection = applyCompactionPolicy(candidateSelection);
99        candidateSelection = checkMinFilesCriteria(candidateSelection);
100     }
101     candidateSelection =
102         removeExcessFiles(candidateSelection, isUserCompaction, majorCompaction);
103     return candidateSelection;
104   }

116   /**
117    * Select the expired store files to compact
118    *
119    * @param candidates the initial set of storeFiles
120    * @param maxExpiredTimeStamp
121    *          The store file will be marked as expired if its max time stamp is
122    *          less than this maxExpiredTimeStamp.
123    * @return A CompactSelection contains the expired store files as
124    *         filesToCompact
125    */
126   private CompactSelection selectExpiredStoreFiles(
127       CompactSelection candidates, long maxExpiredTimeStamp) {
128     List<StoreFile> filesToCompact = candidates.getFilesToCompact();
129     if (filesToCompact == null || filesToCompact.size() == 0)
130       return null;
131     ArrayList<StoreFile> expiredStoreFiles = null;
132     boolean hasExpiredStoreFiles = false;
133     CompactSelection expiredSFSelection = null;
134 
135     for (StoreFile storeFile : filesToCompact) {
136       if (storeFile.getReader().getMaxTimestamp() < maxExpiredTimeStamp) {
137         LOG.info("Deleting the expired store file by compaction: "
138             + storeFile.getPath() + " whose maxTimeStamp is "
139             + storeFile.getReader().getMaxTimestamp()
140             + " while the max expired timestamp is " + maxExpiredTimeStamp);
141         if (!hasExpiredStoreFiles) {
142           expiredStoreFiles = new ArrayList<StoreFile>();
143           hasExpiredStoreFiles = true;
144         }
145         expiredStoreFiles.add(storeFile);
146       }
147     }
148 
149     if (hasExpiredStoreFiles) {
150       expiredSFSelection = new CompactSelection(expiredStoreFiles);
151     }
152     return expiredSFSelection;
153   }

155   /**
156    * @param candidates pre-filtrate
157    * @return filtered subset
158    * exclude all files above maxCompactSize
159    * Also save all references. We MUST compact them
160    */
161   private CompactSelection skipLargeFiles(CompactSelection candidates) {
162     int pos = 0;
163     while (pos < candidates.getFilesToCompact().size() &&
164       candidates.getFilesToCompact().get(pos).getReader().length() >
165         comConf.getMaxCompactSize() &&
166       !candidates.getFilesToCompact().get(pos).isReference()) {
167       ++pos;
168     }
169     if (pos > 0) {
170       LOG.debug("Some files are too large. Excluding " + pos
171           + " files from compaction candidates");
172       candidates.clearSubList(0, pos);
173     }
174     return candidates;
175   }

177   /**
178    * @param candidates pre-filtrate
179    * @return filtered subset
180    * exclude all bulk load files if configured
181    */
182   private CompactSelection filterBulk(CompactSelection candidates) {
183     candidates.getFilesToCompact().removeAll(Collections2.filter(
184         candidates.getFilesToCompact(),
185         new Predicate<StoreFile>() {
186           @Override
187           public boolean apply(StoreFile input) {
188             return input.excludeFromMinorCompaction();
189           }
190         }));
191     return candidates;
192   }

194   /**
195    * @param candidates pre-filtrate
196    * @return filtered subset
197    * take upto maxFilesToCompact from the start
198    */
199   private CompactSelection removeExcessFiles(CompactSelection candidates,
200       boolean isUserCompaction, boolean isMajorCompaction) {
201     int excess = candidates.getFilesToCompact().size() - comConf.getMaxFilesToCompact();
202     if (excess > 0) {
203       if (isMajorCompaction && isUserCompaction) {
204         LOG.debug("Warning, compacting more than " + comConf.getMaxFilesToCompact() +
205             " files because of a user-requested major compaction");
206       } else {
207         LOG.debug("Too many admissible files. Excluding " + excess
208           + " files from compaction candidates");
209         candidates.clearSubList(comConf.getMaxFilesToCompact(),
210           candidates.getFilesToCompact().size());
211       }
212     }
213     return candidates;
214   }
215   /**
216    * @param candidates pre-filtrate
217    * @return filtered subset
218    * forget the compactionSelection if we don't have enough files
219    */
220   private CompactSelection checkMinFilesCriteria(CompactSelection candidates) {
221     int minFiles = comConf.getMinFilesToCompact();
222     if (candidates.getFilesToCompact().size() < minFiles) {
223       if(LOG.isDebugEnabled()) {
224         LOG.debug("Not compacting files because we only have " +
225             candidates.getFilesToCompact().size() +
226           " files ready for compaction.  Need " + minFiles + " to initiate.");
227       }
228       candidates.emptyFileList();
229     }
230     return candidates;
231   }

233   /**
234     * @param candidates pre-filtrate
235     * @return filtered subset
236     * -- Default minor compaction selection algorithm:
237     * choose CompactSelection from candidates --
238     * First exclude bulk-load files if indicated in configuration.
239     * Start at the oldest file and stop when you find the first file that
240     * meets compaction criteria:
241     * (1) a recently-flushed, small file (i.e. <= minCompactSize)
242     * OR
243     * (2) within the compactRatio of sum(newer_files)
244     * Given normal skew, any newer files will also meet this criteria
245     * <p/>
246     * Additional Note:
247     * If fileSizes.size() >> maxFilesToCompact, we will recurse on
248     * compact().  Consider the oldest files first to avoid a
249     * situation where we always compact [end-threshold,end).  Then, the
250     * last file becomes an aggregate of the previous compactions.
251     *
252     * normal skew:
253     *
254     *         older ----> newer (increasing seqID)
255     *     _
256     *    | |   _
257     *    | |  | |   _
258     *  --|-|- |-|- |-|---_-------_-------  minCompactSize
259     *    | |  | |  | |  | |  _  | |
260     *    | |  | |  | |  | | | | | |
261     *    | |  | |  | |  | | | | | |
262     */
263   CompactSelection applyCompactionPolicy(CompactSelection candidates) throws IOException {
264     if (candidates.getFilesToCompact().isEmpty()) {
265       return candidates;
266     }
267 
268     // we're doing a minor compaction, let's see what files are applicable
269     int start = 0;
270     double ratio = comConf.getCompactionRatio();
271     if (isOffPeakHour() && candidates.trySetOffpeak()) {
272       ratio = comConf.getCompactionRatioOffPeak();
273       LOG.info("Running an off-peak compaction, selection ratio = " + ratio
274           + ", numOutstandingOffPeakCompactions is now "
275           + CompactSelection.getNumOutStandingOffPeakCompactions());
276     }
277 
278     // get store file sizes for incremental compacting selection.
279     int countOfFiles = candidates.getFilesToCompact().size();
280     long[] fileSizes = new long[countOfFiles];
281     long[] sumSize = new long[countOfFiles];
282     for (int i = countOfFiles - 1; i >= 0; --i) {
283       StoreFile file = candidates.getFilesToCompact().get(i);
284       fileSizes[i] = file.getReader().length();
285       // calculate the sum of fileSizes[i,i+maxFilesToCompact-1) for algo
286       int tooFar = i + comConf.getMaxFilesToCompact() - 1;
287       sumSize[i] = fileSizes[i]
288         + ((i + 1 < countOfFiles) ? sumSize[i + 1] : 0)
289         - ((tooFar < countOfFiles) ? fileSizes[tooFar] : 0);
290     }
291 
292 
293     while (countOfFiles - start >= comConf.getMinFilesToCompact() &&
294       fileSizes[start] > Math.max(comConf.getMinCompactSize(),
295           (long) (sumSize[start + 1] * ratio))) {
296       ++start;
297     }
298     if (start < countOfFiles) {
299       LOG.info("Default compaction algorithm has selected " + (countOfFiles - start)
300         + " files from " + countOfFiles + " candidates");
301     }
302 
303     candidates = candidates.getSubList(start, countOfFiles);
304 
305     return candidates;
306   }

308   /*
309    * @param filesToCompact Files to compact. Can be null.
310    * @return True if we should run a major compaction.
311    */
312   public boolean isMajorCompaction(final List<StoreFile> filesToCompact)
313       throws IOException {
314     boolean result = false;
315     long mcTime = getNextMajorCompactTime(filesToCompact);
316     if (filesToCompact == null || filesToCompact.isEmpty() || mcTime == 0) {
317       return result;
318     }
319     // TODO: Use better method for determining stamp of last major (HBASE-2990)
320     long lowTimestamp = StoreUtils.getLowestTimestamp(filesToCompact);
321     long now = System.currentTimeMillis();
322     if (lowTimestamp > 0l && lowTimestamp < (now - mcTime)) {
323       // Major compaction time has elapsed.
324       long cfTtl = this.storeConfig.getStoreFileTtl();
325       if (filesToCompact.size() == 1) {
326         // Single file
327         StoreFile sf = filesToCompact.get(0);
328         Long minTimestamp = sf.getMinimumTimestamp();
329         long oldest = (minTimestamp == null)
330             ? Long.MIN_VALUE
331             : now - minTimestamp.longValue();
332         if (sf.isMajorCompaction() &&
333             (cfTtl == HConstants.FOREVER || oldest < cfTtl)) {
334           if (LOG.isDebugEnabled()) {
335             LOG.debug("Skipping major compaction of " + this +
336                 " because one (major) compacted file only and oldestTime " +
337                 oldest + "ms is < ttl=" + cfTtl);
338           }
339         } else if (cfTtl != HConstants.FOREVER && oldest > cfTtl) {
340           LOG.debug("Major compaction triggered on store " + this +
341             ", because keyvalues outdated; time since last major compaction " +
342             (now - lowTimestamp) + "ms");
343           result = true;
344         }
345       } else {
346         if (LOG.isDebugEnabled()) {
347           LOG.debug("Major compaction triggered on store " + this +
348               "; time since last major compaction " + (now - lowTimestamp) + "ms");
349         }
350         result = true;
351       }
352     }
353     return result;
354   }

356   public long getNextMajorCompactTime(final List<StoreFile> filesToCompact) {
357     // default = 24hrs
358     long ret = comConf.getMajorCompactionPeriod();
359     if (ret > 0) {
360       // default = 20% = +/- 4.8 hrs
361       double jitterPct = comConf.getMajorCompactionJitter();
362       if (jitterPct > 0) {
363         long jitter = Math.round(ret * jitterPct);
364         // deterministic jitter avoids a major compaction storm on restart
365         Integer seed = StoreUtils.getDeterministicRandomSeed(filesToCompact);
366         if (seed != null) {
367           double rnd = (new Random(seed)).nextDouble();
368           ret += jitter - Math.round(2L * jitter * rnd);
369         } else {
370           ret = 0; // no storefiles == no major compaction
371         }
372       }
373     }
374     return ret;
375   }

393   /**
394    * @return whether this is off-peak hour
395    */
396   private boolean isOffPeakHour() {
397     int currentHour = calendar.get(Calendar.HOUR_OF_DAY);
398     int startHour = comConf.getOffPeakStartHour();
399     int endHour = comConf.getOffPeakEndHour();
400     // If offpeak time checking is disabled just return false.
401     if (startHour == endHour) {
402       return false;
403     }
404     if (startHour < endHour) {
405       return (currentHour >= startHour && currentHour < endHour);
406     }
407     return (currentHour >= startHour || currentHour < endHour);
408   }
