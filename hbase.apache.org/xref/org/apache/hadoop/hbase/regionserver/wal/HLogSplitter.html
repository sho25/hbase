186   /**
187    * Split up a bunch of regionserver commit log files that are no longer being
188    * written to, into new files, one per region for region to replay on startup.
189    * Delete the old log files when finished.
190    *
191    * @throws IOException will throw if corrupted hlogs aren't tolerated
192    * @return the list of splits
193    */
194   public List<Path> splitLog()
195       throws IOException {
196     Preconditions.checkState(!hasSplit,
197         "An HLogSplitter instance may only be used once");
198     hasSplit = true;
199 
200     status = TaskMonitor.get().createStatus(
201         "Splitting logs in " + srcDir);
202     
203     long startTime = EnvironmentEdgeManager.currentTimeMillis();
204     
205     status.setStatus("Determining files to split...");
206     List<Path> splits = null;
207     if (!fs.exists(srcDir)) {
208       // Nothing to do
209       status.markComplete("No log directory existed to split.");
210       return splits;
211     }
212     FileStatus[] logfiles = fs.listStatus(srcDir);
213     if (logfiles == null || logfiles.length == 0) {
214       // Nothing to do
215       return splits;
216     }
217     logAndReport("Splitting " + logfiles.length + " hlog(s) in "
218     + srcDir.toString());
219     splits = splitLog(logfiles);
220 
221     splitTime = EnvironmentEdgeManager.currentTimeMillis() - startTime;
222     String msg = "hlog file splitting completed in " + splitTime +
223         " ms for " + srcDir.toString();
224     status.markComplete(msg);
225     LOG.info(msg);
226     return splits;
227   }

261   /**
262    * Splits the HLog edits in the given list of logfiles (that are a mix of edits
263    * on multiple regions) by region and then splits them per region directories,
264    * in batches of (hbase.hlog.split.batch.size)
265    * <p>
266    * This process is split into multiple threads. In the main thread, we loop
267    * through the logs to be split. For each log, we:
268    * <ul>
269    *   <li> Recover it (take and drop HDFS lease) to ensure no other process can write</li>
270    *   <li> Read each edit (see {@link #parseHLog}</li>
271    *   <li> Mark as "processed" or "corrupt" depending on outcome</li>
272    * </ul>
273    * <p>
274    * Each edit is passed into the EntryBuffers instance, which takes care of
275    * memory accounting and splitting the edits by region.
276    * <p>
277    * The OutputSink object then manages N other WriterThreads which pull chunks
278    * of edits from EntryBuffers and write them to the output region directories.
279    * <p>
280    * After the process is complete, the log files are archived to a separate
281    * directory.
282    */
283   private List<Path> splitLog(final FileStatus[] logfiles) throws IOException {
284     List<Path> processedLogs = new ArrayList<Path>(logfiles.length);
285     List<Path> corruptedLogs = new ArrayList<Path>(logfiles.length);
286     List<Path> splits;
287 
288     boolean skipErrors = conf.getBoolean("hbase.hlog.split.skip.errors", true);
289 
290     countTotalBytes(logfiles);
291     splitSize = 0;
292 
293     outputSink.startWriterThreads();
294 
295     try {
296       int i = 0;
297       for (FileStatus log : logfiles) {
298        Path logPath = log.getPath();
299         long logLength = log.getLen();
300         splitSize += logLength;
301         logAndReport("Splitting hlog " + (i++ + 1) + " of " + logfiles.length
302             + ": " + logPath + ", length=" + logLength);
303         Reader in = null;
304         try {
305           in = getReader(fs, log, conf, skipErrors);
306           if (in != null) {
307             parseHLog(in, logPath, entryBuffers, fs, conf, skipErrors);
308           }
309           processedLogs.add(logPath);
310         } catch (CorruptedLogFileException e) {
311           LOG.info("Got while parsing hlog " + logPath +
312               ". Marking as corrupted", e);
313           corruptedLogs.add(logPath);
314         } finally {
315           if (in != null) {
316             try {
317               in.close();
318             } catch (IOException e) {
319               LOG.warn("Close log reader threw exception -- continuing", e);
320             }
321           }
322         }
323       }
324       status.setStatus("Log splits complete. Checking for orphaned logs.");
325       
326       if (fs.listStatus(srcDir).length > processedLogs.size()
327           + corruptedLogs.size()) {
328         throw new OrphanHLogAfterSplitException(
329             "Discovered orphan hlog after split. Maybe the "
330             + "HRegionServer was not dead when we started");
331       }
332     } finally {
333       status.setStatus("Finishing writing output logs and closing down.");
334       splits = outputSink.finishWritingAndClose();
335     }
336     status.setStatus("Archiving logs after completed split");
337     archiveLogs(srcDir, corruptedLogs, processedLogs, oldLogDir, fs, conf);
338     return splits;
339   }

395   public boolean splitLogFile(FileStatus logfile,
396       CancelableProgressable reporter) throws IOException {
397     boolean isCorrupted = false;
398     Preconditions.checkState(status == null);
399     status = TaskMonitor.get().createStatus(
400         "Splitting log file " + logfile.getPath() +
401         "into a temporary staging area.");
402     boolean skipErrors = conf.getBoolean("hbase.hlog.split.skip.errors",
403         HLog.SPLIT_SKIP_ERRORS_DEFAULT);
404     int interval = conf.getInt("hbase.splitlog.report.interval.loglines", 1024);
405     Path logPath = logfile.getPath();
406     long logLength = logfile.getLen();
407     LOG.info("Splitting hlog: " + logPath + ", length=" + logLength);
408     status.setStatus("Opening log file");
409     Reader in = null;
410     try {
411       in = getReader(fs, logfile, conf, skipErrors);
412     } catch (CorruptedLogFileException e) {
413       LOG.warn("Could not get reader, corrupted log file " + logPath, e);
414       ZKSplitLog.markCorrupted(rootDir, logfile.getPath().getName(), fs);
415       isCorrupted = true;
416     }
417     if (in == null) {
418       status.markComplete("Was nothing to split in log file");
419       LOG.warn("Nothing to split in log file " + logPath);
420       return true;
421     }
422     this.setDistributedLogSplittingHelper(new DistributedLogSplittingHelper(reporter));
423     if (!reportProgressIfIsDistributedLogSplitting()) {
424       return false;
425     }
426     boolean progress_failed = false;
427     int numOpenedFilesBeforeReporting = conf.getInt("hbase.splitlog.report.openedfiles", 3);
428     int numOpenedFilesLastCheck = 0;
429     outputSink.startWriterThreads();
430     // Report progress every so many edits and/or files opened (opening a file
431     // takes a bit of time).
432     Map<byte[], Long> lastFlushedSequenceIds =
433       new TreeMap<byte[], Long>(Bytes.BYTES_COMPARATOR);
434     Entry entry;
435     int editsCount = 0;
436     int editsSkipped = 0;
437     try {
438       while ((entry = getNextLogLine(in,logPath, skipErrors)) != null) {
439         byte[] region = entry.getKey().getEncodedRegionName();
440         Long lastFlushedSequenceId = -1l;
441         if (sequenceIdChecker != null) {
442           lastFlushedSequenceId = lastFlushedSequenceIds.get(region);
443           if (lastFlushedSequenceId == null) {
444               lastFlushedSequenceId = sequenceIdChecker.getLastSequenceId(region);
445               lastFlushedSequenceIds.put(region, lastFlushedSequenceId);
446           }
447         }
448         if (lastFlushedSequenceId >= entry.getKey().getLogSeqNum()) {
449           editsSkipped++;
450           continue;
451         }
452         entryBuffers.appendEntry(entry);
453         editsCount++;
454         // If sufficient edits have passed, check if we should report progress.
455         if (editsCount % interval == 0
456             || (outputSink.logWriters.size() - numOpenedFilesLastCheck) > numOpenedFilesBeforeReporting) {
457           numOpenedFilesLastCheck = outputSink.logWriters.size();
458           String countsStr = (editsCount - editsSkipped) +
459             " edits, skipped " + editsSkipped + " edits.";
460           status.setStatus("Split " + countsStr);
461           if (!reportProgressIfIsDistributedLogSplitting()) {
462             return false;
463           }
464         }
465       }
466     } catch (InterruptedException ie) {
467       IOException iie = new InterruptedIOException();
468       iie.initCause(ie);
469       throw iie;
470     } catch (CorruptedLogFileException e) {
471       LOG.warn("Could not parse, corrupted log file " + logPath, e);
472       ZKSplitLog.markCorrupted(rootDir, logfile.getPath().getName(), fs);
473       isCorrupted = true;
474     } catch (IOException e) {
475       e = RemoteExceptionHandler.checkIOException(e);
476       throw e;
477     } finally {
478       LOG.info("Finishing writing output logs and closing down.");
479       progress_failed = outputSink.finishWritingAndClose() == null;
480       String msg = "Processed " + editsCount + " edits across "
481           + outputSink.getOutputCounts().size() + " regions; log file="
482           + logPath + " is corrupted = " + isCorrupted + " progress failed = "
483           + progress_failed;
484       ;
485       LOG.info(msg);
486       status.markComplete(msg);
487     }
488     return !progress_failed;
489   }

509   public static void finishSplitLogFile(Path rootdir, Path oldLogDir,
510       String logfile, Configuration conf) throws IOException {
511     List<Path> processedLogs = new ArrayList<Path>();
512     List<Path> corruptedLogs = new ArrayList<Path>();
513     FileSystem fs;
514     fs = rootdir.getFileSystem(conf);
515     Path logPath = new Path(logfile);
516     if (ZKSplitLog.isCorrupted(rootdir, logPath.getName(), fs)) {
517       corruptedLogs.add(logPath);
518     } else {
519       processedLogs.add(logPath);
520     }
521     archiveLogs(null, corruptedLogs, processedLogs, oldLogDir, fs, conf);
522     Path stagingDir = ZKSplitLog.getSplitLogDir(rootdir, logPath.getName());
523     fs.delete(stagingDir, true);
524   }

526   /**
527    * Moves processed logs to a oldLogDir after successful processing Moves
528    * corrupted logs (any log that couldn't be successfully parsed to corruptDir
529    * (.corrupt) for later investigation
530    *
531    * @param corruptedLogs
532    * @param processedLogs
533    * @param oldLogDir
534    * @param fs
535    * @param conf
536    * @throws IOException
537    */
538   private static void archiveLogs(
539       final Path srcDir,
540       final List<Path> corruptedLogs,
541       final List<Path> processedLogs, final Path oldLogDir,
542       final FileSystem fs, final Configuration conf) throws IOException {
543     final Path corruptDir = new Path(conf.get(HConstants.HBASE_DIR), conf.get(
544         "hbase.regionserver.hlog.splitlog.corrupt.dir",  HConstants.CORRUPT_DIR_NAME));
545 
546     if (!fs.mkdirs(corruptDir)) {
547       LOG.info("Unable to mkdir " + corruptDir);
548     }
549     fs.mkdirs(oldLogDir);
550 
551     // this method can get restarted or called multiple times for archiving
552     // the same log files.
553     for (Path corrupted : corruptedLogs) {
554       Path p = new Path(corruptDir, corrupted.getName());
555       if (fs.exists(corrupted)) {
556         if (!fs.rename(corrupted, p)) {
557           LOG.warn("Unable to move corrupted log " + corrupted + " to " + p);
558         } else {
559           LOG.warn("Moved corrupted log " + corrupted + " to " + p);
560         }
561       }
562     }
563 
564     for (Path p : processedLogs) {
565       Path newPath = FSHLog.getHLogArchivePath(oldLogDir, p);
566       if (fs.exists(p)) {
567         if (!fs.rename(p, newPath)) {
568           LOG.warn("Unable to move  " + p + " to " + newPath);
569         } else {
570           LOG.debug("Archived processed log " + p + " to " + newPath);
571         }
572       }
573     }
574 
575     // distributed log splitting removes the srcDir (region's log dir) later
576     // when all the log files in that srcDir have been successfully processed
577     if (srcDir != null && !fs.delete(srcDir, true)) {
578       throw new IOException("Unable to delete src dir: " + srcDir);
579     }
580   }

582   /**
583    * Path to a file under RECOVERED_EDITS_DIR directory of the region found in
584    * <code>logEntry</code> named for the sequenceid in the passed
585    * <code>logEntry</code>: e.g. /hbase/some_table/2323432434/recovered.edits/2332.
586    * This method also ensures existence of RECOVERED_EDITS_DIR under the region
587    * creating it if necessary.
588    * @param fs
589    * @param logEntry
590    * @param rootDir HBase root dir.
591    * @return Path to file into which to dump split log edits.
592    * @throws IOException
593    */
594   static Path getRegionSplitEditsPath(final FileSystem fs,
595       final Entry logEntry, final Path rootDir, boolean isCreate)
596   throws IOException {
597     Path tableDir = HTableDescriptor.getTableDir(rootDir, logEntry.getKey().getTablename());
598     String encodedRegionName = Bytes.toString(logEntry.getKey().getEncodedRegionName());
599     Path regiondir = HRegion.getRegionDir(tableDir, encodedRegionName);
600     Path dir = HLogUtil.getRegionDirRecoveredEditsDir(regiondir);
601 
602     if (!fs.exists(regiondir)) {
603       LOG.info("This region's directory doesn't exist: "
604           + regiondir.toString() + ". It is very likely that it was" +
605           " already split so it's safe to discard those edits.");
606       return null;
607     }
608     if (fs.exists(dir) && fs.isFile(dir)) {
609       Path tmp = new Path("/tmp");
610       if (!fs.exists(tmp)) {
611         fs.mkdirs(tmp);
612       }
613       tmp = new Path(tmp,
614         HLog.RECOVERED_EDITS_DIR + "_" + encodedRegionName);
615       LOG.warn("Found existing old file: " + dir + ". It could be some "
616         + "leftover of an old installation. It should be a folder instead. "
617         + "So moving it to " + tmp);
618       if (!fs.rename(dir, tmp)) {
619         LOG.warn("Failed to sideline old file " + dir);
620       }
621     }
622 
623     if (isCreate && !fs.exists(dir)) {
624       if (!fs.mkdirs(dir)) LOG.warn("mkdir failed on " + dir);
625     }
626     // Append file name ends with RECOVERED_LOG_TMPFILE_SUFFIX to ensure
627     // region's replayRecoveredEdits will not delete it
628     String fileName = formatRecoveredEditsFileName(logEntry.getKey().getLogSeqNum());
629     fileName = getTmpRecoveredEditsFileName(fileName);
630     return new Path(dir, fileName);
631   }
632 
633   static String getTmpRecoveredEditsFileName(String fileName) {
634     return fileName + HLog.RECOVERED_LOG_TMPFILE_SUFFIX;

655   /**
656    * Parse a single hlog and put the edits in entryBuffers
657    *
658    * @param in the hlog reader
659    * @param path the path of the log file
660    * @param entryBuffers the buffer to hold the parsed edits
661    * @param fs the file system
662    * @param conf the configuration
663    * @param skipErrors indicator if CorruptedLogFileException should be thrown instead of IOException
664    * @throws IOException
665    * @throws CorruptedLogFileException if hlog is corrupted
666    */
667   private void parseHLog(final Reader in, Path path,
668 		EntryBuffers entryBuffers, final FileSystem fs,
669     final Configuration conf, boolean skipErrors)
670 	throws IOException, CorruptedLogFileException {
671     int editsCount = 0;
672     try {
673       Entry entry;
674       while ((entry = getNextLogLine(in, path, skipErrors)) != null) {
675         entryBuffers.appendEntry(entry);
676         editsCount++;
677       }
678     } catch (InterruptedException ie) {
679       IOException t = new InterruptedIOException();
680       t.initCause(ie);
681       throw t;
682     } finally {
683       LOG.debug("Pushed=" + editsCount + " entries from " + path);
684     }

687   /**
688    * Create a new {@link Reader} for reading logs to split.
689    *
690    * @param fs
691    * @param file
692    * @param conf
693    * @return A new Reader instance
694    * @throws IOException
695    * @throws CorruptedLogFileException
696    */
697   protected Reader getReader(FileSystem fs, FileStatus file, Configuration conf,
698       boolean skipErrors)
699       throws IOException, CorruptedLogFileException {
700     Path path = file.getPath();
701     long length = file.getLen();
702     Reader in;
703 
704 
705     // Check for possibly empty file. With appends, currently Hadoop reports a
706     // zero length even if the file has been sync'd. Revisit if HDFS-376 or
707     // HDFS-878 is committed.
708     if (length <= 0) {
709       LOG.warn("File " + path + " might be still open, length is 0");
710     }
711 
712     try {
713       FSUtils.getInstance(fs, conf).recoverFileLease(fs, path, conf);
714       try {
715         in = getReader(fs, path, conf);
716       } catch (EOFException e) {
717         if (length <= 0) {
718           // TODO should we ignore an empty, not-last log file if skip.errors
719           // is false? Either way, the caller should decide what to do. E.g.
720           // ignore if this is the last log in sequence.
721           // TODO is this scenario still possible if the log has been
722           // recovered (i.e. closed)
723           LOG.warn("Could not open " + path + " for reading. File is empty", e);
724           return null;
725         } else {
726           // EOFException being ignored
727           return null;
728         }
729       }
730     } catch (IOException e) {
731       if (!skipErrors) {
732         throw e;
733       }
734       CorruptedLogFileException t =
735         new CorruptedLogFileException("skipErrors=true Could not open hlog " +
736             path + " ignoring");
737       t.initCause(e);
738       throw t;
739     }
740     return in;
741   }

743   static private Entry getNextLogLine(Reader in, Path path, boolean skipErrors)
744   throws CorruptedLogFileException, IOException {
745     try {
746       return in.next();
747     } catch (EOFException eof) {
748       // truncated files are expected if a RS crashes (see HBASE-2643)
749       LOG.info("EOF from hlog " + path + ".  continuing");
750       return null;
751     } catch (IOException e) {
752       // If the IOE resulted from bad file format,
753       // then this problem is idempotent and retrying won't help
754       if (e.getCause() != null &&
755           (e.getCause() instanceof ParseException ||
756            e.getCause() instanceof org.apache.hadoop.fs.ChecksumException)) {
757         LOG.warn("Parse exception " + e.getCause().toString() + " from hlog "
758            + path + ".  continuing");
759         return null;
760       }
761       if (!skipErrors) {
762         throw e;
763       }
764       CorruptedLogFileException t =
765         new CorruptedLogFileException("skipErrors=true Ignoring exception" +
766             " while parsing hlog " + path + ". Marking as corrupted");
767       t.initCause(e);
768       throw t;
769     }
770   }

805   /**
806    * Class which accumulates edits and separates them into a buffer per region
807    * while simultaneously accounting RAM usage. Blocks if the RAM usage crosses
808    * a predefined threshold.
809    *
810    * Writer threads then pull region-specific buffers from this class.
811    */
812   class EntryBuffers {
813     Map<byte[], RegionEntryBuffer> buffers =
814       new TreeMap<byte[], RegionEntryBuffer>(Bytes.BYTES_COMPARATOR);
815 
816     /* Track which regions are currently in the middle of writing. We don't allow
817        an IO thread to pick up bytes from a region if we're already writing
818        data for that region in a different IO thread. */
819     Set<byte[]> currentlyWriting = new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR);
820 
821     long totalBuffered = 0;
822     long maxHeapUsage;
823 
824     EntryBuffers(long maxHeapUsage) {
825       this.maxHeapUsage = maxHeapUsage;
826     }
827 
828     /**
829      * Append a log entry into the corresponding region buffer.
830      * Blocks if the total heap usage has crossed the specified threshold.
831      *
832      * @throws InterruptedException
833      * @throws IOException
834      */
835     void appendEntry(Entry entry) throws InterruptedException, IOException {
836       HLogKey key = entry.getKey();
837 
838       RegionEntryBuffer buffer;
839       long incrHeap;
840       synchronized (this) {
841         buffer = buffers.get(key.getEncodedRegionName());
842         if (buffer == null) {
843           buffer = new RegionEntryBuffer(key.getTablename(), key.getEncodedRegionName());
844           buffers.put(key.getEncodedRegionName(), buffer);
845         }
846         incrHeap= buffer.appendEntry(entry);        
847       }
848 
849       // If we crossed the chunk threshold, wait for more space to be available
850       synchronized (dataAvailable) {
851         totalBuffered += incrHeap;
852         while (totalBuffered > maxHeapUsage && thrown.get() == null) {
853           LOG.debug("Used " + totalBuffered + " bytes of buffered edits, waiting for IO threads...");
854           dataAvailable.wait(3000);
855         }
856         dataAvailable.notifyAll();
857       }
858       checkForErrors();
859     }
860 
861     synchronized RegionEntryBuffer getChunkToWrite() {
862       long biggestSize=0;
863       byte[] biggestBufferKey=null;
864 
865       for (Map.Entry<byte[], RegionEntryBuffer> entry : buffers.entrySet()) {
866         long size = entry.getValue().heapSize();
867         if (size > biggestSize && !currentlyWriting.contains(entry.getKey())) {
868           biggestSize = size;
869           biggestBufferKey = entry.getKey();
870         }
871       }
872       if (biggestBufferKey == null) {
873         return null;
874       }
875 
876       RegionEntryBuffer buffer = buffers.remove(biggestBufferKey);
877       currentlyWriting.add(biggestBufferKey);
878       return buffer;
879     }
880 
881     void doneWriting(RegionEntryBuffer buffer) {
882       synchronized (this) {
883         boolean removed = currentlyWriting.remove(buffer.encodedRegionName);
884         assert removed;
885       }
886       long size = buffer.heapSize();
887 
888       synchronized (dataAvailable) {
889         totalBuffered -= size;
890         // We may unblock writers
891         dataAvailable.notifyAll();
892       }
893     }
894 
895     synchronized boolean isRegionCurrentlyWriting(byte[] region) {
896       return currentlyWriting.contains(region);
897     }
898   }

900   /**
901    * A buffer of some number of edits for a given region.
902    * This accumulates edits and also provides a memory optimization in order to
903    * share a single byte array instance for the table and region name.
904    * Also tracks memory usage of the accumulated edits.
905    */
906   static class RegionEntryBuffer implements HeapSize {
907     long heapInBuffer = 0;
908     List<Entry> entryBuffer;
909     byte[] tableName;
910     byte[] encodedRegionName;
911 
912     RegionEntryBuffer(byte[] table, byte[] region) {
913       this.tableName = table;
914       this.encodedRegionName = region;
915       this.entryBuffer = new LinkedList<Entry>();
916     }
917 
918     long appendEntry(Entry entry) {
919       internify(entry);
920       entryBuffer.add(entry);
921       long incrHeap = entry.getEdit().heapSize() +
922         ClassSize.align(2 * ClassSize.REFERENCE) + // HLogKey pointers
923         0; // TODO linkedlist entry
924       heapInBuffer += incrHeap;
925       return incrHeap;
926     }
927 
928     private void internify(Entry entry) {
929       HLogKey k = entry.getKey();
930       k.internTableName(this.tableName);
931       k.internEncodedRegionName(this.encodedRegionName);
932     }
933 
934     public long heapSize() {
935       return heapInBuffer;
936     }
937   }

940   class WriterThread extends Thread {
941     private volatile boolean shouldStop = false;
942 
943     WriterThread(int i) {
944       super("WriterThread-" + i);
945     }
946 
947     public void run()  {
948       try {
949         doRun();
950       } catch (Throwable t) {
951         LOG.error("Error in log splitting write thread", t);
952         writerThreadError(t);
953       }
954     }
955 
956     private void doRun() throws IOException {
957       LOG.debug("Writer thread " + this + ": starting");
958       while (true) {
959         RegionEntryBuffer buffer = entryBuffers.getChunkToWrite();
960         if (buffer == null) {
961           // No data currently available, wait on some more to show up
962           synchronized (dataAvailable) {
963             if (shouldStop) return;
964             try {
965               dataAvailable.wait(1000);
966             } catch (InterruptedException ie) {
967               if (!shouldStop) {
968                 throw new RuntimeException(ie);
969               }
970             }
971           }
972           continue;
973         }
974 
975         assert buffer != null;
976         try {
977           writeBuffer(buffer);
978         } finally {
979           entryBuffers.doneWriting(buffer);
980         }
981       }
982     }
983 
984 
985     private void writeBuffer(RegionEntryBuffer buffer) throws IOException {
986       List<Entry> entries = buffer.entryBuffer;
987       if (entries.isEmpty()) {
988         LOG.warn(this.getName() + " got an empty buffer, skipping");
989         return;
990       }
991 
992       WriterAndPath wap = null;
993 
994       long startTime = System.nanoTime();
995       try {
996         int editsCount = 0;
997 
998         for (Entry logEntry : entries) {
999           if (wap == null) {
1000             wap = outputSink.getWriterAndPath(logEntry);
1001             if (wap == null) {
1002               // getWriterAndPath decided we don't need to write these edits
1003               // Message was already logged
1004               return;
1005             }
1006           }
1007           wap.w.append(logEntry);
1008           outputSink.updateRegionMaximumEditLogSeqNum(logEntry);
1009           editsCount++;
1010         }
1011         // Pass along summary statistics
1012         wap.incrementEdits(editsCount);
1013         wap.incrementNanoTime(System.nanoTime() - startTime);
1014       } catch (IOException e) {
1015         e = RemoteExceptionHandler.checkIOException(e);
1016         LOG.fatal(this.getName() + " Got while writing log entry to log", e);
1017         throw e;
1018       }
1019     }
1020 
1021     void finish() {
1022       synchronized (dataAvailable) {
1023         shouldStop = true;
1024         dataAvailable.notifyAll();
1025       }
1026     }
1027   }

1099     // Report progress, only used in distributed log splitting
1100     private final CancelableProgressable splitReporter;
1101     // How often to send a progress report (default 1/2 master timeout)
1102     private final int report_period;
1103     private long last_report_at = 0;
1104 
1105     public DistributedLogSplittingHelper(CancelableProgressable reporter) {
1106       this.splitReporter = reporter;
1107       report_period = conf.getInt("hbase.splitlog.report.period",
1108           conf.getInt("hbase.splitlog.manager.timeout",
1109               SplitLogManager.DEFAULT_TIMEOUT) / 2);
1110     }
1111 
1112     /***
1113      * @return false if reporter failed progressing
1114      */
1115     private boolean reportProgress() {
1116       if (splitReporter == null) {
1117         return true;
1118       } else {
1119         long t = EnvironmentEdgeManager.currentTimeMillis();
1120         if ((t - last_report_at) > report_period) {
1121           last_report_at = t;

1123             LOG.warn("Failed: reporter.progress asked us to terminate");
1124             return false;
1125           }
1126         }

1132   /**
1133    * Class that manages the output streams from the log splitting process.
1134    */
1135   class OutputSink {
1136     private final Map<byte[], WriterAndPath> logWriters = Collections.synchronizedMap(
1137           new TreeMap<byte[], WriterAndPath>(Bytes.BYTES_COMPARATOR));
1138     private final Map<byte[], Long> regionMaximumEditLogSeqNum = Collections
1139         .synchronizedMap(new TreeMap<byte[], Long>(Bytes.BYTES_COMPARATOR));
1140     private final List<WriterThread> writerThreads = Lists.newArrayList();
1141 
1142     /* Set of regions which we've decided should not output edits */
1143     private final Set<byte[]> blacklistedRegions = Collections.synchronizedSet(
1144         new TreeSet<byte[]>(Bytes.BYTES_COMPARATOR));
1145 
1146     private boolean closeAndCleanCompleted = false;
1147     
1148     private boolean logWritersClosed  = false;
1149 
1150     private final int numThreads;
1151 
1152     public OutputSink() {
1153       // More threads could potentially write faster at the expense
1154       // of causing more disk seeks as the logs are split.
1155       // 3. After a certain setting (probably around 3) the
1156       // process will be bound on the reader in the current
1157       // implementation anyway.
1158       numThreads = conf.getInt(
1159           "hbase.regionserver.hlog.splitlog.writer.threads", 3);
1160     }
1161 
1162     /**
1163      * Start the threads that will pump data from the entryBuffers
1164      * to the output files.
1165      */
1166     synchronized void startWriterThreads() {
1167       for (int i = 0; i < numThreads; i++) {
1168         WriterThread t = new WriterThread(i);
1169         t.start();
1170         writerThreads.add(t);
1171       }
1172     }
1173 
1174     /**
1175      * 
1176      * @return null if failed to report progress
1177      * @throws IOException
1178      */
1179     List<Path> finishWritingAndClose() throws IOException {
1180       LOG.info("Waiting for split writer threads to finish");
1181       boolean progress_failed = false;
1182       try {
1183         for (WriterThread t : writerThreads) {
1184           t.finish();
1185         }
1186         for (WriterThread t : writerThreads) {
1187           if (!progress_failed && !reportProgressIfIsDistributedLogSplitting()) {
1188             progress_failed = true;
1189           }
1190           try {
1191             t.join();
1192           } catch (InterruptedException ie) {
1193             IOException iie = new InterruptedIOException();
1194             iie.initCause(ie);
1195             throw iie;
1196           }
1197           checkForErrors();
1198         }
1199         LOG.info("Split writers finished");
1200         if (progress_failed) {
1201           return null;
1202         }
1203         return closeStreams();
1204       } finally {
1205         List<IOException> thrown = closeLogWriters(null);
1206         if (thrown != null && !thrown.isEmpty()) {
1207           throw MultipleIOException.createIOException(thrown);
1208         }
1209       }
1210     }
1211 
1212     /**
1213      * Close all of the output streams.
1214      * @return the list of paths written.
1215      */
1216     private List<Path> closeStreams() throws IOException {
1217       Preconditions.checkState(!closeAndCleanCompleted);
1218 
1219       final List<Path> paths = new ArrayList<Path>();
1220       final List<IOException> thrown = Lists.newArrayList();
1221       ThreadPoolExecutor closeThreadPool = Threads.getBoundedCachedThreadPool(
1222           numThreads, 30L, TimeUnit.SECONDS, new ThreadFactory() {
1223             private int count = 1;
1224             public Thread newThread(Runnable r) {
1225               Thread t = new Thread(r, "split-log-closeStream-" + count++);
1226               return t;
1227             }
1228           });
1229       CompletionService<Void> completionService = new ExecutorCompletionService<Void>(
1230           closeThreadPool);
1231       for (final Map.Entry<byte[], WriterAndPath> logWritersEntry : logWriters
1232           .entrySet()) {
1233         completionService.submit(new Callable<Void>() {
1234           public Void call() throws Exception {
1235             WriterAndPath wap = logWritersEntry.getValue();
1236             try {
1237               wap.w.close();
1238             } catch (IOException ioe) {
1239               LOG.error("Couldn't close log at " + wap.p, ioe);
1240               thrown.add(ioe);
1241               return null;
1242             }
1243             LOG.info("Closed path " + wap.p + " (wrote " + wap.editsWritten
1244                 + " edits in " + (wap.nanosSpent / 1000 / 1000) + "ms)");
1245             Path dst = getCompletedRecoveredEditsFilePath(wap.p,
1246                 regionMaximumEditLogSeqNum.get(logWritersEntry.getKey()));
1247             try {
1248               if (!dst.equals(wap.p) && fs.exists(dst)) {
1249                 LOG.warn("Found existing old edits file. It could be the "
1250                     + "result of a previous failed split attempt. Deleting "
1251                     + dst + ", length=" + fs.getFileStatus(dst).getLen());
1252                 if (!fs.delete(dst, false)) {
1253                   LOG.warn("Failed deleting of old " + dst);
1254                   throw new IOException("Failed deleting of old " + dst);
1255                 }
1256               }
1257               // Skip the unit tests which create a splitter that reads and
1258               // writes the data without touching disk.
1259               // TestHLogSplit#testThreading is an example.
1260               if (fs.exists(wap.p)) {
1261                 if (!fs.rename(wap.p, dst)) {
1262                   throw new IOException("Failed renaming " + wap.p + " to "
1263                       + dst);
1264                 }
1265                 LOG.debug("Rename " + wap.p + " to " + dst);
1266               }
1267             } catch (IOException ioe) {
1268               LOG.error("Couldn't rename " + wap.p + " to " + dst, ioe);
1269               thrown.add(ioe);
1270               return null;
1271             }
1272             paths.add(dst);
1273             return null;
1274           }
1275         });
1276       }
1277 
1278       boolean progress_failed = false;
1279       try {
1280         for (int i = 0, n = logWriters.size(); i < n; i++) {
1281           Future<Void> future = completionService.take();
1282           future.get();
1283           if (!progress_failed && !reportProgressIfIsDistributedLogSplitting()) {
1284             progress_failed = true;
1285           }
1286         }
1287       } catch (InterruptedException e) {
1288         IOException iie = new InterruptedIOException();
1289         iie.initCause(e);
1290         throw iie;
1291       } catch (ExecutionException e) {
1292         throw new IOException(e.getCause());
1293       } finally {
1294         closeThreadPool.shutdownNow();
1295       }
1296 
1297       if (!thrown.isEmpty()) {
1298         throw MultipleIOException.createIOException(thrown);
1299       }
1300       logWritersClosed = true;
1301       closeAndCleanCompleted = true;
1302       if (progress_failed) {
1303         return null;
1304       }
1305       return paths;
1306     }
1307     
1308     private List<IOException> closeLogWriters(List<IOException> thrown)
1309         throws IOException {
1310       if (!logWritersClosed) {
1311         if (thrown == null) {
1312           thrown = Lists.newArrayList();
1313         }
1314         try {
1315           for (WriterThread t : writerThreads) {
1316             while (t.isAlive()) {
1317               t.shouldStop = true;
1318               t.interrupt();
1319               try {
1320                 t.join(10);
1321               } catch (InterruptedException e) {
1322                 IOException iie = new InterruptedIOException();
1323                 iie.initCause(e);
1324                 throw iie;
1325               }
1326             }
1327           }
1328         } finally {
1329           synchronized (logWriters) {
1330             for (WriterAndPath wap : logWriters.values()) {
1331               try {
1332                 wap.w.close();
1333               } catch (IOException ioe) {
1334                 LOG.error("Couldn't close log at " + wap.p, ioe);
1335                 thrown.add(ioe);
1336                 continue;
1337               }
1338               LOG.info("Closed path " + wap.p + " (wrote " + wap.editsWritten
1339                   + " edits in " + (wap.nanosSpent / 1000 / 1000) + "ms)");
1340             }
1341           }
1342           logWritersClosed = true;
1343         }
1344       }
1345       return thrown;
1346     }

1350      *
1351      * This function is threadsafe so long as multiple threads are always
1352      * acting on different regions.
1353      *
1354      * @return null if this region shouldn't output any logs
1355      */
1356     WriterAndPath getWriterAndPath(Entry entry) throws IOException {
1357       byte region[] = entry.getKey().getEncodedRegionName();
1358       WriterAndPath ret = logWriters.get(region);
1359       if (ret != null) {
1360         return ret;
1361       }
1362       // If we already decided that this region doesn't get any output
1363       // we don't need to check again.
1364       if (blacklistedRegions.contains(region)) {
