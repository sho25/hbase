80  /**
81   * A Store holds a column family in a Region.  Its a memstore and a set of zero
82   * or more StoreFiles, which stretch backwards over time.
83   *
84   * <p>There's no reason to consider append-logging at this level; all logging
85   * and locking is handled at the HRegion level.  Store just provides
86   * services to manage sets of StoreFiles.  One of the most important of those
87   * services is compaction services where files are aggregated once they pass
88   * a configurable threshold.
89   *
90   * <p>The only thing having to do with logs that Store needs to deal with is
91   * the reconstructionLog.  This is a segment of an HRegion's log that might
92   * NOT be present upon startup.  If the param is NULL, there's nothing to do.
93   * If the param is non-NULL, we need to process the log to reconstruct
94   * a TreeMap that might not have been written to disk before the process
95   * died.
96   *
97   * <p>It's assumed that after this constructor returns, the reconstructionLog
98   * file will be deleted (by whoever has instantiated the Store).
99   *
100  * <p>Locking and transactions are handled at a higher level.  This API should
101  * not be called directly but by an HRegion manager.
102  */

156   /**
157    * Constructor
158    * @param basedir qualified path under which the region directory lives;
159    * generally the table subdirectory
160    * @param region
161    * @param family HColumnDescriptor for this column
162    * @param fs file system object
163    * @param confParam configuration object
164    * failed.  Can be null.
165    * @throws IOException
166    */
167   protected HStore(Path basedir, HRegion region, HColumnDescriptor family,
168       FileSystem fs, Configuration confParam)
169   throws IOException {
170 
171     HRegionInfo info = region.getRegionInfo();
172     this.fs = fs;
173     // Assemble the store's home directory.
174     Path p = getStoreHomedir(basedir, info.getEncodedName(), family.getName());
175     // Ensure it exists.
176     this.homedir = createStoreHomeDir(this.fs, p);
177     this.region = region;
178     this.family = family;
179     // 'conf' renamed to 'confParam' b/c we use this.conf in the constructor
180     this.conf = new CompoundConfiguration()
181       .add(confParam)
182       .add(family.getValues());
183     this.blocksize = family.getBlocksize();
184 
185     this.dataBlockEncoder =
186         new HFileDataBlockEncoderImpl(family.getDataBlockEncodingOnDisk(),
187             family.getDataBlockEncoding());
188 
189     this.comparator = info.getComparator();
190     // Get TTL
191     this.ttl = determineTTLFromFamily(family);
192     // used by ScanQueryMatcher
193     long timeToPurgeDeletes =
194         Math.max(conf.getLong("hbase.hstore.time.to.purge.deletes", 0), 0);
195     LOG.trace("Time to purge deletes set to " + timeToPurgeDeletes +
196         "ms in store " + this);
197     // Why not just pass a HColumnDescriptor in here altogether?  Even if have
198     // to clone it?
199     scanInfo = new ScanInfo(family, ttl, timeToPurgeDeletes, this.comparator);
200     this.memstore = new MemStore(conf, this.comparator);
201 
202     // Setting up cache configuration for this family
203     this.cacheConf = new CacheConfig(conf, family);
204     this.blockingStoreFileCount =
205       conf.getInt("hbase.hstore.blockingStoreFiles", 7);
206 
207     this.verifyBulkLoads = conf.getBoolean("hbase.hstore.bulkload.verify", false);
208 
209     if (HStore.closeCheckInterval == 0) {
210       HStore.closeCheckInterval = conf.getInt(
211           "hbase.hstore.close.check.interval", 10*1000*1000 /* 10 MB */);
212     }
213     this.storefiles = sortAndClone(loadStoreFiles());
214 
215     // Initialize checksum type from name. The names are CRC32, CRC32C, etc.
216     this.checksumType = getChecksumType(conf);
217     // initilize bytes per checksum
218     this.bytesPerChecksum = getBytesPerChecksum(conf);
219     // Create a compaction tool instance
220     this.compactor = new Compactor(conf);
221     // Create a compaction manager.
222     this.compactionPolicy = new CompactionPolicy(conf, this);
223   }

225   /**
226    * @param family
227    * @return
228    */
229   private static long determineTTLFromFamily(final HColumnDescriptor family) {
230     // HCD.getTimeToLive returns ttl in seconds.  Convert to milliseconds.
231     long ttl = family.getTimeToLive();
232     if (ttl == HConstants.FOREVER) {
233       // Default is unlimited ttl.
234       ttl = Long.MAX_VALUE;
235     } else if (ttl == -1) {
236       ttl = Long.MAX_VALUE;
237     } else {
238       // Second -> ms adjust for user data
239       ttl *= 1000;
240     }
241     return ttl;
242   }

385   /**
386    * Creates an unsorted list of StoreFile loaded in parallel
387    * from the given directory.
388    * @throws IOException
389    */
390   private List<StoreFile> loadStoreFiles() throws IOException {
391     ArrayList<StoreFile> results = new ArrayList<StoreFile>();
392     FileStatus files[] = getStoreFiles();
393 
394     if (files == null || files.length == 0) {
395       return results;
396     }
397     // initialize the thread pool for opening store files in parallel..
398     ThreadPoolExecutor storeFileOpenerThreadPool =
399       this.region.getStoreFileOpenAndCloseThreadPool("StoreFileOpenerThread-" +
400           this.family.getNameAsString());
401     CompletionService<StoreFile> completionService =
402       new ExecutorCompletionService<StoreFile>(storeFileOpenerThreadPool);
403 
404     int totalValidStoreFile = 0;
405     for (int i = 0; i < files.length; i++) {
406       // Skip directories.
407       if (files[i].isDir()) {
408         continue;
409       }
410       final Path p = files[i].getPath();
411       // Check for empty hfile. Should never be the case but can happen
412       // after data loss in hdfs for whatever reason (upgrade, etc.): HBASE-646
413       // NOTE: that the HFileLink is just a name, so it's an empty file.
414       if (!HFileLink.isHFileLink(p) && this.fs.getFileStatus(p).getLen() <= 0) {
415         LOG.warn("Skipping " + p + " because its empty. HBASE-646 DATA LOSS?");
416         continue;
417       }
418 
419       // open each store file in parallel
420       completionService.submit(new Callable<StoreFile>() {
421         public StoreFile call() throws IOException {
422           StoreFile storeFile = new StoreFile(fs, p, conf, cacheConf,
423               family.getBloomFilterType(), dataBlockEncoder);
424           storeFile.createReader();
425           return storeFile;
426         }
427       });
428       totalValidStoreFile++;
429     }
430 
431     try {
432       for (int i = 0; i < totalValidStoreFile; i++) {
433         Future<StoreFile> future = completionService.take();
434         StoreFile storeFile = future.get();
435         long length = storeFile.getReader().length();
436         this.storeSize += length;
437         this.totalUncompressedBytes +=
438           storeFile.getReader().getTotalUncompressedBytes();
439         if (LOG.isDebugEnabled()) {
440           LOG.debug("loaded " + storeFile.toStringDetailed());
441         }
442         results.add(storeFile);
443       }
444     } catch (InterruptedException e) {
445       throw new IOException(e);
446     } catch (ExecutionException e) {
447       throw new IOException(e.getCause());
448     } finally {
449       storeFileOpenerThreadPool.shutdownNow();
450     }
451 
452     return results;
453   }

498   @Override
499   public void assertBulkLoadHFileOk(Path srcPath) throws IOException {
500     HFile.Reader reader  = null;
501     try {
502       LOG.info("Validating hfile at " + srcPath + " for inclusion in "
503           + "store " + this + " region " + this.region);
504       reader = HFile.createReader(srcPath.getFileSystem(conf),
505           srcPath, cacheConf);
506       reader.loadFileInfo();
507 
508       byte[] firstKey = reader.getFirstRowKey();
509       Preconditions.checkState(firstKey != null, "First key can not be null");
510       byte[] lk = reader.getLastKey();
511       Preconditions.checkState(lk != null, "Last key can not be null");
512       byte[] lastKey =  KeyValue.createKeyValueFromKey(lk).getRow();
513 
514       LOG.debug("HFile bounds: first=" + Bytes.toStringBinary(firstKey) +
515           " last=" + Bytes.toStringBinary(lastKey));
516       LOG.debug("Region bounds: first=" +
517           Bytes.toStringBinary(region.getStartKey()) +
518           " last=" + Bytes.toStringBinary(region.getEndKey()));
519 
520       HRegionInfo hri = region.getRegionInfo();
521       if (!hri.containsRange(firstKey, lastKey)) {
522         throw new WrongRegionException(
523             "Bulk load file " + srcPath.toString() + " does not fit inside region "
524             + this.region);
525       }
526 
527       if (verifyBulkLoads) {
528         KeyValue prevKV = null;
529         HFileScanner scanner = reader.getScanner(false, false, false);
530         scanner.seekTo();
531         do {
532           KeyValue kv = scanner.getKeyValue();
533           if (prevKV != null) {
534             if (Bytes.compareTo(prevKV.getBuffer(), prevKV.getRowOffset(),
535                 prevKV.getRowLength(), kv.getBuffer(), kv.getRowOffset(),
536                 kv.getRowLength()) > 0) {
537               throw new InvalidHFileException("Previous row is greater than"
538                   + " current row: path=" + srcPath + " previous="
539                   + Bytes.toStringBinary(prevKV.getKey()) + " current="
540                   + Bytes.toStringBinary(kv.getKey()));
541             }
542             if (Bytes.compareTo(prevKV.getBuffer(), prevKV.getFamilyOffset(),
543                 prevKV.getFamilyLength(), kv.getBuffer(), kv.getFamilyOffset(),
544                 kv.getFamilyLength()) != 0) {
545               throw new InvalidHFileException("Previous key had different"
546                   + " family compared to current key: path=" + srcPath
547                   + " previous=" + Bytes.toStringBinary(prevKV.getFamily())
548                   + " current=" + Bytes.toStringBinary(kv.getFamily()));
549             }
550           }
551           prevKV = kv;
552         } while (scanner.next());
553       }
554     } finally {
555       if (reader != null) reader.close();
556     }
557   }

559   @Override
560   public void bulkLoadHFile(String srcPathStr, long seqNum) throws IOException {
561     Path srcPath = new Path(srcPathStr);
562 
563     // Copy the file if it's on another filesystem
564     FileSystem srcFs = srcPath.getFileSystem(conf);
565     FileSystem desFs = fs instanceof HFileSystem ? ((HFileSystem)fs).getBackingFs() : fs;
566     if (!srcFs.equals(desFs)) {
567       LOG.info("Bulk-load file " + srcPath + " is on different filesystem than " +
568           "the destination store. Copying file over to destination filesystem.");
569       Path tmpPath = getTmpPath();
570       FileUtil.copy(srcFs, srcPath, fs, tmpPath, false, conf);
571       LOG.info("Copied " + srcPath
572           + " to temporary path on destination filesystem: " + tmpPath);
573       srcPath = tmpPath;
574     }
575 
576     Path dstPath = StoreFile.getRandomFilename(fs, homedir,
577         (seqNum == -1) ? null : "_SeqId_" + seqNum + "_");
578     LOG.debug("Renaming bulk load file " + srcPath + " to " + dstPath);
579     StoreFile.rename(fs, srcPath, dstPath);
580 
581     StoreFile sf = new StoreFile(fs, dstPath, this.conf, this.cacheConf,
582         this.family.getBloomFilterType(), this.dataBlockEncoder);
583 
584     StoreFile.Reader r = sf.createReader();
585     this.storeSize += r.length();
586     this.totalUncompressedBytes += r.getTotalUncompressedBytes();
587 
588     LOG.info("Moved HFile " + srcPath + " into store directory " +
589         homedir + " - updating store file list.");
590 
591     // Append the new storefile into the list
592     this.lock.writeLock().lock();
593     try {
594       ArrayList<StoreFile> newFiles = new ArrayList<StoreFile>(storefiles);
595       newFiles.add(sf);
596       this.storefiles = sortAndClone(newFiles);
597     } finally {
598       // We need the lock, as long as we are updating the storefiles
599       // or changing the memstore. Let us release it before calling
600       // notifyChangeReadersObservers. See HBASE-4485 for a possible
601       // deadlock scenario that could have happened if continue to hold
602       // the lock.
603       this.lock.writeLock().unlock();
604     }
605     notifyChangedReadersObservers();
606     LOG.info("Successfully loaded store file " + srcPath
607         + " into store " + this + " (new location: " + dstPath + ")");
608   }

620   @Override
621   public ImmutableList<StoreFile> close() throws IOException {
622     this.lock.writeLock().lock();
623     try {
624       ImmutableList<StoreFile> result = storefiles;
625 
626       // Clear so metrics doesn't find them.
627       storefiles = ImmutableList.of();
628 
629       if (!result.isEmpty()) {
630         // initialize the thread pool for closing store files in parallel.
631         ThreadPoolExecutor storeFileCloserThreadPool = this.region
632             .getStoreFileOpenAndCloseThreadPool("StoreFileCloserThread-"
633                 + this.family.getNameAsString());
634 
635         // close each store file in parallel
636         CompletionService<Void> completionService =
637           new ExecutorCompletionService<Void>(storeFileCloserThreadPool);
638         for (final StoreFile f : result) {
639           completionService.submit(new Callable<Void>() {
640             public Void call() throws IOException {
641               f.closeReader(true);
642               return null;
643             }
644           });
645         }
646 
647         try {
648           for (int i = 0; i < result.size(); i++) {
649             Future<Void> future = completionService.take();
650             future.get();
651           }
652         } catch (InterruptedException e) {
653           throw new IOException(e);
654         } catch (ExecutionException e) {
655           throw new IOException(e.getCause());
656         } finally {
657           storeFileCloserThreadPool.shutdownNow();
658         }
659       }
660       LOG.info("Closed " + this);
661       return result;
662     } finally {
663       this.lock.writeLock().unlock();
664     }
665   }

699   /*
700    * @param cache
701    * @param logCacheFlushId
702    * @param snapshotTimeRangeTracker
703    * @param flushedSize The number of bytes flushed
704    * @return Path The path name of the tmp file to which the store was flushed
705    * @throws IOException
706    */
707   private Path internalFlushCache(final SortedSet<KeyValue> set,
708       final long logCacheFlushId,
709       TimeRangeTracker snapshotTimeRangeTracker,
710       AtomicLong flushedSize,
711       MonitoredTask status)
712       throws IOException {
713     StoreFile.Writer writer;
714     // Find the smallest read point across all the Scanners.
715     long smallestReadPoint = region.getSmallestReadPoint();
716     long flushed = 0;
717     Path pathName;
718     // Don't flush if there are no entries.
719     if (set.size() == 0) {
720       return null;
721     }
722     // Use a store scanner to find which rows to flush.
723     // Note that we need to retain deletes, hence
724     // treat this as a minor compaction.
725     InternalScanner scanner = null;
726     KeyValueScanner memstoreScanner = new CollectionBackedScanner(set, this.comparator);
727     if (getHRegion().getCoprocessorHost() != null) {
728       scanner = getHRegion().getCoprocessorHost()
729           .preFlushScannerOpen(this, memstoreScanner);
730     }
731     if (scanner == null) {
732       Scan scan = new Scan();
733       scan.setMaxVersions(scanInfo.getMaxVersions());
734       scanner = new StoreScanner(this, scanInfo, scan,
735           Collections.singletonList(memstoreScanner), ScanType.MINOR_COMPACT,
736           this.region.getSmallestReadPoint(), HConstants.OLDEST_TIMESTAMP);
737     }
738     if (getHRegion().getCoprocessorHost() != null) {
739       InternalScanner cpScanner =
740         getHRegion().getCoprocessorHost().preFlush(this, scanner);
741       // NULL scanner returned from coprocessor hooks means skip normal processing
742       if (cpScanner == null) {
743         return null;
744       }
745       scanner = cpScanner;
746     }
747     try {
748       int compactionKVMax = conf.getInt(HConstants.COMPACTION_KV_MAX, 10);
749       // TODO:  We can fail in the below block before we complete adding this
750       // flush to list of store files.  Add cleanup of anything put on filesystem
751       // if we fail.
752       synchronized (flushLock) {
753         status.setStatus("Flushing " + this + ": creating writer");
754         // A. Write the map out to the disk
755         writer = createWriterInTmp(set.size());
756         writer.setTimeRangeTracker(snapshotTimeRangeTracker);
757         pathName = writer.getPath();
758         try {
759           List<KeyValue> kvs = new ArrayList<KeyValue>();
760           boolean hasMore;
761           do {
762             hasMore = scanner.next(kvs, compactionKVMax);
763             if (!kvs.isEmpty()) {
764               for (KeyValue kv : kvs) {
765                 // If we know that this KV is going to be included always, then let us
766                 // set its memstoreTS to 0. This will help us save space when writing to
767                 // disk.
768                 if (kv.getMemstoreTS() <= smallestReadPoint) {
769                   // let us not change the original KV. It could be in the memstore
770                   // changing its memstoreTS could affect other threads/scanners.
771                   kv = kv.shallowCopy();
772                   kv.setMemstoreTS(0);
773                 }
774                 writer.append(kv);
775                 flushed += this.memstore.heapSizeChange(kv, true);
776               }
777               kvs.clear();
778             }
779           } while (hasMore);
780         } finally {
781           // Write out the log sequence number that corresponds to this output
782           // hfile. Also write current time in metadata as minFlushTime.
783           // The hfile is current up to and including logCacheFlushId.
784           status.setStatus("Flushing " + this + ": appending metadata");
785           writer.appendMetadata(logCacheFlushId, false);
786           status.setStatus("Flushing " + this + ": closing flushed file");
787           writer.close();
788         }
789       }
790     } finally {
791       flushedSize.set(flushed);
792       scanner.close();
793     }
794     if (LOG.isInfoEnabled()) {
795       LOG.info("Flushed " +
796                ", sequenceid=" + logCacheFlushId +
797                ", memsize=" + StringUtils.humanReadableInt(flushed) +
798                ", into tmp file " + pathName);
799     }
800     return pathName;
801   }

977   /**
978    * Compact the StoreFiles.  This method may take some time, so the calling
979    * thread must be able to block for long periods.
980    *
981    * <p>During this time, the Store can work as usual, getting values from
982    * StoreFiles and writing new StoreFiles from the memstore.
983    *
984    * Existing StoreFiles are not destroyed until the new compacted StoreFile is
985    * completely written-out to disk.
986    *
987    * <p>The compactLock prevents multiple simultaneous compactions.
988    * The structureLock prevents us from interfering with other write operations.
989    *
990    * <p>We don't want to hold the structureLock for the whole time, as a compact()
991    * can be lengthy and we want to allow cache-flushes during this period.
992    *
993    * @param cr
994    *          compaction details obtained from requestCompaction()
995    * @throws IOException
996    * @return Storefile we compacted into or null if we failed or opted out early.
997    */
998   StoreFile compact(CompactionRequest cr) throws IOException {
999     if (cr == null || cr.getFiles().isEmpty()) return null;
1000     Preconditions.checkArgument(cr.getStore().toString().equals(this.toString()));
1001     List<StoreFile> filesToCompact = cr.getFiles();
1002     synchronized (filesCompacting) {
1003       // sanity check: we're compacting files that this store knows about
1004       // TODO: change this to LOG.error() after more debugging
1005       Preconditions.checkArgument(filesCompacting.containsAll(filesToCompact));
1006     }
1007 
1008     // Max-sequenceID is the last key in the files we're compacting
1009     long maxId = StoreFile.getMaxSequenceIdInList(filesToCompact, true);
1010 
1011     // Ready to go. Have list of files to compact.
1012     LOG.info("Starting compaction of " + filesToCompact.size() + " file(s) in "
1013         + this + " of " + this.region.getRegionInfo().getRegionNameAsString()
1014         + " into tmpdir=" + region.getTmpDir() + ", seqid=" + maxId + ", totalSize="
1015         + StringUtils.humanReadableInt(cr.getSize()));
1016 
1017     StoreFile sf = null;
1018     long compactionStartTime = EnvironmentEdgeManager.currentTimeMillis();
1019     try {
1020       StoreFile.Writer writer =
1021         this.compactor.compact(this, filesToCompact, cr.isMajor(), maxId);
1022       // Move the compaction into place.
1023       if (this.conf.getBoolean("hbase.hstore.compaction.complete", true)) {
1024         sf = completeCompaction(filesToCompact, writer);
1025         if (region.getCoprocessorHost() != null) {
1026           region.getCoprocessorHost().postCompact(this, sf);
1027         }
1028       } else {
1029         // Create storefile around what we wrote with a reader on it.
1030         sf = new StoreFile(this.fs, writer.getPath(), this.conf, this.cacheConf,
1031           this.family.getBloomFilterType(), this.dataBlockEncoder);
1032         sf.createReader();
1033       }
1034     } finally {
1035       synchronized (filesCompacting) {
1036         filesCompacting.removeAll(filesToCompact);
1037       }
1038     }
1039 
1040     long now = EnvironmentEdgeManager.currentTimeMillis();
1041     LOG.info("Completed" + (cr.isMajor() ? " major " : " ") + "compaction of "
1042         + filesToCompact.size() + " file(s) in " + this + " of "
1043         + this.region.getRegionInfo().getRegionNameAsString()
1044         + " into " +
1045         (sf == null ? "none" : sf.getPath().getName()) +
1046         ", size=" + (sf == null ? "none" :
1047           StringUtils.humanReadableInt(sf.getReader().length()))
1048         + "; total size for store is " + StringUtils.humanReadableInt(storeSize)
1049         + ". This selection was in queue for "
1050         + StringUtils.formatTimeDiff(compactionStartTime, cr.getSelectionTime())
1051         + ", and took " + StringUtils.formatTimeDiff(now, compactionStartTime)
1052         + " to execute.");
1053     return sf;
1054   }

1132   public CompactionRequest requestCompaction(int priority) throws IOException {
1133     // don't even select for compaction if writes are disabled
1134     if (!this.region.areWritesEnabled()) {
1135       return null;
1136     }
1137 
1138     CompactionRequest ret = null;
1139     this.lock.readLock().lock();
1140     try {
1141       synchronized (filesCompacting) {
1142         // candidates = all storefiles not already in compaction queue
1143         List<StoreFile> candidates = Lists.newArrayList(storefiles);
1144         if (!filesCompacting.isEmpty()) {
1145           // exclude all files older than the newest file we're currently
1146           // compacting. this allows us to preserve contiguity (HBASE-2856)
1147           StoreFile last = filesCompacting.get(filesCompacting.size() - 1);
1148           int idx = candidates.indexOf(last);
1149           Preconditions.checkArgument(idx != -1);
1150           candidates.subList(0, idx + 1).clear();
1151         }
1152 
1153         boolean override = false;
1154         if (region.getCoprocessorHost() != null) {
1155           override = region.getCoprocessorHost().preCompactSelection(
1156               this, candidates);
1157         }
1158         CompactSelection filesToCompact;
1159         if (override) {
1160           // coprocessor is overriding normal file selection
1161           filesToCompact = new CompactSelection(candidates);
1162         } else {
1163           boolean isUserCompaction = priority == Store.PRIORITY_USER;
1164           filesToCompact = compactionPolicy.selectCompaction(candidates, isUserCompaction,
1165               forceMajor && filesCompacting.isEmpty());
1166         }
1167 
1168         if (region.getCoprocessorHost() != null) {
1169           region.getCoprocessorHost().postCompactSelection(this,
1170               ImmutableList.copyOf(filesToCompact.getFilesToCompact()));
1171         }
1172 
1173         // no files to compact
1174         if (filesToCompact.getFilesToCompact().isEmpty()) {
1175           return null;
1176         }
1177 
1178         // basic sanity check: do not try to compact the same StoreFile twice.
1179         if (!Collections.disjoint(filesCompacting, filesToCompact.getFilesToCompact())) {
1180           // TODO: change this from an IAE to LOG.error after sufficient testing
1181           Preconditions.checkArgument(false, "%s overlaps with %s",
1182               filesToCompact, filesCompacting);
1183         }
1184         filesCompacting.addAll(filesToCompact.getFilesToCompact());
1185         Collections.sort(filesCompacting, StoreFile.Comparators.SEQ_ID);
1186 
1187         // major compaction iff all StoreFiles are included
1188         boolean isMajor =
1189             (filesToCompact.getFilesToCompact().size() == this.storefiles.size());
1190         if (isMajor) {
1191           // since we're enqueuing a major, update the compaction wait interval
1192           this.forceMajor = false;
1193         }
1194 
1195         LOG.debug(getHRegionInfo().getEncodedName() + " - " +
1196             getColumnFamilyName() + ": Initiating " +
1197             (isMajor ? "major" : "minor") + " compaction");
1198 
1199         // everything went better than expected. create a compaction request
1200         int pri = getCompactPriority(priority);
1201         ret = new CompactionRequest(region, this, filesToCompact, isMajor, pri);
1202       }
1203     } finally {
1204       this.lock.readLock().unlock();
1205     }
1206     if (ret != null) {
1207       CompactionRequest.preRequest(ret);
1208     }
1209     return ret;
1210   }

1245   /*
1246    * <p>It works by processing a compaction that's been written to disk.
1247    *
1248    * <p>It is usually invoked at the end of a compaction, but might also be
1249    * invoked at HStore startup, if the prior execution died midway through.
1250    *
1251    * <p>Moving the compacted TreeMap into place means:
1252    * <pre>
1253    * 1) Moving the new compacted StoreFile into place
1254    * 2) Unload all replaced StoreFile, close and collect list to delete.
1255    * 3) Loading the new TreeMap.
1256    * 4) Compute new store size
1257    * </pre>
1258    *
1259    * @param compactedFiles list of files that were compacted
1260    * @param compactedFile StoreFile that is the result of the compaction
1261    * @return StoreFile created. May be null.
1262    * @throws IOException
1263    */
1264   StoreFile completeCompaction(final Collection<StoreFile> compactedFiles,
1265                                        final StoreFile.Writer compactedFile)
1266       throws IOException {
1267     // 1. Moving the new files into place -- if there is a new file (may not
1268     // be if all cells were expired or deleted).
1269     StoreFile result = null;
1270     if (compactedFile != null) {
1271       validateStoreFile(compactedFile.getPath());
1272       // Move the file into the right spot
1273       Path origPath = compactedFile.getPath();
1274       Path destPath = new Path(homedir, origPath.getName());
1275       LOG.info("Renaming compacted file at " + origPath + " to " + destPath);
1276       if (!fs.rename(origPath, destPath)) {
1277         LOG.error("Failed move of compacted file " + origPath + " to " +
1278             destPath);
1279         throw new IOException("Failed move of compacted file " + origPath +
1280             " to " + destPath);
1281       }
1282       result = new StoreFile(this.fs, destPath, this.conf, this.cacheConf,
1283           this.family.getBloomFilterType(), this.dataBlockEncoder);
1284       result.createReader();
1285     }
1286     try {
1287       this.lock.writeLock().lock();
1288       try {
1289         // Change this.storefiles so it reflects new state but do not
1290         // delete old store files until we have sent out notification of
1291         // change in case old files are still being accessed by outstanding
1292         // scanners.
1293         ArrayList<StoreFile> newStoreFiles = Lists.newArrayList(storefiles);
1294         newStoreFiles.removeAll(compactedFiles);
1295         filesCompacting.removeAll(compactedFiles); // safe bc: lock.writeLock()
1296 
1297         // If a StoreFile result, move it into place.  May be null.
1298         if (result != null) {
1299           newStoreFiles.add(result);
1300         }
1301 
1302         this.storefiles = sortAndClone(newStoreFiles);
1303       } finally {
1304         // We need the lock, as long as we are updating the storefiles
1305         // or changing the memstore. Let us release it before calling
1306         // notifyChangeReadersObservers. See HBASE-4485 for a possible
1307         // deadlock scenario that could have happened if continue to hold
1308         // the lock.
1309         this.lock.writeLock().unlock();
1310       }
1311 
1312       // Tell observers that list of StoreFiles has changed.
1313       notifyChangedReadersObservers();
1314 
1315       // let the archive util decide if we should archive or delete the files
1316       LOG.debug("Removing store files after compaction...");
1317       HFileArchiver.archiveStoreFiles(this.fs, this.region, this.conf,
1318         this.family.getName(), compactedFiles);
1319 
1320     } catch (IOException e) {
1321       e = RemoteExceptionHandler.checkIOException(e);
1322       LOG.error("Failed replacing compacted files in " + this +
1323         ". Compacted file is " + (result == null? "none": result.toString()) +
1324         ".  Files replaced " + compactedFiles.toString() +
1325         " some of which may have been already removed", e);
1326     }
1327 
1328     // 4. Compute new store size
1329     this.storeSize = 0L;
1330     this.totalUncompressedBytes = 0L;
1331     for (StoreFile hsf : this.storefiles) {
1332       StoreFile.Reader r = hsf.getReader();
1333       if (r == null) {
1334         LOG.warn("StoreFile " + hsf + " has a null Reader");
1335         continue;
1336       }
1337       this.storeSize += r.length();
1338       this.totalUncompressedBytes += r.getTotalUncompressedBytes();
1339     }
1340     return result;
1341   }

1405   /*
1406    * Check an individual MapFile for the row at or before a given row.
1407    * @param f
1408    * @param state
1409    * @throws IOException
1410    */
1411   private void rowAtOrBeforeFromStoreFile(final StoreFile f,
1412                                           final GetClosestRowBeforeTracker state)
1413       throws IOException {
1414     StoreFile.Reader r = f.getReader();
1415     if (r == null) {
1416       LOG.warn("StoreFile " + f + " has a null Reader");
1417       return;
1418     }
1419     if (r.getEntries() == 0) {
1420       LOG.warn("StoreFile " + f + " is a empty store file");
1421       return;
1422     }
1423     // TODO: Cache these keys rather than make each time?
1424     byte [] fk = r.getFirstKey();
1425     if (fk == null) return;
1426     KeyValue firstKV = KeyValue.createKeyValueFromKey(fk, 0, fk.length);
1427     byte [] lk = r.getLastKey();
1428     KeyValue lastKV = KeyValue.createKeyValueFromKey(lk, 0, lk.length);
1429     KeyValue firstOnRow = state.getTargetKey();
1430     if (this.comparator.compareRows(lastKV, firstOnRow) < 0) {
1431       // If last key in file is not of the target table, no candidates in this
1432       // file.  Return.
1433       if (!state.isTargetTable(lastKV)) return;
1434       // If the row we're looking for is past the end of file, set search key to
1435       // last key. TODO: Cache last and first key rather than make each time.
1436       firstOnRow = new KeyValue(lastKV.getRow(), HConstants.LATEST_TIMESTAMP);
1437     }
1438     // Get a scanner that caches blocks and that uses pread.
1439     HFileScanner scanner = r.getScanner(true, true, false);
1440     // Seek scanner.  If can't seek it, return.
1441     if (!seekToScanner(scanner, firstOnRow, firstKV)) return;
1442     // If we found candidate on firstOnRow, just return. THIS WILL NEVER HAPPEN!
1443     // Unlikely that there'll be an instance of actual first row in table.
1444     if (walkForwardInSingleRow(scanner, firstOnRow, state)) return;
1445     // If here, need to start backing up.
1446     while (scanner.seekBefore(firstOnRow.getBuffer(), firstOnRow.getKeyOffset(),
1447        firstOnRow.getKeyLength())) {
1448       KeyValue kv = scanner.getKeyValue();
1449       if (!state.isTargetTable(kv)) break;
1450       if (!state.isBetterCandidate(kv)) break;
1451       // Make new first on row.
1452       firstOnRow = new KeyValue(kv.getRow(), HConstants.LATEST_TIMESTAMP);
1453       // Seek scanner.  If can't seek it, break.
1454       if (!seekToScanner(scanner, firstOnRow, firstKV)) break;
1455       // If we find something, break;
1456       if (walkForwardInSingleRow(scanner, firstOnRow, state)) break;
1457     }
1458   }

1460   /*
1461    * Seek the file scanner to firstOnRow or first entry in file.
1462    * @param scanner
1463    * @param firstOnRow
1464    * @param firstKV
1465    * @return True if we successfully seeked scanner.
1466    * @throws IOException
1467    */
1468   private boolean seekToScanner(final HFileScanner scanner,
1469                                 final KeyValue firstOnRow,
1470                                 final KeyValue firstKV)
1471       throws IOException {
1472     KeyValue kv = firstOnRow;
1473     // If firstOnRow < firstKV, set to firstKV
1474     if (this.comparator.compareRows(firstKV, firstOnRow) == 0) kv = firstKV;
1475     int result = scanner.seekTo(kv.getBuffer(), kv.getKeyOffset(),
1476       kv.getKeyLength());
1477     return result >= 0;
1478   }

1480   /*
1481    * When we come in here, we are probably at the kv just before we break into
1482    * the row that firstOnRow is on.  Usually need to increment one time to get
1483    * on to the row we are interested in.
1484    * @param scanner
1485    * @param firstOnRow
1486    * @param state
1487    * @return True we found a candidate.
1488    * @throws IOException
1489    */
1490   private boolean walkForwardInSingleRow(final HFileScanner scanner,
1491                                          final KeyValue firstOnRow,
1492                                          final GetClosestRowBeforeTracker state)
1493       throws IOException {
1494     boolean foundCandidate = false;
1495     do {
1496       KeyValue kv = scanner.getKeyValue();
1497       // If we are not in the row, skip.
1498       if (this.comparator.compareRows(kv, firstOnRow) < 0) continue;
1499       // Did we go beyond the target row? If so break.
1500       if (state.isTooFar(kv, firstOnRow)) break;
1501       if (state.isExpired(kv)) {
1502         continue;
1503       }
1504       // If we added something, this row is a contender. break.
1505       if (state.handle(kv)) {
1506         foundCandidate = true;
1507         break;
1508       }
1509     } while(scanner.next());
1510     return foundCandidate;
1511   }

1596     } finally {
