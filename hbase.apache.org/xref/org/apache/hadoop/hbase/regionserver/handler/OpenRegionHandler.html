38  /**
39   * Handles opening of a region on a region server.
40   * <p>
41   * This is executed after receiving an OPEN RPC from the master or client.
42   */
43  @InterfaceAudience.Private
44  public class OpenRegionHandler extends EventHandler {
45    private static final Log LOG = LogFactory.getLog(OpenRegionHandler.class);
46  
47    private final RegionServerServices rsServices;
48  
49    private final HRegionInfo regionInfo;
50    private final HTableDescriptor htd;

52    // We get version of our znode at start of open process and monitor it across
53    // the total open. We'll fail the open if someone hijacks our znode; we can
54    // tell this has happened if version is not as expected.
55    private volatile int version = -1;
56    //version of the offline node that was set by the master
57    private volatile int versionOfOfflineNode = -1;

86    @Override
87    public void process() throws IOException {
88      try {
89        final String name = regionInfo.getRegionNameAsString();
90        if (this.server.isStopped() || this.rsServices.isStopping()) {
91          return;
92        }
93        final String encodedName = regionInfo.getEncodedName();
94  
95        // Check that this region is not already online
96        HRegion region = this.rsServices.getFromOnlineRegions(encodedName);
97  
98        // If fails, just return.  Someone stole the region from under us.
99        // Calling transitionZookeeperOfflineToOpening initalizes this.version.
100       if (!transitionZookeeperOfflineToOpening(encodedName,
101           versionOfOfflineNode)) {
102         LOG.warn("Region was hijacked? It no longer exists, encodedName=" +
103           encodedName);
104         return;
105       }
106 
107       // Open region.  After a successful open, failures in subsequent
108       // processing needs to do a close as part of cleanup.
109       region = openRegion();
110       if (region == null) {
111         tryTransitionToFailedOpen(regionInfo);
112         return;
113       }
114       boolean failed = true;
115       if (tickleOpening("post_region_open")) {
116         if (updateMeta(region)) {
117           failed = false;
118         }
119       }
120       if (failed || this.server.isStopped() ||
121           this.rsServices.isStopping()) {
122         cleanupFailedOpen(region);
123         tryTransitionToFailedOpen(regionInfo);
124         return;
125       }
126 
127       if (!transitionToOpened(region)) {
128         // If we fail to transition to opened, it's because of one of two cases:
129         //    (a) we lost our ZK lease
130         // OR (b) someone else opened the region before us
131         // In either case, we don't need to transition to FAILED_OPEN state.
132         // In case (a), the Master will process us as a dead server. In case
133         // (b) the region is already being handled elsewhere anyway.
134         cleanupFailedOpen(region);
135         return;
136       }
137 
138       // One more check to make sure we are opening instead of closing
139       if (!isRegionStillOpening()) {
140         LOG.warn("Open region aborted since it isn't opening any more");
141         cleanupFailedOpen(region);
142         return;
143       }
144 
145       // Successful region open, and add it to OnlineRegions
146       this.rsServices.addToOnlineRegions(region);
147 
148       // Done!  Successful region open
149       LOG.debug("Opened " + name + " on server:" +
150         this.server.getServerName());
151     } finally {
152       this.rsServices.getRegionsInTransitionInRS().
153           remove(this.regionInfo.getEncodedNameAsBytes());
154     }
155   }

157   /**
158    * Update ZK, ROOT or META.  This can take a while if for example the
159    * .META. is not available -- if server hosting .META. crashed and we are
160    * waiting on it to come back -- so run in a thread and keep updating znode
161    * state meantime so master doesn't timeout our region-in-transition.
162    * Caller must cleanup region if this fails.
163    */
164   boolean updateMeta(final HRegion r) {
165     if (this.server.isStopped() || this.rsServices.isStopping()) {
166       return false;
167     }
168     // Object we do wait/notify on.  Make it boolean.  If set, we're done.
169     // Else, wait.
170     final AtomicBoolean signaller = new AtomicBoolean(false);
171     PostOpenDeployTasksThread t = new PostOpenDeployTasksThread(r,
172       this.server, this.rsServices, signaller);
173     t.start();
174     int assignmentTimeout = this.server.getConfiguration().
175       getInt("hbase.master.assignment.timeoutmonitor.period", 10000);
176     // Total timeout for meta edit.  If we fail adding the edit then close out
177     // the region and let it be assigned elsewhere.
178     long timeout = assignmentTimeout * 10;
179     long now = System.currentTimeMillis();
180     long endTime = now + timeout;
181     // Let our period at which we update OPENING state to be be 1/3rd of the
182     // regions-in-transition timeout period.
183     long period = Math.max(1, assignmentTimeout/ 3);
184     long lastUpdate = now;
185     boolean tickleOpening = true;
186     while (!signaller.get() && t.isAlive() && !this.server.isStopped() &&
187         !this.rsServices.isStopping() && (endTime > now)) {
188       long elapsed = now - lastUpdate;
189       if (elapsed > period) {
190         // Only tickle OPENING if postOpenDeployTasks is taking some time.
191         lastUpdate = now;
192         tickleOpening = tickleOpening("post_open_deploy");
193       }
194       synchronized (signaller) {
195         try {
196           signaller.wait(period);
197         } catch (InterruptedException e) {
198           // Go to the loop check.
199         }
200       }
201       now = System.currentTimeMillis();
202     }
203     // Is thread still alive?  We may have left above loop because server is
204     // stopping or we timed out the edit.  Is so, interrupt it.
205     if (t.isAlive()) {
206       if (!signaller.get()) {
207         // Thread still running; interrupt
208         LOG.debug("Interrupting thread " + t);
209         t.interrupt();
210       }
211       try {
212         t.join();
213       } catch (InterruptedException ie) {
214         LOG.warn("Interrupted joining " +
215           r.getRegionInfo().getRegionNameAsString(), ie);
216         Thread.currentThread().interrupt();
217       }
218     }
219 
220     // Was there an exception opening the region?  This should trigger on
221     // InterruptedException too.  If so, we failed.  Even if tickle opening fails
222     // then it is a failure.
223     return ((!Thread.interrupted() && t.getException() == null) && tickleOpening);
224   }

226   /**
227    * Thread to run region post open tasks. Call {@link #getException()} after
228    * the thread finishes to check for exceptions running
229    * {@link RegionServerServices#postOpenDeployTasks(HRegion, org.apache.hadoop.hbase.catalog.CatalogTracker, boolean)}
230    * .
231    */
232   static class PostOpenDeployTasksThread extends Thread {
233     private Exception exception = null;
234     private final Server server;
235     private final RegionServerServices services;
236     private final HRegion region;
237     private final AtomicBoolean signaller;
238 
239     PostOpenDeployTasksThread(final HRegion region, final Server server,
240         final RegionServerServices services, final AtomicBoolean signaller) {
241       super("PostOpenDeployTasks:" + region.getRegionInfo().getEncodedName());
242       this.setDaemon(true);
243       this.server = server;
244       this.services = services;
245       this.region = region;
246       this.signaller = signaller;
247     }
248 
249     public void run() {
250       try {
251         this.services.postOpenDeployTasks(this.region,
252           this.server.getCatalogTracker(), false);
253       } catch (Exception e) {
254         LOG.warn("Exception running postOpenDeployTasks; region=" +
255           this.region.getRegionInfo().getEncodedName(), e);
256         this.exception = e;
257       }
258       // We're done.  Set flag then wake up anyone waiting on thread to complete.
259       this.signaller.set(true);
260       synchronized (this.signaller) {
261         this.signaller.notify();
262       }
263     }
264 
265     /**
266      * @return Null or the run exception; call this method after thread is done.
267      */
268     Exception getException() {
269       return this.exception;
270     }
271   }

274   /**
275    * @param r Region we're working on.
276    * @return whether znode is successfully transitioned to OPENED state.
277    * @throws IOException
278    */
279   private boolean transitionToOpened(final HRegion r) throws IOException {
280     if (!isRegionStillOpening()) {
281       LOG.warn("Open region aborted since it isn't opening any more");
282       return false;
283     }
284     boolean result = false;
285     HRegionInfo hri = r.getRegionInfo();
286     final String name = hri.getRegionNameAsString();
287     // Finally, Transition ZK node to OPENED
288     try {
289       if (ZKAssign.transitionNodeOpened(this.server.getZooKeeper(), hri,
290           this.server.getServerName(), this.version) == -1) {
291         LOG.warn("Completed the OPEN of region " + name +
292           " but when transitioning from " +
293           " OPENING to OPENED got a version mismatch, someone else clashed " +
294           "so now unassigning -- closing region on server: " +
295           this.server.getServerName());
296       } else {
297         LOG.debug("region transitioned to opened in zookeeper: " +
298           r.getRegionInfo() + ", server: " + this.server.getServerName());
299         result = true;
300       }
301     } catch (KeeperException e) {
302       LOG.error("Failed transitioning node " + name +
303         " from OPENING to OPENED -- closing region", e);
304     }
305     return result;
306   }

308   /**
309    * This is not guaranteed to succeed, we just do our best.
310    * @param hri Region we're working on.
311    * @return whether znode is successfully transitioned to FAILED_OPEN state.
312    */
313   private boolean tryTransitionToFailedOpen(final HRegionInfo hri) {
314     boolean result = false;
315     final String name = hri.getRegionNameAsString();
316     try {
317       LOG.info("Opening of region " + hri + " failed, marking as FAILED_OPEN in ZK");
318       if (ZKAssign.transitionNode(
319           this.server.getZooKeeper(), hri,
320           this.server.getServerName(),
321           EventType.RS_ZK_REGION_OPENING,
322           EventType.RS_ZK_REGION_FAILED_OPEN,
323           this.version) == -1) {
324         LOG.warn("Unable to mark region " + hri + " as FAILED_OPEN. " +
325             "It's likely that the master already timed out this open " +
326             "attempt, and thus another RS already has the region.");
327       } else {
328         result = true;
329       }
330     } catch (KeeperException e) {
331       LOG.error("Failed transitioning node " + name +
332         " from OPENING to FAILED_OPEN", e);
333     }
334     return result;
335   }

337   /**
338    * @return Instance of HRegion if successful open else null.
339    */
340   HRegion openRegion() {
341     HRegion region = null;
342     try {
343       // Instantiate the region.  This also periodically tickles our zk OPENING
344       // state so master doesn't timeout this region in transition.
345       region = HRegion.openHRegion(this.regionInfo, this.htd,
346           this.rsServices.getWAL(), this.server.getConfiguration(),
347           this.rsServices,
348         new CancelableProgressable() {
349           public boolean progress() {
350             // We may lose the znode ownership during the open.  Currently its
351             // too hard interrupting ongoing region open.  Just let it complete
352             // and check we still have the znode after region open.
353             return tickleOpening("open_region_progress");
354           }
355         });
356     } catch (Throwable t) {
357       // We failed open. Our caller will see the 'null' return value
358       // and transition the node back to FAILED_OPEN. If that fails,
359       // we rely on the Timeout Monitor in the master to reassign.
360       LOG.error(
361           "Failed open of region=" + this.regionInfo.getRegionNameAsString()
362               + ", starting to roll back the global memstore size.", t);
363       // Decrease the global memstore size.
364       if (this.rsServices != null) {
365         RegionServerAccounting rsAccounting =
366           this.rsServices.getRegionServerAccounting();
367         if (rsAccounting != null) {
368           rsAccounting.rollbackRegionReplayEditsSize(this.regionInfo.getRegionName());
369         }
370       }
371     }
372     return region;
373   }

375   private void cleanupFailedOpen(final HRegion region) throws IOException {
376     if (region != null) region.close();
377   }

379   private boolean isRegionStillOpening() {
380     byte[] encodedName = regionInfo.getEncodedNameAsBytes();
381     Boolean action = rsServices.getRegionsInTransitionInRS().get(encodedName);
382     return action != null && action.booleanValue();
383   }

385   /**
386    * Transition ZK node from OFFLINE to OPENING.
387    * @param encodedName Name of the znode file (Region encodedName is the znode
388    * name).
389    * @param versionOfOfflineNode - version Of OfflineNode that needs to be compared
390    * before changing the node's state from OFFLINE 
391    * @return True if successful transition.
392    */
393   boolean transitionZookeeperOfflineToOpening(final String encodedName,
394       int versionOfOfflineNode) {
395     if (!isRegionStillOpening()) {
396       LOG.warn("Open region aborted since it isn't opening any more");
397       return false;
398     }
399     // TODO: should also handle transition from CLOSED?
400     try {
401       // Initialize the znode version.
402       this.version = ZKAssign.transitionNode(server.getZooKeeper(), regionInfo,
403           server.getServerName(), EventType.M_ZK_REGION_OFFLINE,
404           EventType.RS_ZK_REGION_OPENING, versionOfOfflineNode);
405     } catch (KeeperException e) {
406       LOG.error("Error transition from OFFLINE to OPENING for region=" +
407         encodedName, e);
408     }
409     boolean b = isGoodVersion();
410     if (!b) {
411       LOG.warn("Failed transition from OFFLINE to OPENING for region=" +
412         encodedName);
413     }
414     return b;
415   }

417   /**
418    * Update our OPENING state in zookeeper.
419    * Do this so master doesn't timeout this region-in-transition.
420    * @param context Some context to add to logs if failure
421    * @return True if successful transition.
422    */
423   boolean tickleOpening(final String context) {
424     if (!isRegionStillOpening()) {
425       LOG.warn("Open region aborted since it isn't opening any more");
426       return false;
427     }
428     // If previous checks failed... do not try again.
429     if (!isGoodVersion()) return false;
430     String encodedName = this.regionInfo.getEncodedName();
431     try {
432       this.version =
433         ZKAssign.retransitionNodeOpening(server.getZooKeeper(),
434           this.regionInfo, this.server.getServerName(), this.version);
435     } catch (KeeperException e) {
436       server.abort("Exception refreshing OPENING; region=" + encodedName +
437         ", context=" + context, e);
438       this.version = -1;
439     }
440     boolean b = isGoodVersion();
441     if (!b) {
442       LOG.warn("Failed refreshing OPENING; region=" + encodedName +
443         ", context=" + context);
444     }
445     return b;
446   }
447 
448   private boolean isGoodVersion() {
449     return this.version != -1;
450   }
