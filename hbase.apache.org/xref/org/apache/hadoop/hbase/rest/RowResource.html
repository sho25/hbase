82    @GET
83    @Produces({MIMETYPE_XML, MIMETYPE_JSON, MIMETYPE_PROTOBUF})
84    public Response get(final @Context UriInfo uriInfo) {
85      if (LOG.isDebugEnabled()) {
86        LOG.debug("GET " + uriInfo.getAbsolutePath());
87      }
88      servlet.getMetrics().incrementRequests(1);
89      try {
90        ResultGenerator generator =
91          ResultGenerator.fromRowSpec(tableResource.getName(), rowspec, null);
92        if (!generator.hasNext()) {
93          throw new WebApplicationException(Response.Status.NOT_FOUND);
94        }
95        int count = 0;
96        CellSetModel model = new CellSetModel();
97        KeyValue value = generator.next();
98        byte[] rowKey = value.getRow();
99        RowModel rowModel = new RowModel(rowKey);
100       do {
101         if (!Bytes.equals(value.getRow(), rowKey)) {
102           model.addRow(rowModel);
103           rowKey = value.getRow();
104           rowModel = new RowModel(rowKey);
105         }
106         rowModel.addCell(new CellModel(value.getFamily(), value.getQualifier(),
107           value.getTimestamp(), value.getValue()));
108         if (++count > rowspec.getMaxValues()) {
109           break;
110         }
111         value = generator.next();
112       } while (value != null);
113       model.addRow(rowModel);
114       servlet.getMetrics().incrementSucessfulGetRequests(1);
115       return Response.ok(model).build();
116     } catch (IOException e) {
117       servlet.getMetrics().incrementFailedGetRequests(1);
118       throw new WebApplicationException(e,
119                   Response.Status.SERVICE_UNAVAILABLE);
120     }
121   }

126     if (LOG.isDebugEnabled()) {
127       LOG.debug("GET " + uriInfo.getAbsolutePath() + " as "+ MIMETYPE_BINARY);
128     }
129     servlet.getMetrics().incrementRequests(1);
130     // doesn't make sense to use a non specific coordinate as this can only

133       throw new WebApplicationException(Response.Status.BAD_REQUEST);
134     }
135     try {
136       ResultGenerator generator =
137         ResultGenerator.fromRowSpec(tableResource.getName(), rowspec, null);
138       if (!generator.hasNext()) {
139         throw new WebApplicationException(Response.Status.NOT_FOUND);
140       }
141       KeyValue value = generator.next();
142       ResponseBuilder response = Response.ok(value.getValue());
143       response.header("X-Timestamp", value.getTimestamp());
144       servlet.getMetrics().incrementSucessfulGetRequests(1);

153   Response update(final CellSetModel model, final boolean replace) {
154     servlet.getMetrics().incrementRequests(1);
155     if (servlet.isReadOnly()) {
156       throw new WebApplicationException(Response.Status.FORBIDDEN);
157     }
158 
159     if (CHECK_PUT.equalsIgnoreCase(check)) {
160       return checkAndPut(model);
161     } else if (CHECK_DELETE.equalsIgnoreCase(check)) {
162       return checkAndDelete(model);
163     } else if (check != null && check.length() > 0) {
164       LOG.warn("Unknown check value: " + check + ", ignored");
165     }
166 
167     HTablePool pool = servlet.getTablePool();
168     HTableInterface table = null;
169     try {
170       List<RowModel> rows = model.getRows();
171       List<Put> puts = new ArrayList<Put>();
172       for (RowModel row: rows) {
173         byte[] key = row.getKey();
174         if (key == null) {
175           key = rowspec.getRow();
176         }
177         if (key == null) {
178           throw new WebApplicationException(Response.Status.BAD_REQUEST);
179         }
180         Put put = new Put(key);
181         int i = 0;
182         for (CellModel cell: row.getCells()) {
183           byte[] col = cell.getColumn();
184           if (col == null) try {
185             col = rowspec.getColumns()[i++];
186           } catch (ArrayIndexOutOfBoundsException e) {
187             col = null;
188           }
189           if (col == null) {
190             throw new WebApplicationException(Response.Status.BAD_REQUEST);
191           }
192           byte [][] parts = KeyValue.parseColumn(col);
193           if (parts.length == 2 && parts[1].length > 0) {
194             put.add(parts[0], parts[1], cell.getTimestamp(), cell.getValue());
195           } else {
196             put.add(parts[0], null, cell.getTimestamp(), cell.getValue());
197           }
198         }
199         puts.add(put);
200         if (LOG.isDebugEnabled()) {
201           LOG.debug("PUT " + put.toString());
202         }
203       }
204       table = pool.getTable(tableResource.getName());
205       table.put(puts);
206       table.flushCommits();
207       ResponseBuilder response = Response.ok();
208       servlet.getMetrics().incrementSucessfulPutRequests(1);
209       return response.build();
210     } catch (IOException e) {
211       servlet.getMetrics().incrementFailedPutRequests(1);
212       throw new WebApplicationException(e,
213                   Response.Status.SERVICE_UNAVAILABLE);
214     } finally {
215       if (table != null) try {
216         table.close();
217       } catch (IOException ioe) { }
218     }
219   }

221   // This currently supports only update of one row at a time.
222   Response updateBinary(final byte[] message, final HttpHeaders headers,
223       final boolean replace) {
224     servlet.getMetrics().incrementRequests(1);
225     if (servlet.isReadOnly()) {
226       throw new WebApplicationException(Response.Status.FORBIDDEN);
227     }
228     HTablePool pool = servlet.getTablePool();
229     HTableInterface table = null;
230     try {
231       byte[] row = rowspec.getRow();
232       byte[][] columns = rowspec.getColumns();
233       byte[] column = null;
234       if (columns != null) {
235         column = columns[0];
236       }
237       long timestamp = HConstants.LATEST_TIMESTAMP;
238       List<String> vals = headers.getRequestHeader("X-Row");
239       if (vals != null && !vals.isEmpty()) {
240         row = Bytes.toBytes(vals.get(0));
241       }
242       vals = headers.getRequestHeader("X-Column");
243       if (vals != null && !vals.isEmpty()) {
244         column = Bytes.toBytes(vals.get(0));
245       }
246       vals = headers.getRequestHeader("X-Timestamp");
247       if (vals != null && !vals.isEmpty()) {
248         timestamp = Long.valueOf(vals.get(0));
249       }
250       if (column == null) {
251         throw new WebApplicationException(Response.Status.BAD_REQUEST);
252       }
253       Put put = new Put(row);
254       byte parts[][] = KeyValue.parseColumn(column);
255       if (parts.length == 2 && parts[1].length > 0) {
256         put.add(parts[0], parts[1], timestamp, message);
257       } else {
258         put.add(parts[0], null, timestamp, message);
259       }
260       table = pool.getTable(tableResource.getName());
261       table.put(put);
262       if (LOG.isDebugEnabled()) {
263         LOG.debug("PUT " + put.toString());
264       }
265       servlet.getMetrics().incrementSucessfulPutRequests(1);
266       return Response.ok().build();
267     } catch (IOException e) {
268       servlet.getMetrics().incrementFailedPutRequests(1);
269       throw new WebApplicationException(e,
270                   Response.Status.SERVICE_UNAVAILABLE);
271     } finally {
272       if (table != null) try {
273         table.close();
274       } catch (IOException ioe) { }
275     }
276   }

320   @DELETE
321   public Response delete(final @Context UriInfo uriInfo) {
322     if (LOG.isDebugEnabled()) {
323       LOG.debug("DELETE " + uriInfo.getAbsolutePath());
324     }
325     servlet.getMetrics().incrementRequests(1);
326     if (servlet.isReadOnly()) {
327       throw new WebApplicationException(Response.Status.FORBIDDEN);
328     }
329     Delete delete = null;
330     if (rowspec.hasTimestamp())
331       delete = new Delete(rowspec.getRow(), rowspec.getTimestamp(), null);
332     else
333       delete = new Delete(rowspec.getRow());
334 
335     for (byte[] column: rowspec.getColumns()) {
336       byte[][] split = KeyValue.parseColumn(column);
337       if (rowspec.hasTimestamp()) {
338         if (split.length == 2 && split[1].length != 0) {
339           delete.deleteColumns(split[0], split[1], rowspec.getTimestamp());
340         } else {
341           delete.deleteFamily(split[0], rowspec.getTimestamp());
342         }
343       } else {
344         if (split.length == 2 && split[1].length != 0) {
345           delete.deleteColumns(split[0], split[1]);
346         } else {
347           delete.deleteFamily(split[0]);
348         }
349       }
350     }
351     HTablePool pool = servlet.getTablePool();
352     HTableInterface table = null;
353     try {
354       table = pool.getTable(tableResource.getName());
355       table.delete(delete);
356       servlet.getMetrics().incrementSucessfulDeleteRequests(1);
357       if (LOG.isDebugEnabled()) {
358         LOG.debug("DELETE " + delete.toString());
359       }
360     } catch (IOException e) {
361       servlet.getMetrics().incrementFailedDeleteRequests(1);
362       throw new WebApplicationException(e, 
363                   Response.Status.SERVICE_UNAVAILABLE);
364     } finally {
365       if (table != null) try {
366         table.close();
367       } catch (IOException ioe) { }
368     }
369     return Response.ok().build();
370   }

372   /**
373    * Validates the input request parameters, parses columns from CellSetModel,
374    * and invokes checkAndPut on HTable.
375    *
376    * @param model instance of CellSetModel
377    * @return Response 200 OK, 304 Not modified, 400 Bad request
378    */
379   Response checkAndPut(final CellSetModel model) {
380     HTablePool pool = servlet.getTablePool();
381     HTableInterface table = null;
382     try {
383       if (model.getRows().size() != 1) {
384         throw new WebApplicationException(Response.Status.BAD_REQUEST);
385       }
386 
387       RowModel rowModel = model.getRows().get(0);
388       byte[] key = rowModel.getKey();
389       if (key == null) {
390         key = rowspec.getRow();
391       }
392 
393       List<CellModel> cellModels = rowModel.getCells();
394       int cellModelCount = cellModels.size();
395       if (key == null || cellModelCount <= 1) {
396         throw new WebApplicationException(Response.Status.BAD_REQUEST);
397       }
398 
399       Put put = new Put(key);
400       CellModel valueToCheckCell = cellModels.get(cellModelCount - 1);
401       byte[] valueToCheckColumn = valueToCheckCell.getColumn();
402       byte[][] valueToPutParts = KeyValue.parseColumn(valueToCheckColumn);
403       if (valueToPutParts.length == 2 && valueToPutParts[1].length > 0) {
404         CellModel valueToPutCell = null;
405         for (int i = 0, n = cellModelCount - 1; i < n ; i++) {
406           if(Bytes.equals(cellModels.get(i).getColumn(),
407               valueToCheckCell.getColumn())) {
408             valueToPutCell = cellModels.get(i);
409             break;
410           }
411         }
412         if (valueToPutCell != null) {
413           put.add(valueToPutParts[0], valueToPutParts[1], valueToPutCell
414             .getTimestamp(), valueToPutCell.getValue());
415         } else {
416           throw new WebApplicationException(Response.Status.BAD_REQUEST);
417         }
418       } else {
419         throw new WebApplicationException(Response.Status.BAD_REQUEST);
420       }
421 
422       table = pool.getTable(this.tableResource.getName());
423       boolean retValue = table.checkAndPut(key, valueToPutParts[0],
424         valueToPutParts[1], valueToCheckCell.getValue(), put);
425       if (LOG.isDebugEnabled()) {
426         LOG.debug("CHECK-AND-PUT " + put.toString() + ", returns " + retValue);
427       }
428       table.flushCommits();
429       ResponseBuilder response = Response.ok();
430       if (!retValue) {
431         response = Response.status(304);
432       }
433       return response.build();
434     } catch (IOException e) {
435       throw new WebApplicationException(e, Response.Status.SERVICE_UNAVAILABLE);
436     } finally {
437       if (table != null) try {
438         table.close();
439       } catch (IOException ioe) { }
440     }
441   }

443   /**
444    * Validates the input request parameters, parses columns from CellSetModel,
445    * and invokes checkAndDelete on HTable.
446    *
447    * @param model instance of CellSetModel
448    * @return Response 200 OK, 304 Not modified, 400 Bad request
449    */
450   Response checkAndDelete(final CellSetModel model) {
451     HTablePool pool = servlet.getTablePool();
452     HTableInterface table = null;
453     Delete delete = null;
454     try {
455       if (model.getRows().size() != 1) {
456         throw new WebApplicationException(Response.Status.BAD_REQUEST);
457       }
458       RowModel rowModel = model.getRows().get(0);
459       byte[] key = rowModel.getKey();
460       if (key == null) {
461         key = rowspec.getRow();
462       }
463       if (key == null) {
464         throw new WebApplicationException(Response.Status.BAD_REQUEST);
465       }
466 
467       delete = new Delete(key);
468       CellModel valueToDeleteCell = rowModel.getCells().get(0);
469       byte[] valueToDeleteColumn = valueToDeleteCell.getColumn();
470       if (valueToDeleteColumn == null) {
471         try {
472           valueToDeleteColumn = rowspec.getColumns()[0];
473         } catch (final ArrayIndexOutOfBoundsException e) {
474           throw new WebApplicationException(Response.Status.BAD_REQUEST);
475         }
476       }
477       byte[][] parts = KeyValue.parseColumn(valueToDeleteColumn);
478       if (parts.length == 2 && parts[1].length > 0) {
479         delete.deleteColumns(parts[0], parts[1]);
480       } else {
481         throw new WebApplicationException(Response.Status.BAD_REQUEST);
482       }
483 
484       table = pool.getTable(tableResource.getName());
485       boolean retValue = table.checkAndDelete(key, parts[0], parts[1],
486         valueToDeleteCell.getValue(), delete);
487       if (LOG.isDebugEnabled()) {
488         LOG.debug("CHECK-AND-DELETE " + delete.toString() + ", returns "
489           + retValue);
490       }
491       table.flushCommits();
492       ResponseBuilder response = Response.ok();
493       if (!retValue) {
494         response = Response.status(304);
495       }
496       return response.build();
497     } catch (IOException e) {
498       throw new WebApplicationException(e, Response.Status.SERVICE_UNAVAILABLE);
499     } finally {
500       if (table != null) try {
501         table.close();
502       } catch (IOException ioe) { }
503     }
504   }
