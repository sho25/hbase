50  /**
51   * Main class for launching REST gateway as a servlet hosted by Jetty.
52   * <p>
53   * The following options are supported:
54   * <ul>
55   * <li>-p --port : service port</li>
56   * <li>-ro --readonly : server mode</li>
57   * </ul>
58   */
59  @InterfaceAudience.Private
60  public class RESTServer implements Constants {
61  
62    private static void printUsageAndExit(Options options, int exitCode) {
63      HelpFormatter formatter = new HelpFormatter();
64      formatter.printHelp("bin/hbase rest start", "", options,
65        "\nTo run the REST server as a daemon, execute " +
66        "bin/hbase-daemon.sh start|stop rest [-p <port>] [-ro]\n", true);
67      System.exit(exitCode);
68    }
69  
70    /**
71     * The main method for the HBase rest server.
72     * @param args command-line arguments
73     * @throws Exception exception
74     */
75    public static void main(String[] args) throws Exception {
76      Log LOG = LogFactory.getLog("RESTServer");
77  
78  	VersionInfo.logVersion();
79      Configuration conf = HBaseConfiguration.create();
80      RESTServlet servlet = RESTServlet.getInstance(conf);
81  
82      Options options = new Options();
83      options.addOption("p", "port", true, "Port to bind to [default: 8080]");
84      options.addOption("ro", "readonly", false, "Respond only to GET HTTP " +
85        "method requests [default: false]");
86  
87      CommandLine commandLine = null;
88      try {
89        commandLine = new PosixParser().parse(options, args);
90      } catch (ParseException e) {
91        LOG.error("Could not parse: ", e);
92        printUsageAndExit(options, -1);
93      }
94  
95      // check for user-defined port setting, if so override the conf
96      if (commandLine != null && commandLine.hasOption("port")) {
97        String val = commandLine.getOptionValue("port");
98        servlet.getConfiguration()
99            .setInt("hbase.rest.port", Integer.valueOf(val));
100       LOG.debug("port set to " + val);
101     }
102 
103     // check if server should only process GET requests, if so override the conf
104     if (commandLine != null && commandLine.hasOption("readonly")) {
105       servlet.getConfiguration().setBoolean("hbase.rest.readonly", true);
106       LOG.debug("readonly set to true");
107     }
108 
109     @SuppressWarnings("unchecked")
110     List<String> remainingArgs = commandLine != null ?
111         commandLine.getArgList() : new ArrayList<String>();
112     if (remainingArgs.size() != 1) {
113       printUsageAndExit(options, 1);
114     }
115 
116     String command = remainingArgs.get(0);
117     if ("start".equals(command)) {
118       // continue and start container
119     } else if ("stop".equals(command)) {
120       System.exit(1);
121     } else {
122       printUsageAndExit(options, 1);
123     }
124 
125     // set up the Jersey servlet container for Jetty
126     ServletHolder sh = new ServletHolder(ServletContainer.class);
127     sh.setInitParameter(
128       "com.sun.jersey.config.property.resourceConfigClass",
129       ResourceConfig.class.getCanonicalName());
130     sh.setInitParameter("com.sun.jersey.config.property.packages",
131       "jetty");
132 
133     // set up Jetty and run the embedded server
134 
135     Server server = new Server();
136 
137     Connector connector = new SelectChannelConnector();
138     connector.setPort(servlet.getConfiguration().getInt("hbase.rest.port", 8080));
139     connector.setHost(servlet.getConfiguration().get("hbase.rest.host", "0.0.0.0"));
140 
141     server.addConnector(connector);
142 
143     server.setSendServerVersion(false);
144     server.setSendDateHeader(false);
145     server.setStopAtShutdown(true);
146       // set up context
147     Context context = new Context(server, "/", Context.SESSIONS);
148     context.addServlet(sh, "/*");
149     context.addFilter(GzipFilter.class, "/*", 0);
150 
151     // login the server principal (if using secure Hadoop)   
152     if (User.isSecurityEnabled() && User.isHBaseSecurityEnabled(conf)) {
153       String machineName = Strings.domainNamePointerToHostName(
154         DNS.getDefaultHost(conf.get("hbase.rest.dns.interface", "default"),
155           conf.get("hbase.rest.dns.nameserver", "default")));
156       User.login(conf, "hbase.rest.keytab.file", "hbase.rest.kerberos.principal",
157         machineName);
158     }
159 
160     // start server
161     server.start();
162     server.join();
163   }
