116   /**
117    * Represents an HFile waiting to be loaded. An queue is used
118    * in this class in order to support the case where a region has
119    * split during the process of the load. When this happens,
120    * the HFile is split into two physical parts across the new
121    * region boundary, and each part is added back into the queue.
122    * The import process finishes when the queue is empty.
123    */
124   static class LoadQueueItem {
125     final byte[] family;
126     final Path hfilePath;
127 
128     public LoadQueueItem(byte[] family, Path hfilePath) {
129       this.family = family;
130       this.hfilePath = hfilePath;
131     }
132 
133     public String toString() {
134       return "family:"+ Bytes.toString(family) + " path:" + hfilePath.toString();
135     }
136   }

138   /**
139    * Walk the given directory for all HFiles, and return a Queue
140    * containing all such files.
141    */
142   private void discoverLoadQueue(Deque<LoadQueueItem> ret, Path hfofDir)
143   throws IOException {
144     FileSystem fs = hfofDir.getFileSystem(getConf());
145 
146     if (!fs.exists(hfofDir)) {
147       throw new FileNotFoundException("HFileOutputFormat dir " +
148           hfofDir + " not found");
149     }
150 
151     FileStatus[] familyDirStatuses = fs.listStatus(hfofDir);
152     if (familyDirStatuses == null) {
153       throw new FileNotFoundException("No families found in " + hfofDir);
154     }
155 
156     for (FileStatus stat : familyDirStatuses) {
157       if (!stat.isDir()) {
158         LOG.warn("Skipping non-directory " + stat.getPath());
159         continue;
160       }
161       Path familyDir = stat.getPath();
162       // Skip _logs, etc
163       if (familyDir.getName().startsWith("_")) continue;
164       byte[] family = familyDir.getName().getBytes();
165       Path[] hfiles = FileUtil.stat2Paths(fs.listStatus(familyDir));
166       for (Path hfile : hfiles) {
167         if (hfile.getName().startsWith("_")) continue;
168         ret.add(new LoadQueueItem(family, hfile));
169       }
170     }
171   }

173   /**
174    * Perform a bulk load of the given directory into the given
175    * pre-existing table.  This method is not threadsafe.
176    * 
177    * @param hfofDir the directory that was provided as the output path
178    * of a job using HFileOutputFormat
179    * @param table the table to load into
180    * @throws TableNotFoundException if table does not yet exist
181    */
182   public void doBulkLoad(Path hfofDir, final HTable table)
183     throws TableNotFoundException, IOException
184   {
185     final HConnection conn = table.getConnection();
186 
187     if (!conn.isTableAvailable(table.getTableName())) {
188       throw new TableNotFoundException("Table " +
189           Bytes.toStringBinary(table.getTableName()) +
190           "is not currently available.");
191     }
192 
193     // initialize thread pools
194     int nrThreads = cfg.getInt("hbase.loadincremental.threads.max",
195         Runtime.getRuntime().availableProcessors());
196     ThreadFactoryBuilder builder = new ThreadFactoryBuilder();
197     builder.setNameFormat("LoadIncrementalHFiles-%1$d");
198     ExecutorService pool = new ThreadPoolExecutor(nrThreads, nrThreads,
199         60, TimeUnit.SECONDS,
200         new LinkedBlockingQueue<Runnable>(),
201         builder.build());
202     ((ThreadPoolExecutor)pool).allowCoreThreadTimeOut(true);
203 
204     // LQI queue does not need to be threadsafe -- all operations on this queue
205     // happen in this thread
206     Deque<LoadQueueItem> queue = new LinkedList<LoadQueueItem>();
207     try {
208       discoverLoadQueue(queue, hfofDir);
209       int count = 0;
210 
211       if (queue.isEmpty()) {
212         LOG.warn("Bulk load operation did not find any files to load in " +
213             "directory " + hfofDir.toUri() + ".  Does it contain files in " +
214             "subdirectories that correspond to column family names?");
215         return;
216       }
217 
218       // Assumes that region splits can happen while this occurs.
219       while (!queue.isEmpty()) {
220         // need to reload split keys each iteration.
221         final Pair<byte[][], byte[][]> startEndKeys = table.getStartEndKeys();
222         if (count != 0) {
223           LOG.info("Split occured while grouping HFiles, retry attempt " +
224               + count + " with " + queue.size() + " files remaining to group or split");
225         }
226 
227         int maxRetries = cfg.getInt("hbase.bulkload.retries.number", 0);
228         if (maxRetries != 0 && count >= maxRetries) {
229           LOG.error("Retry attempted " + count +  " times without completing, bailing out");
230           return;
231         }
232         count++;
233 
234         // Using ByteBuffer for byte[] equality semantics
235         Multimap<ByteBuffer, LoadQueueItem> regionGroups = groupOrSplitPhase(table,
236             pool, queue, startEndKeys);
237 
238         bulkLoadPhase(table, conn, pool, queue, regionGroups);
239 
240         // NOTE: The next iteration's split / group could happen in parallel to
241         // atomic bulkloads assuming that there are splits and no merges, and
242         // that we can atomically pull out the groups we want to retry.
243       }
244 
245     } finally {
246       pool.shutdown();
247       if (queue != null && !queue.isEmpty()) {
248         StringBuilder err = new StringBuilder();
249         err.append("-------------------------------------------------\n");
250         err.append("Bulk load aborted with some files not yet loaded:\n");
251         err.append("-------------------------------------------------\n");
252         for (LoadQueueItem q : queue) {
253           err.append("  ").append(q.hfilePath).append('\n');
254         }
255         LOG.error(err);
256       }
257     }
258   }

260   /**
261    * This takes the LQI's grouped by likely regions and attempts to bulk load
262    * them.  Any failures are re-queued for another pass with the
263    * groupOrSplitPhase.
264    */
265   protected void bulkLoadPhase(final HTable table, final HConnection conn,
266       ExecutorService pool, Deque<LoadQueueItem> queue,
267       final Multimap<ByteBuffer, LoadQueueItem> regionGroups) throws IOException {
268     // atomically bulk load the groups.
269     Set<Future<List<LoadQueueItem>>> loadingFutures = new HashSet<Future<List<LoadQueueItem>>>();
270     for (Entry<ByteBuffer, ? extends Collection<LoadQueueItem>> e: regionGroups.asMap().entrySet()) {
271       final byte[] first = e.getKey().array();
272       final Collection<LoadQueueItem> lqis =  e.getValue();
273 
274       final Callable<List<LoadQueueItem>> call = new Callable<List<LoadQueueItem>>() {
275         public List<LoadQueueItem> call() throws Exception {
276           List<LoadQueueItem> toRetry = tryAtomicRegionLoad(conn, table.getTableName(), first, lqis);
277           return toRetry;
278         }
279       };
280       loadingFutures.add(pool.submit(call));
281     }
282 
283     // get all the results.
284     for (Future<List<LoadQueueItem>> future : loadingFutures) {
285       try {
286         List<LoadQueueItem> toRetry = future.get();
287 
288         // LQIs that are requeued to be regrouped.
289         queue.addAll(toRetry);
290 
291       } catch (ExecutionException e1) {
292         Throwable t = e1.getCause();
293         if (t instanceof IOException) {
294           // At this point something unrecoverable has happened.
295           // TODO Implement bulk load recovery
296           throw new IOException("BulkLoad encountered an unrecoverable problem", t);
297         }
298         LOG.error("Unexpected execution exception during bulk load", e1);
299         throw new IllegalStateException(t);
300       } catch (InterruptedException e1) {
301         LOG.error("Unexpected interrupted exception during bulk load", e1);
302         throw new IllegalStateException(e1);
303       }
304     }
305   }

307   /**
308    * @return A Multimap<startkey, LoadQueueItem> that groups LQI by likely
309    * bulk load region targets.
310    */
311   private Multimap<ByteBuffer, LoadQueueItem> groupOrSplitPhase(final HTable table,
312       ExecutorService pool, Deque<LoadQueueItem> queue,
313       final Pair<byte[][], byte[][]> startEndKeys) throws IOException {
314     // <region start key, LQI> need synchronized only within this scope of this
315     // phase because of the puts that happen in futures.
316     Multimap<ByteBuffer, LoadQueueItem> rgs = HashMultimap.create();
317     final Multimap<ByteBuffer, LoadQueueItem> regionGroups = Multimaps.synchronizedMultimap(rgs);
318 
319     // drain LQIs and figure out bulk load groups
320     Set<Future<List<LoadQueueItem>>> splittingFutures = new HashSet<Future<List<LoadQueueItem>>>();
321     while (!queue.isEmpty()) {
322       final LoadQueueItem item = queue.remove();
323       
324       final Callable<List<LoadQueueItem>> call = new Callable<List<LoadQueueItem>>() {
325         public List<LoadQueueItem> call() throws Exception {
326           List<LoadQueueItem> splits = groupOrSplit(regionGroups, item, table, startEndKeys);
327           return splits;
328         }
329       };
330       splittingFutures.add(pool.submit(call));
331     }
332     // get all the results.  All grouping and splitting must finish before
333     // we can attempt the atomic loads.
334     for (Future<List<LoadQueueItem>> lqis : splittingFutures) {
335       try {
336         List<LoadQueueItem> splits = lqis.get();
337         if (splits != null) {
338           queue.addAll(splits);
339         }
340       } catch (ExecutionException e1) {
341         Throwable t = e1.getCause();
342         if (t instanceof IOException) {
343           LOG.error("IOException during splitting", e1);
344           throw (IOException)t; // would have been thrown if not parallelized,
345         }
346         LOG.error("Unexpected execution exception during splitting", e1);
347         throw new IllegalStateException(t);
348       } catch (InterruptedException e1) {
349         LOG.error("Unexpected interrupted exception during splitting", e1);
350         throw new IllegalStateException(e1);
351       }
352     }
353     return regionGroups;
354   }

362   protected List<LoadQueueItem> splitStoreFile(final LoadQueueItem item,
363       final HTable table, byte[] startKey,
364       byte[] splitKey) throws IOException {
365     final Path hfilePath = item.hfilePath;
366 
367     // We use a '_' prefix which is ignored when walking directory trees
368     // above.
369     final Path tmpDir = new Path(item.hfilePath.getParent(), "_tmp");
370 
371     LOG.info("HFile at " + hfilePath + " no longer fits inside a single " +
372     "region. Splitting...");
373 
374     String uniqueName = getUniqueName(table.getTableName());
375     HColumnDescriptor familyDesc = table.getTableDescriptor().getFamily(item.family);
376     Path botOut = new Path(tmpDir, uniqueName + ".bottom");
377     Path topOut = new Path(tmpDir, uniqueName + ".top");
378     splitStoreFile(getConf(), hfilePath, familyDesc, splitKey,
379         botOut, topOut);
380 
381     // Add these back at the *front* of the queue, so there's a lower
382     // chance that the region will just split again before we get there.
383     List<LoadQueueItem> lqis = new ArrayList<LoadQueueItem>(2);
384     lqis.add(new LoadQueueItem(item.family, botOut));
385     lqis.add(new LoadQueueItem(item.family, topOut));
386 
387     LOG.info("Successfully split into new HFiles " + botOut + " and " + topOut);
388     return lqis;
389   }

391   /**
392    * Attempt to assign the given load queue item into its target region group.
393    * If the hfile boundary no longer fits into a region, physically splits
394    * the hfile such that the new bottom half will fit and returns the list of
395    * LQI's corresponding to the resultant hfiles.
396    *
397    * protected for testing
398    */
399   protected List<LoadQueueItem> groupOrSplit(Multimap<ByteBuffer, LoadQueueItem> regionGroups,
400       final LoadQueueItem item, final HTable table,
401       final Pair<byte[][], byte[][]> startEndKeys)
402       throws IOException {
403     final Path hfilePath = item.hfilePath;
404     final FileSystem fs = hfilePath.getFileSystem(getConf());
405     HFile.Reader hfr = HFile.createReader(fs, hfilePath,
406         new CacheConfig(getConf()));
407     final byte[] first, last;
408     try {
409       hfr.loadFileInfo();
410       first = hfr.getFirstRowKey();
411       last = hfr.getLastRowKey();
412     }  finally {
413       hfr.close();
414     }
415 
416     LOG.info("Trying to load hfile=" + hfilePath +
417         " first=" + Bytes.toStringBinary(first) +
418         " last="  + Bytes.toStringBinary(last));
419     if (first == null || last == null) {
420       assert first == null && last == null;
421       // TODO what if this is due to a bad HFile?
422       LOG.info("hfile " + hfilePath + " has no entries, skipping");
423       return null;
424     }
425     if (Bytes.compareTo(first, last) > 0) {
426       throw new IllegalArgumentException(
427       "Invalid range: " + Bytes.toStringBinary(first) +
428       " > " + Bytes.toStringBinary(last));
429     }
430     int idx = Arrays.binarySearch(startEndKeys.getFirst(), first,
431         Bytes.BYTES_COMPARATOR);
432     if (idx < 0) {
433       // not on boundary, returns -(insertion index).  Calculate region it
434       // would be in.
435       idx = -(idx + 1) - 1;
436     }
437     final int indexForCallable = idx;
438     boolean lastKeyInRange =
439       Bytes.compareTo(last, startEndKeys.getSecond()[idx]) < 0 ||
440       Bytes.equals(startEndKeys.getSecond()[idx], HConstants.EMPTY_BYTE_ARRAY);
441     if (!lastKeyInRange) {
442       List<LoadQueueItem> lqis = splitStoreFile(item, table,
443           startEndKeys.getFirst()[indexForCallable],
444           startEndKeys.getSecond()[indexForCallable]);
445       return lqis;
446     }
447 
448     // group regions.
449     regionGroups.put(ByteBuffer.wrap(startEndKeys.getFirst()[idx]), item);
450     return null;
451   }

453   /**
454    * Attempts to do an atomic load of many hfiles into a region.  If it fails,
455    * it returns a list of hfiles that need to be retried.  If it is successful
456    * it will return an empty list.
457    * 
458    * NOTE: To maintain row atomicity guarantees, region server callable should
459    * succeed atomically and fails atomically.
460    * 
461    * Protected for testing.
462    * 
463    * @return empty list if success, list of items to retry on recoverable
464    * failure
465    */
466   protected List<LoadQueueItem> tryAtomicRegionLoad(final HConnection conn,
467       byte[] tableName, final byte[] first, Collection<LoadQueueItem> lqis) throws IOException {
468 
469     final List<Pair<byte[], String>> famPaths =
470       new ArrayList<Pair<byte[], String>>(lqis.size());
471     for (LoadQueueItem lqi : lqis) {
472       famPaths.add(Pair.newPair(lqi.family, lqi.hfilePath.toString()));
473     }
474 
475     final ServerCallable<Boolean> svrCallable = new ServerCallable<Boolean>(conn,
476         tableName, first) {
477       @Override
478       public Boolean call() throws Exception {
479         LOG.debug("Going to connect to server " + location + " for row "
480             + Bytes.toStringBinary(row));
481         byte[] regionName = location.getRegionInfo().getRegionName();
482         return ProtobufUtil.bulkLoadHFile(server, famPaths, regionName,
483             assignSeqIds);
484       }
485     };
486 
487     try {
488       List<LoadQueueItem> toRetry = new ArrayList<LoadQueueItem>();
489       boolean success = svrCallable.withRetries();
490       if (!success) {
491         LOG.warn("Attempt to bulk load region containing "
492             + Bytes.toStringBinary(first) + " into table "
493             + Bytes.toStringBinary(tableName)  + " with files " + lqis
494             + " failed.  This is recoverable and they will be retried.");
495         toRetry.addAll(lqis); // return lqi's to retry
496       }
497       // success
498       return toRetry;
499     } catch (IOException e) {
500       LOG.error("Encountered unrecoverable error from region server", e);
501       throw e;
502     }
503   }

505   /**
506    * Split a storefile into a top and bottom half, maintaining
507    * the metadata, recreating bloom filters, etc.
508    */
509   static void splitStoreFile(
510       Configuration conf, Path inFile,
511       HColumnDescriptor familyDesc, byte[] splitKey,
512       Path bottomOut, Path topOut) throws IOException
513   {
514     // Open reader with no block cache, and not in-memory
515     Reference topReference = Reference.createTopReference(splitKey);
516     Reference bottomReference = Reference.createBottomReference(splitKey);
517 
518     copyHFileHalf(conf, inFile, topOut, topReference, familyDesc);
519     copyHFileHalf(conf, inFile, bottomOut, bottomReference, familyDesc);
520   }

522   /**
523    * Copy half of an HFile into a new HFile.
524    */
525   private static void copyHFileHalf(
526       Configuration conf, Path inFile, Path outFile, Reference reference,
527       HColumnDescriptor familyDescriptor)
528   throws IOException {
529     FileSystem fs = inFile.getFileSystem(conf);
530     CacheConfig cacheConf = new CacheConfig(conf);
531     HalfStoreFileReader halfReader = null;
532     StoreFile.Writer halfWriter = null;
533     HFileDataBlockEncoder dataBlockEncoder = new HFileDataBlockEncoderImpl(
534         familyDescriptor.getDataBlockEncodingOnDisk(),
535         familyDescriptor.getDataBlockEncoding());
536     try {
537       halfReader = new HalfStoreFileReader(fs, inFile, cacheConf,
538           reference, DataBlockEncoding.NONE);
539       Map<byte[], byte[]> fileInfo = halfReader.loadFileInfo();
540 
541       int blocksize = familyDescriptor.getBlocksize();
542       Algorithm compression = familyDescriptor.getCompression();
543       BloomType bloomFilterType = familyDescriptor.getBloomFilterType();
544 
545       halfWriter = new StoreFile.WriterBuilder(conf, cacheConf,
546           fs, blocksize)
547               .withFilePath(outFile)
548               .withCompression(compression)
549               .withDataBlockEncoder(dataBlockEncoder)
550               .withBloomType(bloomFilterType)
551               .withChecksumType(HStore.getChecksumType(conf))
552               .withBytesPerChecksum(HStore.getBytesPerChecksum(conf))
553               .build();
554       HFileScanner scanner = halfReader.getScanner(false, false, false);
555       scanner.seekTo();
556       do {
557         KeyValue kv = scanner.getKeyValue();
558         halfWriter.append(kv);
559       } while (scanner.next());
560 
561       for (Map.Entry<byte[],byte[]> entry : fileInfo.entrySet()) {
562         if (shouldCopyHFileMetaKey(entry.getKey())) {
563           halfWriter.appendFileInfo(entry.getKey(), entry.getValue());
564         }
565       }
566     } finally {
567       if (halfWriter != null) halfWriter.close();
568       if (halfReader != null) halfReader.close(cacheConf.shouldEvictOnClose());
569     }
570   }

573     return !HFile.isReservedFileInfoKey(key);

580   /*
581    * Infers region boundaries for a new table.
582    * Parameter:
583    *   bdryMap is a map between keys to an integer belonging to {+1, -1}
584    *     If a key is a start key of a file, then it maps to +1
585    *     If a key is an end key of a file, then it maps to -1
586    * Algo:
587    * 1) Poll on the keys in order: 
588    *    a) Keep adding the mapped values to these keys (runningSum) 
589    *    b) Each time runningSum reaches 0, add the start Key from when the runningSum had started to a boundary list.
590    * 2) Return the boundary list. 
591    */
592   public static byte[][] inferBoundaries(TreeMap<byte[], Integer> bdryMap) {
593     ArrayList<byte[]> keysArray = new ArrayList<byte[]>();
594     int runningValue = 0;
595     byte[] currStartKey = null;
596     boolean firstBoundary = true;
597     
598     for (Map.Entry<byte[], Integer> item: bdryMap.entrySet()) {
599       if (runningValue == 0) currStartKey = item.getKey();
600       runningValue += item.getValue();
601       if (runningValue == 0) {
602         if (!firstBoundary) keysArray.add(currStartKey);
603         firstBoundary = false;
604       } 
605     }
606     
607     return keysArray.toArray(new byte[0][0]);
608   }

610   /*
611    * If the table is created for the first time, then "completebulkload" reads the files twice.
612    * More modifications necessary if we want to avoid doing it.
613    */
614   private void createTable(String tableName, String dirPath) throws Exception {
615     Path hfofDir = new Path(dirPath);
616     FileSystem fs = hfofDir.getFileSystem(getConf());
617 
618     if (!fs.exists(hfofDir)) {
619       throw new FileNotFoundException("HFileOutputFormat dir " +
620           hfofDir + " not found");
621     }
622 
623     FileStatus[] familyDirStatuses = fs.listStatus(hfofDir);
624     if (familyDirStatuses == null) {
625       throw new FileNotFoundException("No families found in " + hfofDir);
626     }
627 
628     HTableDescriptor htd = new HTableDescriptor(tableName);
629     HColumnDescriptor hcd = null;
630 
631     // Add column families
632     // Build a set of keys
633     byte[][] keys = null;
634     TreeMap<byte[], Integer> map = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
635     
636     for (FileStatus stat : familyDirStatuses) {
637       if (!stat.isDir()) {
638         LOG.warn("Skipping non-directory " + stat.getPath());
639         continue;
640       }
641       Path familyDir = stat.getPath();
642       // Skip _logs, etc
643       if (familyDir.getName().startsWith("_")) continue;
644       byte[] family = familyDir.getName().getBytes();
645      
646       hcd = new HColumnDescriptor(family);
647       htd.addFamily(hcd);
648       
649       Path[] hfiles = FileUtil.stat2Paths(fs.listStatus(familyDir));
650       for (Path hfile : hfiles) {
651         if (hfile.getName().startsWith("_")) continue;
652         HFile.Reader reader = HFile.createReader(fs, hfile,
653             new CacheConfig(getConf()));
654         final byte[] first, last;
655         try {
656           if (hcd.getCompressionType() != reader.getCompressionAlgorithm()) {
657             hcd.setCompressionType(reader.getCompressionAlgorithm());
658             LOG.info("Setting compression " + hcd.getCompressionType().name() +
659                      " for family " + hcd.toString());
660           }
661           reader.loadFileInfo();
662           first = reader.getFirstRowKey();
663           last =  reader.getLastRowKey();
664 
665           LOG.info("Trying to figure out region boundaries hfile=" + hfile +
666             " first=" + Bytes.toStringBinary(first) +
667             " last="  + Bytes.toStringBinary(last));
668           
669           // To eventually infer start key-end key boundaries
670           Integer value = map.containsKey(first)?(Integer)map.get(first):0;
671           map.put(first, value+1);
672 
673           value = map.containsKey(last)?(Integer)map.get(last):0;
674           map.put(last, value-1);
675         }  finally {
676           reader.close();
677         }
678       }
679     }
680     
681     keys = LoadIncrementalHFiles.inferBoundaries(map);
682     this.hbAdmin.createTable(htd,keys);
683 
684     LOG.info("Table "+ tableName +" is available!!");
685   }
