55  /**
56   * This client class is for invoking the aggregate functions deployed on the
57   * Region Server side via the AggregateService. This class will implement the
58   * supporting functionality for summing/processing the individual results
59   * obtained from the AggregateService for each region.
60   * <p>
61   * This will serve as the client side handler for invoking the aggregate
62   * functions.
63   * <ul>
64   * For all aggregate functions,
65   * <li>start row < end row is an essential condition (if they are not
66   * {@link HConstants#EMPTY_BYTE_ARRAY})
67   * <li>Column family can't be null. In case where multiple families are
68   * provided, an IOException will be thrown. An optional column qualifier can
69   * also be defined.
70   * <li>For methods to find maximum, minimum, sum, rowcount, it returns the
71   * parameter type. For average and std, it returns a double value. For row
72   * count, it returns a long value.
73   */
74  @InterfaceAudience.Public
75  @InterfaceStability.Evolving
76  public class AggregationClient {

89    /**
90     * It gives the maximum value of a column for a given column family for the
91     * given range. In case qualifier is null, a max of all values for the given
92     * family is returned.
93     * @param tableName
94     * @param ci
95     * @param scan
96     * @return max val <R>
97     * @throws Throwable
98     *           The caller is supposed to handle the exception as they are thrown
99     *           & propagated to it.
100    */
101   public <R, S> R max(final byte[] tableName, final ColumnInterpreter<R, S> ci,
102       final Scan scan) throws Throwable {
103     final AggregateArgument requestArg = validateArgAndGetPB(scan, ci);
104     class MaxCallBack implements Batch.Callback<R> {
105       R max = null;
106 
107       R getMax() {
108         return max;
109       }
110 
111       @Override
112       public synchronized void update(byte[] region, byte[] row, R result) {
113         max = (max == null || (result != null && ci.compare(max, result) < 0)) ? result : max;
114       }
115     }
116     MaxCallBack aMaxCallBack = new MaxCallBack();
117     HTable table = null;
118     try {
119       table = new HTable(conf, tableName);
120       table.coprocessorService(AggregateService.class, scan.getStartRow(),
121           scan.getStopRow(), new Batch.Call<AggregateService, R>() {
122             @Override
123             public R call(AggregateService instance) throws IOException {
124               ServerRpcController controller = new ServerRpcController();
125               BlockingRpcCallback<AggregateResponse> rpcCallback =
126                   new BlockingRpcCallback<AggregateResponse>();
127               instance.getMax(controller, requestArg, rpcCallback);
128               AggregateResponse response = rpcCallback.get();
129               if (controller.failedOnException()) {
130                 throw controller.getFailedOn();
131               }
132               if (response.getFirstPartCount() > 0) {
133                 return ci.castToCellType(
134                           ci.parseResponseAsPromotedType(
135                               getBytesFromResponse(response.getFirstPart(0))));
136               }
137               return null;
138             }
139           }, aMaxCallBack);
140     } finally {
141       if (table != null) {
142         table.close();
143       }
144     }
145     return aMaxCallBack.getMax();
146   }

148   private void validateParameters(Scan scan) throws IOException {
149     if (scan == null
150         || (Bytes.equals(scan.getStartRow(), scan.getStopRow()) && !Bytes
151             .equals(scan.getStartRow(), HConstants.EMPTY_START_ROW))
152         || ((Bytes.compareTo(scan.getStartRow(), scan.getStopRow()) > 0) &&
153         	!Bytes.equals(scan.getStopRow(), HConstants.EMPTY_END_ROW))) {
154       throw new IOException(
155           "Agg client Exception: Startrow should be smaller than Stoprow");
156     } else if (scan.getFamilyMap().size() != 1) {
157       throw new IOException("There must be only one family.");
158     }
159   }

336    * It computes average while fetching sum and row count from all the

339    * @param tableName
340    * @param scan

343   private <R, S> Pair<S, Long> getAvgArgs(final byte[] tableName,
344       final ColumnInterpreter<R, S> ci, final Scan scan) throws Throwable {
345     final AggregateArgument requestArg = validateArgAndGetPB(scan, ci);
346     class AvgCallBack implements Batch.Callback<Pair<S, Long>> {
347       S sum = null;
348       Long rowCount = 0l;
349 
350       public Pair<S, Long> getAvgArgs() {
351         return new Pair<S, Long>(sum, rowCount);
352       }
353 
354       @Override
355       public synchronized void update(byte[] region, byte[] row, Pair<S, Long> result) {
356         sum = ci.add(sum, result.getFirst());
357         rowCount += result.getSecond();
358       }
359     }
360     AvgCallBack avgCallBack = new AvgCallBack();
361     HTable table = null;
362     try {
363       table = new HTable(conf, tableName);
364       table.coprocessorService(AggregateService.class, scan.getStartRow(),
365           scan.getStopRow(),
366           new Batch.Call<AggregateService, Pair<S, Long>>() {
367             @Override
368             public Pair<S, Long> call(AggregateService instance)
369                 throws IOException {
370               ServerRpcController controller = new ServerRpcController();
371               BlockingRpcCallback<AggregateResponse> rpcCallback =
372                   new BlockingRpcCallback<AggregateResponse>();
373               instance.getAvg(controller, requestArg, rpcCallback);
374               AggregateResponse response = rpcCallback.get();
375               if (controller.failedOnException()) {
376                 throw controller.getFailedOn();
377               }
378               Pair<S,Long> pair = new Pair<S, Long>(null, 0L);
379               if (response.getFirstPartCount() == 0) {
380                 return pair;
381               }
382               pair.setFirst(ci.parseResponseAsPromotedType(
383                   getBytesFromResponse(response.getFirstPart(0))));
384               ByteBuffer bb = ByteBuffer.allocate(8).put(
385                   getBytesFromResponse(response.getSecondPart()));
386               bb.rewind();
387               pair.setSecond(bb.getLong());
388               return pair;
389             }
390           }, avgCallBack);
391     } finally {
392       if (table != null) {
393         table.close();
394       }
395     }
396     return avgCallBack.getAvgArgs();

417   /**
418    * It computes a global standard deviation for a given column and its value.
419    * Standard deviation is square root of (average of squares -
420    * average*average). From individual regions, it obtains sum, square sum and
421    * number of rows. With these, the above values are computed to get the global
422    * std.
423    * @param tableName
424    * @param scan
425    * @return
426    * @throws Throwable
427    */
428   private <R, S> Pair<List<S>, Long> getStdArgs(final byte[] tableName,
429       final ColumnInterpreter<R, S> ci, final Scan scan) throws Throwable {
430     final AggregateArgument requestArg = validateArgAndGetPB(scan, ci);
431     class StdCallback implements Batch.Callback<Pair<List<S>, Long>> {
432       long rowCountVal = 0l;
433       S sumVal = null, sumSqVal = null;
434 
435       public Pair<List<S>, Long> getStdParams() {
436         List<S> l = new ArrayList<S>();
437         l.add(sumVal);
438         l.add(sumSqVal);
439         Pair<List<S>, Long> p = new Pair<List<S>, Long>(l, rowCountVal);
440         return p;
441       }
442 
443       @Override
444       public synchronized void update(byte[] region, byte[] row, Pair<List<S>, Long> result) {
445         if (result.getFirst().size() > 0) {
446           sumVal = ci.add(sumVal, result.getFirst().get(0));
447           sumSqVal = ci.add(sumSqVal, result.getFirst().get(1));
448           rowCountVal += result.getSecond();
449         }
450       }
451     }
452     StdCallback stdCallback = new StdCallback();
453     HTable table = null;
454     try {
455       table = new HTable(conf, tableName);
456       table.coprocessorService(AggregateService.class, scan.getStartRow(),
457           scan.getStopRow(),
458           new Batch.Call<AggregateService, Pair<List<S>, Long>>() {
459             @Override
460             public Pair<List<S>, Long> call(AggregateService instance)
461                 throws IOException {
462               ServerRpcController controller = new ServerRpcController();
463               BlockingRpcCallback<AggregateResponse> rpcCallback =
464                   new BlockingRpcCallback<AggregateResponse>();
465               instance.getStd(controller, requestArg, rpcCallback);
466               AggregateResponse response = rpcCallback.get();
467               if (controller.failedOnException()) {
468                 throw controller.getFailedOn();
469               }
470               Pair<List<S>,Long> pair = 
471                   new Pair<List<S>, Long>(new ArrayList<S>(), 0L);
472               if (response.getFirstPartCount() == 0) {
473                 return pair;
474               }
475               List<S> list = new ArrayList<S>();
476               for (int i = 0; i < response.getFirstPartCount(); i++) {
477                 list.add(ci.parseResponseAsPromotedType(
478                     getBytesFromResponse(response.getFirstPart(i))));
479               }
480               pair.setFirst(list);
481               ByteBuffer bb = ByteBuffer.allocate(8).put(
482                   getBytesFromResponse(response.getSecondPart()));
483               bb.rewind();
484               pair.setSecond(bb.getLong());
485               return pair;
486             }
487           }, stdCallback);
488     } finally {
489       if (table != null) {
490         table.close();
491       }
492     }
493     return stdCallback.getStdParams();
494   }

496   /**
497    * This is the client side interface/handle for calling the std method for a
498    * given cf-cq combination. It was necessary to add one more call stack as its
499    * return type should be a decimal value, irrespective of what
500    * columninterpreter says. So, this methods collects the necessary parameters
501    * to compute the std and returns the double value.
502    * @param tableName
503    * @param ci
504    * @param scan
505    * @return <R, S>
506    * @throws Throwable
507    */
508   public <R, S> double std(final byte[] tableName, ColumnInterpreter<R, S> ci,
509       Scan scan) throws Throwable {
510     Pair<List<S>, Long> p = getStdArgs(tableName, ci, scan);
511     double res = 0d;
512     double avg = ci.divideForAvg(p.getFirst().get(0), p.getSecond());
513     double avgOfSumSq = ci.divideForAvg(p.getFirst().get(1), p.getSecond());
514     res = avgOfSumSq - (avg) * (avg); // variance
515     res = Math.pow(res, 0.5);
516     return res;
517   }

542         l.add(sumVal);
543         l.add(sumWeights);
544         Pair<NavigableMap<byte[], List<S>>, List<S>> p =

546         return p;
547       }
548 
549       @Override
550       public synchronized void update(byte[] region, byte[] row, List<S> result) {

555     }

566                   new BlockingRpcCallback<AggregateResponse>();
567               instance.getMedian(controller, requestArg, rpcCallback);
568               AggregateResponse response = rpcCallback.get();
569               if (controller.failedOnException()) {
570                 throw controller.getFailedOn();
571               }
572 
573               List<S> list = new ArrayList<S>();
574               for (int i = 0; i < response.getFirstPartCount(); i++) {
575                 list.add(ci.parseResponseAsPromotedType(
576                     getBytesFromResponse(response.getFirstPart(i))));
577               }
578               return list;
579             }
580 
581           }, stdCallback);
582     } finally {
583       if (table != null) {
584         table.close();
585       }
586     }
587     return stdCallback.getMedianParams();
588   }

590   /**
591    * This is the client side interface/handler for calling the median method for a
592    * given cf-cq combination. This method collects the necessary parameters
593    * to compute the median and returns the median.
594    * @param tableName
595    * @param ci
596    * @param scan
597    * @return R the median
598    * @throws Throwable
599    */
600   public <R, S> R median(final byte[] tableName, ColumnInterpreter<R, S> ci,
601       Scan scan) throws Throwable {
602     Pair<NavigableMap<byte[], List<S>>, List<S>> p = getMedianArgs(tableName, ci, scan);
603     byte[] startRow = null;
604     byte[] colFamily = scan.getFamilies()[0];
605     NavigableSet<byte[]> quals = scan.getFamilyMap().get(colFamily);
606     NavigableMap<byte[], List<S>> map = p.getFirst();
607     S sumVal = p.getSecond().get(0);
608     S sumWeights = p.getSecond().get(1);
609     double halfSumVal = ci.divideForAvg(sumVal, 2L);
610     double movingSumVal = 0;
611     boolean weighted = false;
612     if (quals.size() > 1) {
613       weighted = true;
614       halfSumVal = ci.divideForAvg(sumWeights, 2L);
615     }
616     
617     for (Map.Entry<byte[], List<S>> entry : map.entrySet()) {
618       S s = weighted ? entry.getValue().get(1) : entry.getValue().get(0);
619       double newSumVal = movingSumVal + ci.divideForAvg(s, 1L);
620       if (newSumVal > halfSumVal) break;  // we found the region with the median
621       movingSumVal = newSumVal;
622       startRow = entry.getKey();
623     }
624     // scan the region with median and find it
625     Scan scan2 = new Scan(scan);
626     // inherit stop row from method parameter
627     if (startRow != null) scan2.setStartRow(startRow);
628     HTable table = null;
629     ResultScanner scanner = null;
630     try {
