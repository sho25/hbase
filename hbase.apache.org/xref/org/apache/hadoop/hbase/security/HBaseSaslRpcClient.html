51  /**
52   * A utility class that encapsulates SASL logic for RPC client.
53   * Copied from <code>org.apache.hadoop.security</code>
54   */
55  public class HBaseSaslRpcClient {
56    public static final Log LOG = LogFactory.getLog(HBaseSaslRpcClient.class);
57  
58    private final SaslClient saslClient;
59  
60    /**
61     * Create a HBaseSaslRpcClient for an authentication method
62     * 
63     * @param method
64     *          the requested authentication method
65     * @param token
66     *          token to use if needed by the authentication method
67     */
68    public HBaseSaslRpcClient(AuthMethod method,
69        Token<? extends TokenIdentifier> token, String serverPrincipal)
70        throws IOException {
71      switch (method) {
72      case DIGEST:
73        if (LOG.isDebugEnabled())
74          LOG.debug("Creating SASL " + AuthMethod.DIGEST.getMechanismName()
75              + " client to authenticate to service at " + token.getService());
76        saslClient = Sasl.createSaslClient(new String[] { AuthMethod.DIGEST
77            .getMechanismName() }, null, null, HBaseSaslRpcServer.SASL_DEFAULT_REALM,
78            HBaseSaslRpcServer.SASL_PROPS, new SaslClientCallbackHandler(token));
79        break;
80      case KERBEROS:
81        if (LOG.isDebugEnabled()) {
82          LOG
83              .debug("Creating SASL " + AuthMethod.KERBEROS.getMechanismName()
84                  + " client. Server's Kerberos principal name is "
85                  + serverPrincipal);
86        }
87        if (serverPrincipal == null || serverPrincipal.length() == 0) {
88          throw new IOException(
89              "Failed to specify server's Kerberos principal name");
90        }
91        String names[] = HBaseSaslRpcServer.splitKerberosName(serverPrincipal);
92        if (names.length != 3) {
93          throw new IOException(
94            "Kerberos principal does not have the expected format: "
95                  + serverPrincipal);
96        }
97        saslClient = Sasl.createSaslClient(new String[] { AuthMethod.KERBEROS
98            .getMechanismName() }, null, names[0], names[1],
99            HBaseSaslRpcServer.SASL_PROPS, null);
100       break;
101     default:
102       throw new IOException("Unknown authentication method " + method);
103     }
104     if (saslClient == null)
105       throw new IOException("Unable to find SASL client implementation");
106   }

108   private static void readStatus(DataInputStream inStream) throws IOException {
109     int status = inStream.readInt(); // read status
110     if (status != SaslStatus.SUCCESS.state) {
111       throw new RemoteException(WritableUtils.readString(inStream),
112           WritableUtils.readString(inStream));
113     }
114   }

116   /**
117    * Do client side SASL authentication with server via the given InputStream
118    * and OutputStream
119    * 
120    * @param inS
121    *          InputStream to use
122    * @param outS
123    *          OutputStream to use
124    * @return true if connection is set up, or false if needs to switch 
125    *             to simple Auth.
126    * @throws IOException
127    */
128   public boolean saslConnect(InputStream inS, OutputStream outS)
129       throws IOException {
130     DataInputStream inStream = new DataInputStream(new BufferedInputStream(inS));
131     DataOutputStream outStream = new DataOutputStream(new BufferedOutputStream(
132         outS));
133 
134     try {
135       byte[] saslToken = new byte[0];
136       if (saslClient.hasInitialResponse())
137         saslToken = saslClient.evaluateChallenge(saslToken);
138       if (saslToken != null) {
139         outStream.writeInt(saslToken.length);
140         outStream.write(saslToken, 0, saslToken.length);
141         outStream.flush();
142         if (LOG.isDebugEnabled())
143           LOG.debug("Have sent token of size " + saslToken.length
144               + " from initSASLContext.");
145       }
146       if (!saslClient.isComplete()) {
147         readStatus(inStream);
148         int len = inStream.readInt();
149         if (len == HBaseSaslRpcServer.SWITCH_TO_SIMPLE_AUTH) {
150           if (LOG.isDebugEnabled())
151             LOG.debug("Server asks us to fall back to simple auth.");
152           saslClient.dispose();
153           return false;
154         }
155         saslToken = new byte[len];
156         if (LOG.isDebugEnabled())
157           LOG.debug("Will read input token of size " + saslToken.length
158               + " for processing by initSASLContext");
159         inStream.readFully(saslToken);
160       }
161 
162       while (!saslClient.isComplete()) {
163         saslToken = saslClient.evaluateChallenge(saslToken);
164         if (saslToken != null) {
165           if (LOG.isDebugEnabled())
166             LOG.debug("Will send token of size " + saslToken.length
167                 + " from initSASLContext.");
168           outStream.writeInt(saslToken.length);
169           outStream.write(saslToken, 0, saslToken.length);
170           outStream.flush();
171         }
172         if (!saslClient.isComplete()) {
173           readStatus(inStream);
174           saslToken = new byte[inStream.readInt()];
175           if (LOG.isDebugEnabled())
176             LOG.debug("Will read input token of size " + saslToken.length
177                 + " for processing by initSASLContext");
178           inStream.readFully(saslToken);
179         }
180       }
181       if (LOG.isDebugEnabled()) {
182         LOG.debug("SASL client context established. Negotiated QoP: "
183             + saslClient.getNegotiatedProperty(Sasl.QOP));
184       }
185       return true;
186     } catch (IOException e) {
187       try {
188         saslClient.dispose();
189       } catch (SaslException ignored) {
190         // ignore further exceptions during cleanup
191       }
192       throw e;
193     }
194   }

196   /**
197    * Get a SASL wrapped InputStream. Can be called only after saslConnect() has
198    * been called.
199    * 
200    * @param in
201    *          the InputStream to wrap
202    * @return a SASL wrapped InputStream
203    * @throws IOException
204    */
205   public InputStream getInputStream(InputStream in) throws IOException {
206     if (!saslClient.isComplete()) {
207       throw new IOException("Sasl authentication exchange hasn't completed yet");
208     }
209     return new SaslInputStream(in, saslClient);
210   }
211 
212   /**
213    * Get a SASL wrapped OutputStream. Can be called only after saslConnect() has
214    * been called.
215    * 
216    * @param out
217    *          the OutputStream to wrap
218    * @return a SASL wrapped OutputStream
219    * @throws IOException
220    */
221   public OutputStream getOutputStream(OutputStream out) throws IOException {
222     if (!saslClient.isComplete()) {
223       throw new IOException("Sasl authentication exchange hasn't completed yet");
224     }
225     return new SaslOutputStream(out, saslClient);
226   }

228   /** Release resources used by wrapped saslClient */
229   public void dispose() throws SaslException {
230     saslClient.dispose();

233   private static class SaslClientCallbackHandler implements CallbackHandler {
234     private final String userName;
235     private final char[] userPassword;
236 
237     public SaslClientCallbackHandler(Token<? extends TokenIdentifier> token) {
238       this.userName = HBaseSaslRpcServer.encodeIdentifier(token.getIdentifier());
239       this.userPassword = HBaseSaslRpcServer.encodePassword(token.getPassword());
240     }
241 
242     public void handle(Callback[] callbacks)
243         throws UnsupportedCallbackException {
244       NameCallback nc = null;
245       PasswordCallback pc = null;
246       RealmCallback rc = null;
247       for (Callback callback : callbacks) {
248         if (callback instanceof RealmChoiceCallback) {
249           continue;
250         } else if (callback instanceof NameCallback) {
251           nc = (NameCallback) callback;
252         } else if (callback instanceof PasswordCallback) {
253           pc = (PasswordCallback) callback;
254         } else if (callback instanceof RealmCallback) {
255           rc = (RealmCallback) callback;
256         } else {
257           throw new UnsupportedCallbackException(callback,
258               "Unrecognized SASL client callback");
259         }
260       }
261       if (nc != null) {
262         if (LOG.isDebugEnabled())
263           LOG.debug("SASL client callback: setting username: " + userName);
264         nc.setName(userName);
265       }
266       if (pc != null) {
267         if (LOG.isDebugEnabled())
268           LOG.debug("SASL client callback: setting userPassword");
269         pc.setPassword(userPassword);
270       }
271       if (rc != null) {
272         if (LOG.isDebugEnabled())
273           LOG.debug("SASL client callback: setting realm: "
274               + rc.getDefaultText());
275         rc.setText(rc.getDefaultText());
276       }
277     }
278   }
