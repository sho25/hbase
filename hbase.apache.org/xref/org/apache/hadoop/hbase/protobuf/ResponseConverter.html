62    /**
63     * Get the client Results from a protocol buffer ScanResponse
64     *
65     * @param response the protocol buffer ScanResponse
66     * @return the client Results in the response
67     */
68    public static Result[] getResults(final ScanResponse response) {
69      if (response == null) return null;
70      int count = response.getResultCount();
71      Result[] results = new Result[count];
72      for (int i = 0; i < count; i++) {
73        results[i] = ProtobufUtil.toResult(response.getResult(i));
74      }
75      return results;
76    }

78    /**
79     * Get the results from a protocol buffer MultiResponse
80     *
81     * @param proto the protocol buffer MultiResponse to convert
82     * @return the results in the MultiResponse
83     * @throws IOException
84     */
85    public static List<Object> getResults(
86        final ClientProtos.MultiResponse proto) throws IOException {
87      List<Object> results = new ArrayList<Object>();
88      List<ActionResult> resultList = proto.getResultList();
89      for (int i = 0, n = resultList.size(); i < n; i++) {
90        ActionResult result = resultList.get(i);
91        if (result.hasException()) {
92          results.add(ProtobufUtil.toException(result.getException()));
93        } else if (result.hasValue()) {
94          Object value = ProtobufUtil.toObject(result.getValue());
95          if (value instanceof ClientProtos.Result) {
96            results.add(ProtobufUtil.toResult((ClientProtos.Result)value));
97          } else {
98            results.add(value);
99          }
100       } else {
101         results.add(new Result());
102       }
103     }
104     return results;
105   }

107   /**
108    * Wrap a throwable to an action result.
109    *
110    * @param t
111    * @return an action result
112    */
113   public static ActionResult buildActionResult(final Throwable t) {
114     ActionResult.Builder builder = ActionResult.newBuilder();
115     NameBytesPair.Builder parameterBuilder = NameBytesPair.newBuilder();
116     parameterBuilder.setName(t.getClass().getName());
117     parameterBuilder.setValue(
118       ByteString.copyFromUtf8(StringUtils.stringifyException(t)));
119     builder.setException(parameterBuilder.build());
120     return builder.build();
121   }

126   public static UserPermissionsResponse buildUserPermissionsResponse(

138   /**
139    * Get the list of regions to flush from a RollLogWriterResponse
140    *
141    * @param proto the RollLogWriterResponse
142    * @return the the list of regions to flush
143    */
144   public static byte[][] getRegions(final RollWALWriterResponse proto) {
145     if (proto == null || proto.getRegionToFlushCount() == 0) return null;
146     List<byte[]> regions = new ArrayList<byte[]>();
147     for (ByteString region: proto.getRegionToFlushList()) {
148       regions.add(region.toByteArray());
149     }
150     return (byte[][])regions.toArray();
151   }

207    * A utility to build a GetServerInfoResponse.
208    *
209    * @param serverName
210    * @param webuiPort
211    * @return the response
212    */
213   public static GetServerInfoResponse buildGetServerInfoResponse(
214       final ServerName serverName, final int webuiPort) {
215     GetServerInfoResponse.Builder builder = GetServerInfoResponse.newBuilder();
216     ServerInfo.Builder serverInfoBuilder = ServerInfo.newBuilder();
217     serverInfoBuilder.setServerName(ProtobufUtil.toServerName(serverName));
218     if (webuiPort >= 0) {
219       serverInfoBuilder.setWebuiPort(webuiPort);
220     }
221     builder.setServerInfo(serverInfoBuilder.build());
222     return builder.build();
223   }

267   /**
268    * Stores an exception encountered during RPC invocation so it can be passed back
269    * through to the client.
270    * @param controller the controller instance provided by the client when calling the service
271    * @param ioe the exception encountered
272    */
273   public static void setControllerException(RpcController controller, IOException ioe) {
274     if (controller != null) {
275       if (controller instanceof ServerRpcController) {
276         ((ServerRpcController)controller).setFailedOn(ioe);
277       } else {
278         controller.setFailed(StringUtils.stringifyException(ioe));
279       }
280     }
