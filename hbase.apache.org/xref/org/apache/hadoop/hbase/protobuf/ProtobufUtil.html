170   /**
171    * Magic we put ahead of a serialized protobuf message.
172    * For example, all znode content is protobuf messages with the below magic
173    * for preamble.
174    */
175   public static final byte [] PB_MAGIC = new byte [] {'P', 'B', 'U', 'F'};
176   private static final String PB_MAGIC_STR = Bytes.toString(PB_MAGIC);
177 
178   /**
179    * Prepend the passed bytes with four bytes of magic, {@link #PB_MAGIC}, to flag what
180    * follows as a protobuf in hbase.  Prepend these bytes to all content written to znodes, etc.
181    * @param bytes Bytes to decorate
182    * @return The passed <code>bytes</codes> with magic prepended (Creates a new
183    * byte array that is <code>bytes.length</code> plus {@link #PB_MAGIC}.length.
184    */
185   public static byte [] prependPBMagic(final byte [] bytes) {
186     return Bytes.add(PB_MAGIC, bytes);
187   }
188 
189   /**
190    * @param bytes Bytes to check.
191    * @return True if passed <code>bytes</code> has {@link #PB_MAGIC} for a prefix.
192    */
193   public static boolean isPBMagicPrefix(final byte [] bytes) {
194     if (bytes == null || bytes.length < PB_MAGIC.length) return false;
195     return Bytes.compareTo(PB_MAGIC, 0, PB_MAGIC.length, bytes, 0, PB_MAGIC.length) == 0;
196   }
197 
198   /**
199    * @param bytes
200    * @throws DeserializationException if we are missing the pb magic prefix
201    */
202   public static void expectPBMagicPrefix(final byte [] bytes) throws DeserializationException {
203     if (!isPBMagicPrefix(bytes)) {
204       throw new DeserializationException("Missing pb magic " + PB_MAGIC_STR + " prefix");
205     }
206   }
207 
208   /**
209    * @return Length of {@link #PB_MAGIC}
210    */
211   public static int lengthOfPBMagic() {
212     return PB_MAGIC.length;
213   }

231   /**
232    * Convert a protocol buffer Exec to a client Exec
233    *
234    * @param proto the protocol buffer Exec to convert
235    * @return the converted client Exec
236    */
237   @SuppressWarnings("unchecked")
238   public static Exec toExec(
239       final ClientProtos.Exec proto) throws IOException {
240     byte[] row = proto.getRow().toByteArray();
241     String protocolName = proto.getProtocolName();
242     String methodName = proto.getMethodName();
243     List<Object> parameters = new ArrayList<Object>();
244     Class<? extends CoprocessorProtocol> protocol = null;
245     Method method = null;
246     try {
247       List<Class<?>> types = new ArrayList<Class<?>>();
248       for (NameBytesPair parameter: proto.getParameterList()) {
249         String type = parameter.getName();
250         Class<?> declaredClass = PRIMITIVES.get(type);
251         if (declaredClass == null) {
252           declaredClass = Class.forName(parameter.getName());
253         }
254         parameters.add(toObject(parameter));
255         types.add(declaredClass);
256       }
257       Class<?> [] parameterTypes = new Class<?> [types.size()];
258       types.toArray(parameterTypes);
259       protocol = (Class<? extends CoprocessorProtocol>)
260         Class.forName(protocolName);
261       method = protocol.getMethod(methodName, parameterTypes);
262     } catch (NoSuchMethodException nsme) {
263       throw new IOException(nsme);
264     } catch (ClassNotFoundException cnfe) {
265       throw new IOException(cnfe);
266     }
267     Configuration conf = HBaseConfiguration.create();
268     for (NameStringPair p: proto.getPropertyList()) {
269       conf.set(p.getName(), p.getValue());
270     }
271     Object[] parameterObjects = new Object[parameters.size()];
272     parameters.toArray(parameterObjects);
273     return new Exec(conf, row, protocol,
274       method, parameterObjects);
275   }

349   /**
350    * Convert a protocol buffer Get to a client Get
351    *
352    * @param proto the protocol buffer Get to convert
353    * @return the converted client Get
354    * @throws IOException
355    */
356   public static Get toGet(
357       final ClientProtos.Get proto) throws IOException {
358     if (proto == null) return null;
359     byte[] row = proto.getRow().toByteArray();
360     RowLock rowLock = null;
361     if (proto.hasLockId()) {
362       rowLock = new RowLock(proto.getLockId());
363     }
364     Get get = new Get(row, rowLock);
365     if (proto.hasCacheBlocks()) {
366       get.setCacheBlocks(proto.getCacheBlocks());
367     }
368     if (proto.hasMaxVersions()) {
369       get.setMaxVersions(proto.getMaxVersions());
370     }
371     if (proto.hasStoreLimit()) {
372       get.setMaxResultsPerColumnFamily(proto.getStoreLimit());
373     }
374     if (proto.hasStoreOffset()) {
375       get.setRowOffsetPerColumnFamily(proto.getStoreOffset());
376     }
377     if (proto.hasTimeRange()) {
378       HBaseProtos.TimeRange timeRange = proto.getTimeRange();
379       long minStamp = 0;
380       long maxStamp = Long.MAX_VALUE;
381       if (timeRange.hasFrom()) {
382         minStamp = timeRange.getFrom();
383       }
384       if (timeRange.hasTo()) {
385         maxStamp = timeRange.getTo();
386       }
387       get.setTimeRange(minStamp, maxStamp);
388     }
389     if (proto.hasFilter()) {
390       HBaseProtos.Filter filter = proto.getFilter();
391       get.setFilter(ProtobufUtil.toFilter(filter));
392     }
393     for (NameBytesPair attribute: proto.getAttributeList()) {
394       get.setAttribute(attribute.getName(), attribute.getValue().toByteArray());
395     }
396     if (proto.getColumnCount() > 0) {
397       for (Column column: proto.getColumnList()) {
398         byte[] family = column.getFamily().toByteArray();
399         if (column.getQualifierCount() > 0) {
400           for (ByteString qualifier: column.getQualifierList()) {
401             get.addColumn(family, qualifier.toByteArray());
402           }
403         } else {
404           get.addFamily(family);
405         }
406       }
407     }
408     return get;
409   }

411   /**
412    * Convert a protocol buffer Mutate to a Put
413    *
414    * @param proto the protocol buffer Mutate to convert
415    * @return the converted client Put
416    * @throws DoNotRetryIOException
417    */
418   public static Put toPut(
419       final Mutate proto) throws DoNotRetryIOException {
420     MutateType type = proto.getMutateType();
421     assert type == MutateType.PUT : type.name();
422     byte[] row = proto.getRow().toByteArray();
423     long timestamp = HConstants.LATEST_TIMESTAMP;
424     if (proto.hasTimestamp()) {
425       timestamp = proto.getTimestamp();
426     }
427     RowLock lock = null;
428     if (proto.hasLockId()) {
429       lock = new RowLock(proto.getLockId());
430     }
431     Put put = new Put(row, timestamp, lock);
432     put.setWriteToWAL(proto.getWriteToWAL());
433     for (NameBytesPair attribute: proto.getAttributeList()) {
434       put.setAttribute(attribute.getName(),
435         attribute.getValue().toByteArray());
436     }
437     for (ColumnValue column: proto.getColumnValueList()) {
438       byte[] family = column.getFamily().toByteArray();
439       for (QualifierValue qv: column.getQualifierValueList()) {
440         byte[] qualifier = qv.getQualifier().toByteArray();
441         if (!qv.hasValue()) {
442           throw new DoNotRetryIOException(
443             "Missing required field: qualifer value");
444         }
445         byte[] value = qv.getValue().toByteArray();
446         long ts = timestamp;
447         if (qv.hasTimestamp()) {
448           ts = qv.getTimestamp();
449         }
450         put.add(family, qualifier, ts, value);
451       }
452     }
453     return put;
454   }
