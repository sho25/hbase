60  /**
61   * This class serves as a helper for all things related to zookeeper in
62   * replication.
63   * <p/>
64   * The layout looks something like this under zookeeper.znode.parent for the
65   * master cluster:
66   * <p/>
67   *
68   * <pre>
69   * replication/
70   *  state      {contains true or false}
71   *  clusterId  {contains a byte}
72   *  peers/
73   *    1/   {contains a full cluster address}
74   *      peer-state  {contains ENABLED or DISABLED}
75   *    2/
76   *    ...
77   *  rs/ {lists all RS that replicate}
78   *    startcode1/ {lists all peer clusters}
79   *      1/ {lists hlogs to process}
80   *        10.10.1.76%3A53488.123456789 {contains nothing or a position}
81   *        10.10.1.76%3A53488.123456790
82   *        ...
83   *      2/
84   *      ...
85   *    startcode2/
86   *    ...
87   * </pre>
88   */
89  @InterfaceAudience.Private

147   /**
148    * Constructor used by region servers, connects to the peer cluster right away.
149    *
150    * @param server
151    * @param replicating    atomic boolean to start/stop replication
152    * @throws IOException
153    * @throws KeeperException 
154    */
155   public ReplicationZookeeper(final Server server, final AtomicBoolean replicating)
156   throws IOException, KeeperException {
157     this.abortable = server;
158     this.zookeeper = server.getZooKeeper();
159     this.conf = server.getConfiguration();
160     this.replicating = replicating;
161     setZNodes(server);
162 
163     this.peerClusters = new HashMap<String, ReplicationPeer>();
164     ZKUtil.createWithParents(this.zookeeper,
165         ZKUtil.joinZNode(this.replicationZNode, this.replicationStateNodeName));
166     this.rsServerNameZnode = ZKUtil.joinZNode(rsZNode, server.getServerName().toString());
167     ZKUtil.createWithParents(this.zookeeper, this.rsServerNameZnode);
168     connectExistingPeers();
169   }

171   private void setZNodes(Abortable abortable) throws KeeperException {
172     String replicationZNodeName = conf.get("zookeeper.znode.replication", "replication");
173     String peersZNodeName = conf.get("zookeeper.znode.replication.peers", "peers");
174     this.peerStateNodeName = conf.get("zookeeper.znode.replication.peers.state", "peer-state");
175     this.replicationStateNodeName = conf.get("zookeeper.znode.replication.state", "state");
176     String rsZNodeName = conf.get("zookeeper.znode.replication.rs", "rs");
177     this.ourClusterKey = ZKUtil.getZooKeeperClusterKey(this.conf);
178     this.replicationZNode = ZKUtil.joinZNode(this.zookeeper.baseZNode, replicationZNodeName);
179     this.peersZNode = ZKUtil.joinZNode(replicationZNode, peersZNodeName);
180     ZKUtil.createWithParents(this.zookeeper, this.peersZNode);
181     this.rsZNode = ZKUtil.joinZNode(replicationZNode, rsZNodeName);
182     ZKUtil.createWithParents(this.zookeeper, this.rsZNode);
183 
184     // Set a tracker on replicationStateNodeNode
185     this.statusTracker = new ReplicationStatusTracker(this.zookeeper, abortable);
186     statusTracker.start();
187     readReplicationStateZnode();
188   }

190   private void connectExistingPeers() throws IOException, KeeperException {
191     List<String> znodes = ZKUtil.listChildrenNoWatch(this.zookeeper, this.peersZNode);
192     if (znodes != null) {
193       for (String z : znodes) {
194         connectToPeer(z);
195       }
196     }
197   }

213   /**
214    * Map of this cluster's peers for display.
215    * @return A map of peer ids to peer cluster keys
216    */
217   public Map<String,String> listPeers() {
218     Map<String,String> peers = new TreeMap<String,String>();
219     List<String> ids = null;
220     try {
221       ids = ZKUtil.listChildrenNoWatch(this.zookeeper, this.peersZNode);
222       for (String id : ids) {
223         byte[] bytes = ZKUtil.getData(this.zookeeper, ZKUtil.joinZNode(this.peersZNode, id));
224         String clusterKey = null;
225         try {
226           clusterKey = parsePeerFrom(bytes);
227         } catch (DeserializationException de) {
228           LOG.warn("Failed parse of clusterid=" + id + " znode content, continuing.");
229           continue;
230         }
231         peers.put(id, clusterKey);
232       }
233     } catch (KeeperException e) {
234       this.abortable.abort("Cannot get the list of peers ", e);
235     }
236     return peers;
237   }

239   /**
240    * Returns all region servers from given peer
241    *
242    * @param peerClusterId (byte) the cluster to interrogate
243    * @return addresses of all region servers
244    */
245   public List<ServerName> getSlavesAddresses(String peerClusterId) {
246     if (this.peerClusters.size() == 0) {
247       return Collections.emptyList();
248     }
249     ReplicationPeer peer = this.peerClusters.get(peerClusterId);
250     if (peer == null) {
251       return Collections.emptyList();
252     }
253     
254     List<ServerName> addresses;
255     try {
256       addresses = fetchSlavesAddresses(peer.getZkw());
257     } catch (KeeperException ke) {
258       reconnectPeer(ke, peer);
259       addresses = Collections.emptyList();
260     }
261     peer.setRegionServers(addresses);
262     return peer.getRegionServers();
263   }

276   /**
277    * Lists the children of the specified znode, retrieving the data of each
278    * child as a server address.
279    *
280    * Used to list the currently online regionservers and their addresses.
281    *
282    * Sets no watches at all, this method is best effort.
283    *
284    * Returns an empty list if the node has no children.  Returns null if the
285    * parent node itself does not exist.
286    *
287    * @param zkw zookeeper reference
288    * @param znode node to get children of as addresses
289    * @return list of data of children of specified znode, empty if no children,
290    *         null if parent does not exist
291    * @throws KeeperException if unexpected zookeeper exception
292    */
293   public static List<ServerName> listChildrenAndGetAsServerNames(
294       ZooKeeperWatcher zkw, String znode)
295   throws KeeperException {
296     List<String> children = ZKUtil.listChildrenNoWatch(zkw, znode);
297     if(children == null) {
298       return Collections.emptyList();
299     }
300     List<ServerName> addresses = new ArrayList<ServerName>(children.size());
301     for (String child : children) {
302       addresses.add(ServerName.parseServerName(child));
303     }
304     return addresses;
305   }

307   /**
308    * This method connects this cluster to another one and registers it
309    * in this region server's replication znode
310    * @param peerId id of the peer cluster
311    * @throws KeeperException 
312    */
313   public boolean connectToPeer(String peerId)
314       throws IOException, KeeperException {
315     if (peerClusters == null) {
316       return false;
317     }
318     if (this.peerClusters.containsKey(peerId)) {
319       return false;
320     }
321     ReplicationPeer peer = getPeer(peerId);
322     if (peer == null) {
323       return false;
324     }
325     this.peerClusters.put(peerId, peer);
326     ZKUtil.createWithParents(this.zookeeper, ZKUtil.joinZNode(
327         this.rsServerNameZnode, peerId));
328     LOG.info("Added new peer cluster " + peer.getClusterKey());
329     return true;
330   }

332   /**
333    * Helper method to connect to a peer
334    * @param peerId peer's identifier
335    * @return object representing the peer
336    * @throws IOException
337    * @throws KeeperException
338    */
339   public ReplicationPeer getPeer(String peerId) throws IOException, KeeperException{
340     String znode = ZKUtil.joinZNode(this.peersZNode, peerId);
341     byte [] data = ZKUtil.getData(this.zookeeper, znode);
342     String otherClusterKey = "";
343     try {
344       otherClusterKey = parsePeerFrom(data);
345     } catch (DeserializationException e) {
346       LOG.warn("Failed parse of cluster key from peerId=" + peerId
347           + ", specifically the content from the following znode: " + znode);
348     }
349     if (this.ourClusterKey.equals(otherClusterKey)) {
350       LOG.debug("Not connecting to " + peerId + " because it's us");
351       return null;
352     }
353     // Construct the connection to the new peer
354     Configuration otherConf = new Configuration(this.conf);
355     try {
356       ZKUtil.applyClusterKeyToConf(otherConf, otherClusterKey);
357     } catch (IOException e) {
358       LOG.error("Can't get peer because:", e);
359       return null;
360     }
361 
362     ReplicationPeer peer = new ReplicationPeer(otherConf, peerId,
363         otherClusterKey);
364     peer.startStateTracker(this.zookeeper, this.getPeerStateNode(peerId));
365     return peer;
366   }

380   /**
381    * Remove the peer from zookeeper. which will trigger the watchers on every
382    * region server and close their sources
383    * @param id
384    * @throws IllegalArgumentException Thrown when the peer doesn't exist
385    */
386   public void removePeer(String id) throws IOException {
387     try {
388       if (!peerExists(id)) {
389         throw new IllegalArgumentException("Cannot remove inexisting peer");
390       }
391       ZKUtil.deleteNodeRecursively(this.zookeeper,
392           ZKUtil.joinZNode(this.peersZNode, id));
393     } catch (KeeperException e) {
394       throw new IOException("Unable to remove a peer", e);
395     }
396   }

398   /**
399    * Add a new peer to this cluster
400    * @param id peer's identifier
401    * @param clusterKey ZK ensemble's addresses, client port and root znode
402    * @throws IllegalArgumentException Thrown when the peer doesn't exist
403    * @throws IllegalStateException Thrown when a peer already exists, since
404    *         multi-slave isn't supported yet.
405    */
406   public void addPeer(String id, String clusterKey) throws IOException {
407     try {
408       if (peerExists(id)) {
409         throw new IllegalArgumentException("Cannot add existing peer");
410       }
411       ZKUtil.createWithParents(this.zookeeper, this.peersZNode);
412       ZKUtil.createAndWatch(this.zookeeper, ZKUtil.joinZNode(this.peersZNode, id),
413         toByteArray(clusterKey));
414       // A peer is enabled by default
415       ZKUtil.createAndWatch(this.zookeeper, getPeerStateNode(id), ENABLED_ZNODE_BYTES);
416     } catch (KeeperException e) {
417       throw new IOException("Unable to add peer", e);
418     }
419   }

474   /**
475    * @param bytes Content of a peer znode.
476    * @return ClusterKey parsed from the passed bytes.
477    * @throws DeserializationException
478    */
479   static String parsePeerFrom(final byte[] bytes) throws DeserializationException {
480     if (ProtobufUtil.isPBMagicPrefix(bytes)) {
481       int pblen = ProtobufUtil.lengthOfPBMagic();
482       ZooKeeperProtos.ReplicationPeer.Builder builder = ZooKeeperProtos.ReplicationPeer
483           .newBuilder();
484       ZooKeeperProtos.ReplicationPeer peer;
485       try {
486         peer = builder.mergeFrom(bytes, pblen, bytes.length - pblen).build();
487       } catch (InvalidProtocolBufferException e) {
488         throw new DeserializationException(e);
489       }
490       return peer.getClusterkey();
491     } else {
492       if (bytes.length > 0) {
493         return Bytes.toString(bytes);
494       }
495       return "";
496     }
497   }

499   /**
500    * @param bytes Content of a state znode.
501    * @return State parsed from the passed bytes.
502    * @throws DeserializationException
503    */
504   static ZooKeeperProtos.ReplicationState.State parseStateFrom(final byte[] bytes)
505       throws DeserializationException {
506     ProtobufUtil.expectPBMagicPrefix(bytes);
507     int pblen = ProtobufUtil.lengthOfPBMagic();
508     ZooKeeperProtos.ReplicationState.Builder builder = ZooKeeperProtos.ReplicationState
509         .newBuilder();
510     ZooKeeperProtos.ReplicationState state;
511     try {
512       state = builder.mergeFrom(bytes, pblen, bytes.length - pblen).build();
513       return state.getState();
514     } catch (InvalidProtocolBufferException e) {
515       throw new DeserializationException(e);
516     }
517   }

600   private void changePeerState(String id, ZooKeeperProtos.ReplicationState.State state)
601       throws IOException {
602     try {
603       if (!peerExists(id)) {
604         throw new IllegalArgumentException("peer " + id + " is not registered");
605       }
606       String peerStateZNode = getPeerStateNode(id);
607       byte[] stateBytes = (state == ZooKeeperProtos.ReplicationState.State.ENABLED) ? ENABLED_ZNODE_BYTES
608           : DISABLED_ZNODE_BYTES;
609       if (ZKUtil.checkExists(this.zookeeper, peerStateZNode) != -1) {
610         ZKUtil.setData(this.zookeeper, peerStateZNode, stateBytes);
611       } else {
612         ZKUtil.createAndWatch(zookeeper, peerStateZNode, stateBytes);
613       }
614       LOG.info("state of the peer " + id + " changed to " + state.name());
615     } catch (KeeperException e) {
616       throw new IOException("Unable to change state of the peer " + id, e);
617     }
618   }

620   /**
621    * Check whether the peer is enabled or not. This method checks the atomic
622    * boolean of ReplicationPeer locally.
623    *
624    * @param id peer identifier
625    * @return true if the peer is enabled, otherwise false
626    * @throws IllegalArgumentException
627    *           Thrown when the peer doesn't exist
628    */
629   public boolean getPeerEnabled(String id) {
630     if (!this.peerClusters.containsKey(id)) {
631       throw new IllegalArgumentException("peer " + id + " is not registered");
632     }
633     return this.peerClusters.get(id).getPeerEnabled().get();

787   /**
788    * Try to set a lock in another server's znode.
789    * @param znode the server names of the other server
790    * @return true if the lock was acquired, false in every other cases
791    */
792   public boolean lockOtherRS(String znode) {
793     try {
794       String parent = ZKUtil.joinZNode(this.rsZNode, znode);
795       if (parent.equals(rsServerNameZnode)) {
796         LOG.warn("Won't lock because this is us, we're dead!");
797         return false;
798       }
799       String p = ZKUtil.joinZNode(parent, RS_LOCK_ZNODE);
800       ZKUtil.createAndWatch(this.zookeeper, p, lockToByteArray(rsServerNameZnode));
801     } catch (KeeperException e) {
802       // This exception will pop up if the znode under which we're trying to
803       // create the lock is already deleted by another region server, meaning
804       // that the transfer already occurred.
805       // NoNode => transfer is done and znodes are already deleted
806       // NodeExists => lock znode already created by another RS
807       if (e instanceof KeeperException.NoNodeException ||
808           e instanceof KeeperException.NodeExistsException) {
809         LOG.info("Won't transfer the queue," +
810             " another RS took care of it because of: " + e.getMessage());
811       } else {
812         LOG.info("Failed lock other rs", e);
813       }
814       return false;
815     }
816     return true;
817   }

819   /**
820    * This methods copies all the hlogs queues from another region server
821    * and returns them all sorted per peer cluster (appended with the dead
822    * server's znode)
823    * @param znode server names to copy
824    * @return all hlogs for all peers of that cluster, null if an error occurred
825    */
826   public SortedMap<String, SortedSet<String>> copyQueuesFromRS(String znode) {
827     // TODO this method isn't atomic enough, we could start copying and then
828     // TODO fail for some reason and we would end up with znodes we don't want.
829     SortedMap<String,SortedSet<String>> queues =
830         new TreeMap<String,SortedSet<String>>();
831     try {
832       String nodePath = ZKUtil.joinZNode(rsZNode, znode);
833       List<String> clusters =
834         ZKUtil.listChildrenNoWatch(this.zookeeper, nodePath);
835       // We have a lock znode in there, it will count as one.
836       if (clusters == null || clusters.size() <= 1) {
837         return queues;
838       }
839       // The lock isn't a peer cluster, remove it
840       clusters.remove(RS_LOCK_ZNODE);
841       for (String cluster : clusters) {
842         // We add the name of the recovered RS to the new znode, we can even
843         // do that for queues that were recovered 10 times giving a znode like
844         // number-startcode-number-otherstartcode-number-anotherstartcode-etc
845         String newCluster = cluster+"-"+znode;
846         String newClusterZnode = ZKUtil.joinZNode(rsServerNameZnode, newCluster);
847         String clusterPath = ZKUtil.joinZNode(nodePath, cluster);
848         List<String> hlogs = ZKUtil.listChildrenNoWatch(this.zookeeper, clusterPath);
849         // That region server didn't have anything to replicate for this cluster
850         if (hlogs == null || hlogs.size() == 0) {
851           continue;
852         }
853         ZKUtil.createNodeIfNotExistsAndWatch(this.zookeeper, newClusterZnode,
854             HConstants.EMPTY_BYTE_ARRAY);
855         SortedSet<String> logQueue = new TreeSet<String>();
856         queues.put(newCluster, logQueue);
857         for (String hlog : hlogs) {
858           String z = ZKUtil.joinZNode(clusterPath, hlog);
859           byte[] positionBytes = ZKUtil.getData(this.zookeeper, z);
860           long position = 0;
861           try {
862             position = parseHLogPositionFrom(positionBytes);
863           } catch (DeserializationException e) {
864             LOG.warn("Failed parse of hlog position from the following znode: " + z);
865           }
866           LOG.debug("Creating " + hlog + " with data " + position);
867           String child = ZKUtil.joinZNode(newClusterZnode, hlog);
868           // Position doesn't actually change, we are just deserializing it for
869           // logging, so just use the already serialized version
870           ZKUtil.createAndWatch(this.zookeeper, child, positionBytes);
871           logQueue.add(hlog);
872         }
873       }
874     } catch (KeeperException e) {
875       this.abortable.abort("Copy queues from rs", e);
876     }
877     return queues;
878   }

899    * @param znode
900    */
901   public void deleteRsQueues(String znode) {
902     String fullpath = ZKUtil.joinZNode(rsZNode, znode);
903     try {
904       List<String> clusters =
905         ZKUtil.listChildrenNoWatch(this.zookeeper, fullpath);
906       for (String cluster : clusters) {
907         // We'll delete it later
908         if (cluster.equals(RS_LOCK_ZNODE)) {
909           continue;
910         }
911         String fullClusterPath = ZKUtil.joinZNode(fullpath, cluster);
912         ZKUtil.deleteNodeRecursively(this.zookeeper, fullClusterPath);
913       }
914       // Finish cleaning up
915       ZKUtil.deleteNodeRecursively(this.zookeeper, fullpath);
916     } catch (KeeperException e) {
917       if (e instanceof KeeperException.NoNodeException ||
918           e instanceof KeeperException.NotEmptyException) {
919         // Testing a special case where another region server was able to
920         // create a lock just after we deleted it, but then was also able to
921         // delete the RS znode before us or its lock znode is still there.
922         if (e.getPath().equals(fullpath)) {
923           return;
924         }
925       }
926       this.abortable.abort("Failed delete of " + znode, e);
