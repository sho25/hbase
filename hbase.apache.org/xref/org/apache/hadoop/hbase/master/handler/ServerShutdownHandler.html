50  /**
51   * Process server shutdown.
52   * Server-to-handle must be already in the deadservers lists.  See
53   * {@link ServerManager#expireServer(ServerName)}
54   */
55  @InterfaceAudience.Private
56  public class ServerShutdownHandler extends EventHandler {
57    private static final Log LOG = LogFactory.getLog(ServerShutdownHandler.class);
58    private final ServerName serverName;
59    private final MasterServices services;
60    private final DeadServer deadServers;
61    private final boolean shouldSplitHlog; // whether to split HLog or not

63    public ServerShutdownHandler(final Server server, final MasterServices services,
64        final DeadServer deadServers, final ServerName serverName,
65        final boolean shouldSplitHlog) {
66      this(server, services, deadServers, serverName, EventType.M_SERVER_SHUTDOWN,
67          shouldSplitHlog);
68    }

70    ServerShutdownHandler(final Server server, final MasterServices services,
71        final DeadServer deadServers, final ServerName serverName, EventType type,
72        final boolean shouldSplitHlog) {
73      super(server, type);
74      this.serverName = serverName;
75      this.server = server;
76      this.services = services;
77      this.deadServers = deadServers;
78      if (!this.deadServers.contains(this.serverName)) {
79        LOG.warn(this.serverName + " is NOT in deadservers; it should be!");
80      }
81      this.shouldSplitHlog = shouldSplitHlog;
82    }

93    /**
94     * Before assign the ROOT region, ensure it haven't
95     *  been assigned by other place
96     * <p>
97     * Under some scenarios, the ROOT region can be opened twice, so it seemed online
98     * in two regionserver at the same time.
99     * If the ROOT region has been assigned, so the operation can be canceled.
100    * @throws InterruptedException
101    * @throws IOException
102    * @throws KeeperException
103    */
104   private void verifyAndAssignRoot()
105   throws InterruptedException, IOException, KeeperException {
106     long timeout = this.server.getConfiguration().
107       getLong("hbase.catalog.verification.timeout", 1000);
108     if (!this.server.getCatalogTracker().verifyRootRegionLocation(timeout)) {
109       this.services.getAssignmentManager().assignRoot();
110     }
111   }
112 
113   /**
114    * Failed many times, shutdown processing
115    * @throws IOException
116    */
117   private void verifyAndAssignRootWithRetries() throws IOException {
118     int iTimes = this.server.getConfiguration().getInt(
119         "hbase.catalog.verification.retries", 10);
120 
121     long waitTime = this.server.getConfiguration().getLong(
122         "hbase.catalog.verification.timeout", 1000);
123 
124     int iFlag = 0;
125     while (true) {
126       try {
127         verifyAndAssignRoot();
128         break;
129       } catch (KeeperException e) {
130         this.server.abort("In server shutdown processing, assigning root", e);
131         throw new IOException("Aborting", e);
132       } catch (Exception e) {
133         if (iFlag >= iTimes) {
134           this.server.abort("verifyAndAssignRoot failed after" + iTimes
135               + " times retries, aborting", e);
136           throw new IOException("Aborting", e);
137         }
138         try {
139           Thread.sleep(waitTime);
140         } catch (InterruptedException e1) {
141           LOG.warn("Interrupted when is the thread sleep", e1);
142           Thread.currentThread().interrupt();
143           throw new IOException("Interrupted", e1);
144         }
145         iFlag++;
146       }
147     }
148   }

173   @Override
174   public void process() throws IOException {
175     final ServerName serverName = this.serverName;
176     try {
177       try {
178         if (this.shouldSplitHlog) {
179           LOG.info("Splitting logs for " + serverName);
180           this.services.getMasterFileSystem().splitLog(serverName);
181         } else {
182           LOG.info("Skipping log splitting for " + serverName);
183         }
184       } catch (IOException ioe) {
185         this.services.getExecutorService().submit(this);
186         this.deadServers.add(serverName);
187         throw new IOException("failed log splitting for " +
188           serverName + ", will retry", ioe);
189       }
190 
191       // Assign root and meta if we were carrying them.
192       if (isCarryingRoot()) { // -ROOT-
193         LOG.info("Server " + serverName +
194             " was carrying ROOT. Trying to assign.");
195         this.services.getAssignmentManager().
196           regionOffline(HRegionInfo.ROOT_REGIONINFO);
197         verifyAndAssignRootWithRetries();
198       }
199 
200       // Carrying meta?
201       if (isCarryingMeta()) {
202         LOG.info("Server " + serverName +
203           " was carrying META. Trying to assign.");
204         this.services.getAssignmentManager().
205           regionOffline(HRegionInfo.FIRST_META_REGIONINFO);
206         this.services.getAssignmentManager().assignMeta();
207       }
208 
209       // We don't want worker thread in the MetaServerShutdownHandler
210       // executor pool to block by waiting availability of -ROOT-
211       // and .META. server. Otherwise, it could run into the following issue:
212       // 1. The current MetaServerShutdownHandler instance For RS1 waits for the .META.
213       //    to come online.
214       // 2. The newly assigned .META. region server RS2 was shutdown right after
215       //    it opens the .META. region. So the MetaServerShutdownHandler
216       //    instance For RS1 will still be blocked.
217       // 3. The new instance of MetaServerShutdownHandler for RS2 is queued.
218       // 4. The newly assigned .META. region server RS3 was shutdown right after
219       //    it opens the .META. region. So the MetaServerShutdownHandler
220       //    instance For RS1 and RS2 will still be blocked.
221       // 5. The new instance of MetaServerShutdownHandler for RS3 is queued.
222       // 6. Repeat until we run out of MetaServerShutdownHandler worker threads
223       // The solution here is to resubmit a ServerShutdownHandler request to process
224       // user regions on that server so that MetaServerShutdownHandler
225       // executor pool is always available.
226       //
227       // If AssignmentManager hasn't finished rebuilding user regions,
228       // we are not ready to assign dead regions either. So we re-queue up
229       // the dead server for further processing too.
230       if (isCarryingRoot() || isCarryingMeta() // -ROOT- or .META.
231           || !services.getAssignmentManager().isFailoverCleanupDone()) {
232         this.services.getServerManager().processDeadServer(serverName);
233         return;
234       }
235 
236       // Wait on meta to come online; we need it to progress.
237       // TODO: Best way to hold strictly here?  We should build this retry logic
238       // into the MetaReader operations themselves.
239       // TODO: Is the reading of .META. necessary when the Master has state of
240       // cluster in its head?  It should be possible to do without reading .META.
241       // in all but one case. On split, the RS updates the .META.
242       // table and THEN informs the master of the split via zk nodes in
243       // 'unassigned' dir.  Currently the RS puts ephemeral nodes into zk so if
244       // the regionserver dies, these nodes do not stick around and this server
245       // shutdown processing does fixup (see the fixupDaughters method below).
246       // If we wanted to skip the .META. scan, we'd have to change at least the
247       // final SPLIT message to be permanent in zk so in here we'd know a SPLIT
248       // completed (zk is updated after edits to .META. have gone in).  See
249       // {@link SplitTransaction}.  We'd also have to be figure another way for
250       // doing the below .META. daughters fixup.
251       NavigableMap<HRegionInfo, Result> hris = null;
252       while (!this.server.isStopped()) {
253         try {
254           this.server.getCatalogTracker().waitForMeta();
255           hris = MetaReader.getServerUserRegions(this.server.getCatalogTracker(),
256             this.serverName);
257           break;
258         } catch (InterruptedException e) {
259           Thread.currentThread().interrupt();
260           throw new IOException("Interrupted", e);
261         } catch (IOException ioe) {
262           LOG.info("Received exception accessing META during server shutdown of " +
263               serverName + ", retrying META read", ioe);
264         }
265       }
266       if (this.server.isStopped()) {
267         throw new IOException("Server is stopped");
268       }
269 
270       // Clean out anything in regions in transition.  Being conservative and
271       // doing after log splitting.  Could do some states before -- OPENING?
272       // OFFLINE? -- and then others after like CLOSING that depend on log
273       // splitting.
274       AssignmentManager am = services.getAssignmentManager();
275       List<RegionState> regionsInTransition = am.processServerShutdown(serverName);
276       LOG.info("Reassigning " + ((hris == null)? 0: hris.size()) +
277         " region(s) that " + (serverName == null? "null": serverName)  +
278         " was carrying (skipping " + regionsInTransition.size() +
279         " regions(s) that are already in transition)");
280 
281       // Iterate regions that were on this server and assign them
282       if (hris != null) {
283         RegionStates regionStates = am.getRegionStates();
284         List<HRegionInfo> toAssignRegions = new ArrayList<HRegionInfo>();
285         for (Map.Entry<HRegionInfo, Result> e: hris.entrySet()) {
286           HRegionInfo hri = e.getKey();
287           RegionState rit = regionStates.getRegionTransitionState(hri);
288           if (processDeadRegion(hri, e.getValue(), am, server.getCatalogTracker())) {
289             ServerName addressFromAM = regionStates.getRegionServerOfRegion(hri);
290             if (addressFromAM != null && !addressFromAM.equals(this.serverName)) {
291               // If this region is in transition on the dead server, it must be
292               // opening or pending_open, which is covered by AM#processServerShutdown
293               LOG.debug("Skip assigning region " + hri.getRegionNameAsString()
294                 + " because it has been opened in " + addressFromAM.getServerName());
295               continue;
296             }
297             if (rit != null) {
298               if (!rit.isOnServer(serverName)
299                   || rit.isClosed() || rit.isOpened() || rit.isSplit()) {
300                 // Skip regions that are in transition on other server,
301                 // or in state closed/opened/split
302                 LOG.info("Skip assigning region " + rit);
303                 continue;
304               }
305               try{
306                 //clean zk node
307                 LOG.info("Reassigning region with rs = " + rit + " and deleting zk node if exists");
308                 ZKAssign.deleteNodeFailSilent(services.getZooKeeper(), hri);
309               } catch (KeeperException ke) {
310                 this.server.abort("Unexpected ZK exception deleting unassigned node " + hri, ke);
311                 return;
312               }
313             }
314             toAssignRegions.add(hri);
315           } else if (rit != null) {
316             if (rit.isSplitting() || rit.isSplit()) {
317               // This will happen when the RS went down and the call back for the SPLIITING or SPLIT
318               // has not yet happened for node Deleted event. In that case if the region was actually
319               // split
320               // but the RS had gone down before completing the split process then will not try to
321               // assign the parent region again. In that case we should make the region offline and
322               // also delete the region from RIT.
323               am.regionOffline(hri);
324             } else if ((rit.isClosing() || rit.isPendingClose())
325                 && am.getZKTable().isDisablingOrDisabledTable(hri.getTableNameAsString())) {
326               // If the table was partially disabled and the RS went down, we should clear the RIT
327               // and remove the node for the region.
328               // The rit that we use may be stale in case the table was in DISABLING state
329               // but though we did assign we will not be clearing the znode in CLOSING state.
330               // Doing this will have no harm. See HBASE-5927
331               am.deleteClosingOrClosedNode(hri);
332               am.regionOffline(hri);
333             } else {
334               LOG.warn("THIS SHOULD NOT HAPPEN: unexpected region in transition "
335                 + rit + " not to be assigned by SSH of server " + serverName);
336             }
337           }
338         }
339         try {
340           am.assign(toAssignRegions);
341         } catch (InterruptedException ie) {
342           LOG.error("Caught " + ie + " during round-robin assignment");
343           throw new IOException(ie);
344         }
345       }
346     } finally {
347       this.deadServers.finish(serverName);
348     }
349     LOG.info("Finished processing of shutdown of " + serverName);
350   }

352   /**
353    * Process a dead region from a dead RS. Checks if the region is disabled or
354    * disabling or if the region has a partially completed split.
355    * @param hri
356    * @param result
357    * @param assignmentManager
358    * @param catalogTracker
359    * @return Returns true if specified region should be assigned, false if not.
360    * @throws IOException
361    */
362   public static boolean processDeadRegion(HRegionInfo hri, Result result,
363       AssignmentManager assignmentManager, CatalogTracker catalogTracker)
364   throws IOException {
365     boolean tablePresent = assignmentManager.getZKTable().isTablePresent(
366         hri.getTableNameAsString());
367     if (!tablePresent) {
368       LOG.info("The table " + hri.getTableNameAsString()
369           + " was deleted.  Hence not proceeding.");
370       return false;
371     }
372     // If table is not disabled but the region is offlined,
373     boolean disabled = assignmentManager.getZKTable().isDisabledTable(
374         hri.getTableNameAsString());
375     if (disabled){
376       LOG.info("The table " + hri.getTableNameAsString()
377           + " was disabled.  Hence not proceeding.");
378       return false;
379     }
380     if (hri.isOffline() && hri.isSplit()) {
381       LOG.debug("Offlined and split region " + hri.getRegionNameAsString() +
382         "; checking daughter presence");
383       if (MetaReader.getRegion(catalogTracker, hri.getRegionName()) == null) {
384         return false;
385       }
386       fixupDaughters(result, assignmentManager, catalogTracker);
387       return false;
388     }
389     boolean disabling = assignmentManager.getZKTable().isDisablingTable(
390         hri.getTableNameAsString());
391     if (disabling) {
392       LOG.info("The table " + hri.getTableNameAsString()
393           + " is disabled.  Hence not assigning region" + hri.getEncodedName());
394       return false;
395     }
396     return true;
397   }

399   /**
400    * Check that daughter regions are up in .META. and if not, add them.
401    * @param result The contents of the parent row in .META.
402    * @param assignmentManager
403    * @param catalogTracker
404    * @return the number of daughters missing and fixed
405    * @throws IOException
406    */
407   public static int fixupDaughters(final Result result,
408       final AssignmentManager assignmentManager,
409       final CatalogTracker catalogTracker)
410   throws IOException {
411     PairOfSameType<HRegionInfo> daughters = HRegionInfo.getDaughterRegions(result);
412     int fixedA = fixupDaughter(result, daughters.getFirst(),
413       assignmentManager, catalogTracker);
414     int fixedB = fixupDaughter(result, daughters.getSecond(),
415       assignmentManager, catalogTracker);
416     return fixedA + fixedB;
417   }

419   /**
420    * Check individual daughter is up in .META.; fixup if its not.
421    * @param result The contents of the parent row in .META. - not used
422    * @param daughter Which daughter to check for.
423    * @return 1 if the daughter is missing and fixed. Otherwise 0
424    * @throws IOException
425    */
426   static int fixupDaughter(final Result result, HRegionInfo daughter,
427       final AssignmentManager assignmentManager,
428       final CatalogTracker catalogTracker)
429   throws IOException {
430     if (daughter == null) return 0;
431     if (isDaughterMissing(catalogTracker, daughter)) {
432       LOG.info("Fixup; missing daughter " + daughter.getRegionNameAsString());
433       MetaEditor.addDaughter(catalogTracker, daughter, null);
434 
435       // TODO: Log WARN if the regiondir does not exist in the fs.  If its not
436       // there then something wonky about the split -- things will keep going
437       // but could be missing references to parent region.
438 
439       // And assign it.
440       assignmentManager.assign(daughter, true, true);
441       return 1;
442     } else {
443       LOG.debug("Daughter " + daughter.getRegionNameAsString() + " present");
444     }
445     return 0;
446   }

448   /**
449    * Look for presence of the daughter OR of a split of the daughter in .META.
450    * Daughter could have been split over on regionserver before a run of the
451    * catalogJanitor had chance to clear reference from parent.
452    * @param daughter Daughter region to search for.
453    * @throws IOException
454    */
455   private static boolean isDaughterMissing(final CatalogTracker catalogTracker,
456       final HRegionInfo daughter) throws IOException {
457     FindDaughterVisitor visitor = new FindDaughterVisitor(daughter);
458     // Start the scan at what should be the daughter's row in the .META.
459     // We will either 1., find the daughter or some derivative split of the
460     // daughter (will have same table name and start row at least but will sort
461     // after because has larger regionid -- the regionid is timestamp of region
462     // creation), OR, we will not find anything with same table name and start
463     // row.  If the latter, then assume daughter missing and do fixup.
464     byte [] startrow = daughter.getRegionName();
465     MetaReader.fullScan(catalogTracker, visitor, startrow);
466     return !visitor.foundDaughter();
467   }

469   /**
470    * Looks for daughter.  Sets a flag if daughter or some progeny of daughter
471    * is found up in <code>.META.</code>.
472    */
473   static class FindDaughterVisitor implements MetaReader.Visitor {
474     private final HRegionInfo daughter;
475     private boolean found = false;
476 
477     FindDaughterVisitor(final HRegionInfo daughter) {
478       this.daughter = daughter;
479     }
480 
481     /**
482      * @return True if we found a daughter region during our visiting.
483      */
484     boolean foundDaughter() {
485       return this.found;
486     }
487 
488     @Override
489     public boolean visit(Result r) throws IOException {
490       HRegionInfo hri =
491         HRegionInfo.getHRegionInfo(r);
492       if (hri == null) {
493         LOG.warn("No serialized HRegionInfo in " + r);
494         return true;
495       }
496       byte [] value = r.getValue(HConstants.CATALOG_FAMILY,
497           HConstants.SERVER_QUALIFIER);
498       // See if daughter is assigned to some server
499       if (value == null) return false;
500 
501       // Now see if we have gone beyond the daughter's startrow.
502       if (!Bytes.equals(daughter.getTableName(),
503           hri.getTableName())) {
504         // We fell into another table.  Stop scanning.
505         return false;
506       }
507       // If our start rows do not compare, move on.
508       if (!Bytes.equals(daughter.getStartKey(), hri.getStartKey())) {
509         return false;
510       }
511       // Else, table name and start rows compare.  It means that the daughter
512       // or some derivative split of the daughter is up in .META.  Daughter
513       // exists.
514       this.found = true;
515       return false;
516     }
517   }
