49  /**
50   * Handler to run enable of a table.
51   */
52  @InterfaceAudience.Private
53  public class EnableTableHandler extends EventHandler {
54    private static final Log LOG = LogFactory.getLog(EnableTableHandler.class);
55    private final byte [] tableName;
56    private final String tableNameStr;
57    private final AssignmentManager assignmentManager;
58    private final CatalogTracker ct;
59    private boolean retainAssignment = false;
60  
61    public EnableTableHandler(Server server, byte [] tableName,
62        CatalogTracker catalogTracker, AssignmentManager assignmentManager,
63        boolean skipTableStateCheck)
64    throws TableNotFoundException, TableNotDisabledException, IOException {
65      super(server, EventType.C_M_ENABLE_TABLE);
66      this.tableName = tableName;
67      this.tableNameStr = Bytes.toString(tableName);
68      this.ct = catalogTracker;
69      this.assignmentManager = assignmentManager;
70      this.retainAssignment = skipTableStateCheck;
71      // Check if table exists
72      if (!MetaReader.tableExists(catalogTracker, this.tableNameStr)) {
73        throw new TableNotFoundException(Bytes.toString(tableName));
74      }
75  
76      // There could be multiple client requests trying to disable or enable
77      // the table at the same time. Ensure only the first request is honored
78      // After that, no other requests can be accepted until the table reaches
79      // DISABLED or ENABLED.
80      if (!skipTableStateCheck)
81      {
82        try {
83          if (!this.assignmentManager.getZKTable().checkDisabledAndSetEnablingTable
84            (this.tableNameStr)) {
85            LOG.info("Table " + tableNameStr + " isn't disabled; skipping enable");
86            throw new TableNotDisabledException(this.tableNameStr);
87          }
88        } catch (KeeperException e) {
89          throw new IOException("Unable to ensure that the table will be" +
90            " enabling because of a ZooKeeper issue", e);
91        }
92      }
93    }
94  
95    @Override
96    public String toString() {
97      String name = "UnknownServerName";
98      if(server != null && server.getServerName() != null) {
99        name = server.getServerName().toString();
100     }
101     return getClass().getSimpleName() + "-" + name + "-" + getSeqid() + "-" +
102       tableNameStr;
103   }
104 
105   @Override
106   public void process() {
107     try {
108       LOG.info("Attempting to enable the table " + this.tableNameStr);
109       MasterCoprocessorHost cpHost = ((HMaster) this.server)
110           .getCoprocessorHost();
111       if (cpHost != null) {
112         cpHost.preEnableTableHandler(this.tableName);
113       }
114       handleEnableTable();
115       if (cpHost != null) {
116         cpHost.postEnableTableHandler(this.tableName);
117       }
118     } catch (IOException e) {
119       LOG.error("Error trying to enable the table " + this.tableNameStr, e);
120     } catch (KeeperException e) {
121       LOG.error("Error trying to enable the table " + this.tableNameStr, e);
122     } catch (InterruptedException e) {
123       LOG.error("Error trying to enable the table " + this.tableNameStr, e);
124     }
125   }
126 
127   private void handleEnableTable() throws IOException, KeeperException, InterruptedException {
128     // I could check table is disabling and if so, not enable but require
129     // that user first finish disabling but that might be obnoxious.
130 
131     // Set table enabling flag up in zk.
132     this.assignmentManager.getZKTable().setEnablingTable(this.tableNameStr);
133     boolean done = false;
134     // Get the regions of this table. We're done when all listed
135     // tables are onlined.
136     List<Pair<HRegionInfo, ServerName>> tableRegionsAndLocations = MetaReader
137         .getTableRegionsAndLocations(this.ct, tableName, true);
138     int countOfRegionsInTable = tableRegionsAndLocations.size();
139     List<HRegionInfo> regions = regionsToAssignWithServerName(tableRegionsAndLocations);
140     int regionsCount = regions.size();
141     if (regionsCount == 0) {
142       done = true;
143     }
144     LOG.info("Table '" + this.tableNameStr + "' has " + countOfRegionsInTable
145       + " regions, of which " + regionsCount + " are offline.");
146     BulkEnabler bd = new BulkEnabler(this.server, regions, countOfRegionsInTable,
147         this.retainAssignment);
148     try {
149       if (bd.bulkAssign()) {
150         done = true;
151       }
152     } catch (InterruptedException e) {
153       LOG.warn("Enable operation was interrupted when enabling table '"
154         + this.tableNameStr + "'");
155       // Preserve the interrupt.
156       Thread.currentThread().interrupt();
157     }
158     if (done) {
159       // Flip the table to enabled.
160       this.assignmentManager.getZKTable().setEnabledTable(
161         this.tableNameStr);
162       LOG.info("Table '" + this.tableNameStr
163       + "' was successfully enabled. Status: done=" + done);
164     } else {
165       LOG.warn("Table '" + this.tableNameStr
166       + "' wasn't successfully enabled. Status: done=" + done);
167     }
168   }

170   /**
171    * @param regionsInMeta
172    * @return List of regions neither in transition nor assigned.
173    * @throws IOException
174    */
175   private List<HRegionInfo> regionsToAssignWithServerName(
176       final List<Pair<HRegionInfo, ServerName>> regionsInMeta) throws IOException {
177     ServerManager serverManager = ((HMaster) this.server).getServerManager();
178     List<HRegionInfo> regions = new ArrayList<HRegionInfo>();
179     RegionStates regionStates = this.assignmentManager.getRegionStates();
180     for (Pair<HRegionInfo, ServerName> regionLocation : regionsInMeta) {
181       HRegionInfo hri = regionLocation.getFirst();
182       ServerName sn = regionLocation.getSecond();
183       if (!regionStates.isRegionInTransition(hri) && !regionStates.isRegionAssigned(hri)) {
184         if (this.retainAssignment && sn != null && serverManager.isServerOnline(sn)) {
185           this.assignmentManager.addPlan(hri.getEncodedName(), new RegionPlan(hri, null, sn));
186         }
187         regions.add(hri);
188       } else {
189         if (LOG.isDebugEnabled()) {
190           LOG.debug("Skipping assign for the region " + hri + " during enable table "
191               + hri.getTableNameAsString() + " because its already in tranition or assigned.");
192         }
193       }
194     }
195     return regions;
196   }

198   /**
199    * Run bulk enable.
200    */
201   class BulkEnabler extends BulkAssigner {
202     private final List<HRegionInfo> regions;
203     // Count of regions in table at time this assign was launched.
204     private final int countOfRegionsInTable;
205     private final boolean retainAssignment;
206 
207     BulkEnabler(final Server server, final List<HRegionInfo> regions,
208         final int countOfRegionsInTable, boolean retainAssignment) {
209       super(server);
210       this.regions = regions;
211       this.countOfRegionsInTable = countOfRegionsInTable;
212       this.retainAssignment = retainAssignment;
213     }
214 
215     @Override
216     protected void populatePool(ExecutorService pool) throws IOException {
217       boolean roundRobinAssignment = this.server.getConfiguration().getBoolean(
218           "hbase.master.enabletable.roundrobin", false);
219 
220       // In case of masterRestart always go with single assign.  Going thro
221       // roundRobinAssignment will use bulkassign which may lead to double assignment.
222       if (retainAssignment || !roundRobinAssignment) {
223         for (HRegionInfo region : regions) {
224           if (assignmentManager.getRegionStates()
225               .isRegionInTransition(region)) {
226             continue;
227           }
228           final HRegionInfo hri = region;
229           pool.execute(Trace.wrap(new Runnable() {
230             public void run() {
231               assignmentManager.assign(hri, true);
232             }
233           }));
234         }
235       } else {
236         try {
237           assignmentManager.assign(regions);
238         } catch (InterruptedException e) {
239           LOG.warn("Assignment was interrupted");
240           Thread.currentThread().interrupt();
241         }
242       }
243     }
244 
245     @Override
246     protected boolean waitUntilDone(long timeout)
247     throws InterruptedException {
248       long startTime = System.currentTimeMillis();
249       long remaining = timeout;
250       List<HRegionInfo> regions = null;
251       int lastNumberOfRegions = 0;
252       while (!server.isStopped() && remaining > 0) {
253         Thread.sleep(waitingTimeForEvents);
254         regions = assignmentManager.getRegionStates()
255           .getRegionsOfTable(tableName);
256         if (isDone(regions)) break;
257 
258         // Punt on the timeout as long we make progress
259         if (regions.size() > lastNumberOfRegions) {
260           lastNumberOfRegions = regions.size();
261           timeout += waitingTimeForEvents;
262         }
263         remaining = timeout - (System.currentTimeMillis() - startTime);
264       }
265       return isDone(regions);
266     }
267 
268     private boolean isDone(final List<HRegionInfo> regions) {
269       return regions != null && regions.size() >= this.countOfRegionsInTable;
270     }
271   }
