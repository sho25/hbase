170   /**
171    * Add a region to RegionStates. The initial state is OFFLINE.
172    * If it is already in RegionStates, this call has no effect,
173    * and the original state is returned.
174    */
175   public synchronized RegionState createRegionState(final HRegionInfo hri) {
176     String regionName = hri.getEncodedName();
177     RegionState regionState = regionStates.get(regionName);
178     if (regionState != null) {
179       LOG.warn("Tried to create a state of a region already in RegionStates "
180         + hri + ", used existing state: " + regionState
181         + ", ignored new state: state=OFFLINE, server=null");
182     } else {
183       regionState = new RegionState(hri, State.OFFLINE);
184       regionStates.put(regionName, regionState);
185     }
186     return regionState;
187   }

201   /**
202    * Update a region state. If it is not splitting,
203    * it will be put in transition if not already there.
204    *
205    * If we can't find the region info based on the region name in
206    * the transition, log a warning and return null.
207    */
208   public synchronized RegionState updateRegionState(
209       final RegionTransition transition, final State state) {
210     byte[] regionName = transition.getRegionName();
211     HRegionInfo regionInfo = getRegionInfo(regionName);
212     if (regionInfo == null) {
213       String prettyRegionName = HRegionInfo.prettyPrint(
214         HRegionInfo.encodeRegionName(regionName));
215       LOG.warn("Failed to find region " + prettyRegionName
216         + " in updating its state to " + state
217         + " based on region transition " + transition);
218       return null;
219     }
220     return updateRegionState(regionInfo, state,
221       transition.getServerName());
222   }

224   /**
225    * Update a region state. If it is not splitting,
226    * it will be put in transition if not already there.
227    */
228   public synchronized RegionState updateRegionState(
229       final HRegionInfo hri, final State state, final ServerName serverName) {
230     ServerName newServerName = serverName;
231     if (serverName != null &&
232         (state == State.CLOSED || state == State.OFFLINE)) {
233       LOG.warn("Closed region " + hri + " still on "
234         + serverName + "? Ignored, reset it to null");
235       newServerName = null;
236     }
237 
238     String regionName = hri.getEncodedName();
239     RegionState regionState = new RegionState(
240       hri, state, System.currentTimeMillis(), newServerName);
241     RegionState oldState = regionStates.put(regionName, regionState);
242     LOG.info("Region " + hri + " transitioned from " + oldState + " to " + regionState);
243     if (state != State.SPLITTING && (newServerName != null
244         || (state != State.PENDING_CLOSE && state != State.CLOSING))) {
245       regionsInTransition.put(regionName, regionState);
246     }
247 
248     // notify the change
249     this.notifyAll();
250     return regionState;
251   }

253   /**
254    * A region is online, won't be in transition any more.
255    * We can't confirm it is really online on specified region server
256    * because it hasn't been put in region server's online region list yet.
257    */
258   public synchronized void regionOnline(
259       final HRegionInfo hri, final ServerName serverName) {
260     String regionName = hri.getEncodedName();
261     RegionState oldState = regionStates.get(regionName);
262     if (oldState == null) {
263       LOG.warn("Online a region not in RegionStates: " + hri);
264     } else {
265       State state = oldState.getState();
266       ServerName sn = oldState.getServerName();
267       if (state != State.OPEN || sn == null || !sn.equals(serverName)) {
268         LOG.debug("Online a region with current state=" + state + ", expected state=OPEN"
269           + ", assigned to server: " + sn + " expected " + serverName);
270       }
271     }
272     updateRegionState(hri, State.OPEN, serverName);
273     regionsInTransition.remove(regionName);
274 
275     ServerName oldServerName = regionAssignments.put(hri, serverName);
276     if (!serverName.equals(oldServerName)) {
277       LOG.info("Onlined region " + hri + " on " + serverName);
278       Set<HRegionInfo> regions = serverHoldings.get(serverName);
279       if (regions == null) {
280         regions = new HashSet<HRegionInfo>();
281         serverHoldings.put(serverName, regions);
282       }
283       regions.add(hri);
284       if (oldServerName != null) {
285         LOG.info("Offlined region " + hri + " from " + oldServerName);
286         serverHoldings.get(oldServerName).remove(hri);
287       }
288     }
289   }

291   /**
292    * A region is offline, won't be in transition any more.
293    */
294   public synchronized void regionOffline(final HRegionInfo hri) {
295     String regionName = hri.getEncodedName();
296     RegionState oldState = regionStates.get(regionName);
297     if (oldState == null) {
298       LOG.warn("Offline a region not in RegionStates: " + hri);
299     } else {
300       State state = oldState.getState();
301       ServerName sn = oldState.getServerName();
302       if (state != State.OFFLINE || sn != null) {
303         LOG.debug("Online a region with current state=" + state + ", expected state=OFFLINE"
304           + ", assigned to server: " + sn + ", expected null");
305       }
306     }
307     updateRegionState(hri, State.OFFLINE);
308     regionsInTransition.remove(regionName);
309 
310     ServerName oldServerName = regionAssignments.remove(hri);
311     if (oldServerName != null) {
312       LOG.info("Offlined region " + hri + " from " + oldServerName);
313       serverHoldings.get(oldServerName).remove(hri);
314     }
315   }

317   /**
318    * A server is offline, all regions on it are dead.
319    */
320   public synchronized List<RegionState> serverOffline(final ServerName sn) {
321     // Clean up this server from map of servers to regions, and remove all regions
322     // of this server from online map of regions.
323     List<RegionState> rits = new ArrayList<RegionState>();
324     Set<HRegionInfo> assignedRegions = serverHoldings.get(sn);
325     if (assignedRegions == null) {
326       assignedRegions = new HashSet<HRegionInfo>();
327     }
328 
329     for (HRegionInfo region : assignedRegions) {
330       regionAssignments.remove(region);
331     }
332 
333     // See if any of the regions that were online on this server were in RIT
334     // If they are, normal timeouts will deal with them appropriately so
335     // let's skip a manual re-assignment.
336     for (RegionState state : regionsInTransition.values()) {
337       if (assignedRegions.contains(state.getRegion())) {
338         rits.add(state);
339       } else if (sn.equals(state.getServerName())) {
340         // Region is in transition on this region server, and this
341         // region is not open on this server. So the region must be
342         // moving to this server from another one (i.e. opening or
343         // pending open on this server, was open on another one
344         if (state.isPendingOpen() || state.isOpening()) {
345           state.setTimestamp(0); // timeout it, let timeout monitor reassign
346         } else {
347           LOG.warn("THIS SHOULD NOT HAPPEN: unexpected state "
348             + state + " of region in transition on server " + sn);
349         }
350       }
351     }
352     assignedRegions.clear();
353     this.notifyAll();
354     return rits;
355   }

357   /**
358    * Gets the online regions of the specified table.
359    * This method looks at the in-memory state.  It does not go to <code>.META.</code>.
360    * Only returns <em>online</em> regions.  If a region on this table has been
361    * closed during a disable, etc., it will be included in the returned list.
362    * So, the returned list may not necessarily be ALL regions in this table, its
363    * all the ONLINE regions in the table.
364    * @param tableName
365    * @return Online regions from <code>tableName</code>
366    */
367   public synchronized List<HRegionInfo> getRegionsOfTable(byte[] tableName) {
368     List<HRegionInfo> tableRegions = new ArrayList<HRegionInfo>();
369     // boundary needs to have table's name but regionID 0 so that it is sorted
370     // before all table's regions.
371     HRegionInfo boundary = new HRegionInfo(tableName, null, null, false, 0L);
372     for (HRegionInfo hri: regionAssignments.tailMap(boundary).keySet()) {
373       if(!Bytes.equals(hri.getTableName(), tableName)) break;
374       tableRegions.add(hri);
375     }
376     return tableRegions;
377   }

380   /**
381    * Wait on region to clear regions-in-transition.
382    * <p>
383    * If the region isn't in transition, returns immediately.  Otherwise, method
384    * blocks until the region is out of transition.
385    */
386   public synchronized void waitOnRegionToClearRegionsInTransition(
387       final HRegionInfo hri) throws InterruptedException {
388     if (!isRegionInTransition(hri)) return;
389 
390     while(!server.isStopped() && isRegionInTransition(hri)) {
391       RegionState rs = getRegionState(hri);
392       LOG.info("Waiting on " + rs + " to clear regions-in-transition");
393       waitForUpdate(100);
394     }
395 
396     if (server.isStopped()) {
397       LOG.info("Giving up wait on region in " +
398         "transition because stoppable.isStopped is set");
399     }
400   }

402   /**
403    * Waits until the specified region has completed assignment.
404    * <p>
405    * If the region is already assigned, returns immediately.  Otherwise, method
406    * blocks until the region is assigned.
407    */
408   public synchronized void waitForAssignment(
409       final HRegionInfo hri) throws InterruptedException {
410     if (!isRegionAssigned(hri)) return;
411 
412     while(!server.isStopped() && !isRegionAssigned(hri)) {
413       RegionState rs = getRegionState(hri);
414       LOG.info("Waiting on " + rs + " to be assigned");
415       waitForUpdate(100);
416     }
417 
418     if (server.isStopped()) {
419       LOG.info("Giving up wait on region " +
420         "assignment because stoppable.isStopped is set");
421     }
422   }

424   /**
425    * Compute the average load across all region servers.
426    * Currently, this uses a very naive computation - just uses the number of
427    * regions being served, ignoring stats about number of requests.
428    * @return the average load
429    */
430   protected synchronized double getAverageLoad() {
431     int numServers = 0, totalLoad = 0;
432     for (Map.Entry<ServerName, Set<HRegionInfo>> e: serverHoldings.entrySet()) {
433       Set<HRegionInfo> regions = e.getValue();
434       ServerName serverName = e.getKey();
435       int regionCount = regions.size();
436       if (regionCount > 0 || serverManager.isServerOnline(serverName)) {
437         totalLoad += regionCount;
438         numServers++;
439       }
440     }
441     return numServers == 0 ? 0.0 :
442       (double)totalLoad / (double)numServers;
443   }

445   /**
446    * This is an EXPENSIVE clone.  Cloning though is the safest thing to do.
447    * Can't let out original since it can change and at least the load balancer
448    * wants to iterate this exported list.  We need to synchronize on regions
449    * since all access to this.servers is under a lock on this.regions.
450    *
451    * @return A clone of current assignments by table.
452    */
453   protected Map<String, Map<ServerName, List<HRegionInfo>>> getAssignmentsByTable() {
454     Map<String, Map<ServerName, List<HRegionInfo>>> result =
455       new HashMap<String, Map<ServerName,List<HRegionInfo>>>();
456     synchronized (this) {
457       if (!server.getConfiguration().getBoolean("hbase.master.loadbalance.bytable", false)) {
458         Map<ServerName, List<HRegionInfo>> svrToRegions =
459           new HashMap<ServerName, List<HRegionInfo>>(serverHoldings.size());
460         for (Map.Entry<ServerName, Set<HRegionInfo>> e: serverHoldings.entrySet()) {
461           svrToRegions.put(e.getKey(), new ArrayList<HRegionInfo>(e.getValue()));
462         }
463         result.put("ensemble", svrToRegions);
464       } else {
465         for (Map.Entry<ServerName, Set<HRegionInfo>> e: serverHoldings.entrySet()) {
466           for (HRegionInfo hri: e.getValue()) {
467             if (hri.isMetaRegion() || hri.isRootRegion()) continue;
468             String tablename = hri.getTableNameAsString();
469             Map<ServerName, List<HRegionInfo>> svrToRegions = result.get(tablename);
470             if (svrToRegions == null) {
471               svrToRegions = new HashMap<ServerName, List<HRegionInfo>>(serverHoldings.size());
472               result.put(tablename, svrToRegions);
473             }
474             List<HRegionInfo> regions = svrToRegions.get(e.getKey());
475             if (regions == null) {
476               regions = new ArrayList<HRegionInfo>();
477               svrToRegions.put(e.getKey(), regions);
478             }
479             regions.add(hri);
480           }
481         }
482       }
483     }
484 
485     Map<ServerName, ServerLoad>
486       onlineSvrs = serverManager.getOnlineServers();
487     // Take care of servers w/o assignments.
488     for (Map<ServerName, List<HRegionInfo>> map: result.values()) {
489       for (ServerName svr: onlineSvrs.keySet()) {
490         if (!map.containsKey(svr)) {
491           map.put(svr, new ArrayList<HRegionInfo>());
492         }
493       }
494     }
495     return result;
496   }

506   /**
507    * Get the HRegionInfo from cache, if not there, from the META table
508    * @param  regionName
509    * @return HRegionInfo for the region
510    */
511   protected HRegionInfo getRegionInfo(final byte [] regionName) {
512     String encodedName = HRegionInfo.encodeRegionName(regionName);
513     RegionState regionState = regionStates.get(encodedName);
514     if (regionState != null) {
515       return regionState.getRegion();
516     }
517 
518     try {
519       Pair<HRegionInfo, ServerName> p =
520         MetaReader.getRegion(server.getCatalogTracker(), regionName);
521       HRegionInfo hri = p == null ? null : p.getFirst();
522       if (hri != null) {
523         createRegionState(hri);
524       }
525       return hri;
526     } catch (IOException e) {
527       server.abort("Aborting because error occoured while reading " +
528         Bytes.toStringBinary(regionName) + " from .META.", e);
529       return null;
530     }
