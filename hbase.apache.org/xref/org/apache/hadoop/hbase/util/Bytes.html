142    * @throws IOException e
143    */
144   public static byte [] readByteArray(final DataInput in)
145   throws IOException {
146     int len = WritableUtils.readVInt(in);
147     if (len < 0) {
148       throw new NegativeArraySizeException(Integer.toString(len));

151     in.readFully(result, 0, len);
152     return result;

204    * @param srcOffset source offset
205    * @param srcLength source length
206    * @return New offset in src array.

208   public static int writeByteArray(final byte [] tgt, final int tgtOffset,
209       final byte [] src, final int srcOffset, final int srcLength) {
210     byte [] vint = vintToBytes(srcLength);
211     System.arraycopy(vint, 0, tgt, tgtOffset, vint.length);
212     int offset = tgtOffset + vint.length;
213     System.arraycopy(src, srcOffset, tgt, offset, srcLength);
214     return offset + srcLength;

341   /**
342    * Write a printable representation of a byte array. Non-printable
343    * characters are hex escaped in the format \\x%02X, eg:
344    * \x00 \x05 etc
345    *
346    * @param b array to write out
347    * @param off offset to start at
348    * @param len length to write
349    * @return string output
350    */
351   public static String toStringBinary(final byte [] b, int off, int len) {
352     StringBuilder result = new StringBuilder();
353     // Just in case we are passed a 'len' that is > buffer length...
354     if (off >= b.length) return result.toString();
355     if (off + len > b.length) len = b.length - off;
356     for (int i = off; i < off + len ; ++i ) {
357       int ch = b[i] & 0xFF;
358       if ( (ch >= '0' && ch <= '9')
359           || (ch >= 'A' && ch <= 'Z')
360           || (ch >= 'a' && ch <= 'z')
361           || " `~!@#$%^&*()-_=+[]{}|;:'\",.<>/?".indexOf(ch) >= 0 ) {
362         result.append((char)ch);
363       } else {
364         result.append(String.format("\\x%02X", ch));
365       }
366     }
367     return result.toString();
368   }

370   private static boolean isHexDigit(char c) {
371     return
372         (c >= 'A' && c <= 'F') ||
373         (c >= '0' && c <= '9');
374   }

376   /**
377    * Takes a ASCII digit in the range A-F0-9 and returns
378    * the corresponding integer/ordinal value.
379    * @param ch  The hex digit.
380    * @return The converted hex value as a byte.
381    */
382   public static byte toBinaryFromHex(byte ch) {
383     if ( ch >= 'A' && ch <= 'F' )
384       return (byte) ((byte)10 + (byte) (ch - 'A'));
385     // else
386     return (byte) (ch - '0');
387   }

389   public static byte [] toBytesBinary(String in) {
390     // this may be bigger than we need, but let's be safe.
391     byte [] b = new byte[in.length()];
392     int size = 0;
393     for (int i = 0; i < in.length(); ++i) {
394       char ch = in.charAt(i);
395       if (ch == '\\' && in.length() > i+1 && in.charAt(i+1) == 'x') {
396         // ok, take next 2 hex digits.
397         char hd1 = in.charAt(i+2);
398         char hd2 = in.charAt(i+3);
399 
400         // they need to be A-F0-9:
401         if (!isHexDigit(hd1) ||
402             !isHexDigit(hd2)) {
403           // bogus escape code, ignore:
404           continue;
405         }
406         // turn hex ASCII digit -> number
407         byte d = (byte) ((toBinaryFromHex((byte)hd1) << 4) + toBinaryFromHex((byte)hd2));
408 
409         b[size++] = d;
410         i += 3; // skip 3
411       } else {
412         b[size++] = (byte) ch;
413       }
414     }
415     // resize:
416     byte [] b2 = new byte[size];
417     System.arraycopy(b, 0, b2, 0, size);
418     return b2;
419   }

491   /**
492    * Converts a byte array to a long value.
493    *
494    * @param bytes array of bytes
495    * @param offset offset into array
496    * @param length length of data (must be {@link #SIZEOF_LONG})
497    * @return the long value
498    * @throws IllegalArgumentException if length is not {@link #SIZEOF_LONG} or
499    * if there's not enough room in the array at the offset indicated.
500    */
501   public static long toLong(byte[] bytes, int offset, final int length) {
502     if (length != SIZEOF_LONG || offset + length > bytes.length) {
503       throw explainWrongLengthOrOffset(bytes, offset, length, SIZEOF_LONG);
504     }
505     long l = 0;
506     for(int i = offset; i < offset + length; i++) {
507       l <<= 8;
508       l ^= bytes[i] & 0xFF;
509     }
510     return l;
511   }

513   private static IllegalArgumentException
514     explainWrongLengthOrOffset(final byte[] bytes,
515                                final int offset,
516                                final int length,
517                                final int expectedLength) {
518     String reason;
519     if (length != expectedLength) {
520       reason = "Wrong length: " + length + ", expected " + expectedLength;
521     } else {
522      reason = "offset (" + offset + ") + length (" + length + ") exceed the"
523         + " capacity of the array: " + bytes.length;
524     }
525     return new IllegalArgumentException(reason);
526   }

529    * Put a long value out to the specified byte array position.

545     }

547     return offset + SIZEOF_LONG;

670   public static int toInt(byte[] bytes, int offset, final int length) {

678     }

853   /**
854    * @param vint Integer to make a vint of.
855    * @return Vint as bytes array.
856    */
857   public static byte [] vintToBytes(final long vint) {
858     long i = vint;
859     int size = WritableUtils.getVIntSize(i);
860     byte [] result = new byte[size];
861     int offset = 0;
862     if (i >= -112 && i <= 127) {
863       result[offset] = (byte) i;
864       return result;
865     }
866 
867     int len = -112;
868     if (i < 0) {
869       i ^= -1L; // take one's complement'
870       len = -120;
871     }
872 
873     long tmp = i;
874     while (tmp != 0) {
875       tmp = tmp >> 8;
876       len--;
877     }
878 
879     result[offset++] = (byte) len;
880 
881     len = (len < -120) ? -(len + 120) : -(len + 112);
882 
883     for (int idx = len; idx != 0; idx--) {
884       int shiftbits = (idx - 1) * 8;
885       long mask = 0xFFL << shiftbits;
886       result[offset++] = (byte)((i & mask) >> shiftbits);
887     }
888     return result;
889   }

1005       @Override
1006       public int compareTo(byte[] buffer1, int offset1, int length1,
1007           byte[] buffer2, int offset2, int length2) {
1008         // Short circuit equal case
1009         if (buffer1 == buffer2 &&
1010             offset1 == offset2 &&
1011             length1 == length2) {
1012           return 0;
1013         }
1014         // Bring WritableComparator code local
1015         int end1 = offset1 + length1;
1016         int end2 = offset2 + length2;
1017         for (int i = offset1, j = offset2; i < end1 && j < end2; i++, j++) {
1018           int a = (buffer1[i] & 0xff);
1019           int b = (buffer2[j] & 0xff);
1020           if (a != b) {
1021             return a - b;
1022           }
1023         }
1024         return length1 - length2;
1025       }

1028     @VisibleForTesting
1029     enum UnsafeComparer implements Comparer<byte[]> {
1030       INSTANCE;
1031 
1032       static final Unsafe theUnsafe;
1033 
1034       /** The offset to the first element in a byte array. */
1035       static final int BYTE_ARRAY_BASE_OFFSET;
1036 
1037       static {
1038         theUnsafe = (Unsafe) AccessController.doPrivileged(
1039             new PrivilegedAction<Object>() {
1040               @Override
1041               public Object run() {
1042                 try {
1043                   Field f = Unsafe.class.getDeclaredField("theUnsafe");
1044                   f.setAccessible(true);
1045                   return f.get(null);
1046                 } catch (NoSuchFieldException e) {
1047                   // It doesn't matter what we throw;
1048                   // it's swallowed in getBestComparer().
1049                   throw new Error();
1050                 } catch (IllegalAccessException e) {
1051                   throw new Error();
1052                 }
1053               }
1054             });
1055 
1056         BYTE_ARRAY_BASE_OFFSET = theUnsafe.arrayBaseOffset(byte[].class);
1057 
1058         // sanity check - this should never fail
1059         if (theUnsafe.arrayIndexScale(byte[].class) != 1) {
1060           throw new AssertionError();
1061         }
1062       }
1063 
1064       static final boolean littleEndian =
1065         ByteOrder.nativeOrder().equals(ByteOrder.LITTLE_ENDIAN);
1066 
1067       /**
1068        * Returns true if x1 is less than x2, when both values are treated as
1069        * unsigned.
1070        */
1071       static boolean lessThanUnsigned(long x1, long x2) {
1072         return (x1 + Long.MIN_VALUE) < (x2 + Long.MIN_VALUE);
1073       }
1074 
1075       /**
1076        * Lexicographically compare two arrays.
1077        *
1078        * @param buffer1 left operand
1079        * @param buffer2 right operand
1080        * @param offset1 Where to start comparing in the left buffer
1081        * @param offset2 Where to start comparing in the right buffer
1082        * @param length1 How much to compare from the left buffer
1083        * @param length2 How much to compare from the right buffer
1084        * @return 0 if equal, < 0 if left is less than right, etc.
1085        */
1086       @Override
1087       public int compareTo(byte[] buffer1, int offset1, int length1,
1088           byte[] buffer2, int offset2, int length2) {
1089         // Short circuit equal case
1090         if (buffer1 == buffer2 &&
1091             offset1 == offset2 &&
1092             length1 == length2) {
1093           return 0;
1094         }
1095         int minLength = Math.min(length1, length2);
1096         int minWords = minLength / SIZEOF_LONG;
1097         int offset1Adj = offset1 + BYTE_ARRAY_BASE_OFFSET;
1098         int offset2Adj = offset2 + BYTE_ARRAY_BASE_OFFSET;
1099 
1100         /*
1101          * Compare 8 bytes at a time. Benchmarking shows comparing 8 bytes at a
1102          * time is no slower than comparing 4 bytes at a time even on 32-bit.
1103          * On the other hand, it is substantially faster on 64-bit.
1104          */
1105         for (int i = 0; i < minWords * SIZEOF_LONG; i += SIZEOF_LONG) {
1106           long lw = theUnsafe.getLong(buffer1, offset1Adj + (long) i);
1107           long rw = theUnsafe.getLong(buffer2, offset2Adj + (long) i);
1108           long diff = lw ^ rw;
1109 
1110           if (diff != 0) {
1111             if (!littleEndian) {
1112               return lessThanUnsigned(lw, rw) ? -1 : 1;
1113             }
1114 
1115             // Use binary search
1116             int n = 0;
1117             int y;
1118             int x = (int) diff;
1119             if (x == 0) {
1120               x = (int) (diff >>> 32);
1121               n = 32;
1122             }
1123 
1124             y = x << 16;
1125             if (y == 0) {
1126               n += 16;
1127             } else {
1128               x = y;
1129             }
1130 
1131             y = x << 8;
1132             if (y == 0) {
1133               n += 8;
1134             }
1135             return (int) (((lw >>> n) & 0xFFL) - ((rw >>> n) & 0xFFL));
1136           }
1137         }
1138 
1139         // The epilogue to cover the last (minLength % 8) elements.
1140         for (int i = minWords * SIZEOF_LONG; i < minLength; i++) {
1141           int a = (buffer1[offset1 + i] & 0xff);
1142           int b = (buffer2[offset2 + i] & 0xff);
1143           if (a != b) {
1144             return a - b;
1145           }
1146         }
1147         return length1 - length2;
1148       }
1149     }
1150   }

1374   /**
1375    * Iterate over keys within the passed range.
1376    */
1377   public static Iterable<byte[]> iterateOnSplits(
1378       final byte[] a, final byte[]b, boolean inclusive, final int num)
1379   {
1380     byte [] aPadded;
1381     byte [] bPadded;
1382     if (a.length < b.length) {
1383       aPadded = padTail(a, b.length - a.length);
1384       bPadded = b;
1385     } else if (b.length < a.length) {
1386       aPadded = a;
1387       bPadded = padTail(b, a.length - b.length);
1388     } else {
1389       aPadded = a;
1390       bPadded = b;
1391     }
1392     if (compareTo(aPadded,bPadded) >= 0) {
1393       throw new IllegalArgumentException("b <= a");
1394     }
1395     if (num <= 0) {
1396       throw new IllegalArgumentException("num cannot be < 0");
