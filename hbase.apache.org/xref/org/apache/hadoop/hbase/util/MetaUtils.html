120   /**
121    * Open or return cached opened meta region
122    *
123    * @param metaInfo HRegionInfo for meta region
124    * @return meta HRegion
125    * @throws IOException e
126    */
127   public HRegion getMetaRegion(HRegionInfo metaInfo) throws IOException {
128     HRegion meta = metaRegions.get(metaInfo.getRegionName());
129     if (meta == null) {
130       meta = openMetaRegion(metaInfo);
131       LOG.info("OPENING META " + meta.toString());
132       this.metaRegions.put(metaInfo.getRegionName(), meta);
133     }
134     return meta;
135   }

137   /**
138    * Closes catalog regions if open. Also closes and deletes the HLog. You
139    * must call this method if you want to persist changes made during a
140    * MetaUtils edit session.
141    */
142   public void shutdown() {
143     if (this.rootRegion != null) {
144       try {
145         this.rootRegion.close();
146       } catch (IOException e) {
147         LOG.error("closing root region", e);
148       } finally {
149         this.rootRegion = null;
150       }
151     }
152     try {
153       for (HRegion r: metaRegions.values()) {
154         LOG.info("CLOSING META " + r.toString());
155         r.close();
156       }
157     } catch (IOException e) {
158       LOG.error("closing meta region", e);
159     } finally {
160       metaRegions.clear();
161     }
162     try {
163       if (this.log != null) {
164         this.log.rollWriter();
165         this.log.closeAndDelete();
166       }
167     } catch (IOException e) {
168       LOG.error("closing HLog", e);
169     } finally {
170       this.log = null;
171     }
172   }

189   /**
190    * Scans the root region. For every meta region found, calls the listener with
191    * the HRegionInfo of the meta region.
192    *
193    * @param listener method to be called for each meta region found
194    * @throws IOException e
195    */
196   public void scanRootRegion(ScannerListener listener) throws IOException {
197     // Open root region so we can scan it
198     if (this.rootRegion == null) {
199       openRootRegion();
200     }
201     scanMetaRegion(this.rootRegion, listener);
202   }

204   /**
205    * Scan the passed in metaregion <code>m</code> invoking the passed
206    * <code>listener</code> per row found.
207    * @param r region
208    * @param listener scanner listener
209    * @throws IOException e
210    */
211   public void scanMetaRegion(final HRegion r, final ScannerListener listener)
212   throws IOException {
213     Scan scan = new Scan();
214     scan.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
215     InternalScanner s = r.getScanner(scan);
216     try {
217       List<KeyValue> results = new ArrayList<KeyValue>();
218       boolean hasNext = true;
219       do {
220         hasNext = s.next(results);
221         HRegionInfo info = null;
222         for (KeyValue kv: results) {
223           info = HRegionInfo.parseFromOrNull(kv.getValue());
224           if (info == null) {
225             LOG.warn("Region info is null for row " +
226               Bytes.toStringBinary(kv.getRow()) + " in table " +
227               r.getTableDesc().getNameAsString());
228           }
229           continue;
230         }
231         if (!listener.processRow(info)) {
232           break;
233         }
234         results.clear();
235       } while (hasNext);
236     } finally {
237       s.close();
238     }
239   }

241   /**
242    * Scans a meta region. For every region found, calls the listener with
243    * the HRegionInfo of the region.
244    * TODO: Use Visitor rather than Listener pattern.  Allow multiple Visitors.
245    * Use this everywhere we scan meta regions: e.g. in metascanners, in close
246    * handling, etc.  Have it pass in the whole row, not just HRegionInfo.
247    * <p>Use for reading meta only.  Does not close region when done.
248    * Use {@link #getMetaRegion(HRegionInfo)} instead if writing.  Adds
249    * meta region to list that will get a close on {@link #shutdown()}.
250    *
251    * @param metaRegionInfo HRegionInfo for meta region
252    * @param listener method to be called for each meta region found
253    * @throws IOException e
254    */
255   public void scanMetaRegion(HRegionInfo metaRegionInfo,

257   throws IOException {
258     // Open meta region so we can scan it
259     HRegion metaRegion = openMetaRegion(metaRegionInfo);
260     scanMetaRegion(metaRegion, listener);

263   private synchronized HRegion openRootRegion() throws IOException {
264     if (this.rootRegion != null) {
265       return this.rootRegion;
266     }
267     this.rootRegion = HRegion.openHRegion(HRegionInfo.ROOT_REGIONINFO,
268       HTableDescriptor.ROOT_TABLEDESC, getLog(),
269       this.conf);
270     this.rootRegion.compactStores();
271     return this.rootRegion;
272   }

274   private HRegion openMetaRegion(HRegionInfo metaInfo) throws IOException {
275     HRegion meta = HRegion.openHRegion(metaInfo, HTableDescriptor.META_TABLEDESC,
276       getLog(), this.conf);
277     meta.compactStores();
278     return meta;
279   }

281   /**
282    * Set a single region on/offline.
283    * This is a tool to repair tables that have offlined tables in their midst.
284    * Can happen on occasion.  Use at your own risk.  Call from a bit of java
285    * or jython script.  This method is 'expensive' in that it creates a
286    * {@link HTable} instance per invocation to go against <code>.META.</code>
287    * @param c A configuration that has its <code>hbase.master</code>
288    * properly set.
289    * @param row Row in the catalog .META. table whose HRegionInfo's offline
290    * status we want to change.
291    * @param onlineOffline Pass <code>true</code> to OFFLINE the region.
292    * @throws IOException e
293    */
294   public static void changeOnlineStatus (final Configuration c,
295       final byte [] row, final boolean onlineOffline)
296   throws IOException {
297     HTable t = new HTable(c, HConstants.META_TABLE_NAME);
298     Get get = new Get(row);
299     get.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
300     Result res = t.get(get);
301     KeyValue [] kvs = res.raw();
302     if(kvs.length <= 0) {
303       throw new IOException("no information for row " + Bytes.toString(row));
304     }
305     HRegionInfo info = HRegionInfo.getHRegionInfo(res);
306     if (info == null) {
307       throw new IOException("no information for row " + Bytes.toString(row));
308     }
309 
310     info.setOffline(onlineOffline);
311     MetaEditor.addRegionToMeta(t, info);
312 
313     Delete delete = new Delete(row);
314     delete.deleteColumns(HConstants.CATALOG_FAMILY, HConstants.SERVER_QUALIFIER);
315     delete.deleteColumns(HConstants.CATALOG_FAMILY,
316         HConstants.STARTCODE_QUALIFIER);
317 
318     t.delete(delete);
319   }

321   /**
322    * Update COL_REGIONINFO in meta region r with HRegionInfo hri
323    *
324    * @param r region
325    * @param hri region info
326    * @throws IOException e
327    */
328   public void updateMETARegionInfo(HRegion r, final HRegionInfo hri)
329   throws IOException {
330     if (LOG.isDebugEnabled()) {
331       Get get = new Get(hri.getRegionName());
332       get.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
333       Result res = r.get(get, null);
334       KeyValue [] kvs = res.raw();
335       if(kvs.length <= 0) {
336         return;
337       }
338 
339       HRegionInfo h = HRegionInfo.getHRegionInfo(res);
340       if (h == null) {
341         return;
342       }
343       LOG.debug("Old " + Bytes.toString(HConstants.CATALOG_FAMILY) + ":" +
344           Bytes.toString(HConstants.REGIONINFO_QUALIFIER) + " for " +
345           hri.toString() + " in " + r.toString() + " is: " + h.toString());
346     }
347 
348     Put put = MetaEditor.makePutFromRegionInfo(hri);
349     r.put(put);
350 
351     if (LOG.isDebugEnabled()) {
352       Get get = new Get(hri.getRegionName());
353       get.addColumn(HConstants.CATALOG_FAMILY, HConstants.REGIONINFO_QUALIFIER);
354       Result res = r.get(get, null);
355       KeyValue [] kvs = res.raw();
356       if(kvs.length <= 0) {
357         return;
358       }
359       HRegionInfo h = HRegionInfo.getHRegionInfo(res);
360       if (h == null) {
361         return;
362       }
363       LOG.debug("New " + Bytes.toString(HConstants.CATALOG_FAMILY) + ":" +
364             Bytes.toString(HConstants.REGIONINFO_QUALIFIER) + " for " +
365             hri.toString() + " in " + r.toString() + " is: " +  h.toString());
366     }
367   }

369   /**
370    * @return List of {@link HRegionInfo} rows found in the ROOT or META
371    * catalog table.
372    * @param tableName Name of table to go looking for.
373    * @throws IOException e
374    * @see #getMetaRegion(HRegionInfo)
375    */
376   public List<HRegionInfo> getMETARows(final byte [] tableName)
377   throws IOException {
378     final List<HRegionInfo> result = new ArrayList<HRegionInfo>();
379     // If passed table name is META, then  return the root region.
380     if (Bytes.equals(HConstants.META_TABLE_NAME, tableName)) {
381       result.add(openRootRegion().getRegionInfo());
382       return result;
383     }
384     // Return all meta regions that contain the passed tablename.
385     scanRootRegion(new ScannerListener() {
386       private final Log SL_LOG = LogFactory.getLog(this.getClass());
387 
388       public boolean processRow(HRegionInfo info) throws IOException {
389         SL_LOG.debug("Testing " + info);
390         if (Bytes.equals(info.getTableName(),
391             HConstants.META_TABLE_NAME)) {
392           result.add(info);
393           return false;
394         }
395         return true;
396       }});
397     return result;
398   }
