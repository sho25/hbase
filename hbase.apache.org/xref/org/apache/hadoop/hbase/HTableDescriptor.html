228       Map<ImmutableBytesWritable,ImmutableBytesWritable> values) {

279   /**
280    * Construct a table descriptor by cloning the descriptor passed as a parameter.
281    * <p>
282    * Makes a deep copy of the supplied descriptor.
283    * Can make a modifiable descriptor from an UnmodifyableHTableDescriptor.
284    * @param desc The descriptor.
285    */
286   public HTableDescriptor(final HTableDescriptor desc) {
287     super();
288     this.name = desc.name.clone();
289     this.nameAsString = Bytes.toString(this.name);
290     setMetaFlags(this.name);
291     for (HColumnDescriptor c: desc.families.values()) {
292       this.families.put(c.getName(), new HColumnDescriptor(c));
293     }
294     for (Map.Entry<ImmutableBytesWritable, ImmutableBytesWritable> e:
295         desc.values.entrySet()) {
296       this.values.put(e.getKey(), e.getValue());
297     }
298   }

399   /**
400    * Check passed byte buffer, "tableName", is legal user-space table name.
401    * @return Returns passed <code>tableName</code> param
402    * @throws NullPointerException If passed <code>tableName</code> is null
403    * @throws IllegalArgumentException if passed a tableName
404    * that is made of other than 'word' characters or underscores: i.e.
405    * <code>[a-zA-Z_0-9].
406    */
407   public static byte [] isLegalTableName(final byte [] tableName) {
408     if (tableName == null || tableName.length <= 0) {
409       throw new IllegalArgumentException("Name is null or empty");
410     }
411     if (tableName[0] == '.' || tableName[0] == '-') {
412       throw new IllegalArgumentException("Illegal first character <" + tableName[0] +
413           "> at 0. User-space table names can only start with 'word " +
414           "characters': i.e. [a-zA-Z_0-9]: " + Bytes.toString(tableName));
415     }
416     if (HConstants.CLUSTER_ID_FILE_NAME.equalsIgnoreCase(Bytes
417         .toString(tableName))
418         || HConstants.SPLIT_LOGDIR_NAME.equalsIgnoreCase(Bytes
419             .toString(tableName))
420         || HConstants.VERSION_FILE_NAME.equalsIgnoreCase(Bytes
421             .toString(tableName))) {
422       throw new IllegalArgumentException(Bytes.toString(tableName)
423           + " conflicted with system reserved words");
424     }
425     for (int i = 0; i < tableName.length; i++) {
426       if (Character.isLetterOrDigit(tableName[i]) || tableName[i] == '_' || 
427     		  tableName[i] == '-' || tableName[i] == '.') {
428         continue;
429       }
430       throw new IllegalArgumentException("Illegal character <" + tableName[i] +
431         "> at " + i + ". User-space table names can only contain " +
432         "'word characters': i.e. [a-zA-Z_0-9-.]: " + Bytes.toString(tableName));
433     }
434     return tableName;
435   }

581   /**
582    * This is used to defer the log edits syncing to the file system. Everytime 
583    * an edit is sent to the server it is first sync'd to the file system by the 
584    * log writer. This sync is an expensive operation and thus can be deferred so 
585    * that the edits are kept in memory for a specified period of time as represented
586    * by <code> hbase.regionserver.optionallogflushinterval </code> and not flushed
587    * for every edit.
588    * <p>
589    * NOTE:- This option might result in data loss if the region server crashes
590    * before these deferred edits in memory are flushed onto the filesystem. 
591    * </p>
592    * 
593    * @param isDeferredLogFlush
594    */
595   public void setDeferredLogFlush(final boolean isDeferredLogFlush) {
596     setValue(DEFERRED_LOG_FLUSH_KEY, isDeferredLogFlush? TRUE: FALSE);
597     this.deferredLog = isDeferredLogFlush;
598   }

753   private StringBuilder getValues(boolean printDefaults) {
754     StringBuilder s = new StringBuilder();
755 
756     // step 1: set partitioning and pruning
757     Set<ImmutableBytesWritable> reservedKeys = new TreeSet<ImmutableBytesWritable>();
758     Set<ImmutableBytesWritable> configKeys = new TreeSet<ImmutableBytesWritable>();
759     for (ImmutableBytesWritable k : values.keySet()) {
760       if (k == null || k.get() == null) continue;
761       String key = Bytes.toString(k.get());
762       // in this section, print out reserved keywords + coprocessor info
763       if (!RESERVED_KEYWORDS.contains(k) && !key.startsWith("coprocessor$")) {
764         configKeys.add(k);
765         continue;
766       }
767       // only print out IS_ROOT/IS_META if true
768       String value = Bytes.toString(values.get(k).get());
769       if (key.equalsIgnoreCase(IS_ROOT) || key.equalsIgnoreCase(IS_META)) {
770         if (Boolean.valueOf(value) == false) continue;
771       }
772       // see if a reserved key is a default value. may not want to print it out
773       if (printDefaults
774           || !DEFAULT_VALUES.containsKey(key)
775           || !DEFAULT_VALUES.get(key).equalsIgnoreCase(value)) {
776         reservedKeys.add(k);
777       }
778     }
779 
780     // early exit optimization
781     if (reservedKeys.isEmpty() && configKeys.isEmpty()) return s;
782 
783     // step 2: printing
784     s.append(", {TABLE_ATTRIBUTES => {");
785 
786     // print all reserved keys first
787     boolean printCommaForAttr = false;
788     for (ImmutableBytesWritable k : reservedKeys) {
789       String key = Bytes.toString(k.get());
790       String value = Bytes.toString(values.get(k).get());
791       if (printCommaForAttr) s.append(", ");
792       printCommaForAttr = true;
793       s.append(key);
794       s.append(" => ");
795       s.append('\'').append(value).append('\'');
796     }
797 
798     if (!configKeys.isEmpty()) {
799       // print all non-reserved, advanced config keys as a separate subset
800       if (printCommaForAttr) s.append(", ");
801       printCommaForAttr = true;
802       s.append(HConstants.CONFIG).append(" => ");
803       s.append("{");
804       boolean printCommaForCfg = false;
805       for (ImmutableBytesWritable k : configKeys) {
806         String key = Bytes.toString(k.get());
807         String value = Bytes.toString(values.get(k).get());
808         if (printCommaForCfg) s.append(", ");
809         printCommaForCfg = true;
810         s.append('\'').append(key).append('\'');
811         s.append(" => ");
812         s.append('\'').append(value).append('\'');
813       }
814       s.append("}");
815     }
816 
817     s.append("}}"); // end METHOD
818     return s;
819   }

932    * This compares the content of the two descriptors and not the reference.
933    *

935    * 		 1 if there is a mismatch in the contents

937   @Override
938   public int compareTo(final HTableDescriptor other) {
939     int result = Bytes.compareTo(this.name, other.name);
940     if (result == 0) {
941       result = families.size() - other.families.size();
942     }
943     if (result == 0 && families.size() != other.families.size()) {
944       result = Integer.valueOf(families.size()).compareTo(
945           Integer.valueOf(other.families.size()));
946     }
947     if (result == 0) {
948       for (Iterator<HColumnDescriptor> it = families.values().iterator(),
949           it2 = other.families.values().iterator(); it.hasNext(); ) {
950         result = it.next().compareTo(it2.next());
951         if (result != 0) {
952           break;
953         }
954       }
955     }
956     if (result == 0) {
957       // punt on comparison for ordering, just calculate difference
958       result = this.values.hashCode() - other.values.hashCode();
959       if (result < 0)
960         result = -1;
961       else if (result > 0)

1045    * Add a table coprocessor to this table. The coprocessor
1046    * type must be {@link org.apache.hadoop.hbase.coprocessor.RegionObserver}
1047    * or Endpoint.
1048    * It won't check if the class can be loaded or not.
1049    * Whether a coprocessor is loadable or not will be determined when
1050    * a region is opened.
1051    * @param jarFilePath Path of the jar file. If it's null, the class will be
1052    * loaded from default classloader.
1053    * @param className Full class name.
1054    * @param priority Priority
1055    * @param kvs Arbitrary key-value parameter pairs passed into the coprocessor.
1056    * @throws IOException
1057    */
1058   public void addCoprocessor(String className, Path jarFilePath,
1059                              int priority, final Map<String, String> kvs)
1060   throws IOException {
1061     if (hasCoprocessor(className)) {
1062       throw new IOException("Coprocessor " + className + " already exists.");
1063     }
1064     // validate parameter kvs
1065     StringBuilder kvString = new StringBuilder();
1066     if (kvs != null) {
1067       for (Map.Entry<String, String> e: kvs.entrySet()) {
1068         if (!e.getKey().matches(HConstants.CP_HTD_ATTR_VALUE_PARAM_KEY_PATTERN)) {
1069           throw new IOException("Illegal parameter key = " + e.getKey());
1070         }
1071         if (!e.getValue().matches(HConstants.CP_HTD_ATTR_VALUE_PARAM_VALUE_PATTERN)) {
1072           throw new IOException("Illegal parameter (" + e.getKey() +
1073               ") value = " + e.getValue());
1074         }
1075         if (kvString.length() != 0) {
1076           kvString.append(',');
1077         }
1078         kvString.append(e.getKey());
1079         kvString.append('=');
1080         kvString.append(e.getValue());
1081       }
1082     }
1083 
1084     // generate a coprocessor key
1085     int maxCoprocessorNumber = 0;
1086     Matcher keyMatcher;
1087     for (Map.Entry<ImmutableBytesWritable, ImmutableBytesWritable> e:
1088         this.values.entrySet()) {
1089       keyMatcher =
1090           HConstants.CP_HTD_ATTR_KEY_PATTERN.matcher(
1091               Bytes.toString(e.getKey().get()));
1092       if (!keyMatcher.matches()) {
1093         continue;
1094       }
1095       maxCoprocessorNumber = Math.max(Integer.parseInt(keyMatcher.group(1)),
1096           maxCoprocessorNumber);
1097     }
1098     maxCoprocessorNumber++;
1099 
1100     String key = "coprocessor$" + Integer.toString(maxCoprocessorNumber);
1101     String value = ((jarFilePath == null)? "" : jarFilePath.toString()) +
1102         "|" + className + "|" + Integer.toString(priority) + "|" +
1103         kvString.toString();
1104     setValue(key, value);
1105   }
