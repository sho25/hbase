181    * @throws IOException
182    * @throws KeeperException

186   @Test(timeout = 5000)
187   public void testBalanceOnMasterFailoverScenarioWithOpenedNode()
188   throws IOException, KeeperException, InterruptedException, ServiceException, DeserializationException {
189     AssignmentManagerWithExtrasForTesting am =
190       setUpMockedAssignmentManager(this.server, this.serverManager);
191     try {
192       createRegionPlanAndBalance(am, SERVERNAME_A, SERVERNAME_B, REGIONINFO);
193       startFakeFailedOverMasterAssignmentManager(am, this.watcher);
194       while (!am.processRITInvoked) Thread.sleep(1);
195       // As part of the failover cleanup, the balancing region plan is removed.
196       // So a random server will be used to open the region. For testing purpose,
197       // let's assume it is going to open on server b:
198       am.addPlan(REGIONINFO.getEncodedName(), new RegionPlan(REGIONINFO, null, SERVERNAME_B));
199 
200       // Now fake the region closing successfully over on the regionserver; the
201       // regionserver will have set the region in CLOSED state. This will
202       // trigger callback into AM. The below zk close call is from the RS close
203       // region handler duplicated here because its down deep in a private
204       // method hard to expose.
205       int versionid =
206         ZKAssign.transitionNodeClosed(this.watcher, REGIONINFO, SERVERNAME_A, -1);
207       assertNotSame(versionid, -1);
208       Mocking.waitForRegionPendingOpenInRIT(am, REGIONINFO.getEncodedName());
209 
210       // Get current versionid else will fail on transition from OFFLINE to
211       // OPENING below
212       versionid = ZKAssign.getVersion(this.watcher, REGIONINFO);
213       assertNotSame(-1, versionid);
214       // This uglyness below is what the openregionhandler on RS side does.
215       versionid = ZKAssign.transitionNode(server.getZooKeeper(), REGIONINFO,
216         SERVERNAME_B, EventType.M_ZK_REGION_OFFLINE,
217         EventType.RS_ZK_REGION_OPENING, versionid);
218       assertNotSame(-1, versionid);
219       // Move znode from OPENING to OPENED as RS does on successful open.
220       versionid = ZKAssign.transitionNodeOpened(this.watcher, REGIONINFO,
221         SERVERNAME_B, versionid);
222       assertNotSame(-1, versionid);
223       am.gate.set(false);
224       // Block here until our znode is cleared or until this test times out.
225       ZKAssign.blockUntilNoRIT(watcher);
226     } finally {
227       am.getExecutorService().shutdown();
228       am.shutdown();
229     }
230   }

337   /**
338    * Tests AssignmentManager balance function.  Runs a balance moving a region
339    * from one server to another mocking regionserver responding over zk.
340    * @throws IOException
341    * @throws KeeperException
342    * @throws DeserializationException 
343    */
344   @Test
345   public void testBalance()
346     throws IOException, KeeperException, DeserializationException, InterruptedException {
347     // Create and startup an executor.  This is used by AssignmentManager
348     // handling zk callbacks.
349     ExecutorService executor = startupMasterExecutor("testBalanceExecutor");
350 
351     // We need a mocked catalog tracker.
352     CatalogTracker ct = Mockito.mock(CatalogTracker.class);
353     LoadBalancer balancer = LoadBalancerFactory.getLoadBalancer(server
354         .getConfiguration());
355     // Create an AM.
356     AssignmentManager am = new AssignmentManager(this.server,
357       this.serverManager, ct, balancer, executor, null);
358     am.failoverCleanupDone.set(true);
359     try {
360       // Make sure our new AM gets callbacks; once registered, can't unregister.
361       // Thats ok because we make a new zk watcher for each test.
362       this.watcher.registerListenerFirst(am);
363       // Call the balance function but fake the region being online first at
364       // SERVERNAME_A.  Create a balance plan.
365       am.regionOnline(REGIONINFO, SERVERNAME_A);
366       // Balance region from A to B.
367       RegionPlan plan = new RegionPlan(REGIONINFO, SERVERNAME_A, SERVERNAME_B);
368       am.balance(plan);
369 
370       // Now fake the region closing successfully over on the regionserver; the
371       // regionserver will have set the region in CLOSED state.  This will
372       // trigger callback into AM. The below zk close call is from the RS close
373       // region handler duplicated here because its down deep in a private
374       // method hard to expose.
375       int versionid =
376         ZKAssign.transitionNodeClosed(this.watcher, REGIONINFO, SERVERNAME_A, -1);
377       assertNotSame(versionid, -1);
378       // AM is going to notice above CLOSED and queue up a new assign.  The
379       // assign will go to open the region in the new location set by the
380       // balancer.  The zk node will be OFFLINE waiting for regionserver to
381       // transition it through OPENING, OPENED.  Wait till we see the OFFLINE
382       // zk node before we proceed.
383       Mocking.waitForRegionPendingOpenInRIT(am, REGIONINFO.getEncodedName());
384 
385       // Get current versionid else will fail on transition from OFFLINE to OPENING below
386       versionid = ZKAssign.getVersion(this.watcher, REGIONINFO);
387       assertNotSame(-1, versionid);
388       // This uglyness below is what the openregionhandler on RS side does.
389       versionid = ZKAssign.transitionNode(server.getZooKeeper(), REGIONINFO,
390         SERVERNAME_A, EventType.M_ZK_REGION_OFFLINE,
391         EventType.RS_ZK_REGION_OPENING, versionid);
392       assertNotSame(-1, versionid);
393       // Move znode from OPENING to OPENED as RS does on successful open.
394       versionid =
395         ZKAssign.transitionNodeOpened(this.watcher, REGIONINFO, SERVERNAME_B, versionid);
396       assertNotSame(-1, versionid);
397       // Wait on the handler removing the OPENED znode.
398       while(am.getRegionStates().isRegionInTransition(REGIONINFO)) Threads.sleep(1);
399     } finally {
400       executor.shutdown();
401       am.shutdown();
402       // Clean up all znodes
403       ZKAssign.deleteAllNodes(this.watcher);
404     }
405   }

466   private void testCaseWithSplitRegionPartial(boolean regionSplitDone) throws KeeperException,
467       IOException, NodeExistsException, InterruptedException, ServiceException {
468     // Create and startup an executor. This is used by AssignmentManager
469     // handling zk callbacks.
470     ExecutorService executor = startupMasterExecutor("testSSHWhenSplitRegionInProgress");
471     // We need a mocked catalog tracker.
472     CatalogTracker ct = Mockito.mock(CatalogTracker.class);
473     ZKAssign.deleteAllNodes(this.watcher);
474 
475     // Create an AM.
476     AssignmentManagerWithExtrasForTesting am = setUpMockedAssignmentManager(
477       this.server, this.serverManager);
478     // adding region to regions and servers maps.
479     am.regionOnline(REGIONINFO, SERVERNAME_A);
480     // adding region in pending close.
481     am.getRegionStates().updateRegionState(
482       REGIONINFO, State.SPLITTING, SERVERNAME_A);
483     am.getZKTable().setEnabledTable(REGIONINFO.getTableNameAsString());
484     RegionTransition data = RegionTransition.createRegionTransition(EventType.RS_ZK_REGION_SPLITTING,
485         REGIONINFO.getRegionName(), SERVERNAME_A);
486     String node = ZKAssign.getNodeName(this.watcher, REGIONINFO.getEncodedName());
487     // create znode in M_ZK_REGION_CLOSING state.
488     ZKUtil.createAndWatch(this.watcher, node, data.toByteArray());
489 
490     try {
491       processServerShutdownHandler(ct, am, regionSplitDone);
492       // check znode deleted or not.
493       // In both cases the znode should be deleted.
494 
495       if (regionSplitDone) {
496         assertFalse("Region state of region in SPLITTING should be removed from rit.",
497             am.getRegionStates().isRegionsInTransition());
498       } else {
499         while (!am.assignInvoked) {
500           Thread.sleep(1);
501         }
502         assertTrue("Assign should be invoked.", am.assignInvoked);
503       }
504     } finally {
505       REGIONINFO.setOffline(false);
506       REGIONINFO.setSplit(false);
507       executor.shutdown();
508       am.shutdown();
509       // Clean up all znodes
510       ZKAssign.deleteAllNodes(this.watcher);
511     }
512   }

514   private void testCaseWithPartiallyDisabledState(Table.State state) throws KeeperException,
515       IOException, NodeExistsException, ServiceException {
516     // Create and startup an executor. This is used by AssignmentManager
517     // handling zk callbacks.
518     ExecutorService executor = startupMasterExecutor("testSSHWhenDisableTableInProgress");
519     // We need a mocked catalog tracker.

521     LoadBalancer balancer = LoadBalancerFactory.getLoadBalancer(server.getConfiguration());
522     ZKAssign.deleteAllNodes(this.watcher);
523 
524     // Create an AM.
525     AssignmentManager am = new AssignmentManager(this.server,
526       this.serverManager, ct, balancer, executor, null);
527     // adding region to regions and servers maps.
528     am.regionOnline(REGIONINFO, SERVERNAME_A);
529     // adding region in pending close.
530     am.getRegionStates().updateRegionState(REGIONINFO, State.PENDING_CLOSE);
531     if (state == Table.State.DISABLING) {
532       am.getZKTable().setDisablingTable(REGIONINFO.getTableNameAsString());
533     } else {
534       am.getZKTable().setDisabledTable(REGIONINFO.getTableNameAsString());
535     }
536     RegionTransition data = RegionTransition.createRegionTransition(EventType.M_ZK_REGION_CLOSING,
537         REGIONINFO.getRegionName(), SERVERNAME_A);
538     // RegionTransitionData data = new
539     // RegionTransitionData(EventType.M_ZK_REGION_CLOSING,
540     // REGIONINFO.getRegionName(), SERVERNAME_A);
