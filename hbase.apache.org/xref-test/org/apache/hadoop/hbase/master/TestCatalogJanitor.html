320   @Test
321   public void testCleanParent() throws IOException, InterruptedException {
322     HBaseTestingUtility htu = new HBaseTestingUtility();
323     setRootDirAndCleanIt(htu, "testCleanParent");
324     Server server = new MockServer(htu);
325     try {
326       MasterServices services = new MockMasterServices(server);
327       CatalogJanitor janitor = new CatalogJanitor(server, services);
328       // Create regions.
329       HTableDescriptor htd = new HTableDescriptor("table");
330       htd.addFamily(new HColumnDescriptor("f"));
331       HRegionInfo parent =
332         new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
333             Bytes.toBytes("eee"));
334       HRegionInfo splita =
335         new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
336             Bytes.toBytes("ccc"));
337       HRegionInfo splitb =
338         new HRegionInfo(htd.getName(), Bytes.toBytes("ccc"),
339             Bytes.toBytes("eee"));
340       // Test that when both daughter regions are in place, that we do not
341       // remove the parent.
342       Result r = createResult(parent, splita, splitb);
343       // Add a reference under splitA directory so we don't clear out the parent.
344       Path rootdir = services.getMasterFileSystem().getRootDir();
345       Path tabledir =
346         HTableDescriptor.getTableDir(rootdir, htd.getName());
347       Path storedir = HStore.getStoreHomedir(tabledir, splita.getEncodedName(),
348           htd.getColumnFamilies()[0].getName());
349       Reference ref = Reference.createTopReference(Bytes.toBytes("ccc"));
350       long now = System.currentTimeMillis();
351       // Reference name has this format: StoreFile#REF_NAME_PARSER
352       Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
353       FileSystem fs = services.getMasterFileSystem().getFileSystem();
354       Path path = ref.write(fs, p);
355       assertTrue(fs.exists(path));
356       assertFalse(janitor.cleanParent(parent, r));
357       // Remove the reference file and try again.
358       assertTrue(fs.delete(p, true));
359       assertTrue(janitor.cleanParent(parent, r));
360     } finally {
361       server.stop("shutdown");
362     }
363   }

389   /**
390    * Make sure parent with specified end key gets cleaned up even if daughter is cleaned up before it.
391    *
392    * @param rootDir the test case name, used as the HBase testing utility root
393    * @param lastEndKey the end key of the split parent
394    * @throws IOException
395    * @throws InterruptedException
396    */
397   private void parentWithSpecifiedEndKeyCleanedEvenIfDaughterGoneFirst(
398   final String rootDir, final byte[] lastEndKey)
399   throws IOException, InterruptedException {
400     HBaseTestingUtility htu = new HBaseTestingUtility();
401     setRootDirAndCleanIt(htu, rootDir);
402     Server server = new MockServer(htu);
403     MasterServices services = new MockMasterServices(server);
404     CatalogJanitor janitor = new CatalogJanitor(server, services);
405     final HTableDescriptor htd = createHTableDescriptor();
406 
407     // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
408 
409     // Parent
410     HRegionInfo parent = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
411       lastEndKey);
412     // Sleep a second else the encoded name on these regions comes out
413     // same for all with same start key and made in same second.
414     Thread.sleep(1001);
415 
416     // Daughter a
417     HRegionInfo splita = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
418       Bytes.toBytes("ccc"));
419     Thread.sleep(1001);
420     // Make daughters of daughter a; splitaa and splitab.
421     HRegionInfo splitaa = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
422       Bytes.toBytes("bbb"));
423     HRegionInfo splitab = new HRegionInfo(htd.getName(), Bytes.toBytes("bbb"),
424       Bytes.toBytes("ccc"));
425 
426     // Daughter b
427     HRegionInfo splitb = new HRegionInfo(htd.getName(), Bytes.toBytes("ccc"),
428       lastEndKey);
429     Thread.sleep(1001);
430     // Make Daughters of daughterb; splitba and splitbb.
431     HRegionInfo splitba = new HRegionInfo(htd.getName(), Bytes.toBytes("ccc"),
432       Bytes.toBytes("ddd"));
433     HRegionInfo splitbb = new HRegionInfo(htd.getName(), Bytes.toBytes("ddd"),
434     lastEndKey);
435 
436     // First test that our Comparator works right up in CatalogJanitor.
437     // Just fo kicks.
438     SortedMap<HRegionInfo, Result> regions =
439       new TreeMap<HRegionInfo, Result>(new CatalogJanitor.SplitParentFirstComparator());
440     // Now make sure that this regions map sorts as we expect it to.
441     regions.put(parent, createResult(parent, splita, splitb));
442     regions.put(splitb, createResult(splitb, splitba, splitbb));
443     regions.put(splita, createResult(splita, splitaa, splitab));
444     // Assert its properly sorted.
445     int index = 0;
446     for (Map.Entry<HRegionInfo, Result> e: regions.entrySet()) {
447       if (index == 0) {
448         assertTrue(e.getKey().getEncodedName().equals(parent.getEncodedName()));
449       } else if (index == 1) {
450         assertTrue(e.getKey().getEncodedName().equals(splita.getEncodedName()));
451       } else if (index == 2) {
452         assertTrue(e.getKey().getEncodedName().equals(splitb.getEncodedName()));
453       }
454       index++;
455     }
456 
457     // Now play around with the cleanParent function.  Create a ref from splita
458     // up to the parent.
459     Path splitaRef =
460       createReferences(services, htd, parent, splita, Bytes.toBytes("ccc"), false);
461     // Make sure actual super parent sticks around because splita has a ref.
462     assertFalse(janitor.cleanParent(parent, regions.get(parent)));
463 
464     //splitba, and split bb, do not have dirs in fs.  That means that if
465     // we test splitb, it should get cleaned up.
466     assertTrue(janitor.cleanParent(splitb, regions.get(splitb)));
467 
468     // Now remove ref from splita to parent... so parent can be let go and so
469     // the daughter splita can be split (can't split if still references).
470     // BUT make the timing such that the daughter gets cleaned up before we
471     // can get a chance to let go of the parent.
472     FileSystem fs = FileSystem.get(htu.getConfiguration());
473     assertTrue(fs.delete(splitaRef, true));
474     // Create the refs from daughters of splita.
475     Path splitaaRef =
476       createReferences(services, htd, splita, splitaa, Bytes.toBytes("bbb"), false);
477     Path splitabRef =
478       createReferences(services, htd, splita, splitab, Bytes.toBytes("bbb"), true);
479 
480     // Test splita.  It should stick around because references from splitab, etc.
481     assertFalse(janitor.cleanParent(splita, regions.get(splita)));
482 
483     // Now clean up parent daughter first.  Remove references from its daughters.
484     assertTrue(fs.delete(splitaaRef, true));
485     assertTrue(fs.delete(splitabRef, true));
486     assertTrue(janitor.cleanParent(splita, regions.get(splita)));
487 
488     // Super parent should get cleaned up now both splita and splitb are gone.
489     assertTrue(janitor.cleanParent(parent, regions.get(parent)));
490 
491     services.stop("test finished");
492     janitor.join();
493   }

496    * CatalogJanitor.scan() should not clean parent regions if their own
497    * parents are still referencing them. This ensures that grandfather regions
498    * do not point to deleted parent regions.

500   @Test
501   public void testScanDoesNotCleanRegionsWithExistingParents() throws Exception {
502     HBaseTestingUtility htu = new HBaseTestingUtility();
503     setRootDirAndCleanIt(htu, "testScanDoesNotCleanRegionsWithExistingParents");
504     Server server = new MockServer(htu);
505     MasterServices services = new MockMasterServices(server);
506 
507     final HTableDescriptor htd = createHTableDescriptor();
508 
509     // Create regions: aaa->{lastEndKey}, aaa->ccc, aaa->bbb, bbb->ccc, etc.
510 
511     // Parent
512     HRegionInfo parent = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
513       new byte[0], true);
514     // Sleep a second else the encoded name on these regions comes out
515     // same for all with same start key and made in same second.
516     Thread.sleep(1001);
517 
518     // Daughter a
519     HRegionInfo splita = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
520       Bytes.toBytes("ccc"), true);
521     Thread.sleep(1001);
522     // Make daughters of daughter a; splitaa and splitab.
523     HRegionInfo splitaa = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"),
524       Bytes.toBytes("bbb"), false);
525     HRegionInfo splitab = new HRegionInfo(htd.getName(), Bytes.toBytes("bbb"),
526       Bytes.toBytes("ccc"), false);
527 
528     // Daughter b
529     HRegionInfo splitb = new HRegionInfo(htd.getName(), Bytes.toBytes("ccc"),
530         new byte[0]);
531     Thread.sleep(1001);
532 
533     final Map<HRegionInfo, Result> splitParents =
534         new TreeMap<HRegionInfo, Result>(new SplitParentFirstComparator());
535     splitParents.put(parent, createResult(parent, splita, splitb));
536     splita.setOffline(true); //simulate that splita goes offline when it is split
537     splitParents.put(splita, createResult(splita, splitaa,splitab));
538 
539     CatalogJanitor janitor = spy(new CatalogJanitor(server, services));
540     doReturn(new Pair<Integer, Map<HRegionInfo, Result>>(
541         10, splitParents)).when(janitor).getSplitParents();
542 
543     //create ref from splita to parent
544     Path splitaRef =
545         createReferences(services, htd, parent, splita, Bytes.toBytes("ccc"), false);
546 
547     //parent and A should not be removed
548     assertEquals(0, janitor.scan());
549 
550     //now delete the ref
551     FileSystem fs = FileSystem.get(htu.getConfiguration());
552     assertTrue(fs.delete(splitaRef, true));
553 
554     //now, both parent, and splita can be deleted
555     assertEquals(2, janitor.scan());
556 
557     services.stop("test finished");
558     janitor.join();
559   }

580     HRegionInfo splita = new HRegionInfo(htd.getName(), Bytes.toBytes("aaa"), Bytes.toBytes("ccc"));
581     HRegionInfo splitb = new HRegionInfo(htd.getName(), Bytes.toBytes("ccc"), Bytes.toBytes("eee"));

584     // remove the parent.

586     FileSystem fs = FileSystem.get(htu.getConfiguration());
587     Path rootdir = services.getMasterFileSystem().getRootDir();
588     // have to set the root directory since we use it in HFileDisposer to figure out to get to the
589     // archive directory. Otherwise, it just seems to pick the first root directory it can find (so
590     // the single test passes, but when the full suite is run, things get borked).
591     FSUtils.setRootDir(fs.getConf(), rootdir);
592     Path tabledir = HTableDescriptor.getTableDir(rootdir, htd.getName());
593     Path storedir = HStore.getStoreHomedir(tabledir, parent.getEncodedName(),
594       htd.getColumnFamilies()[0].getName());
595     Path storeArchive = HFileArchiveUtil.getStoreArchivePath(services.getConfiguration(), parent,
596       tabledir, htd.getColumnFamilies()[0].getName());
597     LOG.debug("Table dir:" + tabledir);
598     LOG.debug("Store dir:" + storedir);
599     LOG.debug("Store archive dir:" + storeArchive);
600 
601     // add a couple of store files that we can check for
602     FileStatus[] mockFiles = addMockStoreFiles(2, services, storedir);
603     // get the current store files for comparison
604     FileStatus[] storeFiles = fs.listStatus(storedir);
605     int index = 0;
606     for (FileStatus file : storeFiles) {
607       LOG.debug("Have store file:" + file.getPath());
608       assertEquals("Got unexpected store file", mockFiles[index].getPath(),
609         storeFiles[index].getPath());
610       index++;
611     }
612 
613     // do the cleaning of the parent
614     assertTrue(janitor.cleanParent(parent, parentMetaRow));
615     LOG.debug("Finished cleanup of parent region");
616 
617     // and now check to make sure that the files have actually been archived
618     FileStatus[] archivedStoreFiles = fs.listStatus(storeArchive);
619     logFiles("archived files", storeFiles);
620     logFiles("archived files", archivedStoreFiles);
621 
622     assertArchiveEqualToOriginal(storeFiles, archivedStoreFiles, fs);
623 
624     // cleanup
625     FSUtils.delete(fs, rootdir, true);
626     services.stop("Test finished");
627     server.stop("Test finished");
628     janitor.join();
629   }

753   private Path createReferences(final MasterServices services,
754       final HTableDescriptor htd, final HRegionInfo parent,
755       final HRegionInfo daughter, final byte [] midkey, final boolean top)
756   throws IOException {
757     Path rootdir = services.getMasterFileSystem().getRootDir();
758     Path tabledir = HTableDescriptor.getTableDir(rootdir, parent.getTableName());
759     Path storedir = HStore.getStoreHomedir(tabledir, daughter.getEncodedName(),
760       htd.getColumnFamilies()[0].getName());
761     Reference ref =
762       top? Reference.createTopReference(midkey): Reference.createBottomReference(midkey);
763     long now = System.currentTimeMillis();
764     // Reference name has this format: StoreFile#REF_NAME_PARSER
765     Path p = new Path(storedir, Long.toString(now) + "." + parent.getEncodedName());
766     FileSystem fs = services.getMasterFileSystem().getFileSystem();
767     ref.write(fs, p);
