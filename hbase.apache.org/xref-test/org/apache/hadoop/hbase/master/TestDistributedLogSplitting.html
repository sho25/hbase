86    private void startCluster(int num_rs) throws Exception{
87      SplitLogCounters.resetCounters();
88      LOG.info("Starting cluster");
89      conf = HBaseConfiguration.create();
90      conf.getLong("hbase.splitlog.max.resubmit", 0);
91      // Make the failure test faster
92      conf.setInt("zookeeper.recovery.retry", 0);
93      conf.setInt(HConstants.REGIONSERVER_INFO_PORT, -1);
94      conf.setFloat(HConstants.LOAD_BALANCER_SLOP_KEY, (float) 100.0); // no load balancing
95      conf.setBoolean(HConstants.DISTRIBUTED_LOG_SPLITTING_KEY, true);
96      TEST_UTIL = new HBaseTestingUtility(conf);
97      TEST_UTIL.startMiniCluster(NUM_MASTERS, num_rs);
98      cluster = TEST_UTIL.getHBaseCluster();
99      LOG.info("Waiting for active/ready master");
100     cluster.waitForActiveAndReadyMaster();
101     master = cluster.getMaster();
102     while (cluster.getLiveRegionServerThreads().size() < num_rs) {
103       Threads.sleep(1);
104     }
105   }

112   @Test (timeout=300000)
113   public void testRecoveredEdits() throws Exception {
114     LOG.info("testRecoveredEdits");
115     startCluster(NUM_RS);
116     final int NUM_LOG_LINES = 1000;
117     final SplitLogManager slm = master.getMasterFileSystem().splitLogManager;
118     // turn off load balancing to prevent regions from moving around otherwise
119     // they will consume recovered.edits
120     master.balanceSwitch(false);
121     FileSystem fs = master.getMasterFileSystem().getFileSystem();
122 
123     List<RegionServerThread> rsts = cluster.getLiveRegionServerThreads();
124 
125     Path rootdir = FSUtils.getRootDir(conf);
126 
127     installTable(new ZooKeeperWatcher(conf, "table-creation", null),
128         "table", "family", 40);
129     byte[] table = Bytes.toBytes("table");
130     List<HRegionInfo> regions = null;
131     HRegionServer hrs = null;
132     for (int i = 0; i < NUM_RS; i++) {
133       hrs = rsts.get(i).getRegionServer();
134       regions = ProtobufUtil.getOnlineRegions(hrs);
135       if (regions.size() != 0) break;
136     }
137     final Path logDir = new Path(rootdir, HLogUtil.getHLogDirectoryName(hrs
138         .getServerName().toString()));
139 
140     LOG.info("#regions = " + regions.size());
141     Iterator<HRegionInfo> it = regions.iterator();
142     while (it.hasNext()) {
143       HRegionInfo region = it.next();
144       if (region.isMetaTable()) {
145         it.remove();
146       }
147     }
148     makeHLog(hrs.getWAL(), regions, "table",
149         NUM_LOG_LINES, 100);
150 
151     slm.splitLogDistributed(logDir);
152 
153     int count = 0;
154     for (HRegionInfo hri : regions) {
155 
156       Path tdir = HTableDescriptor.getTableDir(rootdir, table);
157       Path editsdir =
158         HLogUtil.getRegionDirRecoveredEditsDir(HRegion.getRegionDir(tdir, hri.getEncodedName()));
159       LOG.debug("checking edits dir " + editsdir);
160       FileStatus[] files = fs.listStatus(editsdir);
161       assertEquals(1, files.length);
162       int c = countHLog(files[0].getPath(), fs, conf);
163       count += c;
164       LOG.info(c + " edits in " + files[0].getPath());
165     }
166     assertEquals(NUM_LOG_LINES, count);
167   }

179   public void testWorkerAbort() throws Exception {
180     LOG.info("testWorkerAbort");
181     startCluster(1);
182     final int NUM_LOG_LINES = 10000;
183     final SplitLogManager slm = master.getMasterFileSystem().splitLogManager;
184     FileSystem fs = master.getMasterFileSystem().getFileSystem();
185 
186     final List<RegionServerThread> rsts = cluster.getLiveRegionServerThreads();
187     HRegionServer hrs = rsts.get(0).getRegionServer();
188     Path rootdir = FSUtils.getRootDir(conf);
189     final Path logDir = new Path(rootdir,
190         HLogUtil.getHLogDirectoryName(hrs.getServerName().toString()));
191 
192     installTable(new ZooKeeperWatcher(conf, "table-creation", null),
193         "table", "family", 40);
194     makeHLog(hrs.getWAL(), ProtobufUtil.getOnlineRegions(hrs), "table",
195         NUM_LOG_LINES, 100);
196 
197     new Thread() {
198       public void run() {
199         waitForCounter(tot_wkr_task_acquired, 0, 1, 1000);
200         for (RegionServerThread rst : rsts) {
201           rst.getRegionServer().abort("testing");
202         }
203       }
204     }.start();
205     // slm.splitLogDistributed(logDir);
206     FileStatus[] logfiles = fs.listStatus(logDir);
207     TaskBatch batch = new TaskBatch();
208     slm.enqueueSplitTask(logfiles[0].getPath().toString(), batch);
209     //waitForCounter but for one of the 2 counters
210     long curt = System.currentTimeMillis();
211     long waitTime = 80000;
212     long endt = curt + waitTime;
213     while (curt < endt) {
214       if ((tot_wkr_task_resigned.get() + tot_wkr_task_err.get() +
215           tot_wkr_final_transistion_failed.get() + tot_wkr_task_done.get() +
216           tot_wkr_preempt_task.get()) == 0) {
217         Thread.yield();
218         curt = System.currentTimeMillis();
219       } else {
220         assertEquals(1, (tot_wkr_task_resigned.get() + tot_wkr_task_err.get() +
221             tot_wkr_final_transistion_failed.get() + tot_wkr_task_done.get() +
222             tot_wkr_preempt_task.get()));
223         return;
224       }
225     }
226     fail("none of the following counters went up in " + waitTime +
227         " milliseconds - " +
228         "tot_wkr_task_resigned, tot_wkr_task_err, " +
229         "tot_wkr_final_transistion_failed, tot_wkr_task_done, " +
230         "tot_wkr_preempt_task");
231   }

237     final int NUM_ROWS_PER_REGION = 100;
238 
239     startCluster(NUM_RS); // NUM_RS=6.

242         "distributed log splitting test", null);
243 
244     HTable ht = installTable(zkw, "table", "family", NUM_REGIONS_TO_CREATE);
245     populateDataInTable(NUM_ROWS_PER_REGION, "family");
246 
247 
248     List<RegionServerThread> rsts = cluster.getLiveRegionServerThreads();
249     assertEquals(NUM_RS, rsts.size());
250     rsts.get(0).getRegionServer().abort("testing");
251     rsts.get(1).getRegionServer().abort("testing");
252     rsts.get(2).getRegionServer().abort("testing");
253 
254     long start = EnvironmentEdgeManager.currentTimeMillis();
255     while (cluster.getLiveRegionServerThreads().size() > (NUM_RS - 3)) {
256       if (EnvironmentEdgeManager.currentTimeMillis() - start > 60000) {
257         assertTrue(false);
258       }
259       Thread.sleep(200);
260     }
261 
262     start = EnvironmentEdgeManager.currentTimeMillis();
263     while (getAllOnlineRegions(cluster).size() < (NUM_REGIONS_TO_CREATE + 2)) {
264       if (EnvironmentEdgeManager.currentTimeMillis() - start > 60000) {
265         assertTrue("Timedout", false);
266       }
267       Thread.sleep(200);
268     }
269 
270     assertEquals(NUM_REGIONS_TO_CREATE * NUM_ROWS_PER_REGION,
271         TEST_UTIL.countRows(ht));
272     ht.close();
273   }

277   @Test(timeout=30000)
278   public void testDelayedDeleteOnFailure() throws Exception {
279     LOG.info("testDelayedDeleteOnFailure");
280     startCluster(1);
281     final SplitLogManager slm = master.getMasterFileSystem().splitLogManager;
282     final FileSystem fs = master.getMasterFileSystem().getFileSystem();
283     final Path logDir = new Path(FSUtils.getRootDir(conf), "x");
284     fs.mkdirs(logDir);
285     ExecutorService executor = null;
286     try {
287       final Path corruptedLogFile = new Path(logDir, "x");
288       FSDataOutputStream out;
289       out = fs.create(corruptedLogFile);
290       out.write(0);
291       out.write(Bytes.toBytes("corrupted bytes"));
292       out.close();
293       slm.ignoreZKDeleteForTesting = true;
294       executor = Executors.newSingleThreadExecutor();
295       Runnable runnable = new Runnable() {
296        @Override
297        public void run() {
298           try {
299             // since the logDir is a fake, corrupted one, so the split log worker
300             // will finish it quickly with error, and this call will fail and throw
301             // an IOException.
302             slm.splitLogDistributed(logDir);
303           } catch (IOException ioe) {
304             try {
305               assertTrue(fs.exists(corruptedLogFile));
306               // this call will block waiting for the task to be removed from the
307               // tasks map which is not going to happen since ignoreZKDeleteForTesting
308               // is set to true, until it is interrupted.
309               slm.splitLogDistributed(logDir);
310             } catch (IOException e) {
311               assertTrue(Thread.currentThread().isInterrupted());
312               return;
313             }
314             fail("did not get the expected IOException from the 2nd call");
315           }
316           fail("did not get the expected IOException from the 1st call");
317         }
318       };
319       Future<?> result = executor.submit(runnable);
320       try {
321         result.get(2000, TimeUnit.MILLISECONDS);
322       } catch (TimeoutException te) {
323         // it is ok, expected.
324       }
325       waitForCounter(tot_mgr_wait_for_zk_delete, 0, 1, 10000);
326       executor.shutdownNow();
327       executor = null;
328 
329       // make sure the runnable is finished with no exception thrown.
330       result.get();
331     } finally {
332       if (executor != null) {
333         // interrupt the thread in case the test fails in the middle.
334         // it has no effect if the thread is already terminated.
335         executor.shutdownNow();
336       }
337       fs.delete(logDir, true);
338     }
339   }

341   HTable installTable(ZooKeeperWatcher zkw, String tname, String fname,
342       int nrs ) throws Exception {
343     // Create a table with regions
344     byte [] table = Bytes.toBytes(tname);
345     byte [] family = Bytes.toBytes(fname);
346     LOG.info("Creating table with " + nrs + " regions");
347     HTable ht = TEST_UTIL.createTable(table, family);
348     int numRegions = TEST_UTIL.createMultiRegions(conf, ht, family, nrs);
349     assertEquals(nrs, numRegions);
350       LOG.info("Waiting for no more RIT\n");
351     blockUntilNoRIT(zkw, master);
352     // disable-enable cycle to get rid of table's dead regions left behind
353     // by createMultiRegions
354     LOG.debug("Disabling table\n");
355     TEST_UTIL.getHBaseAdmin().disableTable(table);
356     LOG.debug("Waiting for no more RIT\n");
357     blockUntilNoRIT(zkw, master);
358     NavigableSet<String> regions = getAllOnlineRegions(cluster);
359     LOG.debug("Verifying only catalog regions are assigned\n");
360     if (regions.size() != 2) {
361       for (String oregion : regions)
362         LOG.debug("Region still online: " + oregion);
363     }
364     assertEquals(2, regions.size());
365     LOG.debug("Enabling table\n");
366     TEST_UTIL.getHBaseAdmin().enableTable(table);
367     LOG.debug("Waiting for no more RIT\n");
368     blockUntilNoRIT(zkw, master);
369     LOG.debug("Verifying there are " + numRegions + " assigned on cluster\n");
370     regions = getAllOnlineRegions(cluster);
371     assertEquals(numRegions + 2, regions.size());
372     return ht;
373   }

375   void populateDataInTable(int nrows, String fname) throws Exception {
376     byte [] family = Bytes.toBytes(fname);
377 
378     List<RegionServerThread> rsts = cluster.getLiveRegionServerThreads();
379     assertEquals(NUM_RS, rsts.size());
380 
381     for (RegionServerThread rst : rsts) {
382       HRegionServer hrs = rst.getRegionServer();
383       List<HRegionInfo> hris = ProtobufUtil.getOnlineRegions(hrs);
384       for (HRegionInfo hri : hris) {
385         if (hri.isMetaTable()) {
386           continue;
387         }
388         LOG.debug("adding data to rs = " + rst.getName() +
389             " region = "+ hri.getRegionNameAsString());
390         HRegion region = hrs.getOnlineRegion(hri.getRegionName());
391         assertTrue(region != null);
392         putData(region, hri.getStartKey(), nrows, Bytes.toBytes("q"), family);
393       }
394     }
395   }

397   public void makeHLog(HLog log,
398       List<HRegionInfo> hris, String tname,
399       int num_edits, int edit_size) throws IOException {
400 
401     // remove root and meta region
402     hris.remove(HRegionInfo.ROOT_REGIONINFO);
403     hris.remove(HRegionInfo.FIRST_META_REGIONINFO);
404     byte[] table = Bytes.toBytes(tname);
405     HTableDescriptor htd = new HTableDescriptor(tname);
406     byte[] value = new byte[edit_size];
407     for (int i = 0; i < edit_size; i++) {
408       value[i] = (byte)('a' + (i % 26));
409     }
410     int n = hris.size();
411     int[] counts = new int[n];
412     int j = 0;
413     if (n > 0) {
414       for (int i = 0; i < num_edits; i += 1) {
415         WALEdit e = new WALEdit();
416         byte [] row = Bytes.toBytes("r" + Integer.toString(i));
417         byte [] family = Bytes.toBytes("f");
418         byte [] qualifier = Bytes.toBytes("c" + Integer.toString(i));
419         e.add(new KeyValue(row, family, qualifier,
420             System.currentTimeMillis(), value));
421         j++;
422         log.append(hris.get(j % n), table, e, System.currentTimeMillis(), htd);
423         counts[j % n] += 1;
424       }
425     }
426     log.sync();
427     log.close();
428     for (int i = 0; i < n; i++) {
429       LOG.info("region " + hris.get(i).getRegionNameAsString() +
430           " has " + counts[i] + " edits");
431     }
432     return;
433   }

451   private void putData(HRegion region, byte[] startRow, int numRows, byte [] qf,
452       byte [] ...families)
453   throws IOException {
454     for(int i = 0; i < numRows; i++) {
455       Put put = new Put(Bytes.add(startRow, Bytes.toBytes(i)));

463   private NavigableSet<String> getAllOnlineRegions(MiniHBaseCluster cluster)

474   private void waitForCounter(AtomicLong ctr, long oldval, long newval,
475       long timems) {
476     long curt = System.currentTimeMillis();
477     long endt = curt + timems;
478     while (curt < endt) {
479       if (ctr.get() == oldval) {
480         Thread.yield();
481         curt = System.currentTimeMillis();
482       } else {
483         assertEquals(newval, ctr.get());
484         return;
485       }
486     }
487     assertTrue(false);
488   }
