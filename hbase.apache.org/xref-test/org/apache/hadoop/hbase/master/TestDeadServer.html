18  package org.apache.hadoop.hbase.master;

20  import static org.junit.Assert.assertFalse;
21  import static org.junit.Assert.assertTrue;
22  
23  import org.apache.hadoop.hbase.*;
24  import org.junit.Test;
25  import org.junit.experimental.categories.Category;
26  
27  @Category(SmallTests.class)
28  public class TestDeadServer {
29    @Test public void testIsDead() {
30      DeadServer ds = new DeadServer();
31      final ServerName hostname123 = new ServerName("127.0.0.1", 123, 3L);
32      ds.add(hostname123);
33      assertTrue(ds.areDeadServersInProgress());
34      ds.finish(hostname123);
35      assertFalse(ds.areDeadServersInProgress());
36      final ServerName hostname1234 = new ServerName("127.0.0.2", 1234, 4L);
37      ds.add(hostname1234);
38      assertTrue(ds.areDeadServersInProgress());
39      ds.finish(hostname1234);
40      assertFalse(ds.areDeadServersInProgress());
41      final ServerName hostname12345 = new ServerName("127.0.0.2", 12345, 4L);
42      ds.add(hostname12345);
43      assertTrue(ds.areDeadServersInProgress());
44      ds.finish(hostname12345);
45      assertFalse(ds.areDeadServersInProgress());
46  
47      // Already dead =       127.0.0.1,9090,112321
48      // Coming back alive =  127.0.0.1,9090,223341
49  
50      final ServerName deadServer = new ServerName("127.0.0.1", 9090, 112321L);
51      assertFalse(ds.cleanPreviousInstance(deadServer));
52      ds.add(deadServer);
53      assertTrue(ds.isDeadServer(deadServer));
54      final ServerName deadServerHostComingAlive =
55        new ServerName("127.0.0.1", 9090, 112321L);
56      assertTrue(ds.cleanPreviousInstance(deadServerHostComingAlive));
57      assertFalse(ds.isDeadServer(deadServer));
58      assertFalse(ds.cleanPreviousInstance(deadServerHostComingAlive));
59    }

61  }
