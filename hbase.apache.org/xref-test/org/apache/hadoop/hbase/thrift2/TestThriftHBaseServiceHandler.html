89    public void assertTColumnValuesEqual(List<TColumnValue> columnValuesA, List<TColumnValue> columnValuesB) {
90      assertEquals(columnValuesA.size(), columnValuesB.size());
91      Comparator<TColumnValue> comparator = new Comparator<TColumnValue>() {
92        @Override
93        public int compare(TColumnValue o1, TColumnValue o2) {
94          return Bytes.compareTo(Bytes.add(o1.getFamily(), o1.getQualifier()),
95              Bytes.add(o2.getFamily(), o2.getQualifier()));
96        }
97      };
98      Collections.sort(columnValuesA, comparator);
99      Collections.sort(columnValuesB, comparator);
100 
101     for (int i = 0; i < columnValuesA.size(); i++) {
102       TColumnValue a = columnValuesA.get(i);
103       TColumnValue b = columnValuesB.get(i);
104       assertArrayEquals(a.getFamily(), b.getFamily());
105       assertArrayEquals(a.getQualifier(), b.getQualifier());
106       assertArrayEquals(a.getValue(), b.getValue());
107     }
108   }
