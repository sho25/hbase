53   * Test the draining servers feature.
54   * @see <a href="https://issues.apache.org/jira/browse/HBASE-4298">HBASE-4298</a>
55   */
56  @Category(MediumTests.class)
57  public class TestDrainingServer {
58    private static final Log LOG = LogFactory.getLog(TestDrainingServer.class);
59    private static final HBaseTestingUtility TEST_UTIL =
60      new HBaseTestingUtility();
61    private static final byte [] TABLENAME = Bytes.toBytes("t");
62    private static final byte [] FAMILY = Bytes.toBytes("f");
63    private static final int COUNT_OF_REGIONS = HBaseTestingUtility.KEYS.length;
64    private static final int NB_SLAVES = 5;
65  
66    /**
67     * Spin up a cluster with a bunch of regions on it.
68     */
69    @BeforeClass
70    public static void setUpBeforeClass() throws Exception {
71      TEST_UTIL.startMiniCluster(NB_SLAVES);
72      TEST_UTIL.getConfiguration().setBoolean("hbase.master.enabletable.roundrobin", true);
73      ZooKeeperWatcher zkw = HBaseTestingUtility.getZooKeeperWatcher(TEST_UTIL);
74      HTableDescriptor htd = new HTableDescriptor(TABLENAME);
75      htd.addFamily(new HColumnDescriptor(FAMILY));
76      TEST_UTIL.createMultiRegionsInMeta(TEST_UTIL.getConfiguration(), htd,
77          HBaseTestingUtility.KEYS);
78      // Make a mark for the table in the filesystem.
79      FileSystem fs = FileSystem.get(TEST_UTIL.getConfiguration());
80      FSTableDescriptors.
81        createTableDescriptor(fs, FSUtils.getRootDir(TEST_UTIL.getConfiguration()), htd);
82      // Assign out the regions we just created.
83      HBaseAdmin admin = new HBaseAdmin(TEST_UTIL.getConfiguration());
84      MiniHBaseCluster cluster = TEST_UTIL.getMiniHBaseCluster();
85      admin.disableTable(TABLENAME);
86      admin.enableTable(TABLENAME);
87  
88      boolean ready = false;
89      while (!ready){
90        ZKAssign.blockUntilNoRIT(zkw);
91  
92        // Assert that every regionserver has some regions on it.
93        int i = 0;
94        ready = true;
95        while (i < NB_SLAVES && ready){
96          HRegionServer hrs = cluster.getRegionServer(i);
97          if (ProtobufUtil.getOnlineRegions(hrs).isEmpty()){
98            ready = false;
99          }
100         i++;
101       }
102 
103       if (!ready){
104         admin.balancer();
105         Thread.sleep(100);
106       }
107     }
108   }

110   private static HRegionServer setDrainingServer(final HRegionServer hrs)
111   throws KeeperException {
112     LOG.info("Making " + hrs.getServerName() + " the draining server; " +
113       "it has " + hrs.getNumberOfOnlineRegions() + " online regions");
114     ZooKeeperWatcher zkw = hrs.getZooKeeper();
115     String hrsDrainingZnode =
116       ZKUtil.joinZNode(zkw.drainingZNode, hrs.getServerName().toString());
117     ZKUtil.createWithParents(zkw, hrsDrainingZnode);
118     return hrs;
119   }
120 
121   private static HRegionServer unsetDrainingServer(final HRegionServer hrs)
122   throws KeeperException {
123     ZooKeeperWatcher zkw = hrs.getZooKeeper();
124     String hrsDrainingZnode =
125       ZKUtil.joinZNode(zkw.drainingZNode, hrs.getServerName().toString());
126     ZKUtil.deleteNode(zkw, hrsDrainingZnode);
127     return hrs;
128   }

135   /**
136    * Test adding server to draining servers and then move regions off it.
137    * Make sure that no regions are moved back to the draining server.
138    * @throws IOException 
139    * @throws KeeperException 
140    */
141   @Test  // (timeout=30000)
142   public void testDrainingServerOffloading()
143   throws IOException, KeeperException, ServiceException, DeserializationException {
144     // I need master in the below.
145     HMaster master = TEST_UTIL.getMiniHBaseCluster().getMaster();
146     HRegionInfo hriToMoveBack = null;
147     // Set first server as draining server.
148     HRegionServer drainingServer =
149       setDrainingServer(TEST_UTIL.getMiniHBaseCluster().getRegionServer(0));
150     try {
151       final int regionsOnDrainingServer =
152         drainingServer.getNumberOfOnlineRegions();
153       Assert.assertTrue(regionsOnDrainingServer > 0);
154       List<HRegionInfo> hris = ProtobufUtil.getOnlineRegions(drainingServer);
155       for (HRegionInfo hri : hris) {
156         // Pass null and AssignmentManager will chose a random server BUT it
157         // should exclude draining servers.
158         master.moveRegion(null,
159           RequestConverter.buildMoveRegionRequest(hri.getEncodedNameAsBytes(), null));
160         // Save off region to move back.
161         hriToMoveBack = hri;
162       }
163       // Wait for regions to come back on line again.
164       waitForAllRegionsOnline();
165       Assert.assertEquals(0, drainingServer.getNumberOfOnlineRegions());
166     } finally {
167       unsetDrainingServer(drainingServer);
168     }
169     // Now we've unset the draining server, we should be able to move a region
170     // to what was the draining server.
171     master.moveRegion(null,
172       RequestConverter.buildMoveRegionRequest(hriToMoveBack.getEncodedNameAsBytes(),
173       Bytes.toBytes(drainingServer.getServerName().toString())));
174     // Wait for regions to come back on line again.
175     waitForAllRegionsOnline();
176     Assert.assertEquals(1, drainingServer.getNumberOfOnlineRegions());
177   }

179   /**
180    * Test that draining servers are ignored even after killing regionserver(s).
181    * Verify that the draining server is not given any of the dead servers regions.
182    * @throws KeeperException
183    * @throws IOException
184    */
185   @Test  (timeout=30000)
186   public void testDrainingServerWithAbort() throws KeeperException, Exception {
187     HMaster master = TEST_UTIL.getHBaseCluster().getMaster();
188 
189     // Ensure a stable env
190     TEST_UTIL.getHBaseAdmin().setBalancerRunning(false, false);
191     waitForAllRegionsOnline();
192 
193     final long regionCount = TEST_UTIL.getMiniHBaseCluster().countServedRegions();
194 
195     // Let's get a copy of the regions today.
196     Collection<HRegion> regions = new ArrayList<HRegion>();
197     for (int i = 0; i < NB_SLAVES; i++) {
198       HRegionServer hrs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(i);
199       regions.addAll( hrs.getCopyOfOnlineRegionsSortedBySize().values() );
200     }
201 
202       // Choose the draining server
203     HRegionServer drainingServer = TEST_UTIL.getMiniHBaseCluster().getRegionServer(0);
204     final int regionsOnDrainingServer = drainingServer.getNumberOfOnlineRegions();
205     Assert.assertTrue(regionsOnDrainingServer > 0);
206 
207     ServerManager sm = master.getServerManager();
208 
209     Collection<HRegion> regionsBefore = drainingServer.
210       getCopyOfOnlineRegionsSortedBySize().values();
211     LOG.info("Regions of drained server are: "+ regionsBefore );
212 
213     try {
214       // Add first server to draining servers up in zk.
215       setDrainingServer(drainingServer);
216 
217       //wait for the master to receive and manage the event
218       while  (sm.createDestinationServersList().contains(drainingServer.getServerName())) ;
219 
220       LOG.info("The available servers are: "+ sm.createDestinationServersList());
221 
222       Assert.assertEquals("Nothing should have happened here.", regionsOnDrainingServer,
223         drainingServer.getNumberOfOnlineRegions());
224       Assert.assertFalse("We should not have regions in transition here. List is: "+
225         master.getAssignmentManager().getRegionStates().getRegionsInTransition(),
226         master.getAssignmentManager().getRegionStates().isRegionsInTransition());
227 
228       // Kill a few regionservers.
229       for (int aborted = 0; aborted <= 2; aborted++) {
230         HRegionServer hrs = TEST_UTIL.getMiniHBaseCluster().getRegionServer(aborted+1);
231         hrs.abort("Aborting");
232       }
233 
234       // Wait for regions to come back online again.
235       waitForAllRegionsOnline();
236 
237       Collection<HRegion> regionsAfter =
238         drainingServer.getCopyOfOnlineRegionsSortedBySize().values();
239       LOG.info("Regions of drained server are: "+ regionsAfter );
240 
241       Assert.assertEquals("Test conditions are not met: regions were" +
242         " created/deleted during the test. ",
243         regionCount, TEST_UTIL.getMiniHBaseCluster().countServedRegions());
244 
245       // Assert the draining server still has the same regions.
246       StringBuilder result = new StringBuilder();
247       for (HRegion r: regionsAfter){
248         if (!regionsBefore.contains(r)){
249           result.append(r).append(" was added after the drain");
250           if (regions.contains(r)){
251             result.append("(existing region");
252           } else {
253             result.append("(new region)");
254           }
255           result.append("; ");
256         }
257       }
258       for (HRegion r: regionsBefore){
259         if (!regionsAfter.contains(r)){
260           result.append(r).append(" was removed after the drain; ");
261         }
262       }
263       Assert.assertTrue("Errors are: "+ result.toString(), result.length()==0);
264 
265     } finally {
266       unsetDrainingServer(drainingServer);
267     }
268   }

270   private void waitForAllRegionsOnline() {
271     // Wait for regions to come back on line again.
272     while (!isAllRegionsOnline()) {
273     }
274 
275     while (TEST_UTIL.getMiniHBaseCluster().getMaster().
276       getAssignmentManager().getRegionStates().isRegionsInTransition()) {
277     }
278   }
279 
280   private boolean isAllRegionsOnline() {
281     return TEST_UTIL.getMiniHBaseCluster().countServedRegions() ==
282       (COUNT_OF_REGIONS + 2 /*catalog regions*/);
283   }
