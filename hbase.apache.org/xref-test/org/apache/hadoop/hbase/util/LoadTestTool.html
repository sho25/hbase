137   private String[] splitColonSeparated(String option,
138       int minNumCols, int maxNumCols) {
139     String optVal = cmd.getOptionValue(option);
140     String[] cols = optVal.split(":");
141     if (cols.length < minNumCols || cols.length > maxNumCols) {
142       throw new IllegalArgumentException("Expected at least "
143           + minNumCols + " columns but no more than " + maxNumCols +
144           " in the colon-separated value '" + optVal + "' of the " +
145           "-" + option + " option");
146     }
147     return cols;
148   }

154   /**
155    * Apply column family options such as Bloom filters, compression, and data
156    * block encoding.
157    */
158   private void applyColumnFamilyOptions(byte[] tableName,
159       byte[][] columnFamilies) throws IOException {
160     HBaseAdmin admin = new HBaseAdmin(conf);
161     HTableDescriptor tableDesc = admin.getTableDescriptor(tableName);
162     LOG.info("Disabling table " + Bytes.toString(tableName));
163     admin.disableTable(tableName);
164     for (byte[] cf : columnFamilies) {
165       HColumnDescriptor columnDesc = tableDesc.getFamily(cf);
166       boolean isNewCf = columnDesc == null;
167       if (isNewCf) {
168         columnDesc = new HColumnDescriptor(cf);
169       }
170       if (bloomType != null) {
171         columnDesc.setBloomFilterType(bloomType);
172       }
173       if (compressAlgo != null) {
174         columnDesc.setCompressionType(compressAlgo);
175       }
176       if (dataBlockEncodingAlgo != null) {
177         columnDesc.setDataBlockEncoding(dataBlockEncodingAlgo);
178         columnDesc.setEncodeOnDisk(!encodeInCacheOnly);
179       }
180       if (isNewCf) {
181         admin.addColumn(tableName, columnDesc);
182       } else {
183         admin.modifyColumn(tableName, columnDesc);
184       }
185     }
186     LOG.info("Enabling table " + Bytes.toString(tableName));
187     admin.enableTable(tableName);
188   }

220   @Override
221   protected void processOptions(CommandLine cmd) {
222     this.cmd = cmd;
223 
224     tableName = Bytes.toBytes(cmd.getOptionValue(OPT_TABLE_NAME,
225         DEFAULT_TABLE_NAME));
226 
227     isWrite = cmd.hasOption(OPT_WRITE);
228     isRead = cmd.hasOption(OPT_READ);
229     isInitOnly = cmd.hasOption(OPT_INIT_ONLY);
230 
231     if (!isWrite && !isRead && !isInitOnly) {
232       throw new IllegalArgumentException("Either -" + OPT_WRITE + " or " +
233           "-" + OPT_READ + " has to be specified");
234     }
235 
236     if (isInitOnly && (isRead || isWrite)) {
237       throw new IllegalArgumentException(OPT_INIT_ONLY + " cannot be specified with"
238           + " either -" + OPT_WRITE + " or -" + OPT_READ);
239     }
240 
241     if (!isInitOnly) {
242       if (!cmd.hasOption(OPT_NUM_KEYS)) {
243         throw new IllegalArgumentException(OPT_NUM_KEYS + " must be specified in "
244             + "read or write mode");
245       }
246       startKey = parseLong(cmd.getOptionValue(OPT_START_KEY,
247           String.valueOf(DEFAULT_START_KEY)), 0, Long.MAX_VALUE);
248       long numKeys = parseLong(cmd.getOptionValue(OPT_NUM_KEYS), 1,
249           Long.MAX_VALUE - startKey);
250       endKey = startKey + numKeys;
251       isSkipInit = cmd.hasOption(OPT_SKIP_INIT);
252       System.out.println("Key range: [" + startKey + ".." + (endKey - 1) + "]");
253     }
254 
255     encodeInCacheOnly = cmd.hasOption(OPT_ENCODE_IN_CACHE_ONLY);
256     parseColumnFamilyOptions(cmd);
257 
258     if (isWrite) {
259       String[] writeOpts = splitColonSeparated(OPT_WRITE, 2, 3);
260 
261       int colIndex = 0;
262       minColsPerKey = 1;
263       maxColsPerKey = 2 * Long.parseLong(writeOpts[colIndex++]);
264       int avgColDataSize =
265           parseInt(writeOpts[colIndex++], 1, Integer.MAX_VALUE);
266       minColDataSize = avgColDataSize / 2;
267       maxColDataSize = avgColDataSize * 3 / 2;
268 
269       if (colIndex < writeOpts.length) {
270         numWriterThreads = getNumThreads(writeOpts[colIndex++]);
271       }
272 
273       isMultiPut = cmd.hasOption(OPT_MULTIPUT);
274 
275       System.out.println("Multi-puts: " + isMultiPut);
276       System.out.println("Columns per key: " + minColsPerKey + ".."
277           + maxColsPerKey);
278       System.out.println("Data size per column: " + minColDataSize + ".."
279           + maxColDataSize);
280     }
281 
282     if (isRead) {
283       String[] readOpts = splitColonSeparated(OPT_READ, 1, 2);
284       int colIndex = 0;
285       verifyPercent = parseInt(readOpts[colIndex++], 0, 100);
286       if (colIndex < readOpts.length) {
287         numReaderThreads = getNumThreads(readOpts[colIndex++]);
288       }
289 
290       if (cmd.hasOption(OPT_MAX_READ_ERRORS)) {
291         maxReadErrors = parseInt(cmd.getOptionValue(OPT_MAX_READ_ERRORS),
292             0, Integer.MAX_VALUE);
293       }
294 
295       if (cmd.hasOption(OPT_KEY_WINDOW)) {
296         keyWindow = parseInt(cmd.getOptionValue(OPT_KEY_WINDOW),
297             0, Integer.MAX_VALUE);
298       }
299 
300       System.out.println("Percent of keys to verify: " + verifyPercent);
301       System.out.println("Reader threads: " + numReaderThreads);
302     }
303   }

305   private void parseColumnFamilyOptions(CommandLine cmd) {
306     String dataBlockEncodingStr = cmd.getOptionValue(OPT_DATA_BLOCK_ENCODING);
307     dataBlockEncodingAlgo = dataBlockEncodingStr == null ? null :
308         DataBlockEncoding.valueOf(dataBlockEncodingStr);
309     if (dataBlockEncodingAlgo == DataBlockEncoding.NONE && encodeInCacheOnly) {
310       throw new IllegalArgumentException("-" + OPT_ENCODE_IN_CACHE_ONLY + " " +
311           "does not make sense when data block encoding is not used");
312     }
313 
314     String compressStr = cmd.getOptionValue(OPT_COMPRESSION);
315     compressAlgo = compressStr == null ? Compression.Algorithm.NONE :
316         Compression.Algorithm.valueOf(compressStr);
317 
318     String bloomStr = cmd.getOptionValue(OPT_BLOOM);
319     bloomType = bloomStr == null ? null :
320         StoreFile.BloomType.valueOf(bloomStr);
321   }

324     HBaseTestingUtility.createPreSplitLoadTestTable(conf, tableName,
325         COLUMN_FAMILY, compressAlgo, dataBlockEncodingAlgo);
326     applyColumnFamilyOptions(tableName, COLUMN_FAMILIES);

329   @Override
330   protected int doWork() throws IOException {
331     if (cmd.hasOption(OPT_ZK_QUORUM)) {
332       conf.set(HConstants.ZOOKEEPER_QUORUM, cmd.getOptionValue(OPT_ZK_QUORUM));
333     }
334 
335     if (isInitOnly) {
336       LOG.info("Initializing only; no reads or writes");
337       initTestTable();
338       return 0;
339     }
340 
341     if (!isSkipInit) {
342       initTestTable();
343     }
344 
345     if (isWrite) {
346       writerThreads = new MultiThreadedWriter(conf, tableName, COLUMN_FAMILY);
347       writerThreads.setMultiPut(isMultiPut);
348       writerThreads.setColumnsPerKey(minColsPerKey, maxColsPerKey);
349       writerThreads.setDataSize(minColDataSize, maxColDataSize);
350     }
351 
352     if (isRead) {
353       readerThreads = new MultiThreadedReader(conf, tableName, COLUMN_FAMILY,
354           verifyPercent);
355       readerThreads.setMaxErrors(maxReadErrors);
356       readerThreads.setKeyWindow(keyWindow);
357     }
358 
359     if (isRead && isWrite) {
360       LOG.info("Concurrent read/write workload: making readers aware of the " +
361           "write point");
362       readerThreads.linkToWriter(writerThreads);
363     }
364 
365     if (isWrite) {
366       System.out.println("Starting to write data...");
367       writerThreads.start(startKey, endKey, numWriterThreads);
368     }
369 
370     if (isRead) {
371       System.out.println("Starting to read data...");
372       readerThreads.start(startKey, endKey, numReaderThreads);
373     }
374 
375     if (isWrite) {
376       writerThreads.waitForFinish();
377     }
378 
379     if (isRead) {
380       readerThreads.waitForFinish();
381     }
382 
383     boolean success = true;
384     if (isWrite) {
385       success = success && writerThreads.getNumWriteFailures() == 0;
386     }
387     if (isRead) {
388       success = success && readerThreads.getNumReadErrors() == 0
389           && readerThreads.getNumReadFailures() == 0;
390     }
391     return success ? 0 : 1;
392   }
