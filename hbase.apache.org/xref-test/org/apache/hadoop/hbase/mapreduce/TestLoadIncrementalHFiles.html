127   private void runTest(String testName, BloomType bloomType, 
128           byte[][][] hfileRanges) throws Exception {
129     Path dir = util.getDataTestDirOnTestFS(testName);
130     FileSystem fs = util.getTestFileSystem();
131     dir = dir.makeQualified(fs);
132     Path familyDir = new Path(dir, Bytes.toString(FAMILY));
133 
134     int hfileIdx = 0;
135     for (byte[][] range : hfileRanges) {
136       byte[] from = range[0];
137       byte[] to = range[1];
138       createHFile(util.getConfiguration(), fs, new Path(familyDir, "hfile_"
139           + hfileIdx++), FAMILY, QUALIFIER, from, to, 1000);
140     }
141     int expectedRows = hfileIdx * 1000;
142 
143     final byte[] TABLE = Bytes.toBytes("mytable_"+testName);
144 
145     HBaseAdmin admin = new HBaseAdmin(util.getConfiguration());
146     HTableDescriptor htd = new HTableDescriptor(TABLE);
147     HColumnDescriptor familyDesc = new HColumnDescriptor(FAMILY);
148     familyDesc.setBloomFilterType(bloomType);
149     htd.addFamily(familyDesc);
150     admin.createTable(htd, SPLIT_KEYS);
151 
152     HTable table = new HTable(util.getConfiguration(), TABLE);
153     util.waitTableAvailable(TABLE, 30000);
154     LoadIncrementalHFiles loader = new LoadIncrementalHFiles(
155       util.getConfiguration());
156     loader.doBulkLoad(dir, table);
157 
158     assertEquals(expectedRows, util.countRows(table));
159   }

161   private void verifyAssignedSequenceNumber(String testName,

163     Path dir = util.getDataTestDir(testName);
164     FileSystem fs = util.getTestFileSystem();
165     dir = dir.makeQualified(fs);

168     int hfileIdx = 0;
169     for (byte[][] range : hfileRanges) {
170       byte[] from = range[0];
171       byte[] to = range[1];
172       createHFile(util.getConfiguration(), fs, new Path(familyDir, "hfile_"
173           + hfileIdx++), FAMILY, QUALIFIER, from, to, 1000);
174     }
175 
176     final byte[] TABLE = Bytes.toBytes("mytable_"+testName);
177 
178     HBaseAdmin admin = new HBaseAdmin(util.getConfiguration());
179     HTableDescriptor htd = new HTableDescriptor(TABLE);
180     HColumnDescriptor familyDesc = new HColumnDescriptor(FAMILY);
181     htd.addFamily(familyDesc);
182     admin.createTable(htd, SPLIT_KEYS);
183 
184     HTable table = new HTable(util.getConfiguration(), TABLE);
185     util.waitTableAvailable(TABLE, 30000);
186     LoadIncrementalHFiles loader = new LoadIncrementalHFiles(
187       util.getConfiguration());
188 
189     // Do a dummy put to increase the hlog sequence number
190     Put put = new Put(Bytes.toBytes("row"));
191     put.add(FAMILY, QUALIFIER, Bytes.toBytes("value"));
192     table.put(put);
193 
194     loader.doBulkLoad(dir, table);
195 
196     // Get the store files
197     List<StoreFile> files = util.getHBaseCluster().
198         getRegions(TABLE).get(0).getStore(FAMILY).getStorefiles();
199     for (StoreFile file: files) {
200       // the sequenceId gets initialized during createReader
201       file.createReader();
202 
203       if (nonZero)
204         assertTrue(file.getMaxSequenceId() > 0);
205       else
206         assertTrue(file.getMaxSequenceId() == -1);
207     }
208   }

233   private int verifyHFile(Path p) throws IOException {
234     Configuration conf = util.getConfiguration();
235     HFile.Reader reader = HFile.createReader(
236         p.getFileSystem(conf), p, new CacheConfig(conf));
237     reader.loadFileInfo();
238     HFileScanner scanner = reader.getScanner(false, false);
239     scanner.seekTo();
240     int count = 0;
241     do {
242       count++;
243     } while (scanner.next());
244     assertTrue(count > 0);
245     reader.close();
246     return count;
247   }

250   /**
251    * Create an HFile with the given number of rows between a given
252    * start key and end key.
253    * TODO put me in an HFileTestUtil or something?
254    */
255   static void createHFile(
256       Configuration conf,
257       FileSystem fs, Path path,
258       byte[] family, byte[] qualifier,
259       byte[] startKey, byte[] endKey, int numRows) throws IOException
260   {
261     HFile.Writer writer = HFile.getWriterFactory(conf, new CacheConfig(conf))
262         .withPath(fs, path)
263         .withBlockSize(BLOCKSIZE)
264         .withCompression(COMPRESSION)
265         .withComparator(KeyValue.KEY_COMPARATOR)
266         .create();
267     long now = System.currentTimeMillis();
268     try {
269       // subtract 2 since iterateOnSplits doesn't include boundary keys
270       for (byte[] key : Bytes.iterateOnSplits(startKey, endKey, numRows-2)) {
271         KeyValue kv = new KeyValue(key, family, qualifier, now, key);
272         writer.append(kv);
273       }
274     } finally {
275       writer.appendFileInfo(StoreFile.BULKLOAD_TIME_KEY,
276           Bytes.toBytes(System.currentTimeMillis()));
277       writer.close();
278     }
279   }

281   private void addStartEndKeysForTest(TreeMap<byte[], Integer> map, byte[] first, byte[] last) {
282     Integer value = map.containsKey(first)?(Integer)map.get(first):0;
283     map.put(first, value+1);

289   @Test 
290   public void testInferBoundaries() {
291     TreeMap<byte[], Integer> map = new TreeMap<byte[], Integer>(Bytes.BYTES_COMPARATOR);
292 
293     /* Toy example
294      *     c---------i            o------p          s---------t     v------x
295      * a------e    g-----k   m-------------q   r----s            u----w
296      *
297      * Should be inferred as:
298      * a-----------------k   m-------------q   r--------------t  u---------x
299      * 
300      * The output should be (m,r,u) 
301      */
302 
303     String first;
304     String last;
305 
306     first = "a"; last = "e";
307     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
308     
309     first = "r"; last = "s";
310     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
311 
312     first = "o"; last = "p";
313     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
314 
315     first = "g"; last = "k";
316     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
317 
318     first = "v"; last = "x";
319     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
320 
321     first = "c"; last = "i";
322     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
323 
324     first = "m"; last = "q";
325     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
326 
327     first = "s"; last = "t";
328     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
329     
330     first = "u"; last = "w";
331     addStartEndKeysForTest(map, first.getBytes(), last.getBytes());
332 
333     byte[][] keysArray = LoadIncrementalHFiles.inferBoundaries(map);
334     byte[][] compare = new byte[3][];
335     compare[0] = "m".getBytes();
336     compare[1] = "r".getBytes(); 
337     compare[2] = "u".getBytes();
338 
339     assertEquals(keysArray.length, 3);
340 
341     for (int row = 0; row<keysArray.length; row++){
342       assertArrayEquals(keysArray[row], compare[row]);
343     }
344   }
