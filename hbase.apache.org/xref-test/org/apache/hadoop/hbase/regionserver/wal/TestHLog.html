214   @Test
215   public void Broken_testSync() throws Exception {
216     byte [] bytes = Bytes.toBytes(getName());
217     // First verify that using streams all works.
218     Path p = new Path(dir, getName() + ".fsdos");
219     FSDataOutputStream out = fs.create(p);
220     out.write(bytes);
221     Method syncMethod = null;
222     try {
223       syncMethod = out.getClass().getMethod("hflush", new Class<?> []{});
224     } catch (NoSuchMethodException e) {
225       try {
226         syncMethod = out.getClass().getMethod("sync", new Class<?> []{});
227       } catch (NoSuchMethodException ex) {
228         fail("This version of Hadoop supports neither Syncable.sync() " +
229             "nor Syncable.hflush().");
230       }
231     }
232     syncMethod.invoke(out, new Object[]{});
233     FSDataInputStream in = fs.open(p);
234     assertTrue(in.available() > 0);
235     byte [] buffer = new byte [1024];
236     int read = in.read(buffer);
237     assertEquals(bytes.length, read);
238     out.close();
239     in.close();
240 
241     HLog wal = HLogFactory.createHLog(fs, dir, "hlogdir", conf);
242     
243     final int total = 20;
244     HLog.Reader reader = null;
245 
246     try {
247       HRegionInfo info = new HRegionInfo(bytes,
248                   null,null, false);
249       HTableDescriptor htd = new HTableDescriptor();
250       htd.addFamily(new HColumnDescriptor(bytes));
251 
252       for (int i = 0; i < total; i++) {
253         WALEdit kvs = new WALEdit();
254         kvs.add(new KeyValue(Bytes.toBytes(i), bytes, bytes));
255         wal.append(info, bytes, kvs, System.currentTimeMillis(), htd);
256       }
257       // Now call sync and try reading.  Opening a Reader before you sync just
258       // gives you EOFE.
259       wal.sync();
260       // Open a Reader.
261       Path walPath = ((FSHLog) wal).computeFilename();
262       reader = HLogFactory.createReader(fs, walPath, conf);
263       int count = 0;
264       HLog.Entry entry = new HLog.Entry();
265       while ((entry = reader.next(entry)) != null) count++;
266       assertEquals(total, count);
267       reader.close();
268       // Add test that checks to see that an open of a Reader works on a file
269       // that has had a sync done on it.
270       for (int i = 0; i < total; i++) {
271         WALEdit kvs = new WALEdit();
272         kvs.add(new KeyValue(Bytes.toBytes(i), bytes, bytes));
273         wal.append(info, bytes, kvs, System.currentTimeMillis(), htd);
274       }
275       reader = HLogFactory.createReader(fs, walPath, conf);
276       count = 0;
277       while((entry = reader.next(entry)) != null) count++;
278       assertTrue(count >= total);
279       reader.close();
280       // If I sync, should see double the edits.
281       wal.sync();
282       reader = HLogFactory.createReader(fs, walPath, conf);
283       count = 0;
284       while((entry = reader.next(entry)) != null) count++;
285       assertEquals(total * 2, count);
286       // Now do a test that ensures stuff works when we go over block boundary,
287       // especially that we return good length on file.
288       final byte [] value = new byte[1025 * 1024];  // Make a 1M value.
289       for (int i = 0; i < total; i++) {
290         WALEdit kvs = new WALEdit();
291         kvs.add(new KeyValue(Bytes.toBytes(i), bytes, value));
292         wal.append(info, bytes, kvs, System.currentTimeMillis(), htd);
293       }
294       // Now I should have written out lots of blocks.  Sync then read.
295       wal.sync();
296       reader = HLogFactory.createReader(fs, walPath, conf);
297       count = 0;
298       while((entry = reader.next(entry)) != null) count++;
299       assertEquals(total * 3, count);
300       reader.close();
301       // Close it and ensure that closed, Reader gets right length also.
302       wal.close();
303       reader = HLogFactory.createReader(fs, walPath, conf);
304       count = 0;
305       while((entry = reader.next(entry)) != null) count++;
306       assertEquals(total * 3, count);
307       reader.close();
308     } finally {
309       if (wal != null) wal.closeAndDelete();
310       if (reader != null) reader.close();
311     }
312   }

342   private void verifySplits(List<Path> splits, final int howmany)
343   throws IOException {
344     assertEquals(howmany, splits.size());
345     for (int i = 0; i < splits.size(); i++) {
346       LOG.info("Verifying=" + splits.get(i));
347       HLog.Reader reader = HLogFactory.createReader(fs, splits.get(i), conf);
348       try {
349         int count = 0;
350         String previousRegion = null;
351         long seqno = -1;
352         HLog.Entry entry = new HLog.Entry();
353         while((entry = reader.next(entry)) != null) {
354           HLogKey key = entry.getKey();
355           String region = Bytes.toString(key.getEncodedRegionName());
356           // Assert that all edits are for same region.
357           if (previousRegion != null) {
358             assertEquals(previousRegion, region);
359           }
360           LOG.info("oldseqno=" + seqno + ", newseqno=" + key.getLogSeqNum());
361           assertTrue(seqno < key.getLogSeqNum());
362           seqno = key.getLogSeqNum();
363           previousRegion = region;
364           count++;
365         }
366         assertEquals(howmany * howmany, count);
367       } finally {
368         reader.close();
369       }
370     }

373   // For this test to pass, requires:
374   // 1. HDFS-200 (append support)
375   // 2. HDFS-988 (SafeMode should freeze file operations
376   //              [FSNamesystem.nextGenerationStampForBlock])
377   // 3. HDFS-142 (on restart, maintain pendingCreates)
378   @Test
379   public void testAppendClose() throws Exception {
380     byte [] tableName = Bytes.toBytes(getName());
381     HRegionInfo regioninfo = new HRegionInfo(tableName,
382              HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, false);
383 
384     HLog wal = HLogFactory.createHLog(fs, dir, "hlogdir", 
385         "hlogdir_archive", conf);
386     final int total = 20;
387 
388     HTableDescriptor htd = new HTableDescriptor();
389     htd.addFamily(new HColumnDescriptor(tableName));
390 
391     for (int i = 0; i < total; i++) {
392       WALEdit kvs = new WALEdit();
393       kvs.add(new KeyValue(Bytes.toBytes(i), tableName, tableName));
394       wal.append(regioninfo, tableName, kvs, System.currentTimeMillis(), htd);
395     }
396     // Now call sync to send the data to HDFS datanodes
397     wal.sync();
398      int namenodePort = cluster.getNameNodePort();
399     final Path walPath = ((FSHLog) wal).computeFilename();
400     
401 
402     // Stop the cluster.  (ensure restart since we're sharing MiniDFSCluster)
403     try {
404       DistributedFileSystem dfs = (DistributedFileSystem) cluster.getFileSystem();
405       dfs.setSafeMode(FSConstants.SafeModeAction.SAFEMODE_ENTER);
406       TEST_UTIL.shutdownMiniDFSCluster();
407       try {
408         // wal.writer.close() will throw an exception,
409         // but still call this since it closes the LogSyncer thread first
410         wal.close();
411       } catch (IOException e) {
412         LOG.info(e);
413       }
414       fs.close(); // closing FS last so DFSOutputStream can't call close
415       LOG.info("STOPPED first instance of the cluster");
416     } finally {
417       // Restart the cluster
418       while (cluster.isClusterUp()){
419         LOG.error("Waiting for cluster to go down");
420         Thread.sleep(1000);
421       }
422       assertFalse(cluster.isClusterUp());
423 
424       // Workaround a strange issue with Hadoop's RPC system - if we don't
425       // sleep here, the new datanodes will pick up a cached IPC connection to
426       // the old (dead) NN and fail to start. Sleeping 2 seconds goes past
427       // the idle time threshold configured in the conf above
428       Thread.sleep(2000);
429 
430       LOG.info("Waiting a few seconds before re-starting HDFS");
431       Thread.sleep(5000);
432       cluster = TEST_UTIL.startMiniDFSClusterForTestHLog(namenodePort);
433       cluster.waitActive();
434       fs = cluster.getFileSystem();
435       LOG.info("STARTED second instance.");
436     }
437 
438     // set the lease period to be 1 second so that the
439     // namenode triggers lease recovery upon append request
440     Method setLeasePeriod = cluster.getClass()
441       .getDeclaredMethod("setLeasePeriod", new Class[]{Long.TYPE, Long.TYPE});
442     setLeasePeriod.setAccessible(true);
443     setLeasePeriod.invoke(cluster,
444                           new Object[]{new Long(1000), new Long(1000)});
445     try {
446       Thread.sleep(1000);
447     } catch (InterruptedException e) {
448       LOG.info(e);
449     }
450     
451     // Now try recovering the log, like the HMaster would do
452     final FileSystem recoveredFs = fs;
453     final Configuration rlConf = conf;
454     
455     class RecoverLogThread extends Thread {
456       public Exception exception = null;
457       public void run() {
458           try {
459             FSUtils.getInstance(fs, rlConf)
460               .recoverFileLease(recoveredFs, walPath, rlConf);
461           } catch (IOException e) {
462             exception = e;
463           }
464       }
465     }
466 
467     RecoverLogThread t = new RecoverLogThread();
468     t.start();
469     // Timeout after 60 sec. Without correct patches, would be an infinite loop
470     t.join(60 * 1000);
471     if(t.isAlive()) {
472       t.interrupt();
473       throw new Exception("Timed out waiting for HLog.recoverLog()");
474     }
475 
476     if (t.exception != null)
477       throw t.exception;
478 
479     // Make sure you can read all the content
480     SequenceFile.Reader reader
481       = new SequenceFile.Reader(this.fs, walPath, this.conf);
482     int count = 0;
483     HLogKey key = HLogUtil.newKey(conf);
484     WALEdit val = new WALEdit();
485     while (reader.next(key, val)) {
486       count++;
487       assertTrue("Should be one KeyValue per WALEdit",
488                  val.getKeyValues().size() == 1);
489     }
490     assertEquals(total, count);
491     reader.close();
492   }

498   @Test
499   public void testEditAdd() throws IOException {
500     final int COL_COUNT = 10;
501     final byte [] tableName = Bytes.toBytes("tablename");
502     final byte [] row = Bytes.toBytes("row");
503     HLog.Reader reader = null;
504     HLog log = null;
505     try {
506       log = HLogFactory.createHLog(fs, hbaseDir, getName(), conf);
507       
508       // Write columns named 1, 2, 3, etc. and then values of single byte
509       // 1, 2, 3...
510       long timestamp = System.currentTimeMillis();
511       WALEdit cols = new WALEdit();
512       for (int i = 0; i < COL_COUNT; i++) {
513         cols.add(new KeyValue(row, Bytes.toBytes("column"),
514             Bytes.toBytes(Integer.toString(i)),
515           timestamp, new byte[] { (byte)(i + '0') }));
516       }
517       HRegionInfo info = new HRegionInfo(tableName,
518         row,Bytes.toBytes(Bytes.toString(row) + "1"), false);
519       HTableDescriptor htd = new HTableDescriptor();
520       htd.addFamily(new HColumnDescriptor("column"));
521 
522       log.append(info, tableName, cols, System.currentTimeMillis(), htd);
523       long logSeqId = log.startCacheFlush(info.getEncodedNameAsBytes());
524       log.completeCacheFlush(info.getEncodedNameAsBytes(), tableName, logSeqId,
525           info.isMetaRegion());
526       log.close();
527       Path filename = ((FSHLog) log).computeFilename();
528       log = null;
529       // Now open a reader on the log and assert append worked.
530       reader = HLogFactory.createReader(fs, filename, conf);
531       // Above we added all columns on a single row so we only read one
532       // entry in the below... thats why we have '1'.
533       for (int i = 0; i < 1; i++) {
534         HLog.Entry entry = reader.next(null);
535         if (entry == null) break;
536         HLogKey key = entry.getKey();
537         WALEdit val = entry.getEdit();
538         assertTrue(Bytes.equals(info.getEncodedNameAsBytes(), key.getEncodedRegionName()));
539         assertTrue(Bytes.equals(tableName, key.getTablename()));
540         KeyValue kv = val.getKeyValues().get(0);
541         assertTrue(Bytes.equals(row, kv.getRow()));
542         assertEquals((byte)(i + '0'), kv.getValue()[0]);
543         System.out.println(key + " " + val);
544       }
545       HLog.Entry entry = null;
546       while ((entry = reader.next(null)) != null) {
547         HLogKey key = entry.getKey();
548         WALEdit val = entry.getEdit();
549         // Assert only one more row... the meta flushed row.
550         assertTrue(Bytes.equals(info.getEncodedNameAsBytes(), key.getEncodedRegionName()));
551         assertTrue(Bytes.equals(tableName, key.getTablename()));
552         KeyValue kv = val.getKeyValues().get(0);
553         assertTrue(Bytes.equals(HLog.METAROW, kv.getRow()));
554         assertTrue(Bytes.equals(HLog.METAFAMILY, kv.getFamily()));
555         assertEquals(0, Bytes.compareTo(HLogUtil.COMPLETE_CACHE_FLUSH,
556           val.getKeyValues().get(0).getValue()));
557         System.out.println(key + " " + val);
558       }
559     } finally {
560       if (log != null) {
561         log.closeAndDelete();
562       }
563       if (reader != null) {
564         reader.close();
565       }
566     }

678     LOG.info("testLogCleaning");
679     final byte [] tableName = Bytes.toBytes("testLogCleaning");
680     final byte [] tableName2 = Bytes.toBytes("testLogCleaning2");
681 
682     HLog log = HLogFactory.createHLog(fs, hbaseDir, 
683         getName(), conf);
684     try {
685       HRegionInfo hri = new HRegionInfo(tableName,
686           HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
687       HRegionInfo hri2 = new HRegionInfo(tableName2,
688           HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
689 
690       // Add a single edit and make sure that rolling won't remove the file
691       // Before HBASE-3198 it used to delete it
692       addEdits(log, hri, tableName, 1);
693       log.rollWriter();
694       assertEquals(1, ((FSHLog) log).getNumLogFiles());
695 
696       // See if there's anything wrong with more than 1 edit
697       addEdits(log, hri, tableName, 2);
698       log.rollWriter();
699       assertEquals(2, ((FSHLog) log).getNumLogFiles());
700 
701       // Now mix edits from 2 regions, still no flushing
702       addEdits(log, hri, tableName, 1);
703       addEdits(log, hri2, tableName2, 1);
704       addEdits(log, hri, tableName, 1);
705       addEdits(log, hri2, tableName2, 1);
706       log.rollWriter();
707       assertEquals(3, ((FSHLog) log).getNumLogFiles());
708 
709       // Flush the first region, we expect to see the first two files getting
710       // archived
711       long seqId = log.startCacheFlush(hri.getEncodedNameAsBytes());
712       log.completeCacheFlush(hri.getEncodedNameAsBytes(), tableName, seqId, false);
713       log.rollWriter();
714       assertEquals(2, ((FSHLog) log).getNumLogFiles());
715 
716       // Flush the second region, which removes all the remaining output files
717       // since the oldest was completely flushed and the two others only contain
718       // flush information
719       seqId = log.startCacheFlush(hri2.getEncodedNameAsBytes());
720       log.completeCacheFlush(hri2.getEncodedNameAsBytes(), tableName2, seqId, false);
721       log.rollWriter();
722       assertEquals(0, ((FSHLog) log).getNumLogFiles());
723     } finally {
724       if (log != null) log.closeAndDelete();
725     }
