48  @Category(MediumTests.class)
49  public class TestPriorityRpc {
50    static HRegionServer regionServer = null;
51    static QosFunction qosFunction = null;
52    @BeforeClass
53    public static void onetimeSetup() {
54      Configuration conf = HBaseConfiguration.create();
55      regionServer =
56          HRegionServer.constructRegionServer(HRegionServer.class, conf);
57      qosFunction = regionServer.getQosFunction();
58    }
59    @Test
60    public void testQosFunctionForMeta() throws IOException {
61      qosFunction = regionServer.getQosFunction();
62      RpcRequestBody.Builder rpcRequestBuilder = RpcRequestBody.newBuilder();
63      //create a rpc request that has references to META region and also
64      //uses one of the known argument classes (known argument classes are
65      //listed in HRegionServer.QosFunction.knownArgumentClasses)
66      rpcRequestBuilder = RpcRequestBody.newBuilder();
67      rpcRequestBuilder.setMethodName("foo");
68  
69      GetRequest.Builder getRequestBuilder = GetRequest.newBuilder();
70      RegionSpecifier.Builder regionSpecifierBuilder = RegionSpecifier.newBuilder();
71      regionSpecifierBuilder.setType(RegionSpecifierType.REGION_NAME);
72      ByteString name =
73          ByteString.copyFrom(HRegionInfo.FIRST_META_REGIONINFO.getRegionName());
74      regionSpecifierBuilder.setValue(name);
75      RegionSpecifier regionSpecifier = regionSpecifierBuilder.build();
76      getRequestBuilder.setRegion(regionSpecifier);
77      Get.Builder getBuilder = Get.newBuilder();
78      getBuilder.setRow(ByteString.copyFrom("somerow".getBytes()));
79      getRequestBuilder.setGet(getBuilder.build());
80      rpcRequestBuilder.setRequest(getRequestBuilder.build().toByteString());
81      rpcRequestBuilder.setRequestClassName(GetRequest.class.getCanonicalName());
82      RpcRequestBody rpcRequest = rpcRequestBuilder.build();
83      HRegion mockRegion = Mockito.mock(HRegion.class);
84      HRegionServer mockRS = Mockito.mock(HRegionServer.class);
85      HRegionInfo mockRegionInfo = Mockito.mock(HRegionInfo.class);
86      Mockito.when(mockRS.getRegion((RegionSpecifier)Mockito.any())).thenReturn(mockRegion);
87      Mockito.when(mockRegion.getRegionInfo()).thenReturn(mockRegionInfo);
88      Mockito.when(mockRegionInfo.isMetaRegion()).thenReturn(true);
89      qosFunction.setRegionServer(mockRS);
90      assertTrue (qosFunction.apply(rpcRequest) == HConstants.HIGH_QOS);
91    }

95      //The request is not using any of the

107   @Test
108   public void testQosFunctionForScanMethod() throws IOException {
109     RpcRequestBody.Builder rpcRequestBuilder = RpcRequestBody.newBuilder();
110     rpcRequestBuilder.setMethodName("scan");
111 
112     //build an empty scan request
113     ScanRequest.Builder scanBuilder = ScanRequest.newBuilder();
114     ByteString requestBody = scanBuilder.build().toByteString();
115     rpcRequestBuilder.setRequest(requestBody);
116     RpcRequestBody rpcRequest = rpcRequestBuilder.build();
117     assertTrue (qosFunction.apply(rpcRequest) == HConstants.NORMAL_QOS);
118 
119     //build a scan request with scannerID
120     scanBuilder = ScanRequest.newBuilder();
121     scanBuilder.setScannerId(12345);
122     requestBody = scanBuilder.build().toByteString();
123     rpcRequestBuilder.setRequest(requestBody);
124     rpcRequestBuilder.setRequestClassName(ScanRequest.class.getCanonicalName());
125     rpcRequest = rpcRequestBuilder.build();
126     //mock out a high priority type handling and see the QoS returned
127     HRegionServer mockRS = Mockito.mock(HRegionServer.class);
128     RegionScanner mockRegionScanner = Mockito.mock(RegionScanner.class);
129     HRegionInfo mockRegionInfo = Mockito.mock(HRegionInfo.class);
130     HRegion mockRegion = Mockito.mock(HRegion.class);
131     Mockito.when(mockRS.getScanner(12345)).thenReturn(mockRegionScanner);
132     Mockito.when(mockRegionScanner.getRegionInfo()).thenReturn(mockRegionInfo);
133     Mockito.when(mockRS.getRegion((RegionSpecifier)Mockito.any())).thenReturn(mockRegion);
134     Mockito.when(mockRegion.getRegionInfo()).thenReturn(mockRegionInfo);
135     Mockito.when(mockRegionInfo.isMetaRegion()).thenReturn(true);
136 
137     qosFunction.setRegionServer(mockRS);
138 
139     assertTrue (qosFunction.apply(rpcRequest) == HConstants.HIGH_QOS);
140 
141     //the same as above but with non-meta region
142     Mockito.when(mockRegionInfo.isMetaRegion()).thenReturn(false);
143     assertTrue (qosFunction.apply(rpcRequest) == HConstants.NORMAL_QOS);
144   }
