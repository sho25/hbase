19  package org.apache.hadoop.hbase.catalog;

21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  import org.apache.hadoop.hbase.*;
24  import org.apache.hadoop.hbase.zookeeper.RootRegionTracker;
25  import org.apache.hadoop.hbase.zookeeper.ZooKeeperWatcher;
26  import org.apache.zookeeper.KeeperException;
27  import org.junit.Test;
28  import org.junit.experimental.categories.Category;
29  
30  /**
31   * Do {@link CatalogTracker} tests on running cluster.
32   */
33  @Category(LargeTests.class)
34  public class TestCatalogTrackerOnCluster {
35    private static final HBaseTestingUtility UTIL = new HBaseTestingUtility();
36    private static final Log LOG =
37      LogFactory.getLog(TestCatalogTrackerOnCluster.class);
38  
39    /**
40     * @throws Exception 
41     * @see {https://issues.apache.org/jira/browse/HBASE-3445}
42     */
43    @Test public void testBadOriginalRootLocation() throws Exception {
44      UTIL.getConfiguration().setInt("ipc.socket.timeout", 3000);
45      // Launch cluster so it does bootstrapping.
46      UTIL.startMiniCluster();
47      // Shutdown hbase.
48      UTIL.shutdownMiniHBaseCluster();
49      // Mess with the root location in the running zk.  Set it to be nonsense.
50      ZooKeeperWatcher zookeeper = new ZooKeeperWatcher(UTIL.getConfiguration(),
51        "Bad Root Location Writer", new Abortable() {
52          @Override
53          public void abort(String why, Throwable e) {
54            LOG.error("Abort was called on 'bad root location writer'", e);
55          }
56          
57          @Override
58          public boolean isAborted() {
59            return false;
60          }
61      });
62      ServerName nonsense =
63        new ServerName("example.org", 1234, System.currentTimeMillis());
64      RootRegionTracker.setRootLocation(zookeeper, nonsense);
65  
66      // Bring back up the hbase cluster.  See if it can deal with nonsense root
67      // location. The cluster should start and be fully available.
68      UTIL.startMiniHBaseCluster(1, 1);
69  
70      // if we can create a table, it's a good sign that it's working
71      UTIL.createTable(
72        getClass().getSimpleName().getBytes(), "family".getBytes());
73  
74      UTIL.shutdownMiniCluster();
75    }
76  
77  }
