160   @Test
161   public void testLogRoll() throws Exception {
162     long seq = 0;
163     long baseline = 1000;
164     long time = baseline;
165     KeyValue kv = new KeyValue(r1, f1, r1);
166     WALEdit edit = new WALEdit();
167     edit.add(kv);
168 
169     List<WALActionsListener> listeners = new ArrayList<WALActionsListener>();
170     listeners.add(replication);
171     HLog hlog = HLogFactory.createHLog(fs, utility.getDataTestDir(), logName,
172         conf, listeners, URLEncoder.encode("regionserver:60020", "UTF8"));
173 
174     manager.init();
175     HTableDescriptor htd = new HTableDescriptor();
176     htd.addFamily(new HColumnDescriptor(f1));
177     // Testing normal log rolling every 20
178     for(long i = 1; i < 101; i++) {
179       if(i > 1 && i % 20 == 0) {
180         hlog.rollWriter();
181       }
182       LOG.info(i);
183       HLogKey key = new HLogKey(hri.getRegionName(), test, seq++,
184           System.currentTimeMillis(), HConstants.DEFAULT_CLUSTER_ID);
185       hlog.append(hri, key, edit, htd, true);
186     }
187 
188     // Simulate a rapid insert that's followed
189     // by a report that's still not totally complete (missing last one)
190     LOG.info(baseline + " and " + time);
191     baseline += 101;
192     time = baseline;
193     LOG.info(baseline + " and " + time);
194 
195     for (int i = 0; i < 3; i++) {
196       HLogKey key = new HLogKey(hri.getRegionName(), test, seq++,
197           System.currentTimeMillis(), HConstants.DEFAULT_CLUSTER_ID);
198       hlog.append(hri, key, edit, htd, true);
199     }
200 
201     assertEquals(6, manager.getHLogs().get(slaveId).size());
202 
203     hlog.rollWriter();
204 
205     manager.logPositionAndCleanOldLogs(manager.getSources().get(0).getCurrentPath(),
206         "1", 0, false, false);
207 
208     HLogKey key = new HLogKey(hri.getRegionName(), test, seq++,
209         System.currentTimeMillis(), HConstants.DEFAULT_CLUSTER_ID);
210     hlog.append(hri, key, edit, htd, true);
211 
212     assertEquals(1, manager.getHLogs().size());
213 
214 
215     // TODO Need a case with only 2 HLogs and we only want to delete the first one
216   }
