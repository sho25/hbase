136   @Test(timeout=300000)
137   public void testCyclicReplication() throws Exception {
138     LOG.info("testCyclicReplication");
139     utility1.startMiniCluster();
140     utility2.startMiniCluster();
141     utility3.startMiniCluster();
142     ReplicationAdmin admin1 = new ReplicationAdmin(conf1);
143     ReplicationAdmin admin2 = new ReplicationAdmin(conf2);
144     ReplicationAdmin admin3 = new ReplicationAdmin(conf3);
145 
146     new HBaseAdmin(conf1).createTable(table);
147     new HBaseAdmin(conf2).createTable(table);
148     new HBaseAdmin(conf3).createTable(table);
149     HTable htable1 = new HTable(conf1, tableName);
150     htable1.setWriteBufferSize(1024);
151     HTable htable2 = new HTable(conf2, tableName);
152     htable2.setWriteBufferSize(1024);
153     HTable htable3 = new HTable(conf3, tableName);
154     htable3.setWriteBufferSize(1024);
155     
156     admin1.addPeer("1", utility2.getClusterKey());
157     admin2.addPeer("1", utility3.getClusterKey());
158     admin3.addPeer("1", utility1.getClusterKey());
159 
160     // put "row" and wait 'til it got around
161     putAndWait(row, famName, htable1, htable3);
162     // it should have passed through table2
163     check(row,famName,htable2);
164 
165     putAndWait(row1, famName, htable2, htable1);
166     check(row,famName,htable3);
167     putAndWait(row2, famName, htable3, htable2);
168     check(row,famName,htable1);
169     
170     deleteAndWait(row,htable1,htable3);
171     deleteAndWait(row1,htable2,htable1);
172     deleteAndWait(row2,htable3,htable2);
173 
174     assertEquals("Puts were replicated back ", 3, getCount(htable1, put));
175     assertEquals("Puts were replicated back ", 3, getCount(htable2, put));
176     assertEquals("Puts were replicated back ", 3, getCount(htable3, put));
177     assertEquals("Deletes were replicated back ", 3, getCount(htable1, delete));
178     assertEquals("Deletes were replicated back ", 3, getCount(htable2, delete));
179     assertEquals("Deletes were replicated back ", 3, getCount(htable3, delete));
180     utility3.shutdownMiniCluster();
181     utility2.shutdownMiniCluster();
182     utility1.shutdownMiniCluster();
183   }

185   /**
186    * Add a row to a table in each cluster, check it's replicated,
187    * delete it, check's gone
188    * Also check the puts and deletes are not replicated back to
189    * the originating cluster.
190    */
191   @Test(timeout=300000)
192   public void testSimplePutDelete() throws Exception {
193     LOG.info("testSimplePutDelete");
194     utility1.startMiniCluster();
195     utility2.startMiniCluster();
196 
197     ReplicationAdmin admin1 = new ReplicationAdmin(conf1);
198     ReplicationAdmin admin2 = new ReplicationAdmin(conf2);
199 
200     new HBaseAdmin(conf1).createTable(table);
201     new HBaseAdmin(conf2).createTable(table);
202     HTable htable1 = new HTable(conf1, tableName);
203     htable1.setWriteBufferSize(1024);
204     HTable htable2 = new HTable(conf2, tableName);
205     htable2.setWriteBufferSize(1024);
206 
207     // set M-M
208     admin1.addPeer("1", utility2.getClusterKey());
209     admin2.addPeer("1", utility1.getClusterKey());
210 
211     // add rows to both clusters,
212     // make sure they are both replication
213     putAndWait(row, famName, htable1, htable2);
214     putAndWait(row1, famName, htable2, htable1);
215 
216     // make sure "row" did not get replicated back.
217     assertEquals("Puts were replicated back ", 2, getCount(htable1, put));
218 
219     // delete "row" and wait
220     deleteAndWait(row, htable1, htable2);
221 
222     // make the 2nd cluster replicated back
223     assertEquals("Puts were replicated back ", 2, getCount(htable2, put));
224 
225     deleteAndWait(row1, htable2, htable1);
226 
227     assertEquals("Deletes were replicated back ", 2, getCount(htable1, delete));
228     utility2.shutdownMiniCluster();
229     utility1.shutdownMiniCluster();
230   }

232   private int getCount(HTable t, byte[] type)  throws IOException {
233     Get test = new Get(row);
234     test.setAttribute("count", new byte[]{});
235     Result res = t.get(test);
236     return Bytes.toInt(res.getValue(count, type));

239   private void deleteAndWait(byte[] row, HTable source, HTable target)
240   throws Exception {
241     Delete del = new Delete(row);
242     source.delete(del);
243 
244     Get get = new Get(row);
245     for (int i = 0; i < NB_RETRIES; i++) {
246       if (i==NB_RETRIES-1) {
247         fail("Waited too much time for del replication");
248       }
249       Result res = target.get(get);
250       if (res.size() >= 1) {
251         LOG.info("Row not deleted");
252         Thread.sleep(SLEEP_TIME);
253       } else {
254         break;
255       }
256     }
257   }
258 
259   private void check(byte[] row, byte[] fam, HTable t) throws IOException {
260     Get get = new Get(row);
261     Result res = t.get(get);
262     if (res.size() == 0) {
263       fail("Row is missing");
264     }
265   }
266 
267   private void putAndWait(byte[] row, byte[] fam, HTable source, HTable target)
268   throws Exception {
269     Put put = new Put(row);
270     put.add(fam, row, row);
271     source.put(put);
272 
273     Get get = new Get(row);
274     for (int i = 0; i < NB_RETRIES; i++) {
275       if (i==NB_RETRIES-1) {
276         fail("Waited too much time for put replication");
277       }
278       Result res = target.get(get);
279       if (res.size() == 0) {
280         LOG.info("Row not available");
281         Thread.sleep(SLEEP_TIME);
282       } else {
283         assertArrayEquals(res.value(), row);
284         break;
285       }
286     }

289   /**
290    * Use a coprocessor to count puts and deletes.
291    * as KVs would be replicated back with the same timestamp
292    * there is otherwise no way to count them.
293    */
294   public static class CoprocessorCounter extends BaseRegionObserver {
295     private int nCount = 0;
296     private int nDelete = 0;
297 
298     @Override
299     public void prePut(final ObserverContext<RegionCoprocessorEnvironment> e,
300         final Put put, final WALEdit edit, 
301         final boolean writeToWAL)
302         throws IOException {
303       nCount++;
304     }
305     @Override
306     public void postDelete(final ObserverContext<RegionCoprocessorEnvironment> c,
307         final Delete delete, final WALEdit edit, 
308         final boolean writeToWAL)
309         throws IOException {
310       nDelete++;
311     }
312     @Override
313     public void preGet(final ObserverContext<RegionCoprocessorEnvironment> c,
314         final Get get, final List<KeyValue> result) throws IOException {
315       if (get.getAttribute("count") != null) {
316         result.clear();
317         // order is important!
318         result.add(new KeyValue(count, count, delete, Bytes.toBytes(nDelete)));
319         result.add(new KeyValue(count, count, put, Bytes.toBytes(nCount)));
320         c.bypass();
321       }
322     }
323   }
